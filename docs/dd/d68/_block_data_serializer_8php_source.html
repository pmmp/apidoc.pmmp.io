38 public static function readFacing(int $raw) : int{
39 $result = [ //this is for redundancy, for when/if the FACING constant values change
40 0 => Facing::DOWN,
41 1 => Facing::UP,
42 2 => Facing::NORTH,
43 3 => Facing::SOUTH,
44 4 => Facing::WEST,
45 5 => Facing::EAST
46 ][$raw] ?? null;
47 if($result === null){
48 throw new InvalidBlockStateException("Invalid facing $raw");
49 }
50 return $result;
51 }
