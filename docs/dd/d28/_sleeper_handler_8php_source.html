105 public function processNotifications() : void{
106 while(true){
107 $notifierIds = $this->sharedObject->synchronized(function() : array{
108 $notifierIds = [];
109 foreach($this->sharedObject as $notifierId => $_){
110 $notifierIds[$notifierId] = $notifierId;
111 unset($this->sharedObject[$notifierId]);
112 }
113 return $notifierIds;
114 });
115 if(count($notifierIds) === 0){
116 break;
117 }
118 foreach($notifierIds as $notifierId){
119 if(!isset($this->notifiers[$notifierId])){
120 //a previously-removed notifier might still be sending notifications; ignore them
121 continue;
122 }
123 $this->notifiers[$notifierId]();
124 }
125 }
