336 public function handleEncoded(string $payload) : void{
337 if(!$this->connected){
338 return;
339 }
340
341 if($this->cipher !== null){
342 Timings::$playerNetworkReceiveDecrypt->startTiming();
343 try{
344 $payload = $this->cipher->decrypt($payload);
345 }catch(DecryptionException $e){
346 $this->logger->debug("Encrypted packet: " . base64_encode($payload));
347 throw PacketHandlingException::wrap($e, "Packet decryption error");
348 }finally{
349 Timings::$playerNetworkReceiveDecrypt->stopTiming();
350 }
351 }
352
353 if($this->enableCompression){
354 Timings::$playerNetworkReceiveDecompress->startTiming();
355 try{
356 $decompressed = $this->compressor->decompress($payload);
357 }catch(DecompressionException $e){
358 $this->logger->debug("Failed to decompress packet: " . base64_encode($payload));
359 throw PacketHandlingException::wrap($e, "Compressed packet batch decode error");
360 }finally{
361 Timings::$playerNetworkReceiveDecompress->stopTiming();
362 }
363 }else{
364 $decompressed = $payload;
365 }
366
367 try{
368 foreach((new PacketBatch($decompressed))->getPackets($this->packetPool, $this->packetSerializerContext, 500) as [$packet, $buffer]){
369 if($packet === null){
370 $this->logger->debug("Unknown packet: " . base64_encode($buffer));
371 throw new PacketHandlingException("Unknown packet received");
372 }
373 try{
374 $this->handleDataPacket($packet, $buffer);
375 }catch(PacketHandlingException $e){
376 $this->logger->debug($packet->getName() . ": " . base64_encode($buffer));
377 throw PacketHandlingException::wrap($e, "Error processing " . $packet->getName());
378 }
379 }
380 }catch(PacketDecodeException $e){
381 $this->logger->logException($e);
382 throw PacketHandlingException::wrap($e, "Packet batch decode error");
383 }
384 }

389 public function handleDataPacket(Packet $packet, string $buffer) : void{
390 if(!($packet instanceof ServerboundPacket)){
391 throw new PacketHandlingException("Unexpected non-serverbound packet");
392 }
393
394 $timings = Timings::getDecodeDataPacketTimings($packet);
395 $timings->startTiming();
396 try{
397 $stream = PacketSerializer::decoder($buffer, 0, $this->packetSerializerContext);
398 try{
399 $packet->decode($stream);
400 }catch(PacketDecodeException $e){
401 throw PacketHandlingException::wrap($e);
402 }
403 if(!$stream->feof()){
404 $remains = substr($stream->getBuffer(), $stream->getOffset());
405 $this->logger->debug("Still " . strlen($remains) . " bytes unread in " . $packet->getName() . ": " . bin2hex($remains));
406 }
407 }finally{
408 $timings->stopTiming();
409 }
410
411 $timings = Timings::getHandleDataPacketTimings($packet);
412 $timings->startTiming();
413 try{
414 //TODO: I'm not sure DataPacketReceiveEvent should be included in the handler timings, but it needs to be
415 //included for now to ensure the receivePacket timings are counted the way they were before
416 $ev = new DataPacketReceiveEvent($this, $packet);
417 $ev->call();
418 if(!$ev->isCancelled() && ($this->handler === null || !$packet->handle($this->handler))){
419 $this->logger->debug("Unhandled " . $packet->getName() . ": " . base64_encode($stream->getBuffer()));
420 }
421 }finally{
422 $timings->stopTiming();
423 }
424 }

426 public function sendDataPacket(ClientboundPacket $packet, bool $immediate = false) : bool{
427 //Basic safety restriction. TODO: improve this
428 if(!$this->loggedIn && !$packet->canBeSentBeforeLogin()){
429 throw new \InvalidArgumentException("Attempted to send " . get_class($packet) . " to " . $this->getDisplayName() . " too early");
430 }
431
432 $timings = Timings::getSendDataPacketTimings($packet);
433 $timings->startTiming();
434 try{
435 $ev = new DataPacketSendEvent([$this], [$packet]);
436 $ev->call();
437 if($ev->isCancelled()){
438 return false;
439 }
440
441 $this->addToSendBuffer($packet);
442 if($immediate){
443 $this->flushSendBuffer(true);
444 }
445
446 return true;
447 }finally{
448 $timings->stopTiming();
449 }

465 private function flushSendBuffer(bool $immediate = false) : void{
466 if(count($this->sendBuffer) > 0){
467 $syncMode = null; //automatic
468 if($immediate){
469 $syncMode = true;
470 }elseif($this->forceAsyncCompression){
471 $syncMode = false;
472 }
473
474 $batch = PacketBatch::fromPackets($this->packetSerializerContext, ...$this->sendBuffer);
475 if($this->enableCompression){
476 $promise = $this->server->prepareBatch($batch, $this->compressor, $syncMode);
477 }else{
478 $promise = new CompressBatchPromise();
479 $promise->resolve($batch->getBuffer());
480 }
481 $this->sendBuffer = [];
482 $this->queueCompressedNoBufferFlush($promise, $immediate);
483 }
484 }

499 private function queueCompressedNoBufferFlush(CompressBatchPromise $payload, bool $immediate = false) : void{
500 if($immediate){
501 //Skips all queues
502 $this->sendEncoded($payload->getResult(), true);
503 }else{
504 $this->compressedQueue->enqueue($payload);
505 $payload->onResolve(function(CompressBatchPromise $payload) : void{
506 if($this->connected && $this->compressedQueue->bottom() === $payload){
507 $this->compressedQueue->dequeue(); //result unused
508 $this->sendEncoded($payload->getResult());
509
510 while(!$this->compressedQueue->isEmpty()){
512 $current = $this->compressedQueue->bottom();
513 if($current->hasResult()){
514 $this->compressedQueue->dequeue();
515
516 $this->sendEncoded($current->getResult());
517 }else{
518 //can't send any more queued until this one is ready
519 break;
520 }
521 }
522 }
523 });
524 }

614 private function setAuthenticationStatus(bool $authenticated, bool $authRequired, ?string $error, ?string $clientPubKey) : void{
615 if(!$this->connected){
616 return;
617 }
618 if($error === null){
619 if($authenticated && !($this->info instanceof XboxLivePlayerInfo)){
620 $error = "Expected XUID but none found";
621 }elseif($clientPubKey === null){
622 $error = "Missing client public key"; //failsafe
623 }
624 }
625
626 if($error !== null){
627 $this->disconnect($this->server->getLanguage()->translate(KnownTranslationFactory::pocketmine_disconnect_invalidSession($this->server->getLanguage()->translateString($error))));
628
629 return;
630 }
631
632 $this->authenticated = $authenticated;
633
634 if(!$this->authenticated){
635 if($authRequired){
636 $this->disconnect(KnownTranslationKeys::DISCONNECTIONSCREEN_NOTAUTHENTICATED);
637 return;
638 }
639 if($this->info instanceof XboxLivePlayerInfo){
640 $this->logger->warning("Discarding unexpected XUID for non-authenticated player");
641 $this->info = $this->info->withoutXboxData();
642 }
643 }
644 $this->logger->debug("Xbox Live authenticated: " . ($this->authenticated ? "YES" : "NO"));
645
646 $checkXUID = $this->server->getConfigGroup()->getPropertyBool("player.verify-xuid", true);
647 $myXUID = $this->info instanceof XboxLivePlayerInfo ? $this->info->getXuid() : "";
648 $kickForXUIDMismatch = function(string $xuid) use ($checkXUID, $myXUID) : bool{
649 if($checkXUID && $myXUID !== $xuid){
650 $this->logger->debug("XUID mismatch: expected '$xuid', but got '$myXUID'");
651 //TODO: Longer term, we should be identifying playerdata using something more reliable, like XUID or UUID.
652 //However, that would be a very disruptive change, so this will serve as a stopgap for now.
653 //Side note: this will also prevent offline players hijacking XBL playerdata on online servers, since their
654 //XUID will always be empty.
655 $this->disconnect("XUID does not match (possible impersonation attempt)");
656 return true;
657 }
658 return false;
659 };
660
661 foreach($this->manager->getSessions() as $existingSession){
662 if($existingSession === $this){
663 continue;
664 }
665 $info = $existingSession->getPlayerInfo();
666 if($info !== null && (strcasecmp($info->getUsername(), $this->info->getUsername()) === 0 || $info->getUuid()->equals($this->info->getUuid()))){
667 if($kickForXUIDMismatch($info instanceof XboxLivePlayerInfo ? $info->getXuid() : "")){
668 return;
669 }
670 $ev = new PlayerDuplicateLoginEvent($this, $existingSession);
671 $ev->call();
672 if($ev->isCancelled()){
673 $this->disconnect($ev->getDisconnectMessage());
674 return;
675 }
676
677 $existingSession->disconnect($ev->getDisconnectMessage());
678 }
679 }
680
681 //TODO: make player data loading async
682 //TODO: we shouldn't be loading player data here at all, but right now we don't have any choice :(
683 $this->cachedOfflinePlayerData = $this->server->getOfflinePlayerData($this->info->getUsername());
684 if($checkXUID){
685 $recordedXUID = $this->cachedOfflinePlayerData !== null ? $this->cachedOfflinePlayerData->getTag("LastKnownXUID") : null;
686 if(!($recordedXUID instanceof StringTag)){
687 $this->logger->debug("No previous XUID recorded, no choice but to trust this player");
688 }elseif(!$kickForXUIDMismatch($recordedXUID->getValue())){
689 $this->logger->debug("XUID match");
690 }
691 }
692
693 if(EncryptionContext::$ENABLED){
694 $this->server->getAsyncPool()->submitTask(new PrepareEncryptionTask($clientPubKey, function(string $encryptionKey, string $handshakeJwt) : void{
695 if(!$this->connected){
696 return;
697 }
698 $this->sendDataPacket(ServerToClientHandshakePacket::create($handshakeJwt), true); //make sure this gets sent before encryption is enabled
699
