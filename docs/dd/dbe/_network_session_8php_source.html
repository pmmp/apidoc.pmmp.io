356 public function handleEncoded(string $payload) : void{
357 if(!$this->connected){
358 return;
359 }
360
361 if($this->incomingPacketBatchBudget <= 0){
362 $this->updatePacketBudget();
363 if($this->incomingPacketBatchBudget <= 0){
364 throw new PacketHandlingException("Receiving packets too fast");
365 }
366 }
367 $this->incomingPacketBatchBudget--;
368
369 if($this->cipher !== null){
370 Timings::$playerNetworkReceiveDecrypt->startTiming();
371 try{
372 $payload = $this->cipher->decrypt($payload);
373 }catch(DecryptionException $e){
374 $this->logger->debug("Encrypted packet: " . base64_encode($payload));
375 throw PacketHandlingException::wrap($e, "Packet decryption error");
376 }finally{
377 Timings::$playerNetworkReceiveDecrypt->stopTiming();
378 }
379 }
380
381 if($this->enableCompression){
382 Timings::$playerNetworkReceiveDecompress->startTiming();
383 try{
384 $decompressed = $this->compressor->decompress($payload);
385 }catch(DecompressionException $e){
386 $this->logger->debug("Failed to decompress packet: " . base64_encode($payload));
387 throw PacketHandlingException::wrap($e, "Compressed packet batch decode error");
388 }finally{
389 Timings::$playerNetworkReceiveDecompress->stopTiming();
390 }
391 }else{
392 $decompressed = $payload;
393 }
394
395 try{
396 foreach((new PacketBatch($decompressed))->getPackets($this->packetPool, $this->packetSerializerContext, 1300) as [$packet, $buffer]){
397 if($packet === null){
398 $this->logger->debug("Unknown packet: " . base64_encode($buffer));
399 throw new PacketHandlingException("Unknown packet received");
400 }
401 try{
402 $this->handleDataPacket($packet, $buffer);
403 }catch(PacketHandlingException $e){
404 $this->logger->debug($packet->getName() . ": " . base64_encode($buffer));
405 throw PacketHandlingException::wrap($e, "Error processing " . $packet->getName());
406 }
407 }
408 }catch(PacketDecodeException $e){
409 $this->logger->logException($e);
410 throw PacketHandlingException::wrap($e, "Packet batch decode error");
411 }
412 }

417 public function handleDataPacket(Packet $packet, string $buffer) : void{
418 if(!($packet instanceof ServerboundPacket)){
419 throw new PacketHandlingException("Unexpected non-serverbound packet");
420 }
421
422 $timings = Timings::getDecodeDataPacketTimings($packet);
423 $timings->startTiming();
424 try{
425 $stream = PacketSerializer::decoder($buffer, 0, $this->packetSerializerContext);
426 try{
427 $packet->decode($stream);
428 }catch(PacketDecodeException $e){
429 throw PacketHandlingException::wrap($e);
430 }
431 if(!$stream->feof()){
432 $remains = substr($stream->getBuffer(), $stream->getOffset());
433 $this->logger->debug("Still " . strlen($remains) . " bytes unread in " . $packet->getName() . ": " . bin2hex($remains));
434 }
435 }finally{
436 $timings->stopTiming();
437 }
438
439 $timings = Timings::getHandleDataPacketTimings($packet);
440 $timings->startTiming();
441 try{
442 //TODO: I'm not sure DataPacketReceiveEvent should be included in the handler timings, but it needs to be
443 //included for now to ensure the receivePacket timings are counted the way they were before
444 $ev = new DataPacketReceiveEvent($this, $packet);
445 $ev->call();
446 if(!$ev->isCancelled() && ($this->handler === null || !$packet->handle($this->handler))){
447 $this->logger->debug("Unhandled " . $packet->getName() . ": " . base64_encode($stream->getBuffer()));
448 }
449 }finally{
450 $timings->stopTiming();
451 }
452 }

454 public function sendDataPacket(ClientboundPacket $packet, bool $immediate = false) : bool{
455 if(!$this->connected){
456 return false;
457 }
458 //Basic safety restriction. TODO: improve this
459 if(!$this->loggedIn && !$packet->canBeSentBeforeLogin()){
460 throw new \InvalidArgumentException("Attempted to send " . get_class($packet) . " to " . $this->getDisplayName() . " too early");
461 }
462
463 $timings = Timings::getSendDataPacketTimings($packet);
464 $timings->startTiming();
465 try{
466 $ev = new DataPacketSendEvent([$this], [$packet]);
467 $ev->call();
468 if($ev->isCancelled()){
469 return false;
470 }
471
472 $this->addToSendBuffer($packet);
473 if($immediate){
474 $this->flushSendBuffer(true);
475 }
476
477 return true;
478 }finally{
479 $timings->stopTiming();
480 }

496 private function flushSendBuffer(bool $immediate = false) : void{
497 if(count($this->sendBuffer) > 0){
498 $syncMode = null; //automatic
499 if($immediate){
500 $syncMode = true;
501 }elseif($this->forceAsyncCompression){
502 $syncMode = false;
503 }
504
505 $batch = PacketBatch::fromPackets($this->packetSerializerContext, ...$this->sendBuffer);
506 if($this->enableCompression){
507 $promise = $this->server->prepareBatch($batch, $this->compressor, $syncMode);
508 }else{
509 $promise = new CompressBatchPromise();
510 $promise->resolve($batch->getBuffer());
511 }
512 $this->sendBuffer = [];
513 $this->queueCompressedNoBufferFlush($promise, $immediate);
514 }

530 private function queueCompressedNoBufferFlush(CompressBatchPromise $payload, bool $immediate = false) : void{
531 if($immediate){
532 //Skips all queues
533 $this->sendEncoded($payload->getResult(), true);
534 }else{
535 $this->compressedQueue->enqueue($payload);
536 $payload->onResolve(function(CompressBatchPromise $payload) : void{
537 if($this->connected && $this->compressedQueue->bottom() === $payload){
538 $this->compressedQueue->dequeue(); //result unused
539 $this->sendEncoded($payload->getResult());
540
541 while(!$this->compressedQueue->isEmpty()){
543 $current = $this->compressedQueue->bottom();
544 if($current->hasResult()){
545 $this->compressedQueue->dequeue();
546
547 $this->sendEncoded($current->getResult());
548 }else{
549 //can't send any more queued until this one is ready
550 break;
551 }
552 }
553 }
554 });
555 }

642 private function setAuthenticationStatus(bool $authenticated, bool $authRequired, ?string $error, ?string $clientPubKey) : void{
643 if(!$this->connected){
644 return;
645 }
646 if($error === null){
647 if($authenticated && !($this->info instanceof XboxLivePlayerInfo)){
648 $error = "Expected XUID but none found";
649 }elseif($clientPubKey === null){
650 $error = "Missing client public key"; //failsafe
651 }
652 }
653
654 if($error !== null){
655 $this->disconnect($this->server->getLanguage()->translate(KnownTranslationFactory::pocketmine_disconnect_invalidSession($this->server->getLanguage()->translateString($error))));
656
657 return;
658 }
659
660 $this->authenticated = $authenticated;
661
662 if(!$this->authenticated){
663 if($authRequired){
664 $this->disconnect(KnownTranslationKeys::DISCONNECTIONSCREEN_NOTAUTHENTICATED);
665 return;
666 }
667 if($this->info instanceof XboxLivePlayerInfo){
668 $this->logger->warning("Discarding unexpected XUID for non-authenticated player");
669 $this->info = $this->info->withoutXboxData();
670 }
671 }
672 $this->logger->debug("Xbox Live authenticated: " . ($this->authenticated ? "YES" : "NO"));
673
674 $checkXUID = $this->server->getConfigGroup()->getPropertyBool("player.verify-xuid", true);
675 $myXUID = $this->info instanceof XboxLivePlayerInfo ? $this->info->getXuid() : "";
676 $kickForXUIDMismatch = function(string $xuid) use ($checkXUID, $myXUID) : bool{
677 if($checkXUID && $myXUID !== $xuid){
678 $this->logger->debug("XUID mismatch: expected '$xuid', but got '$myXUID'");
679 //TODO: Longer term, we should be identifying playerdata using something more reliable, like XUID or UUID.
680 //However, that would be a very disruptive change, so this will serve as a stopgap for now.
681 //Side note: this will also prevent offline players hijacking XBL playerdata on online servers, since their
682 //XUID will always be empty.
683 $this->disconnect("XUID does not match (possible impersonation attempt)");
684 return true;
685 }
686 return false;
687 };
688
689 foreach($this->manager->getSessions() as $existingSession){
690 if($existingSession === $this){
691 continue;
692 }
693 $info = $existingSession->getPlayerInfo();
694 if($info !== null && (strcasecmp($info->getUsername(), $this->info->getUsername()) === 0 || $info->getUuid()->equals($this->info->getUuid()))){
695 if($kickForXUIDMismatch($info instanceof XboxLivePlayerInfo ? $info->getXuid() : "")){
696 return;
697 }
698 $ev = new PlayerDuplicateLoginEvent($this, $existingSession);
699 $ev->call();
700 if($ev->isCancelled()){
701 $this->disconnect($ev->getDisconnectMessage());
702 return;
703 }
704
705 $existingSession->disconnect($ev->getDisconnectMessage());
706 }
707 }
708
709 //TODO: make player data loading async
710 //TODO: we shouldn't be loading player data here at all, but right now we don't have any choice :(

1141 private function updatePacketBudget() : void{
1142 $nowNs = hrtime(true);
1143 $timeSinceLastUpdateNs = $nowNs - $this->lastPacketBudgetUpdateTimeNs;
1144 if($timeSinceLastUpdateNs > 50_000_000){
1145 $ticksSinceLastUpdate = intdiv($timeSinceLastUpdateNs, 50_000_000);
1146 /*
1147 * If the server takes an abnormally long time to process a tick, add the budget for time difference to
1148 * compensate. This extra budget may be very large, but it will disappear the next time a normal update
1149 * occurs. This ensures that backlogs during a large lag spike don't cause everyone to get kicked.
1150 * As long as all the backlogged packets are processed before the next tick, everything should be OK for
1151 * clients behaving normally.
1152 */
1153 $this->incomingPacketBatchBudget = min($this->incomingPacketBatchBudget, self::INCOMING_PACKET_BATCH_MAX_BUDGET) + (self::INCOMING_PACKET_BATCH_PER_TICK * 2 * $ticksSinceLastUpdate);
1154 $this->lastPacketBudgetUpdateTimeNs = $nowNs;
1155 }
