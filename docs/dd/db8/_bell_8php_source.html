122 if($face === Facing::UP){

124 return false;
125 }
126 if($player !== null){

132 return false;
133 }
134 $this->setAttachmentType(BellAttachmentType::CEILING());

136 $this->setFacing($face);
137 if($this->canBeSupportedBy($tx->fetchBlock($this->position->getSide(Facing::opposite($face))), $face)){
138 $this->setAttachmentType(BellAttachmentType::ONE_WALL());

140 return false;

160 public function onInteract(Item $item, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
161 if($player !== null){
162 $faceHit = Facing::opposite($player->getHorizontalFacing());
163 if(
164 $this->attachmentType->equals(BellAttachmentType::CEILING()) ||
165 ($this->attachmentType->equals(BellAttachmentType::FLOOR()) && Facing::axis($faceHit) === Facing::axis($this->facing)) ||
166 (
167 ($this->attachmentType->equals(BellAttachmentType::ONE_WALL()) || $this->attachmentType->equals(BellAttachmentType::TWO_WALLS())) &&
168 ($faceHit === Facing::rotateY($this->facing, false) || $faceHit === Facing::rotateY($this->facing, true))
169 )
170 ){
171 $this->ring($faceHit);
172 return true;
173 }
174 }
175
176 return false;
