57 protected function recalculateCollisionBoxes() : array{
58 if($this->attachmentType->equals(BellAttachmentType::FLOOR())){
59 return [
60 AxisAlignedBB::one()->squash(Facing::axis($this->facing), 1 / 4)->trim(Facing::UP, 3 / 16)

62 }
63 if($this->attachmentType->equals(BellAttachmentType::CEILING())){
64 return [
65 AxisAlignedBB::one()->contract(1 / 4, 0, 1 / 4)->trim(Facing::DOWN, 1 / 4)
66 ];
67 }
68
69 $box = AxisAlignedBB::one()

71 ->trim(Facing::UP, 1 / 16)
72 ->trim(Facing::DOWN, 1 / 4);
73
74 return [
75 $this->attachmentType->equals(BellAttachmentType::ONE_WALL()) ? $box->trim($this->facing, 3 / 16) : $box
76 ];

95 public function place(BlockTransaction $tx, Item $item, Block $blockReplace, Block $blockClicked, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
96 if(!$this->canBeSupportedAt($blockReplace, Facing::opposite($face))){
97 return false;
98 }
99 if($face === Facing::UP){
100 if($player !== null){
101 $this->setFacing(Facing::opposite($player->getHorizontalFacing()));
102 }
103 $this->setAttachmentType(BellAttachmentType::FLOOR());
104 }elseif($face === Facing::DOWN){
105 $this->setAttachmentType(BellAttachmentType::CEILING());
106 }else{
107 $this->setFacing($face);
108 $this->setAttachmentType(
109 $this->canBeSupportedAt($blockReplace, $face) ?
110 BellAttachmentType::TWO_WALLS() :
111 BellAttachmentType::ONE_WALL()
112 );
113 }
114 return parent::place($tx, $item, $blockReplace, $blockClicked, $face, $clickVector, $player);
115 }

117 public function onNearbyBlockChange() : void{
118 foreach(match($this->attachmentType){
119 BellAttachmentType::CEILING() => [Facing::UP],
120 BellAttachmentType::FLOOR() => [Facing::DOWN],
121 BellAttachmentType::ONE_WALL() => [Facing::opposite($this->facing)],
122 BellAttachmentType::TWO_WALLS() => [$this->facing, Facing::opposite($this->facing)],
123 default => throw new AssumptionFailedError("All cases of BellAttachmentType must be handled")
124 } as $supportBlockDirection){
125 if(!$this->canBeSupportedAt($this, $supportBlockDirection)){
126 $this->position->getWorld()->useBreakOn($this->position);

152 $world = $this->position->getWorld();
153 $world->addSound($this->position, new BellRingSound());
154 $tile = $world->getTile($this->position);
155 if($tile instanceof TileBell){
156 $world->broadcastPacketToViewers($this->position, $tile->createFakeUpdatePacket($faceHit));

160 private function isValidFaceToRing(int $faceHit) : bool{
161 return match($this->attachmentType){
162 BellAttachmentType::CEILING() => true,
163 BellAttachmentType::FLOOR() => Facing::axis($faceHit) === Facing::axis($this->facing),
164 BellAttachmentType::ONE_WALL(), BellAttachmentType::TWO_WALLS() => $faceHit === Facing::rotateY($this->facing, false) || $faceHit === Facing::rotateY($this->facing, true),
165 default => throw new AssumptionFailedError("All cases of BellAttachmentType must be handled")
166 };
