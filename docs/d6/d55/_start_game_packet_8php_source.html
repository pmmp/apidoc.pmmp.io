137 protected function decodePayload(PacketSerializer $in) : void{
138 $this->actorUniqueId = $in->getActorUniqueId();
139 $this->actorRuntimeId = $in->getActorRuntimeId();
140 $this->playerGamemode = $in->getVarInt();
141
142 $this->playerPosition = $in->getVector3();
143
144 $this->pitch = $in->getLFloat();
145 $this->yaw = $in->getLFloat();
146
147 $this->levelSettings = LevelSettings::read($in);
148
149 $this->levelId = $in->getString();
150 $this->worldName = $in->getString();
151 $this->premiumWorldTemplateId = $in->getString();
152 $this->isTrial = $in->getBool();
153 $this->playerMovementSettings = PlayerMovementSettings::read($in);
154 $this->currentTick = $in->getLLong();
155
156 $this->enchantmentSeed = $in->getVarInt();
157
158 $this->blockPalette = [];
159 for($i = 0, $len = $in->getUnsignedVarInt(); $i < $len; ++$i){
160 $blockName = $in->getString();
161 $state = $in->getNbtCompoundRoot();
162 $this->blockPalette[] = new BlockPaletteEntry($blockName, new CacheableNbt($state));
163 }
164
165 $this->itemTable = [];
166 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
167 $stringId = $in->getString();
168 $numericId = $in->getSignedLShort();
169 $isComponentBased = $in->getBool();
170
171 $this->itemTable[] = new ItemTypeEntry($stringId, $numericId, $isComponentBased);
172 }
173
174 $this->multiplayerCorrelationId = $in->getString();
175 $this->enableNewInventorySystem = $in->getBool();
176 $this->serverSoftwareVersion = $in->getString();
177 $this->playerActorProperties = new CacheableNbt($in->getNbtCompoundRoot());
178 $this->blockPaletteChecksum = $in->getLLong();
179 $this->worldTemplateId = $in->getUUID();
180 $this->enableClientSideChunkGeneration = $in->getBool();
181 }

183 protected function encodePayload(PacketSerializer $out) : void{
184 $out->putActorUniqueId($this->actorUniqueId);
185 $out->putActorRuntimeId($this->actorRuntimeId);
186 $out->putVarInt($this->playerGamemode);
187
188 $out->putVector3($this->playerPosition);
189
190 $out->putLFloat($this->pitch);
191 $out->putLFloat($this->yaw);
192
193 $this->levelSettings->write($out);
194
195 $out->putString($this->levelId);
196 $out->putString($this->worldName);
197 $out->putString($this->premiumWorldTemplateId);
198 $out->putBool($this->isTrial);
199 $this->playerMovementSettings->write($out);
200 $out->putLLong($this->currentTick);
201
202 $out->putVarInt($this->enchantmentSeed);
203
204 $out->putUnsignedVarInt(count($this->blockPalette));
205 foreach($this->blockPalette as $entry){
206 $out->putString($entry->getName());
207 $out->put($entry->getStates()->getEncodedNbt());
208 }
209
210 $out->putUnsignedVarInt(count($this->itemTable));
211 foreach($this->itemTable as $entry){
212 $out->putString($entry->getStringId());
213 $out->putLShort($entry->getNumericId());
214 $out->putBool($entry->isComponentBased());
215 }
216
217 $out->putString($this->multiplayerCorrelationId);
218 $out->putBool($this->enableNewInventorySystem);
219 $out->putString($this->serverSoftwareVersion);
220 $out->put($this->playerActorProperties->getEncodedNbt());
221 $out->putLLong($this->blockPaletteChecksum);
222 $out->putUUID($this->worldTemplateId);
223 $out->putBool($this->enableClientSideChunkGeneration);
224 }
