133 protected function decodePayload(PacketSerializer $in) : void{
134 $this->actorUniqueId = $in->getActorUniqueId();
135 $this->actorRuntimeId = $in->getActorRuntimeId();
136 $this->playerGamemode = $in->getVarInt();
137
138 $this->playerPosition = $in->getVector3();
139
140 $this->pitch = $in->getLFloat();
141 $this->yaw = $in->getLFloat();
142
143 $this->levelSettings = LevelSettings::read($in);
144
145 $this->levelId = $in->getString();
146 $this->worldName = $in->getString();
147 $this->premiumWorldTemplateId = $in->getString();
148 $this->isTrial = $in->getBool();
149 $this->playerMovementSettings = PlayerMovementSettings::read($in);
150 $this->currentTick = $in->getLLong();
151
152 $this->enchantmentSeed = $in->getVarInt();
153
154 $this->blockPalette = [];
155 for($i = 0, $len = $in->getUnsignedVarInt(); $i < $len; ++$i){
156 $blockName = $in->getString();
157 $state = $in->getNbtCompoundRoot();
158 $this->blockPalette[] = new BlockPaletteEntry($blockName, new CacheableNbt($state));
159 }
160
161 $this->multiplayerCorrelationId = $in->getString();
162 $this->enableNewInventorySystem = $in->getBool();
163 $this->serverSoftwareVersion = $in->getString();
164 $this->playerActorProperties = new CacheableNbt($in->getNbtCompoundRoot());
165 $this->blockPaletteChecksum = $in->getLLong();
166 $this->worldTemplateId = $in->getUUID();
167 $this->enableClientSideChunkGeneration = $in->getBool();
168 $this->blockNetworkIdsAreHashes = $in->getBool();
169 $this->networkPermissions = NetworkPermissions::decode($in);
170 }

172 protected function encodePayload(PacketSerializer $out) : void{
173 $out->putActorUniqueId($this->actorUniqueId);
174 $out->putActorRuntimeId($this->actorRuntimeId);
175 $out->putVarInt($this->playerGamemode);
176
177 $out->putVector3($this->playerPosition);
178
179 $out->putLFloat($this->pitch);
180 $out->putLFloat($this->yaw);
181
182 $this->levelSettings->write($out);
183
184 $out->putString($this->levelId);
185 $out->putString($this->worldName);
186 $out->putString($this->premiumWorldTemplateId);
187 $out->putBool($this->isTrial);
188 $this->playerMovementSettings->write($out);
189 $out->putLLong($this->currentTick);
190
191 $out->putVarInt($this->enchantmentSeed);
192
193 $out->putUnsignedVarInt(count($this->blockPalette));
194 foreach($this->blockPalette as $entry){
195 $out->putString($entry->getName());
196 $out->put($entry->getStates()->getEncodedNbt());
197 }
198
199 $out->putString($this->multiplayerCorrelationId);
200 $out->putBool($this->enableNewInventorySystem);
201 $out->putString($this->serverSoftwareVersion);
202 $out->put($this->playerActorProperties->getEncodedNbt());
203 $out->putLLong($this->blockPaletteChecksum);
204 $out->putUUID($this->worldTemplateId);
205 $out->putBool($this->enableClientSideChunkGeneration);
206 $out->putBool($this->blockNetworkIdsAreHashes);
207 $this->networkPermissions->encode($out);
208 }
