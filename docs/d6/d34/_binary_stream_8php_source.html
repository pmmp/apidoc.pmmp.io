67 public function get(int $len) : string{
68 if($len === 0){
69 return "";
70 }
71 if($len < 0){
72 throw new \InvalidArgumentException("Length must be positive");
73 }
74
75 $remaining = strlen($this->buffer) - $this->offset;
76 if($remaining < $len){
77 throw new BinaryDataException("Not enough bytes left in buffer: need $len, have $remaining");
78 }
79
80 return $len === 1 ? $this->buffer[$this->offset++] : substr($this->buffer, ($this->offset += $len) - $len, $len);
81 }

87 public function getRemaining() : string{
88 $buflen = strlen($this->buffer);

90 throw new BinaryDataException("No bytes left to read");
91 }
92 $str = substr($this->buffer, $this->offset);
93 $this->offset = $buflen;
