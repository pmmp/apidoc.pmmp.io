81 public function handleLogin(LoginPacket $packet) : bool{
82 if(!$this->isCompatibleProtocol($packet->protocol)){
83 $this->session->sendDataPacket(PlayStatusPacket::create($packet->protocol < ProtocolInfo::CURRENT_PROTOCOL ? PlayStatusPacket::LOGIN_FAILED_CLIENT : PlayStatusPacket::LOGIN_FAILED_SERVER), true);
84
85 //This pocketmine disconnect message will only be seen by the console (PlayStatusPacket causes the messages to be shown for the client)
86 $this->session->disconnect(
87 $this->server->getLanguage()->translate(KnownTranslationFactory::pocketmine_disconnect_incompatibleProtocol((string) $packet->protocol)),
88 false
89 );
90
91 return true;
92 }
93
94 $extraData = $this->fetchAuthData($packet->chainDataJwt);
95
96 if(!Player::isValidUserName($extraData->displayName)){
97 $this->session->disconnect(KnownTranslationKeys::DISCONNECTIONSCREEN_INVALIDNAME);
98
99 return true;
100 }
101
102 $clientData = $this->parseClientData($packet->clientDataJwt);
103 try{
104 $skin = SkinAdapterSingleton::get()->fromSkinData(ClientDataToSkinDataHelper::fromClientData($clientData));
105 }catch(\InvalidArgumentException | InvalidSkinException $e){
106 $this->session->getLogger()->debug("Invalid skin: " . $e->getMessage());
107 $this->session->disconnect(KnownTranslationKeys::DISCONNECTIONSCREEN_INVALIDSKIN);
108
109 return true;
110 }
111
112 if(!Uuid::isValid($extraData->identity)){
113 throw new PacketHandlingException("Invalid login UUID");
114 }
115 $uuid = Uuid::fromString($extraData->identity);
116 if($extraData->XUID !== ""){
117 $playerInfo = new XboxLivePlayerInfo(
118 $extraData->XUID,
119 $extraData->displayName,
120 $uuid,
121 $skin,
122 $clientData->LanguageCode,
123 (array) $clientData
124 );
125 }else{
126 $playerInfo = new PlayerInfo(
127 $extraData->displayName,
128 $uuid,
129 $skin,
130 $clientData->LanguageCode,
131 (array) $clientData
132 );
133 }
134 ($this->playerInfoConsumer)($playerInfo);
135
136 $ev = new PlayerPreLoginEvent(
137 $playerInfo,
138 $this->session->getIp(),
139 $this->session->getPort(),
140 $this->server->requiresAuthentication()
141 );
142 if($this->server->getNetwork()->getConnectionCount() > $this->server->getMaxPlayers()){
143 $ev->setKickReason(PlayerPreLoginEvent::KICK_REASON_SERVER_FULL, KnownTranslationKeys::DISCONNECTIONSCREEN_SERVERFULL);
144 }
145 if(!$this->server->isWhitelisted($playerInfo->getUsername())){
146 $ev->setKickReason(PlayerPreLoginEvent::KICK_REASON_SERVER_WHITELISTED, "Server is whitelisted");
147 }
148 if($this->server->getNameBans()->isBanned($playerInfo->getUsername()) || $this->server->getIPBans()->isBanned($this->session->getIp())){
149 $ev->setKickReason(PlayerPreLoginEvent::KICK_REASON_BANNED, "You are banned");
150 }
151
152 $ev->call();
153 if(!$ev->isAllowed()){
154 $this->session->disconnect($ev->getFinalKickMessage());
155 return true;
156 }
157
158 $this->processLogin($packet, $ev->isAuthRequired());
159
160 return true;
161 }

166 protected function fetchAuthData(JwtChain $chain) : AuthenticationData{
168 $extraData = null;
169 foreach($chain->chain as $k => $jwt){
170 //validate every chain element
171 try{
172 [, $claims, ] = JwtUtils::parse($jwt);
173 }catch(JwtException $e){
174 throw PacketHandlingException::wrap($e);
175 }
176 if(isset($claims["extraData"])){
177 if($extraData !== null){
178 throw new PacketHandlingException("Found 'extraData' more than once in chainData");
179 }
180
181 if(!is_array($claims["extraData"])){
182 throw new PacketHandlingException("'extraData' key should be an array");
183 }
184 $mapper = new \JsonMapper;
185 $mapper->bEnforceMapType = false; //TODO: we don't really need this as an array, but right now we don't have enough models
186 $mapper->bExceptionOnMissingData = true;
187 $mapper->bExceptionOnUndefinedProperty = true;
188 try{
190 $extraData = $mapper->map($claims["extraData"], new AuthenticationData);
191 }catch(\JsonMapper_Exception $e){
192 throw PacketHandlingException::wrap($e);
193 }
194 }
195 }
196 if($extraData === null){
197 throw new PacketHandlingException("'extraData' not found in chain data");
198 }
199 return $extraData;
200 }

206 try{
207 [, $clientDataClaims, ] = JwtUtils::parse($clientDataJwt);
208 }catch(JwtException $e){
209 throw PacketHandlingException::wrap($e);
210 }
211
212 $mapper = new \JsonMapper;
213 $mapper->bEnforceMapType = false; //TODO: we don't really need this as an array, but right now we don't have enough models
214 $mapper->bExceptionOnMissingData = true;
215 $mapper->bExceptionOnUndefinedProperty = true;
216 try{
217 $clientData = $mapper->map($clientDataClaims, new ClientData);
218 }catch(\JsonMapper_Exception $e){
219 throw PacketHandlingException::wrap($e);
220 }
221 return $clientData;
222 }

230 protected function processLogin(LoginPacket $packet, bool $authRequired) : void{
