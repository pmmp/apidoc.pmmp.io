130 protected function decodePayload(PacketSerializer $in) : void{
132 $enumValues = [];
133 for($i = 0, $enumValuesCount = $in->getUnsignedVarInt(); $i < $enumValuesCount; ++$i){
134 $enumValues[] = $in->getString();
135 }
136
138 $chainedSubcommandValueNames = [];
139 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
140 $chainedSubcommandValueNames[] = $in->getString();
141 }
142
144 $postfixes = [];
145 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
146 $postfixes[] = $in->getString();
147 }
148
150 $enums = [];
151 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
152 $enums[] = $enum = $this->getEnum($enumValues, $in);
153 //TODO: Bedrock may provide some enums which are not referenced by any command, and can't reasonably be
154 //considered "hardcoded". This happens with various Edu command enums, and other enums which are probably
155 //intended to be used by commands which aren't present in public releases.
156 //We should probably store these somewhere, since we'll need them to be able to correctly re-encode the
157 //packet for testing.
158 if(isset(self::HARDCODED_ENUM_NAMES[$enum->getName()])){
159 $this->hardcodedEnums[] = $enum;
160 }
161 }
162
163 $chainedSubCommandData = [];
164 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
165 $name = $in->getString();
166 $values = [];
167 for($j = 0, $valueCount = $in->getUnsignedVarInt(); $j < $valueCount; ++$j){
168 $valueName = $chainedSubcommandValueNames[$in->getLShort()];
169 $valueType = $in->getLShort();
170 $values[] = new ChainedSubCommandValue($valueName, $valueType);
171 }
172 $chainedSubCommandData[] = new ChainedSubCommandData($name, $values);
173 }
174
175 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
176 $this->commandData[] = $this->getCommandData($enums, $postfixes, $chainedSubCommandData, $in);
177 }
178
179 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
180 $this->softEnums[] = $this->getSoftEnum($in);
181 }
182
183 $this->initSoftEnumsInCommandData();
184
185 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
186 $this->enumConstraints[] = $this->getEnumConstraint($enums, $enumValues, $in);
187 }
188 }

197 foreach($overload->getParameters() as $parameter){
198 if(($parameter->paramType & self::ARG_FLAG_SOFT_ENUM) !== 0){

203 }

306 protected function getEnumConstraint(array $enums, array $enumValues, PacketSerializer $in) : CommandEnumConstraint{
307 //wtf, what was wrong with an offset inside the enum? :(
308 $valueIndex = $in->getLInt();
309 if(!isset($enumValues[$valueIndex])){
310 throw new PacketDecodeException("Enum constraint refers to unknown enum value index $valueIndex");
311 }
312 $enumIndex = $in->getLInt();
313 if(!isset($enums[$enumIndex])){
314 throw new PacketDecodeException("Enum constraint refers to unknown enum index $enumIndex");
315 }
316 $enum = $enums[$enumIndex];
317 $valueOffset = array_search($enumValues[$valueIndex], $enum->getValues(), true);
318 if($valueOffset === false){
319 throw new PacketDecodeException("Value \"" . $enumValues[$valueIndex] . "\" does not belong to enum \"" . $enum->getName() . "\"");
320 }
321
322 $constraintIds = [];
323 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
324 $constraintIds[] = $in->getByte();
325 }
326
327 return new CommandEnumConstraint($enum, $valueOffset, $constraintIds);

351 protected function getCommandData(array $enums, array $postfixes, array $allChainedSubCommandData, PacketSerializer $in) : CommandData{
352 $name = $in->getString();
353 $description = $in->getString();
354 $flags = $in->getLShort();
355 $permission = $in->getByte();
356 $aliases = $enums[$in->getLInt()] ?? null;
357
358 $chainedSubCommandData = [];
359 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
360 $index = $in->getLShort();
361 $chainedSubCommandData[] = $allChainedSubCommandData[$index] ?? throw new PacketDecodeException("Unknown chained subcommand data index $index");
362 }
363 $overloads = [];
364
365 for($overloadIndex = 0, $overloadCount = $in->getUnsignedVarInt(); $overloadIndex < $overloadCount; ++$overloadIndex){
366 $parameters = [];
367 $isChaining = $in->getBool();
368 for($paramIndex = 0, $paramCount = $in->getUnsignedVarInt(); $paramIndex < $paramCount; ++$paramIndex){
369 $parameter = new CommandParameter();
370 $parameter->paramName = $in->getString();
371 $parameter->paramType = $in->getLInt();
372 $parameter->isOptional = $in->getBool();
373 $parameter->flags = $in->getByte();
374
375 if(($parameter->paramType & self::ARG_FLAG_ENUM) !== 0){
376 $index = ($parameter->paramType & 0xffff);
377 $parameter->enum = $enums[$index] ?? null;
378 if($parameter->enum === null){
379 throw new PacketDecodeException("deserializing $name parameter $parameter->paramName: expected enum at $index, but got none");
380 }
381 }elseif(($parameter->paramType & self::ARG_FLAG_POSTFIX) !== 0){
382 $index = ($parameter->paramType & 0xffff);
383 $parameter->postfix = $postfixes[$index] ?? null;
384 if($parameter->postfix === null){
385 throw new PacketDecodeException("deserializing $name parameter $parameter->paramName: expected postfix at $index, but got none");
386 }
387 }elseif(($parameter->paramType & self::ARG_FLAG_VALID) === 0){
388 throw new PacketDecodeException("deserializing $name parameter $parameter->paramName: Invalid parameter type 0x" . dechex($parameter->paramType));
389 }
390
391 $parameters[$paramIndex] = $parameter;
392 }
393 $overloads[$overloadIndex] = new CommandOverload($isChaining, $parameters);
394 }
395
396 return new CommandData($name, $description, $flags, $permission, $aliases, $overloads, $chainedSubCommandData);
397 }

405 protected function putCommandData(CommandData $data, array $enumIndexes, array $softEnumIndexes, array $postfixIndexes, array $chainedSubCommandDataIndexes, PacketSerializer $out) : void{
406 $out->putString($data->name);
407 $out->putString($data->description);
408 $out->putLShort($data->flags);
409 $out->putByte($data->permission);
410
411 if($data->aliases !== null){
412 $out->putLInt($enumIndexes[$data->aliases->getName()] ?? -1);
413 }else{
414 $out->putLInt(-1);
415 }
416
417 $out->putUnsignedVarInt(count($data->chainedSubCommandData));
418 foreach($data->chainedSubCommandData as $chainedSubCommandData){
419 $index = $chainedSubCommandDataIndexes[$chainedSubCommandData->getName()] ??
420 throw new \LogicException("Chained subcommand data {$chainedSubCommandData->getName()} does not have an index (this should be impossible)");
421 $out->putLShort($index);
422 }
423
424 $out->putUnsignedVarInt(count($data->overloads));
425 foreach($data->overloads as $overload){
426 $out->putBool($overload->isChaining());
427 $out->putUnsignedVarInt(count($overload->getParameters()));
428 foreach($overload->getParameters() as $parameter){
429 $out->putString($parameter->paramName);
430
431 if($parameter->enum !== null){
432 if($parameter->enum->isSoft()){
433 $type = self::ARG_FLAG_SOFT_ENUM | self::ARG_FLAG_VALID | ($softEnumIndexes[$parameter->enum->getName()] ?? -1);
434 }else{
435 $type = self::ARG_FLAG_ENUM | self::ARG_FLAG_VALID | ($enumIndexes[$parameter->enum->getName()] ?? -1);
436 }
437 }elseif($parameter->postfix !== null){
438 if(!isset($postfixIndexes[$parameter->postfix])){
439 throw new \LogicException("Postfix '$parameter->postfix' not in postfixes array");
440 }
441 $type = self::ARG_FLAG_POSTFIX | $postfixIndexes[$parameter->postfix];
442 }else{
443 $type = $parameter->paramType;
444 }
445
446 $out->putLInt($type);
447 $out->putBool($parameter->isOptional);
448 $out->putByte($parameter->flags);
449 }
450 }
