120 protected function decodePayload(PacketSerializer $in) : void{
122 $enumValues = [];
123 for($i = 0, $enumValuesCount = $in->getUnsignedVarInt(); $i < $enumValuesCount; ++$i){
124 $enumValues[] = $in->getString();
125 }
126
128 $postfixes = [];
129 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
130 $postfixes[] = $in->getString();
131 }
132
134 $enums = [];
135 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
136 $enums[] = $enum = $this->getEnum($enumValues, $in);
137 if(isset(self::HARDCODED_ENUM_NAMES[$enum->getName()])){
138 $this->hardcodedEnums[] = $enum;
139 }
140 }
141
142 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
143 $this->commandData[] = $this->getCommandData($enums, $postfixes, $in);
144 }
145
146 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
147 $this->softEnums[] = $this->getSoftEnum($in);
148 }
149
150 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
151 $this->enumConstraints[] = $this->getEnumConstraint($enums, $enumValues, $in);
152 }

161 protected function getEnum(array $enumValueList, PacketSerializer $in) : CommandEnum{
162 $enumName = $in->getString();
163 $enumValues = [];
164
165 $listSize = count($enumValueList);
166
167 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
168 $index = $this->getEnumValueIndex($listSize, $in);
169 if(!isset($enumValueList[$index])){
170 throw new PacketDecodeException("Invalid enum value index $index");
171 }
172 //Get the enum value from the initial pile of mess
173 $enumValues[] = $enumValueList[$index];
174 }
175
176 return new CommandEnum($enumName, $enumValues);

251 protected function getEnumConstraint(array $enums, array $enumValues, PacketSerializer $in) : CommandEnumConstraint{
252 //wtf, what was wrong with an offset inside the enum? :(
253 $valueIndex = $in->getLInt();
254 if(!isset($enumValues[$valueIndex])){
255 throw new PacketDecodeException("Enum constraint refers to unknown enum value index $valueIndex");
256 }
257 $enumIndex = $in->getLInt();
258 if(!isset($enums[$enumIndex])){
259 throw new PacketDecodeException("Enum constraint refers to unknown enum index $enumIndex");
260 }
261 $enum = $enums[$enumIndex];
262 $valueOffset = array_search($enumValues[$valueIndex], $enum->getValues(), true);
263 if($valueOffset === false){
264 throw new PacketDecodeException("Value \"" . $enumValues[$valueIndex] . "\" does not belong to enum \"" . $enum->getName() . "\"");
265 }
266
267 $constraintIds = [];
268 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
269 $constraintIds[] = $in->getByte();
270 }
271
272 return new CommandEnumConstraint($enum, $valueOffset, $constraintIds);
273 }

295 protected function getCommandData(array $enums, array $postfixes, PacketSerializer $in) : CommandData{
296 $name = $in->getString();
297 $description = $in->getString();
298 $flags = $in->getLShort();
299 $permission = $in->getByte();
300 $aliases = $enums[$in->getLInt()] ?? null;
301 $overloads = [];
302
303 for($overloadIndex = 0, $overloadCount = $in->getUnsignedVarInt(); $overloadIndex < $overloadCount; ++$overloadIndex){
304 $overloads[$overloadIndex] = [];
305 for($paramIndex = 0, $paramCount = $in->getUnsignedVarInt(); $paramIndex < $paramCount; ++$paramIndex){
306 $parameter = new CommandParameter();
307 $parameter->paramName = $in->getString();
308 $parameter->paramType = $in->getLInt();
309 $parameter->isOptional = $in->getBool();
310 $parameter->flags = $in->getByte();
311
312 if(($parameter->paramType & self::ARG_FLAG_ENUM) !== 0){
313 $index = ($parameter->paramType & 0xffff);
314 $parameter->enum = $enums[$index] ?? null;
315 if($parameter->enum === null){
316 throw new PacketDecodeException("deserializing $name parameter $parameter->paramName: expected enum at $index, but got none");
317 }
318 }elseif(($parameter->paramType & self::ARG_FLAG_POSTFIX) !== 0){
319 $index = ($parameter->paramType & 0xffff);
320 $parameter->postfix = $postfixes[$index] ?? null;
321 if($parameter->postfix === null){
322 throw new PacketDecodeException("deserializing $name parameter $parameter->paramName: expected postfix at $index, but got none");
323 }
324 }elseif(($parameter->paramType & self::ARG_FLAG_VALID) === 0){
325 throw new PacketDecodeException("deserializing $name parameter $parameter->paramName: Invalid parameter type 0x" . dechex($parameter->paramType));
326 }
327
328 $overloads[$overloadIndex][$paramIndex] = $parameter;
329 }
330 }
331
332 return new CommandData($name, $description, $flags, $permission, $aliases, $overloads);
333 }

339 protected function putCommandData(CommandData $data, array $enumIndexes, array $postfixIndexes, PacketSerializer $out) : void{
340 $out->putString($data->name);
341 $out->putString($data->description);
342 $out->putLShort($data->flags);
343 $out->putByte($data->permission);
344
345 if($data->aliases !== null){
346 $out->putLInt($enumIndexes[$data->aliases->getName()] ?? -1);
347 }else{
348 $out->putLInt(-1);
349 }
350
351 $out->putUnsignedVarInt(count($data->overloads));
352 foreach($data->overloads as $overload){
354 $out->putUnsignedVarInt(count($overload));
355 foreach($overload as $parameter){
356 $out->putString($parameter->paramName);
357
358 if($parameter->enum !== null){
359 $type = self::ARG_FLAG_ENUM | self::ARG_FLAG_VALID | ($enumIndexes[$parameter->enum->getName()] ?? -1);
360 }elseif($parameter->postfix !== null){
361 if(!isset($postfixIndexes[$parameter->postfix])){
362 throw new \LogicException("Postfix '$parameter->postfix' not in postfixes array");
363 }
364 $type = self::ARG_FLAG_POSTFIX | $postfixIndexes[$parameter->postfix];
365 }else{
366 $type = $parameter->paramType;
367 }
368
369 $out->putLInt($type);
370 $out->putBool($parameter->isOptional);
371 $out->putByte($parameter->flags);
372 }
373 }

386 $addEnumFn = static function(CommandEnum $enum) use (&$enums, &$enumIndexes, &$enumValueIndexes) : void{
387 if(!isset($enumIndexes[$enum->getName()])){
388 $enums[$enumIndexes[$enum->getName()] = count($enumIndexes)] = $enum;
389 }
390 foreach($enum->getValues() as $str){
391 $enumValueIndexes[$str] = $enumValueIndexes[$str] ?? count($enumValueIndexes); //latest index
392 }
393 };
394 foreach($this->hardcodedEnums as $enum){
395 $addEnumFn($enum);
396 }
397 foreach($this->commandData as $commandData){
398 if($commandData->aliases !== null){
399 $addEnumFn($commandData->aliases);
400 }
402 foreach($commandData->overloads as $overload){
404 foreach($overload as $parameter){
405 if($parameter->enum !== null){
406 $addEnumFn($parameter->enum);
407 }
408
409 if($parameter->postfix !== null){
410 $postfixIndexes[$parameter->postfix] = $postfixIndexes[$parameter->postfix] ?? count($postfixIndexes);
411 }
412 }
413 }
414 }
415
416 $out->putUnsignedVarInt(count($enumValueIndexes));
417 foreach($enumValueIndexes as $enumValue => $index){
418 $out->putString((string) $enumValue); //stupid PHP key casting D:
419 }
420
421 $out->putUnsignedVarInt(count($postfixIndexes));
422 foreach($postfixIndexes as $postfix => $index){
423 $out->putString((string) $postfix); //stupid PHP key casting D:
424 }
425
426 $out->putUnsignedVarInt(count($enums));
427 foreach($enums as $enum){
428 $this->putEnum($enum, $enumValueIndexes, $out);
429 }
430
431 $out->putUnsignedVarInt(count($this->commandData));
432 foreach($this->commandData as $data){
433 $this->putCommandData($data, $enumIndexes, $postfixIndexes, $out);
434 }
435
436 $out->putUnsignedVarInt(count($this->softEnums));
437 foreach($this->softEnums as $enum){
438 $this->putSoftEnum($enum, $out);
439 }
440
441 $out->putUnsignedVarInt(count($this->enumConstraints));
442 foreach($this->enumConstraints as $constraint){
443 $this->putEnumConstraint($constraint, $enumIndexes, $enumValueIndexes, $out);
444 }
