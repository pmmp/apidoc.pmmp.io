142 private function getWorker(int $worker) : AsyncWorker{
143 if(!isset($this->workers[$worker])){
144 $notifier = new SleeperNotifier();
145 $this->workers[$worker] = new AsyncWorker($this->logger, $worker, $this->workerMemoryLimit, $notifier);
146 $this->eventLoop->addNotifier($notifier, function() use ($worker) : void{
147 $this->collectTasksFromWorker($worker);
148 });
149 $this->workers[$worker]->setClassLoaders([$this->classLoader]);
150 $this->workers[$worker]->start(self::WORKER_START_OPTIONS);
151
152 $this->taskQueues[$worker] = new \SplQueue();
153
154 foreach($this->workerStartHooks as $hook){
155 $hook($worker);
156 }
157 }
158
159 return $this->workers[$worker];
160 }

165 public function submitTaskToWorker(AsyncTask $task, int $worker) : void{
166 if($worker < 0 || $worker >= $this->size){
167 throw new \InvalidArgumentException("Invalid worker $worker");
168 }
169 if($task->isSubmitted()){
170 throw new \InvalidArgumentException("Cannot submit the same AsyncTask instance more than once");
171 }
172
173 $task->progressUpdates = new \Threaded;
174 $task->setSubmitted();
175
176 $this->getWorker($worker)->stack($task);
177 $this->taskQueues[$worker]->enqueue($task);
178 $this->workerLastUsed[$worker] = time();
179 }

188 public function selectWorker() : int{
189 $worker = null;
190 $minUsage = PHP_INT_MAX;
191 foreach($this->taskQueues as $i => $queue){
192 if(($usage = $queue->count()) < $minUsage){
193 $worker = $i;
194 $minUsage = $usage;
195 if($usage === 0){
196 break;
197 }
198 }
199 }
200 if($worker === null || ($minUsage > 0 && count($this->workers) < $this->size)){
201 //select a worker to start on the fly
202 for($i = 0; $i < $this->size; ++$i){
203 if(!isset($this->workers[$i])){
204 $worker = $i;
205 break;
206 }
207 }
208 }
209
210 assert($worker !== null);
211 return $worker;
212 }

234 public function collectTasks() : bool{

236 $this->collectTasksFromWorker($worker);

248 public function collectTasksFromWorker(int $worker) : bool{
249 if(!isset($this->taskQueues[$worker])){
250 throw new \InvalidArgumentException("No such worker $worker");
251 }
252 $queue = $this->taskQueues[$worker];
253 $more = false;
254 while(!$queue->isEmpty()){
256 $task = $queue->bottom();
257 $task->checkProgressUpdates();
258 if($task->isFinished()){ //make sure the task actually executed before trying to collect
259 $queue->dequeue();
260
261 if($task->isCrashed()){
262 $this->logger->critical("Could not execute asynchronous task " . (new \ReflectionClass($task))->getShortName() . ": Task crashed");
263 $task->onError();
264 }elseif(!$task->hasCancelledRun()){
265 /*
266 * It's possible for a task to submit a progress update and then finish before the progress
267 * update is detected by the parent thread, so here we consume any missed updates.
268 *
269 * When this happens, it's possible for a progress update to arrive between the previous
270 * checkProgressUpdates() call and the next isGarbage() call, causing progress updates to be
271 * lost. Thus, it's necessary to do one last check here to make sure all progress updates have
272 * been consumed before completing.
273 */
274 $task->checkProgressUpdates();
275 $task->onCompletion();
276 }
277 }else{
278 $more = true;
279 break; //current task is still running, skip to next worker
280 }
281 }
282 $this->workers[$worker]->collect();
283 return $more;
284 }

296 public function shutdownUnusedWorkers() : int{
297 $ret = 0;
298 $time = time();
299 foreach($this->taskQueues as $i => $queue){
300 if((!isset($this->workerLastUsed[$i]) || $this->workerLastUsed[$i] + 300 < $time) && $queue->isEmpty()){
301 $this->workers[$i]->quit();
302 $this->eventLoop->removeNotifier($this->workers[$i]->getNotifier());
303 unset($this->workers[$i], $this->taskQueues[$i], $this->workerLastUsed[$i]);
304 $ret++;
305 }
306 }
307
308 return $ret;
309 }
