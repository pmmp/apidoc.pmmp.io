35final class ZlibCompressor implements Compressor{
36 use SingletonTrait;
37
38 public const DEFAULT_LEVEL = 7;
39 public const DEFAULT_THRESHOLD = 256;
40 public const DEFAULT_MAX_DECOMPRESSION_SIZE = 2 * 1024 * 1024;
41
45 private static function make() : self{
46 return new self(self::DEFAULT_LEVEL, self::DEFAULT_THRESHOLD, self::DEFAULT_MAX_DECOMPRESSION_SIZE);
47 }
48
50 private $level;
52 private $threshold;
54 private $maxDecompressionSize;
55
56 public function __construct(int $level, int $minCompressionSize, int $maxDecompressionSize){
57 $this->level = $level;
58 $this->threshold = $minCompressionSize;
59 $this->maxDecompressionSize = $maxDecompressionSize;
60 }
61
62 public function willCompress(string $data) : bool{
63 return $this->threshold > -1 && strlen($data) >= $this->threshold;
64 }
65
69 public function decompress(string $payload) : string{
70 $result = @zlib_decode($payload, $this->maxDecompressionSize);
71 if($result === false){
72 throw new DecompressionException("Failed to decompress data");
73 }
74 return $result;
75 }
76
77 private static function zlib_encode(string $data, int $level) : string{
78 return Utils::assumeNotFalse(zlib_encode($data, ZLIB_ENCODING_RAW, $level), "ZLIB compression failed");
79 }
80
81 public function compress(string $payload) : string{
82 if(function_exists('libdeflate_deflate_compress')){
83 return $this->willCompress($payload) ?
84 libdeflate_deflate_compress($payload, $this->level) :
85 self::zlib_encode($payload, 0);
86 }
87 return self::zlib_encode($payload, $this->willCompress($payload) ? $this->level : 0);
88 }
89}
