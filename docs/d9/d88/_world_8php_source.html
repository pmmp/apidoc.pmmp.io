291 private const MORTON3D_BIT_SIZE = 21;
292 private const BLOCKHASH_Y_BITS = 9;
293 private const BLOCKHASH_Y_PADDING = 128; //size (in blocks) of padding after both boundaries of the Y axis
294 private const BLOCKHASH_Y_OFFSET = self::BLOCKHASH_Y_PADDING - self::Y_MIN;
295 private const BLOCKHASH_Y_MASK = (1 << self::BLOCKHASH_Y_BITS) - 1;
296 private const BLOCKHASH_XZ_MASK = (1 << self::MORTON3D_BIT_SIZE) - 1;
297 private const BLOCKHASH_XZ_EXTRA_BITS = (self::MORTON3D_BIT_SIZE - self::BLOCKHASH_Y_BITS) >> 1;
298 private const BLOCKHASH_XZ_EXTRA_MASK = (1 << self::BLOCKHASH_XZ_EXTRA_BITS) - 1;
299 private const BLOCKHASH_XZ_SIGN_SHIFT = 64 - self::MORTON3D_BIT_SIZE - self::BLOCKHASH_XZ_EXTRA_BITS;
300 private const BLOCKHASH_X_SHIFT = self::BLOCKHASH_Y_BITS;
301 private const BLOCKHASH_Z_SHIFT = self::BLOCKHASH_X_SHIFT + self::BLOCKHASH_XZ_EXTRA_BITS;

303 public static function blockHash(int $x, int $y, int $z) : int{
304 $shiftedY = $y + self::BLOCKHASH_Y_OFFSET;
305 if(($shiftedY & (~0 << self::BLOCKHASH_Y_BITS)) !== 0){
306 throw new \InvalidArgumentException("Y coordinate $y is out of range!");
307 }
308 //morton3d gives us 21 bits on each axis, but the Y axis only requires 9
309 //so we use the extra space on Y (12 bits) and add 6 extra bits from X and Z instead.
310 //if we ever need more space for Y (e.g. due to expansion), take bits from X/Z to compensate.
311 return morton3d_encode(
312 $x & self::BLOCKHASH_XZ_MASK,
313 ($shiftedY /* & self::BLOCKHASH_Y_MASK */) |
314 ((($x >> self::MORTON3D_BIT_SIZE) & self::BLOCKHASH_XZ_EXTRA_MASK) << self::BLOCKHASH_X_SHIFT) |
315 ((($z >> self::MORTON3D_BIT_SIZE) & self::BLOCKHASH_XZ_EXTRA_MASK) << self::BLOCKHASH_Z_SHIFT),
316 $z & self::BLOCKHASH_XZ_MASK
317 );
318 }

327 public static function getBlockXYZ(int $hash, ?int &$x, ?int &$y, ?int &$z) : void{
328 [$baseX, $baseY, $baseZ] = morton3d_decode($hash);
329
330 $extraX = ((($baseY >> self::BLOCKHASH_X_SHIFT) & self::BLOCKHASH_XZ_EXTRA_MASK) << self::MORTON3D_BIT_SIZE);
331 $extraZ = ((($baseY >> self::BLOCKHASH_Z_SHIFT) & self::BLOCKHASH_XZ_EXTRA_MASK) << self::MORTON3D_BIT_SIZE);
332
333 $x = (($baseX & self::BLOCKHASH_XZ_MASK) | $extraX) << self::BLOCKHASH_XZ_SIGN_SHIFT >> self::BLOCKHASH_XZ_SIGN_SHIFT;
334 $y = ($baseY & self::BLOCKHASH_Y_MASK) - self::BLOCKHASH_Y_OFFSET;
335 $z = (($baseZ & self::BLOCKHASH_XZ_MASK) | $extraZ) << self::BLOCKHASH_XZ_SIGN_SHIFT >> self::BLOCKHASH_XZ_SIGN_SHIFT;
336 }

439 $name = (string) $name;
440 $item = $parser->parse($name);

442 $block = $item->getBlock();
443 }elseif(preg_match("/^-?\d+$/", $name) === 1){
444 $block = BlockFactory::getInstance()->get((int) $name, 0);

451 $dontTickBlocks[$block->getTypeId()] = $name;

508 public function onUnload() : void{
509 if($this->unloaded){
510 throw new \LogicException("Tried to close a world which is already closed");
511 }
512
513 foreach($this->unloadCallbacks as $callback){
514 $callback();
515 }
516 $this->unloadCallbacks = [];
517
518 foreach($this->chunks as $chunkHash => $chunk){
519 self::getXZ($chunkHash, $chunkX, $chunkZ);
520 $this->unloadChunk($chunkX, $chunkZ, false);
521 }
522 foreach($this->entitiesByChunk as $chunkHash => $entities){
523 self::getXZ($chunkHash, $chunkX, $chunkZ);
524
525 $leakedEntities = 0;
526 foreach($entities as $entity){
527 if(!$entity->isFlaggedForDespawn()){
528 $leakedEntities++;
529 }
530 $entity->close();
531 }
532 if($leakedEntities !== 0){
533 $this->logger->warning("$leakedEntities leaked entities found in ungenerated chunk $chunkX $chunkZ during unload, they won't be saved!");
534 }
535 }
536
537 $this->save();
538
539 $this->unregisterGenerator();
540
541 $this->provider->close();
542 $this->blockCache = [];
543
544 $this->unloaded = true;
545 }

642 public function registerChunkLoader(ChunkLoader $loader, int $chunkX, int $chunkZ, bool $autoLoad = true) : void{
643 $loaderId = spl_object_id($loader);
644
645 if(!isset($this->chunkLoaders[$chunkHash = World::chunkHash($chunkX, $chunkZ)])){
646 $this->chunkLoaders[$chunkHash] = [];
647 }elseif(isset($this->chunkLoaders[$chunkHash][$loaderId])){
648 return;
649 }
650
651 $this->chunkLoaders[$chunkHash][$loaderId] = $loader;
652
653 if($loader instanceof TickingChunkLoader){
654 if(!isset($this->tickingLoaders[$loaderId])){
655 $this->tickingLoaderCounter[$loaderId] = 1;
656 $this->tickingLoaders[$loaderId] = $loader;
657 }else{
658 ++$this->tickingLoaderCounter[$loaderId];
659 }
660 }
661
662 $this->cancelUnloadChunkRequest($chunkX, $chunkZ);
663
664 if($autoLoad){
665 $this->loadChunk($chunkX, $chunkZ);
666 }
667 }

669 public function unregisterChunkLoader(ChunkLoader $loader, int $chunkX, int $chunkZ) : void{
670 $chunkHash = World::chunkHash($chunkX, $chunkZ);
671 $loaderId = spl_object_id($loader);
672 if(isset($this->chunkLoaders[$chunkHash][$loaderId])){
673 unset($this->chunkLoaders[$chunkHash][$loaderId]);
674 if(count($this->chunkLoaders[$chunkHash]) === 0){
675 unset($this->chunkLoaders[$chunkHash]);
676 $this->unloadChunkRequest($chunkX, $chunkZ, true);
677 if(isset($this->chunkPopulationRequestMap[$chunkHash]) && !isset($this->activeChunkPopulationTasks[$chunkHash])){
678 $this->chunkPopulationRequestMap[$chunkHash]->reject();
679 unset($this->chunkPopulationRequestMap[$chunkHash]);
680 }
681 }
682
683 if(isset($this->tickingLoaderCounter[$loaderId]) && --$this->tickingLoaderCounter[$loaderId] === 0){
684 unset($this->tickingLoaderCounter[$loaderId]);
685 unset($this->tickingLoaders[$loaderId]);
686 }
687 }

787
