358 private const MORTON3D_BIT_SIZE = 21;
359 private const BLOCKHASH_Y_BITS = 9;
360 private const BLOCKHASH_Y_PADDING = 128; //size (in blocks) of padding after both boundaries of the Y axis
361 private const BLOCKHASH_Y_OFFSET = self::BLOCKHASH_Y_PADDING - self::Y_MIN;
362 private const BLOCKHASH_Y_MASK = (1 << self::BLOCKHASH_Y_BITS) - 1;
363 private const BLOCKHASH_XZ_MASK = (1 << self::MORTON3D_BIT_SIZE) - 1;
364 private const BLOCKHASH_XZ_EXTRA_BITS = (self::MORTON3D_BIT_SIZE - self::BLOCKHASH_Y_BITS) >> 1;
365 private const BLOCKHASH_XZ_EXTRA_MASK = (1 << self::BLOCKHASH_XZ_EXTRA_BITS) - 1;
366 private const BLOCKHASH_XZ_SIGN_SHIFT = 64 - self::MORTON3D_BIT_SIZE - self::BLOCKHASH_XZ_EXTRA_BITS;
367 private const BLOCKHASH_X_SHIFT = self::BLOCKHASH_Y_BITS;
368 private const BLOCKHASH_Z_SHIFT = self::BLOCKHASH_X_SHIFT + self::BLOCKHASH_XZ_EXTRA_BITS;

373 public static function blockHash(int $x, int $y, int $z) : int{
374 $shiftedY = $y + self::BLOCKHASH_Y_OFFSET;
375 if(($shiftedY & (~0 << self::BLOCKHASH_Y_BITS)) !== 0){
376 throw new \InvalidArgumentException("Y coordinate $y is out of range!");
377 }
378 //morton3d gives us 21 bits on each axis, but the Y axis only requires 9
379 //so we use the extra space on Y (12 bits) and add 6 extra bits from X and Z instead.
380 //if we ever need more space for Y (e.g. due to expansion), take bits from X/Z to compensate.
381 return morton3d_encode(
382 $x & self::BLOCKHASH_XZ_MASK,
383 ($shiftedY /* & self::BLOCKHASH_Y_MASK */) |
384 ((($x >> self::MORTON3D_BIT_SIZE) & self::BLOCKHASH_XZ_EXTRA_MASK) << self::BLOCKHASH_X_SHIFT) |
385 ((($z >> self::MORTON3D_BIT_SIZE) & self::BLOCKHASH_XZ_EXTRA_MASK) << self::BLOCKHASH_Z_SHIFT),
386 $z & self::BLOCKHASH_XZ_MASK
387 );
388 }

403 public static function getBlockXYZ(int $hash, ?int &$x, ?int &$y, ?int &$z) : void{
404 [$baseX, $baseY, $baseZ] = morton3d_decode($hash);
405
406 $extraX = ((($baseY >> self::BLOCKHASH_X_SHIFT) & self::BLOCKHASH_XZ_EXTRA_MASK) << self::MORTON3D_BIT_SIZE);
407 $extraZ = ((($baseY >> self::BLOCKHASH_Z_SHIFT) & self::BLOCKHASH_XZ_EXTRA_MASK) << self::MORTON3D_BIT_SIZE);
408
409 $x = (($baseX & self::BLOCKHASH_XZ_MASK) | $extraX) << self::BLOCKHASH_XZ_SIGN_SHIFT >> self::BLOCKHASH_XZ_SIGN_SHIFT;
410 $y = ($baseY & self::BLOCKHASH_Y_MASK) - self::BLOCKHASH_Y_OFFSET;
411 $z = (($baseZ & self::BLOCKHASH_XZ_MASK) | $extraZ) << self::BLOCKHASH_XZ_SIGN_SHIFT >> self::BLOCKHASH_XZ_SIGN_SHIFT;
412 }

589 public function onUnload() : void{
590 if($this->unloaded){
591 throw new \LogicException("Tried to close a world which is already closed");
592 }
593
594 foreach($this->unloadCallbacks as $callback){
595 $callback();
596 }
597 $this->unloadCallbacks = [];
598
599 foreach($this->chunks as $chunkHash => $chunk){
600 self::getXZ($chunkHash, $chunkX, $chunkZ);
601 $this->unloadChunk($chunkX, $chunkZ, false);
602 }
603 foreach($this->entitiesByChunk as $chunkHash => $entities){
604 self::getXZ($chunkHash, $chunkX, $chunkZ);
605
606 $leakedEntities = 0;
607 foreach($entities as $entity){
608 if(!$entity->isFlaggedForDespawn()){
609 $leakedEntities++;
610 }
611 $entity->close();
612 }
613 if($leakedEntities !== 0){
614 $this->logger->warning("$leakedEntities leaked entities found in ungenerated chunk $chunkX $chunkZ during unload, they won't be saved!");
615 }
616 }
617
618 $this->save();
619
620 $this->unregisterGenerator();
621
622 $this->provider->close();
623 $this->blockCache = [];
624
625 $this->unloaded = true;
626 }

760 if(!isset($this->tickingLoaders[$loaderId])){
761 $this->tickingLoaderCounter[$loaderId] = 1;
762 $this->tickingLoaders[$loaderId] = $loader;
763 }else{
764 ++$this->tickingLoaderCounter[$loaderId];
765 }
766 }
767
768 $this->cancelUnloadChunkRequest($chunkX, $chunkZ);
769
770 if($autoLoad){
771 $this->loadChunk($chunkX, $chunkZ);
772 }

784 $this->chunkPopulationRequestMap[$chunkHash]->reject();
785 unset($this->chunkPopulationRequestMap[$chunkHash]);
786 }
787 }
788
789 if(isset($this->tickingLoaderCounter[$loaderId]) && --$this->tickingLoaderCounter[$loaderId] === 0){
790 unset($this->tickingLoaderCounter[$loaderId]);
791 unset($this->tickingLoaders[$loaderId]);
792 }
