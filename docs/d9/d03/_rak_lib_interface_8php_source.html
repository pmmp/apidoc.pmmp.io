120 public function start() : void{
121 $this->server->getTickSleeper()->addNotifier($this->sleeper, function() : void{
122 while($this->eventReceiver->handle($this));

124 $this->server->getLogger()->debug("Waiting for RakLib to start...");
125 try{
126 $this->rakLib->startAndWait();
127 }catch(SocketException $e){
128 throw new NetworkInterfaceStartException($e->getMessage(), 0, $e);
129 }
130 $this->server->getLogger()->debug("RakLib booted successfully");
131 }

181 public function onPacketReceive(int $sessionId, string $packet) : void{
182 if(isset($this->sessions[$sessionId])){
183 if($packet === "" || $packet[0] !== self::MCPE_RAKNET_PACKET_ID){
184 $this->sessions[$sessionId]->getLogger()->debug("Non-FE packet received: " . base64_encode($packet));
185 return;
186 }
187 //get this now for blocking in case the player was closed before the exception was raised
188 $session = $this->sessions[$sessionId];
189 $address = $session->getIp();
190 $buf = substr($packet, 1);
191 try{
192 $session->handleEncoded($buf);
193 }catch(PacketHandlingException $e){
194 $errorId = bin2hex(random_bytes(6));
195
196 $logger = $session->getLogger();
197 $logger->error("Bad packet (error ID $errorId): " . $e->getMessage());
198
199 //intentionally doesn't use logException, we don't want spammy packet error traces to appear in release mode
200 $logger->debug(implode("\n", Utils::printableExceptionInfo($e)));
201 $session->disconnect("Packet processing error (Error ID: $errorId)");
202 $this->interface->blockAddress($address, 5);
203 }
204 }
