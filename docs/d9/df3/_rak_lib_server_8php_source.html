104 public function shutdownHandler(){

106 $error = error_get_last();
107
108 if($error !== null){
109 $this->logger->emergency("Fatal error: " . $error["message"] . " in " . $error["file"] . " on line " . $error["line"]);
110 $this->setCrashInfo(RakLibThreadCrashInfo::fromLastErrorInfo($error));

128 public function startAndWait(int $options = PTHREADS_INHERIT_NONE) : void{
129 $this->start($options);
130 $this->synchronized(function() : void{
131 while(!$this->ready && $this->crashInfo === null){
132 $this->wait();
133 }
134 $crashInfo = $this->crashInfo;
135 if($crashInfo !== null){
136 if($crashInfo->getClass() === SocketException::class){
137 throw new SocketException($crashInfo->getMessage());
138 }
139 throw new \RuntimeException("RakLib failed to start: " . $crashInfo->makePrettyMessage());
140 }
141 });

144 protected function onRun() : void{
145 try{
146 gc_enable();
147 ini_set("display_errors", '1');
148 ini_set("display_startup_errors", '1');
149
150 register_shutdown_function([$this, "shutdownHandler"]);
151
152 try{
153 $socket = new Socket($this->address);
154 }catch(SocketException $e){
155 $this->setCrashInfo(RakLibThreadCrashInfo::fromThrowable($e));
156 return;
157 }
158 $manager = new Server(
159 $this->serverId,
160 $this->logger,
161 $socket,
162 $this->maxMtuSize,
163 new SimpleProtocolAcceptor($this->protocolVersion),
164 new UserToRakLibThreadMessageReceiver(new PthreadsChannelReader($this->mainToThreadBuffer)),
165 new RakLibToUserThreadMessageSender(new SnoozeAwarePthreadsChannelWriter($this->threadToMainBuffer, $this->mainThreadNotifier)),
166 new ExceptionTraceCleaner($this->mainPath)
167 );
168 $this->synchronized(function() : void{
169 $this->ready = true;
170 $this->notify();
171 });
172 while(!$this->isKilled){
173 $manager->tickProcessor();
174 }
175 $manager->waitShutdown();
176 $this->cleanShutdown = true;
177 }catch(\Throwable $e){
178 $this->setCrashInfo(RakLibThreadCrashInfo::fromThrowable($e));
179 $this->logger->logException($e);
180 }
