49 public function execute(CommandSender $sender, string $commandLabel, array $args){
50 if(isset($args[0])){
51 $args[0] = strtolower($args[0]);
52 if($args[0] === "ips"){
53 $list = $sender->getServer()->getIPBans();
54 }elseif($args[0] === "players"){
55 $list = $sender->getServer()->getNameBans();
56 }else{
57 throw new InvalidCommandSyntaxException();
58 }
59 }else{
60 $list = $sender->getServer()->getNameBans();
61 $args[0] = "players";
62 }
63
64 $list = array_map(function(BanEntry $entry) : string{
65 return $entry->getName();
66 }, $list->getEntries());
67 sort($list, SORT_STRING);
68 $message = implode(", ", $list);
69
70 if($args[0] === "ips"){
71 $sender->sendMessage(KnownTranslationFactory::commands_banlist_ips((string) count($list)));
72 }else{
73 $sender->sendMessage(KnownTranslationFactory::commands_banlist_players((string) count($list)));
74 }
75
76 $sender->sendMessage($message);
77
78 return true;
