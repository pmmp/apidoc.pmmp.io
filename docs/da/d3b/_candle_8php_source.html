39class Candle extends Transparent implements Lightable{
40 use CandleTrait {
41 describeBlockOnlyState as encodeLitState;
42 getLightLevel as getBaseLightLevel;
43 }
44
45 public const MIN_COUNT = 1;
46 public const MAX_COUNT = 4;
47
48 private int $count = self::MIN_COUNT;
49
50 protected function describeBlockOnlyState(RuntimeDataDescriber $w) : void{
51 $this->encodeLitState($w);
52 $w->boundedIntAuto(self::MIN_COUNT, self::MAX_COUNT, $this->count);
53 }
54
55 public function getCount() : int{ return $this->count; }
56
58 public function setCount(int $count) : self{
59 if($count < self::MIN_COUNT || $count > self::MAX_COUNT){
60 throw new \InvalidArgumentException("Count must be in range " . self::MIN_COUNT . " ... " . self::MAX_COUNT);
61 }
62 $this->count = $count;
63 return $this;
64 }
65
66 public function getLightLevel() : int{
67 return $this->getBaseLightLevel() * $this->count;
68 }
69
70 protected function recalculateCollisionBoxes() : array{
71 return [
72 (match($this->count){
73 1 => AxisAlignedBB::one()
74 ->squash(Axis::X, 7 / 16)
75 ->squash(Axis::Z, 7 / 16),
76 2 => AxisAlignedBB::one()
77 ->squash(Axis::X, 5 / 16)
78 ->trim(Facing::NORTH, 7 / 16) //0.3 thick on the Z axis
79 ->trim(Facing::SOUTH, 6 / 16),
80 3 => AxisAlignedBB::one()
81 ->trim(Facing::WEST, 5 / 16)
82 ->trim(Facing::EAST, 6 / 16)
83 ->trim(Facing::NORTH, 6 / 16)
84 ->trim(Facing::SOUTH, 5 / 16),
85 4 => AxisAlignedBB::one()
86 ->squash(Axis::X, 5 / 16)
87 ->trim(Facing::NORTH, 5 / 16)
88 ->trim(Facing::SOUTH, 6 / 16),
89 default => throw new AssumptionFailedError("Unreachable")
90 })->trim(Facing::UP, 10 / 16)
91 ];
92 }
93
94 public function getSupportType(int $facing) : SupportType{
95 return SupportType::NONE;
96 }
97
98 protected function getCandleIfCompatibleType(Block $block) : ?Candle{
99 return $block instanceof Candle && $block->hasSameTypeId($this) ? $block : null;
100 }
101
102 public function canBePlacedAt(Block $blockReplace, Vector3 $clickVector, int $face, bool $isClickedBlock) : bool{
103 $candle = $this->getCandleIfCompatibleType($blockReplace);
104 return $candle !== null ? $candle->count < self::MAX_COUNT : parent::canBePlacedAt($blockReplace, $clickVector, $face, $isClickedBlock);
105 }
106
107 public function place(BlockTransaction $tx, Item $item, Block $blockReplace, Block $blockClicked, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
108 if(!$blockReplace->getAdjacentSupportType(Facing::DOWN)->hasCenterSupport()){
109 return false;
110 }
111 $existing = $this->getCandleIfCompatibleType($blockReplace);
112 if($existing !== null){
113 if($existing->count >= self::MAX_COUNT){
114 return false;
115 }
116
117 $this->count = $existing->count + 1;
118 $this->lit = $existing->lit;
119 }
120 return parent::place($tx, $item, $blockReplace, $blockClicked, $face, $clickVector, $player);
121 }
122
123 public function getDropsForCompatibleTool(Item $item) : array{
124 return [$this->asItem()->setCount($this->count)];
125 }
