555 public function createPlayer(NetworkSession $session, PlayerInfo $playerInfo, bool $authenticated, ?CompoundTag $offlinePlayerData) : Promise{
556 $ev = new PlayerCreationEvent($session);
557 $ev->call();
558 $class = $ev->getPlayerClass();
559
560 if($offlinePlayerData !== null && ($world = $this->worldManager->getWorldByName($offlinePlayerData->getString(Player::TAG_LEVEL, ""))) !== null){
561 $playerPos = EntityDataHelper::parseLocation($offlinePlayerData, $world);
562 }else{
563 $world = $this->worldManager->getDefaultWorld();
564 if($world === null){
565 throw new AssumptionFailedError("Default world should always be loaded");
566 }
567 $playerPos = null;
568 }
570 $playerPromiseResolver = new PromiseResolver();
571
572 $createPlayer = function(Location $location) use ($playerPromiseResolver, $class, $session, $playerInfo, $authenticated, $offlinePlayerData) : void{
573 $player = new $class($this, $session, $playerInfo, $authenticated, $location, $offlinePlayerData);
574 if(!$player->hasPlayedBefore()){
575 $player->onGround = true; //TODO: this hack is needed for new players in-air ticks - they don't get detected as on-ground until they move
576 }
577 $playerPromiseResolver->resolve($player);
578 };
579
580 if($playerPos === null){ //new player or no valid position due to world not being loaded
581 $world->requestSafeSpawn()->onCompletion(
582 function(Position $spawn) use ($createPlayer, $playerPromiseResolver, $session, $world) : void{
583 if(!$session->isConnected()){
584 $playerPromiseResolver->reject();
585 return;
586 }
587 $createPlayer(Location::fromObject($spawn, $world));
588 },
589 function() use ($playerPromiseResolver, $session) : void{
590 if($session->isConnected()){
591 //TODO: this needs to be localized - this might be reached if the spawn world was unloaded while the player was logging in
592 $session->disconnect("Failed to find a safe spawn location");
593 }
594 $playerPromiseResolver->reject();
595 }
596 );
597 }else{ //returning player with a valid position - safe spawn not required
598 $createPlayer($playerPos);
599 }
600
601 return $playerPromiseResolver->getPromise();
602 }

614 public function getPlayerByPrefix(string $name) : ?Player{
615 $found = null;
616 $name = strtolower($name);
617 $delta = PHP_INT_MAX;
618 foreach($this->getOnlinePlayers() as $player){
619 if(stripos($player->getName(), $name) === 0){
620 $curDelta = strlen($player->getName()) - strlen($name);
621 if($curDelta < $delta){
622 $found = $player;
623 $delta = $curDelta;
624 }
625 if($curDelta === 0){
626 break;
627 }
628 }
629 }
630
631 return $found;
632 }
