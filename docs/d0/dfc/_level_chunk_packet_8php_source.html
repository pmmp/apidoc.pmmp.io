89 protected function decodePayload(PacketSerializer $in) : void{
90 $this->chunkX = $in->getVarInt();
91 $this->chunkZ = $in->getVarInt();
92
93 $subChunkCountButNotReally = $in->getUnsignedVarInt();
94 if($subChunkCountButNotReally === self::CLIENT_REQUEST_FULL_COLUMN_FAKE_COUNT){
95 $this->clientSubChunkRequestsEnabled = true;
96 $this->subChunkCount = PHP_INT_MAX;
97 }elseif($subChunkCountButNotReally === self::CLIENT_REQUEST_TRUNCATED_COLUMN_FAKE_COUNT){
98 $this->clientSubChunkRequestsEnabled = true;
99 $this->subChunkCount = $in->getLShort();
100 }else{
101 $this->clientSubChunkRequestsEnabled = false;
102 $this->subChunkCount = $subChunkCountButNotReally;
103 }
104
105 $cacheEnabled = $in->getBool();
106 if($cacheEnabled){
107 $this->usedBlobHashes = [];
108 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
109 $this->usedBlobHashes[] = $in->getLLong();
110 }
111 }
112 $this->extraPayload = $in->getString();
113 }

115 protected function encodePayload(PacketSerializer $out) : void{
116 $out->putVarInt($this->chunkX);
117 $out->putVarInt($this->chunkZ);
118
119 if($this->clientSubChunkRequestsEnabled){
120 if($this->subChunkCount === PHP_INT_MAX){
121 $out->putUnsignedVarInt(self::CLIENT_REQUEST_FULL_COLUMN_FAKE_COUNT);
122 }else{
123 $out->putUnsignedVarInt(self::CLIENT_REQUEST_TRUNCATED_COLUMN_FAKE_COUNT);
124 $out->putLShort($this->subChunkCount);
125 }
126 }else{
127 $out->putUnsignedVarInt($this->subChunkCount);
128 }
129
130 $out->putBool($this->usedBlobHashes !== null);
131 if($this->usedBlobHashes !== null){
132 $out->putUnsignedVarInt(count($this->usedBlobHashes));
133 foreach($this->usedBlobHashes as $hash){
134 $out->putLLong($hash);
135 }
136 }
137 $out->putString($this->extraPayload);
138 }
