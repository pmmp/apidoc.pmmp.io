58 public static function init() : void{
59 $timezone = Utils::assumeNotFalse(ini_get("date.timezone"), "date.timezone should always be set in ini");
60 if($timezone !== ""){
61 /*
62 * This is here so that people don't come to us complaining and fill up the issue tracker when they put
63 * an incorrect timezone abbreviation in php.ini apparently.
64 */
65 if(!str_contains($timezone, "/")){
66 $default_timezone = timezone_name_from_abbr($timezone);
67 if($default_timezone !== false){
68 ini_set("date.timezone", $default_timezone);
69 date_default_timezone_set($default_timezone);
70 return;
71 }
72
73 //Bad php.ini value, try another method to detect timezone
74 \GlobalLogger::get()->warning("Timezone \"$timezone\" could not be parsed as a valid timezone from php.ini, falling back to auto-detection");
75 }else{
76 date_default_timezone_set($timezone);
77 return;
78 }
79 }
80
81 if(($timezone = self::detectSystemTimezone()) !== false && date_default_timezone_set($timezone)){
82 //Success! Timezone has already been set and validated in the if statement.
83 //This here is just for redundancy just in case some program wants to read timezone data from the ini.
84 ini_set("date.timezone", $timezone);
85 return;
86 }
87
88 if(($response = Internet::getURL("http://ip-api.com/json")) !== null //If system timezone detection fails or timezone is an invalid value.
89 && is_array($ip_geolocation_data = json_decode($response->getBody(), true))
90 && isset($ip_geolocation_data['status'])
91 && $ip_geolocation_data['status'] !== 'fail'
92 && is_string($ip_geolocation_data['timezone'])
93 && date_default_timezone_set($ip_geolocation_data['timezone'])
94 ){
95 //Again, for redundancy.
96 ini_set("date.timezone", $ip_geolocation_data['timezone']);
97 return;
98 }
99
100 ini_set("date.timezone", "UTC");
101 date_default_timezone_set("UTC");
102 \GlobalLogger::get()->warning("Timezone could not be automatically determined or was set to an invalid value. An incorrect timezone will result in incorrect timestamps on console logs. It has been set to \"UTC\" by default. You can change it on the php.ini file.");
103 }

108 public static function detectSystemTimezone(){
109 switch(Utils::getOS()){
110 case Utils::OS_WINDOWS:
111 $regex = '/(UTC)(\+*\-*\d*\d*\:*\d*\d*)/';
112
113 /*
114 * wmic timezone get Caption
115 * Get the timezone offset
116 *
117 * Sample Output var_dump
118 * array(3) {
119 * [0] =>
120 * string(7) "Caption"
121 * [1] =>
122 * string(20) "(UTC+09:30) Adelaide"
123 * [2] =>
124 * string(0) ""
125 * }
126 */
127 exec("wmic timezone get Caption", $output);
128
129 $string = trim(implode("\n", $output));
130
131 //Detect the Time Zone string
132 preg_match($regex, $string, $matches);
133
134 if(!isset($matches[2])){
135 return false;
136 }
137
138 $offset = $matches[2];
139
140 if($offset == ""){
141 return "UTC";
142 }
143
144 return self::parseOffset($offset);
145 case Utils::OS_LINUX:
146 // Ubuntu / Debian.
147 $data = @file_get_contents('/etc/timezone');
148 if($data !== false){
149 return trim($data);
150 }
151
152 // RHEL / CentOS
153 $data = @parse_ini_file('/etc/sysconfig/clock');
154 if($data !== false && isset($data['ZONE']) && is_string($data['ZONE'])){
155 return trim($data['ZONE']);
156 }
157
158 //Portable method for incompatible linux distributions.
159
160 $offset = trim(exec('date +%:z'));
161
162 if($offset == "+00:00"){
163 return "UTC";
164 }
165
166 return self::parseOffset($offset);
167 case Utils::OS_MACOS:
168 $filename = @readlink('/etc/localtime');
169 if($filename !== false && str_starts_with($filename, '/usr/share/zoneinfo/')){
170 $timezone = substr($filename, 20);
171 return trim($timezone);
172 }
173
174 return false;
175 default:
176 return false;
177 }
178 }

185 private static function parseOffset($offset){
186 //Make signed offsets unsigned for date_parse
187 if(str_starts_with($offset, '-')){
188 $negative_offset = true;
189 $offset = str_replace('-', '', $offset);
190 }else{
191 if(str_starts_with($offset, '+')){
192 $negative_offset = false;
193 $offset = str_replace('+', '', $offset);
194 }else{
195 return false;
196 }
197 }
198
199 $parsed = date_parse($offset);
200 $offset = $parsed['hour'] * 3600 + $parsed['minute'] * 60 + $parsed['second'];
201
202 //After date_parse is done, put the sign back
203 if($negative_offset == true){
204 $offset = -abs($offset);
205 }
206
207 //And then, look the offset up.
208 //timezone_name_from_abbr is not used because it returns false on some(most) offsets because it's mapping function is weird.
209 //That's been a bug in PHP since 2008!
210 foreach(timezone_abbreviations_list() as $zones){
211 foreach($zones as $timezone){
212 if($timezone['timezone_id'] !== null && $timezone['offset'] == $offset){
213 return $timezone['timezone_id'];
214 }
215 }
216 }
217
218 return false;
219 }
