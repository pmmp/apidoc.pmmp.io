101 private static function getOffsetSeed(int $x, int $y, int $z) : int{
102 $p1 = gmp_mul($z, 0x6ebfff5);
103 $p2 = gmp_mul($x, 0x2fc20f);
104 $p3 = $y;
105
106 $xord = gmp_xor(gmp_xor($p1, $p2), $p3);
107
108 $fullResult = gmp_mul(gmp_add(gmp_mul($xord, 0x285b825), 0xb), $xord);
109 return gmp_intval(gmp_and($fullResult, 0xffffffff));
110 }

112 private static function getMaxHeight(int $x, int $z) : int{

116 public function getModelPositionOffset() : ?Vector3{
117 $seed = self::getOffsetSeed($this->position->getFloorX(), 0, $this->position->getFloorZ());
118 $retX = (($seed % 12) + 1) / 16;
119 $retZ = ((($seed >> 8) % 12) + 1) / 16;

164 private function grow(int $maxHeight, int $growAmount, ?Player $player) : bool{
165 $world = $this->position->getWorld();
166 if(!$world->getBlock($this->position->up())->canBeReplaced()){
167 return false;
168 }
169
170 $height = 1;
171 while($world->getBlock($this->position->subtract(0, $height, 0))->hasSameTypeId($this)){
172 if(++$height >= $maxHeight){
173 return false;
174 }
175 }
176
177 $newHeight = $height + $growAmount;
178
179 $stemBlock = (clone $this)->setReady(false)->setLeafSize(self::NO_LEAVES);
180 if($newHeight >= 4 && !$stemBlock->isThick()){ //don't change it to false if height is less, because it might have been chopped
181 $stemBlock = $stemBlock->setThick(true);
182 }
183 $smallLeavesBlock = (clone $stemBlock)->setLeafSize(self::SMALL_LEAVES);
184 $bigLeavesBlock = (clone $stemBlock)->setLeafSize(self::LARGE_LEAVES);
185
186 $newBlocks = [];
187 if($newHeight === 2){
188 $newBlocks[] = $smallLeavesBlock;
189 }elseif($newHeight === 3){
190 $newBlocks[] = $smallLeavesBlock;
191 $newBlocks[] = $smallLeavesBlock;
192 }elseif($newHeight === 4){
193 $newBlocks[] = $bigLeavesBlock;
194 $newBlocks[] = $smallLeavesBlock;
195 $newBlocks[] = $stemBlock;
196 $newBlocks[] = $stemBlock;
197 }elseif($newHeight > 4){
198 $newBlocks[] = $bigLeavesBlock;
199 $newBlocks[] = $bigLeavesBlock;
200 $newBlocks[] = $smallLeavesBlock;
201 for($i = 0, $max = min($growAmount, $newHeight - count($newBlocks)); $i < $max; ++$i){
202 $newBlocks[] = $stemBlock; //to replace the bottom blocks that currently have leaves
203 }
204 }
205
206 $tx = new BlockTransaction($world);
207 foreach($newBlocks as $idx => $newBlock){
208 $tx->addBlock($this->position->subtract(0, $idx - $growAmount, 0), $newBlock);
209 }
210
211 $ev = new StructureGrowEvent($this, $tx, $player);
212 $ev->call();
213 if($ev->isCancelled()){
214 return false;
215 }
216
217 return $tx->apply();
218 }
