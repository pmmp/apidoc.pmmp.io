975 public function register(Block $block, bool $override = false) : void{
976 $variant = $block->getIdInfo()->getVariant();
977
978 $stateMask = $block->getStateBitmask();
979 if(($variant & $stateMask) !== 0){
980 throw new \InvalidArgumentException("Block variant collides with state bitmask");
981 }
982
983 foreach($block->getIdInfo()->getAllBlockIds() as $id){
984 if(!$override && $this->isRegistered($id, $variant)){
985 throw new \InvalidArgumentException("Block registration $id:$variant conflicts with an existing block");
986 }
987
988 for($m = $variant; $m <= ($variant | $stateMask); ++$m){
989 if(($m & ~$stateMask) !== $variant){

991 }
