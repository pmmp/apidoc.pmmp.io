164 protected function deserializePaletted(BinaryStream $stream) : PalettedBlockArray{
165 $bitsPerBlock = $stream->getByte() >> 1;
166
167 try{
168 $words = $stream->get(PalettedBlockArray::getExpectedWordArraySize($bitsPerBlock));
169 }catch(\InvalidArgumentException $e){
170 throw new CorruptedChunkException("Failed to deserialize paletted storage: " . $e->getMessage(), 0, $e);
171 }
172 $nbt = new LittleEndianNbtSerializer();
173 $palette = [];
174 $idMap = LegacyBlockIdToStringIdMap::getInstance();
175 for($i = 0, $paletteSize = $stream->getLInt(); $i < $paletteSize; ++$i){
176 try{
177 $offset = $stream->getOffset();
178
179 $tag = $nbt->read($stream->getBuffer(), $offset)->mustGetCompoundTag();
180 $stream->setOffset($offset);
181
182 $id = $idMap->stringToLegacy($tag->getString("name")) ?? BlockLegacyIds::INFO_UPDATE;
183 $data = $tag->getShort("val");
184 $palette[] = ($id << Block::INTERNAL_METADATA_BITS) | $data;
185 }catch(NbtException $e){
186 throw new CorruptedChunkException("Invalid blockstate NBT at offset $i in paletted storage: " . $e->getMessage(), 0, $e);
187 }
188 }
189
190 //TODO: exceptions
191 return PalettedBlockArray::fromData($bitsPerBlock, $words, $palette);
192 }

199 protected static function deserializeExtraDataKey(int $chunkVersion, int $key, ?int &$x, ?int &$y, ?int &$z) : void{
200 if($chunkVersion >= ChunkVersion::v1_0_0){
201 $x = ($key >> 12) & 0xf;
202 $z = ($key >> 8) & 0xf;
203 $y = $key & 0xff;
204 }else{ //pre-1.0, 7 bits were used because the build height limit was lower
205 $x = ($key >> 11) & 0xf;
206 $z = ($key >> 7) & 0xf;
207 $y = $key & 0x7f;
208 }
209 }

214 protected function deserializeLegacyExtraData(string $index, int $chunkVersion) : array{
215 if(($extraRawData = $this->db->get($index . ChunkDataKey::LEGACY_BLOCK_EXTRA_DATA)) === false || $extraRawData === ""){
216 return [];
217 }
218
220 $extraDataLayers = [];
221 $binaryStream = new BinaryStream($extraRawData);
222 $count = $binaryStream->getLInt();
223 for($i = 0; $i < $count; ++$i){
224 $key = $binaryStream->getLInt();
225 $value = $binaryStream->getLShort();
226
227 self::deserializeExtraDataKey($chunkVersion, $key, $x, $fullY, $z);
228
229 $ySub = ($fullY >> SubChunk::COORD_BIT_SIZE);
230 $y = $key & SubChunk::COORD_MASK;
231
232 $blockId = $value & 0xff;
233 $blockData = ($value >> 8) & 0xf;
234 if(!isset($extraDataLayers[$ySub])){
235 $extraDataLayers[$ySub] = new PalettedBlockArray(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS);
236 }
237 $extraDataLayers[$ySub]->set($x, $y, $z, ($blockId << Block::INTERNAL_METADATA_BITS) | $blockData);
238 }
239
240 return $extraDataLayers;
241 }

246 public function loadChunk(int $chunkX, int $chunkZ) : ?ChunkData{
247 $index = LevelDB::chunkIndex($chunkX, $chunkZ);
248
249 $chunkVersionRaw = $this->db->get($index . ChunkDataKey::OLD_VERSION);
250 if($chunkVersionRaw === false){
251 return null;
252 }
253
255 $subChunks = [];
256
258 $biomeArray = null;
259
260 $chunkVersion = ord($chunkVersionRaw);
261 $hasBeenUpgraded = $chunkVersion < self::CURRENT_LEVEL_CHUNK_VERSION;
262
263 switch($chunkVersion){
264 case ChunkVersion::v1_12_0_4_beta:
265 case ChunkVersion::v1_11_1:
266 case ChunkVersion::v1_11_0_4_beta:
267 case ChunkVersion::v1_11_0_3_beta:
268 case ChunkVersion::v1_11_0_1_beta:
269 case ChunkVersion::v1_9_0:
270 case ChunkVersion::v1_8_0:
271 case ChunkVersion::v1_2_13:
272 case ChunkVersion::v1_2_0:
273 case ChunkVersion::v1_2_0_2_beta:
274 case ChunkVersion::v1_1_0:
275 //TODO: check beds
276 case ChunkVersion::v1_0_0:
277 $convertedLegacyExtraData = $this->deserializeLegacyExtraData($index, $chunkVersion);
278
279 for($y = Chunk::MIN_SUBCHUNK_INDEX; $y <= Chunk::MAX_SUBCHUNK_INDEX; ++$y){
280 if(($data = $this->db->get($index . ChunkDataKey::SUBCHUNK . chr($y))) === false){
281 continue;
282 }
283
284 $binaryStream = new BinaryStream($data);
285 if($binaryStream->feof()){
286 throw new CorruptedChunkException("Unexpected empty data for subchunk $y");
287 }
288 $subChunkVersion = $binaryStream->getByte();
289 if($subChunkVersion < self::CURRENT_LEVEL_SUBCHUNK_VERSION){
290 $hasBeenUpgraded = true;
291 }
292
293 switch($subChunkVersion){
294 case SubChunkVersion::CLASSIC:
295 case SubChunkVersion::CLASSIC_BUG_2: //these are all identical to version 0, but vanilla respects these so we should also
296 case SubChunkVersion::CLASSIC_BUG_3:
297 case SubChunkVersion::CLASSIC_BUG_4:
298 case SubChunkVersion::CLASSIC_BUG_5:
299 case SubChunkVersion::CLASSIC_BUG_6:
300 case SubChunkVersion::CLASSIC_BUG_7:
301 try{
302 $blocks = $binaryStream->get(4096);
303 $blockData = $binaryStream->get(2048);
304
305 if($chunkVersion < ChunkVersion::v1_1_0){
306 $binaryStream->get(4096); //legacy light info, discard it
307 $hasBeenUpgraded = true;
308 }
309 }catch(BinaryDataException $e){
310 throw new CorruptedChunkException($e->getMessage(), 0, $e);
311 }
312
313 $storages = [SubChunkConverter::convertSubChunkXZY($blocks, $blockData)];
314 if(isset($convertedLegacyExtraData[$y])){
315 $storages[] = $convertedLegacyExtraData[$y];
316 }
317
318 $subChunks[$y] = new SubChunk(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS, $storages);
319 break;
320 case SubChunkVersion::PALETTED_SINGLE:
321 $storages = [$this->deserializePaletted($binaryStream)];
322 if(isset($convertedLegacyExtraData[$y])){
323 $storages[] = $convertedLegacyExtraData[$y];
324 }
325 $subChunks[$y] = new SubChunk(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS, $storages);
326 break;
327 case SubChunkVersion::PALETTED_MULTI:
328 //legacy extradata layers intentionally ignored because they aren't supposed to exist in v8
329 $storageCount = $binaryStream->getByte();
330 if($storageCount > 0){
331 $storages = [];
332
333 for($k = 0; $k < $storageCount; ++$k){
334 $storages[] = $this->deserializePaletted($binaryStream);
335 }
336 $subChunks[$y] = new SubChunk(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS, $storages);
337 }
338 break;
339 default:
340 //TODO: set chunks read-only so the version on disk doesn't get overwritten
341 throw new CorruptedChunkException("don't know how to decode LevelDB subchunk format version $subChunkVersion");
342 }
343 }
344
345 if(($maps2d = $this->db->get($index . ChunkDataKey::HEIGHTMAP_AND_2D_BIOMES)) !== false){
346 $binaryStream = new BinaryStream($maps2d);
347
348 try{
349 $binaryStream->get(512); //heightmap, discard it
350 $biomeArray = new BiomeArray($binaryStream->get(256)); //never throws
351 }catch(BinaryDataException $e){
352 throw new CorruptedChunkException($e->getMessage(), 0, $e);
353 }
354 }
355 break;
356 case ChunkVersion::v0_9_5:
357 case ChunkVersion::v0_9_2:
358 case ChunkVersion::v0_9_0:
359 $convertedLegacyExtraData = $this->deserializeLegacyExtraData($index, $chunkVersion);
360
361 $legacyTerrain = $this->db->get($index . ChunkDataKey::LEGACY_TERRAIN);
362 if($legacyTerrain === false){
363 throw new CorruptedChunkException("Missing expected LEGACY_TERRAIN tag for format version $chunkVersion");
364 }
365 $binaryStream = new BinaryStream($legacyTerrain);
366 try{
367 $fullIds = $binaryStream->get(32768);
368 $fullData = $binaryStream->get(16384);
369 $binaryStream->get(32768); //legacy light info, discard it
370 }catch(BinaryDataException $e){
371 throw new CorruptedChunkException($e->getMessage(), 0, $e);
372 }
373
374 for($yy = 0; $yy < 8; ++$yy){
375 $storages = [SubChunkConverter::convertSubChunkFromLegacyColumn($fullIds, $fullData, $yy)];
376 if(isset($convertedLegacyExtraData[$yy])){
377 $storages[] = $convertedLegacyExtraData[$yy];
378 }
379 $subChunks[$yy] = new SubChunk(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS, $storages);
380 }
381
382 try{
383 $binaryStream->get(256); //heightmap, discard it
385 $unpackedBiomeArray = unpack("N*", $binaryStream->get(1024)); //unpack() will never fail here
386 $biomeArray = new BiomeArray(ChunkUtils::convertBiomeColors(array_values($unpackedBiomeArray))); //never throws
387 }catch(BinaryDataException $e){
388 throw new CorruptedChunkException($e->getMessage(), 0, $e);
389 }
390 break;
391 default:
392 //TODO: set chunks read-only so the version on disk doesn't get overwritten
393 throw new CorruptedChunkException("don't know how to decode chunk format version $chunkVersion");
394 }
395
396 $nbt = new LittleEndianNbtSerializer();
397
399 $entities = [];
400 if(($entityData = $this->db->get($index . ChunkDataKey::ENTITIES)) !== false && $entityData !== ""){
401 try{
402 $entities = array_map(fn(TreeRoot $root) => $root->mustGetCompoundTag(), $nbt->readMultiple($entityData));
403 }catch(NbtDataException $e){
404 throw new CorruptedChunkException($e->getMessage(), 0, $e);
405 }
406 }
407
409 $tiles = [];
410 if(($tileData = $this->db->get($index . ChunkDataKey::BLOCK_ENTITIES)) !== false && $tileData !== ""){
411 try{
412 $tiles = array_map(fn(TreeRoot $root) => $root->mustGetCompoundTag(), $nbt->readMultiple($tileData));
413 }catch(NbtDataException $e){
414 throw new CorruptedChunkException($e->getMessage(), 0, $e);
415 }
416 }
417
418 $finalisationChr = $this->db->get($index . ChunkDataKey::FINALIZATION);
419 if($finalisationChr !== false){
420 $finalisation = ord($finalisationChr);
421 $terrainPopulated = $finalisation === self::FINALISATION_DONE;
422 }else{ //older versions didn't have this tag
423 $terrainPopulated = true;
424 }
425
426 //TODO: tile ticks, biome states (?)
427
428 $chunk = new Chunk(
429 $subChunks,
430 $biomeArray ?? BiomeArray::fill(BiomeIds::OCEAN), //TODO: maybe missing biomes should be an error?
431 $terrainPopulated
432 );
433
434 if($hasBeenUpgraded){
435 $chunk->setTerrainDirty(); //trigger rewriting chunk to disk if it was converted from an older format
436 }
437
438 return new ChunkData($chunk, $entities, $tiles);
439 }

441 public function saveChunk(int $chunkX, int $chunkZ, ChunkData $chunkData) : void{
442 $idMap = LegacyBlockIdToStringIdMap::getInstance();
443 $index = LevelDB::chunkIndex($chunkX, $chunkZ);
444
445 $write = new \LevelDBWriteBatch();
446 $write->put($index . ChunkDataKey::OLD_VERSION, chr(self::CURRENT_LEVEL_CHUNK_VERSION));
447
448 $chunk = $chunkData->getChunk();
449 if($chunk->getTerrainDirtyFlag(Chunk::DIRTY_FLAG_BLOCKS)){
450 $subChunks = $chunk->getSubChunks();
451 foreach($subChunks as $y => $subChunk){
452 $key = $index . ChunkDataKey::SUBCHUNK . chr($y);
453 if($subChunk->isEmptyAuthoritative()){
454 $write->delete($key);
455 }else{
456 $subStream = new BinaryStream();
457 $subStream->putByte(self::CURRENT_LEVEL_SUBCHUNK_VERSION);
458
459 $layers = $subChunk->getBlockLayers();
460 $subStream->putByte(count($layers));
461 foreach($layers as $blocks){
462 if($blocks->getBitsPerBlock() !== 0){
463 $subStream->putByte($blocks->getBitsPerBlock() << 1);
464 $subStream->put($blocks->getWordArray());
465 }else{
466 //TODO: we use these in-memory, but they aren't supported on disk by the game yet
467 //polyfill them with a zero'd 1-bpb instead
468 $subStream->putByte(1 << 1);
469 $subStream->put(str_repeat("\x00", PalettedBlockArray::getExpectedWordArraySize(1)));
470 }
471
472 $palette = $blocks->getPalette();
473 $subStream->putLInt(count($palette));
474 $tags = [];
475 foreach($palette as $p){
476 $tags[] = new TreeRoot(CompoundTag::create()
477 ->setString("name", $idMap->legacyToString($p >> Block::INTERNAL_METADATA_BITS) ?? "minecraft:info_update")
478 ->setInt("oldid", $p >> Block::INTERNAL_METADATA_BITS) //PM only (debugging), vanilla doesn't have this
479 ->setShort("val", $p & Block::INTERNAL_METADATA_MASK));
480 }
481
482 $subStream->put((new LittleEndianNbtSerializer())->writeMultiple($tags));
483 }
484
485 $write->put($key, $subStream->getBuffer());
486 }
487 }
488 }
489
490 if($chunk->getTerrainDirtyFlag(Chunk::DIRTY_FLAG_BIOMES)){
491 $write->put($index . ChunkDataKey::HEIGHTMAP_AND_2D_BIOMES, str_repeat("\x00", 512) . $chunk->getBiomeIdArray());
492 }
493
494 //TODO: use this properly
495 $write->put($index . ChunkDataKey::FINALIZATION, chr($chunk->isPopulated() ? self::FINALISATION_DONE : self::FINALISATION_NEEDS_POPULATION));
496
497 $this->writeTags($chunkData->getTileNBT(), $index . ChunkDataKey::BLOCK_ENTITIES, $write);
498 $this->writeTags($chunkData->getEntityNBT(), $index . ChunkDataKey::ENTITIES, $write);
499
500 $write->delete($index . ChunkDataKey::HEIGHTMAP_AND_2D_BIOME_COLORS);
501 $write->delete($index . ChunkDataKey::LEGACY_TERRAIN);
502
503 $this->db->write($write);
504 }

511 $nbt = new LittleEndianNbtSerializer();
512 $write->put($index, $nbt->writeMultiple(array_map(fn(CompoundTag $tag) => new TreeRoot($tag), $targets)));
513 }else{
514 $write->delete($index);
515 }

534 public function getAllChunks(bool $skipCorrupted = false, ?\Logger $logger = null) : \Generator{
535 foreach($this->db->getIterator() as $key => $_){
536 if(strlen($key) === 9 && substr($key, -1) === ChunkDataKey::OLD_VERSION){
537 $chunkX = Binary::readLInt(substr($key, 0, 4));
538 $chunkZ = Binary::readLInt(substr($key, 4, 4));
539 try{
540 if(($chunk = $this->loadChunk($chunkX, $chunkZ)) !== null){
541 yield [$chunkX, $chunkZ] => $chunk;
542 }
543 }catch(CorruptedChunkException $e){
544 if(!$skipCorrupted){
545 throw $e;
546 }
547 if($logger !== null){
548 $logger->error("Skipped corrupted chunk $chunkX $chunkZ (" . $e->getMessage() . ")");
549 }
550 }
551 }
552 }
