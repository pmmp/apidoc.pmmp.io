93 public function __construct(Server $server, \Logger $logger, InternetAddress $address, int $clientId, int $mtuSize, int $internalId){
94 if($mtuSize < self::MIN_MTU_SIZE){
95 throw new \InvalidArgumentException("MTU size must be at least " . self::MIN_MTU_SIZE . ", got $mtuSize");
96 }
97 $this->server = $server;
98 $this->logger = new \PrefixedLogger($logger, "Session: " . $address->toString());
99 $this->address = $address;
100 $this->id = $clientId;
101
102 $this->lastUpdate = microtime(true);
103
104 $this->internalId = $internalId;
105
106 $this->recvLayer = new ReceiveReliabilityLayer(
107 $this->logger,
108 function(EncapsulatedPacket $pk) : void{
109 $this->handleEncapsulatedPacketRoute($pk);
110 },
111 function(AcknowledgePacket $pk) : void{
112 $this->sendPacket($pk);
113 },
114 self::MAX_SPLIT_PART_COUNT,
115 self::MAX_CONCURRENT_SPLIT_COUNT
116 );
117 $this->sendLayer = new SendReliabilityLayer(
118 $mtuSize,
119 function(Datagram $datagram) : void{
120 $this->sendPacket($datagram);
121 },
122 function(int $identifierACK) : void{
123 $this->server->getEventListener()->onPacketAck($this->internalId, $identifierACK);
124 }
125 );
126 }

158 public function update(float $time) : void{
159 if(!$this->isActive and ($this->lastUpdate + 10) < $time){
160 $this->forciblyDisconnect("timeout");
161
162 return;
163 }
164
165 if($this->state === self::STATE_DISCONNECT_PENDING || $this->state === self::STATE_DISCONNECT_NOTIFIED){
166 //by this point we already told the event listener that the session is closing, so we don't need to do it again
167 if(!$this->sendLayer->needsUpdate() and !$this->recvLayer->needsUpdate()){
168 if($this->state === self::STATE_DISCONNECT_PENDING){
169 $this->queueConnectedPacket(new DisconnectionNotification(), PacketReliability::RELIABLE_ORDERED, 0, true);
170 $this->state = self::STATE_DISCONNECT_NOTIFIED;
171 $this->logger->debug("All pending traffic flushed, sent disconnect notification");
172 }else{
173 $this->state = self::STATE_DISCONNECTED;
174 $this->logger->debug("Client cleanly disconnected, marking session for destruction");
175 return;
176 }
177 }elseif($this->disconnectionTime + 10 < $time){
178 $this->state = self::STATE_DISCONNECTED;
179 $this->logger->debug("Timeout during graceful disconnect, forcibly closing session");
180 return;
181 }
182 }
183
184 $this->isActive = false;
185
186 $this->recvLayer->update();
187 $this->sendLayer->update();
188
189 if($this->lastPingTime + 5 < $time){
190 $this->sendPing();
191 $this->lastPingTime = $time;
192 }
193 }

195 private function queueConnectedPacket(ConnectedPacket $packet, int $reliability, int $orderChannel, bool $immediate = false) : void{
196 $out = new PacketSerializer(); //TODO: reuse streams to reduce allocations
197 $packet->encode($out);
198
199 $encapsulated = new EncapsulatedPacket();
200 $encapsulated->reliability = $reliability;
201 $encapsulated->orderChannel = $orderChannel;
202 $encapsulated->buffer = $out->getBuffer();
203
204 $this->sendLayer->addEncapsulatedToQueue($encapsulated, $immediate);
205 }

219 private function handleEncapsulatedPacketRoute(EncapsulatedPacket $packet) : void{
220 if($this->server === null){
221 return;
222 }
223
224 $id = ord($packet->buffer[0]);
225 if($id < MessageIdentifiers::ID_USER_PACKET_ENUM){ //internal data packet
226 if($this->state === self::STATE_CONNECTING){
227 if($id === ConnectionRequest::$ID){
228 $dataPacket = new ConnectionRequest();
229 $dataPacket->decode(new PacketSerializer($packet->buffer));
230 $this->queueConnectedPacket(ConnectionRequestAccepted::create(
231 $this->address,
232 [],
233 $dataPacket->sendPingTime,
234 $this->server->getRakNetTimeMS()
235 ), PacketReliability::UNRELIABLE, 0, true);
236 }elseif($id === NewIncomingConnection::$ID){
237 $dataPacket = new NewIncomingConnection();
238 $dataPacket->decode(new PacketSerializer($packet->buffer));
239
240 if($dataPacket->address->getPort() === $this->server->getPort() or !$this->server->portChecking){
241 $this->state = self::STATE_CONNECTED; //FINALLY!
242 $this->isTemporal = false;
243 $this->server->openSession($this);
244
245 //$this->handlePong($dataPacket->sendPingTime, $dataPacket->sendPongTime); //can't use this due to system-address count issues in MCPE >.<
246 $this->sendPing();
247 }
248 }
249 }elseif($id === DisconnectionNotification::$ID){
250 $this->onClientDisconnect();
251 }elseif($id === ConnectedPing::$ID){
252 $dataPacket = new ConnectedPing();
253 $dataPacket->decode(new PacketSerializer($packet->buffer));
254 $this->queueConnectedPacket(ConnectedPong::create(
255 $dataPacket->sendPingTime,
256 $this->server->getRakNetTimeMS()
257 ), PacketReliability::UNRELIABLE, 0);
258 }elseif($id === ConnectedPong::$ID){
259 $dataPacket = new ConnectedPong();
260 $dataPacket->decode(new PacketSerializer($packet->buffer));
261
262 $this->handlePong($dataPacket->sendPingTime, $dataPacket->sendPongTime);
263 }
264 }elseif($this->state === self::STATE_CONNECTED){
265 $this->server->getEventListener()->onPacketReceive($this->internalId, $packet->buffer);
266 }else{
267 //$this->logger->notice("Received packet before connection: " . bin2hex($packet->buffer));
268 }
269 }

274 private function handlePong(int $sendPingTime, int $sendPongTime) : void{
275 $currentTime = $this->server->getRakNetTimeMS();
276 if($currentTime < $sendPingTime){
277 $this->logger->debug("Received invalid pong: timestamp is in the future by " . ($sendPingTime - $currentTime) . " ms");
278 }else{
279 $this->lastPingMeasure = $currentTime - $sendPingTime;
280 $this->server->getEventListener()->onPingMeasure($this->internalId, $this->lastPingMeasure);
281 }
282 }

284 public function handlePacket(Packet $packet) : void{
285 $this->isActive = true;
286 $this->lastUpdate = microtime(true);
287
288 if($packet instanceof Datagram){ //In reality, ALL of these packets are datagrams.
289 $this->recvLayer->onDatagram($packet);
290 }elseif($packet instanceof ACK){
291 $this->sendLayer->onACK($packet);
292 }elseif($packet instanceof NACK){
293 $this->sendLayer->onNACK($packet);
294 }
295 }

300 public function initiateDisconnect(string $reason) : void{
301 if($this->isConnected()){
302 $this->state = self::STATE_DISCONNECT_PENDING;
303 $this->disconnectionTime = microtime(true);
304 $this->server->getEventListener()->onClientDisconnect($this->internalId, $reason);
305 $this->logger->debug("Requesting graceful disconnect because \"$reason\"");
306 }
307 }

312 public function forciblyDisconnect(string $reason) : void{
313 $this->state = self::STATE_DISCONNECTED;
314 $this->server->getEventListener()->onClientDisconnect($this->internalId, $reason);
315 $this->logger->debug("Forcibly disconnecting session due to \"$reason\"");
316 }

318 private function onClientDisconnect() : void{
319 //the client will expect an ACK for this; make sure it gets sent, because after forcible termination
320 //there won't be any session ticks to update it
321 $this->recvLayer->update();
322
323 if($this->isConnected()){
324 //the client might have disconnected after the server sent a disconnect notification, but before the client
325 //received it - in this case, we don't want to notify the event handler twice
326 $this->server->getEventListener()->onClientDisconnect($this->internalId, "client disconnect");
327 }
328 $this->state = self::STATE_DISCONNECTED;
329 $this->logger->debug("Terminating session due to client disconnect");
330 }
