107 public static function decode(int $recipeType, PacketSerializer $in) : self{
108 $recipeId = $in->getString();
109 $width = $in->getVarInt();
110 $height = $in->getVarInt();
111 $input = [];
112 for($row = 0; $row < $height; ++$row){
113 for($column = 0; $column < $width; ++$column){
114 $input[$row][$column] = $in->getRecipeIngredient();
115 }
116 }
117
118 $output = [];
119 for($k = 0, $resultCount = $in->getUnsignedVarInt(); $k < $resultCount; ++$k){
120 $output[] = $in->getItemStackWithoutStackId();
121 }
122 $uuid = $in->getUUID();
123 $block = $in->getString();
124 $priority = $in->getVarInt();
125 $symmetric = $in->getBool();
126 $unlockingRequirement = RecipeUnlockingRequirement::read($in);
127
128 $recipeNetId = $in->readRecipeNetId();
129
130 return new self($recipeType, $recipeId, $input, $output, $uuid, $block, $priority, $symmetric, $unlockingRequirement, $recipeNetId);
131 }
