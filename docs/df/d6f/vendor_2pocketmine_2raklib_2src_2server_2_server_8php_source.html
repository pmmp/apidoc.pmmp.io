145 public function tickProcessor() : void{
146 $start = microtime(true);
147
148 /*
149 * The below code is designed to allow co-op between sending and receiving to avoid slowing down either one
150 * when high traffic is coming either way. Yielding will occur after 100 messages.
151 */
152 do{
153 $stream = !$this->shutdown;
154 for($i = 0; $i < 100 && $stream && !$this->shutdown; ++$i){ //if we received a shutdown event, we don't care about any more messages from the event source
155 $stream = $this->eventSource->process($this);
156 }
157
158 $socket = true;
159 for($i = 0; $i < 100 && $socket; ++$i){
160 $socket = $this->receivePacket();
161 }
162 }while($stream || $socket);
163
164 $this->tick();
165
166 $time = microtime(true) - $start;
167 if($time < self::RAKLIB_TIME_PER_TICK){
168 @time_sleep_until(microtime(true) + self::RAKLIB_TIME_PER_TICK - $time);
169 }
170 }

175 public function waitShutdown() : void{
176 $this->shutdown = true;
177
178 while($this->eventSource->process($this)){
179 //Ensure that any late messages are processed before we start initiating server disconnects, so that if the
180 //server implementation used a custom disconnect mechanism (e.g. a server transfer), we don't break it in
181 //race conditions.
182 }
183
184 foreach($this->sessions as $session){
185 $session->initiateDisconnect("server shutdown");
186 }
187
188 while(count($this->sessions) > 0){
189 $this->tickProcessor();
190 }
191
192 $this->socket->close();
193 $this->logger->debug("Graceful shutdown complete");
194 }

196 private function tick() : void{
197 $time = microtime(true);
198 foreach($this->sessions as $session){
199 $session->update($time);
200 if($session->isFullyDisconnected()){
201 $this->removeSessionInternal($session);
202 }
203 }
204
205 $this->ipSec = [];
206
207 if(!$this->shutdown and ($this->ticks % self::RAKLIB_TPS) === 0){
208 if($this->sendBytes > 0 or $this->receiveBytes > 0){
209 $this->eventListener->onBandwidthStatsUpdate($this->sendBytes, $this->receiveBytes);
210 $this->sendBytes = 0;
211 $this->receiveBytes = 0;
212 }
213
214 if(count($this->block) > 0){
215 asort($this->block);
216 $now = time();
217 foreach($this->block as $address => $timeout){
218 if($timeout <= $now){
219 unset($this->block[$address]);
220 }else{
221 break;
222 }
223 }
224 }
225 }
226
227 ++$this->ticks;
228 }

231 private function receivePacket() : bool{
232 try{
233 $buffer = $this->socket->readPacket($addressIp, $addressPort);
234 }catch(SocketException $e){
235 $error = $e->getCode();
236 if($error === SOCKET_ECONNRESET){ //client disconnected improperly, maybe crash or lost connection
237 return true;
238 }
239
240 $this->logger->debug($e->getMessage());
241 return false;
242 }
243 if($buffer === null){
244 return false; //no data
245 }
246 $len = strlen($buffer);
247
248 $this->receiveBytes += $len;
249 if(isset($this->block[$addressIp])){
250 return true;
251 }
252
253 if(isset($this->ipSec[$addressIp])){
254 if(++$this->ipSec[$addressIp] >= $this->packetLimit){
255 $this->blockAddress($addressIp);
256 return true;
257 }
258 }else{
259 $this->ipSec[$addressIp] = 1;
260 }
261
262 if($len < 1){
263 return true;
264 }
265
266 $address = new InternetAddress($addressIp, $addressPort, $this->socket->getBindAddress()->getVersion());
267 try{
268 $session = $this->getSessionByAddress($address);
269 if($session !== null){
270 $header = ord($buffer[0]);
271 if(($header & Datagram::BITFLAG_VALID) !== 0){
272 if(($header & Datagram::BITFLAG_ACK) !== 0){
273 $packet = new ACK();
274 }elseif(($header & Datagram::BITFLAG_NAK) !== 0){
275 $packet = new NACK();
276 }else{
277 $packet = new Datagram();
278 }
279 $packet->decode(new PacketSerializer($buffer));
280 $session->handlePacket($packet);
281 return true;
282 }elseif($session->isConnected()){
283 //allows unconnected packets if the session is stuck in DISCONNECTING state, useful if the client
284 //didn't disconnect properly for some reason (e.g. crash)
285 $this->logger->debug("Ignored unconnected packet from $address due to session already opened (0x" . bin2hex($buffer[0]) . ")");
286 return true;
287 }
288 }
289
290 if(!$this->shutdown){
291 if(!($handled = $this->unconnectedMessageHandler->handleRaw($buffer, $address))){
292 foreach($this->rawPacketFilters as $pattern){
293 if(preg_match($pattern, $buffer) > 0){
294 $handled = true;
295 $this->eventListener->onRawPacketReceive($address->getIp(), $address->getPort(), $buffer);
296 break;
297 }
298 }
299 }
300
301 if(!$handled){
302 $this->logger->debug("Ignored packet from $address due to no session opened (0x" . bin2hex($buffer[0]) . ")");
303 }
304 }
305 }catch(BinaryDataException $e){
306 $logFn = function() use ($address, $e, $buffer) : void{
307 $this->logger->debug("Packet from $address (" . strlen($buffer) . " bytes): 0x" . bin2hex($buffer));
308 $this->logger->debug(get_class($e) . ": " . $e->getMessage() . " in " . $e->getFile() . " on line " . $e->getLine());
309 foreach($this->traceCleaner->getTrace(0, $e->getTrace()) as $line){
310 $this->logger->debug($line);
311 }
312 $this->logger->error("Bad packet from $address: " . $e->getMessage());
313 };
314 if($this->logger instanceof \BufferedLogger){
315 $this->logger->buffer($logFn);
316 }else{
317 $logFn();
318 }
319 $this->blockAddress($address->getIp(), 5);
320 }
321
322 return true;
323 }

372 public function blockAddress(string $address, int $timeout = 300) : void{
373 $final = time() + $timeout;

403 public function createSession(InternetAddress $address, int $clientId, int $mtuSize) : Session{
404 $existingSession = $this->sessionsByAddress[$address->toString()] ?? null;
405 if($existingSession !== null){
406 $existingSession->forciblyDisconnect("client reconnect");
407 $this->removeSessionInternal($existingSession);
408 }
409
410 $this->checkSessions();
411
412 while(isset($this->sessions[$this->nextSessionId])){
413 $this->nextSessionId++;
414 $this->nextSessionId &= 0x7fffffff; //we don't expect more than 2 billion simultaneous connections, and this fits in 4 bytes
415 }
416
417 $session = new Session($this, $this->logger, clone $address, $clientId, $mtuSize, $this->nextSessionId);
418 $this->sessionsByAddress[$address->toString()] = $session;
419 $this->sessions[$this->nextSessionId] = $session;
420 $this->logger->debug("Created session for $address with MTU size $mtuSize");
421
422 return $session;
423 }
