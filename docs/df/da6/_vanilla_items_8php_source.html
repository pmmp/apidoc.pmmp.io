358 protected static function register(string $name, \Closure $createItem) : Item{
359 //this sketchy hack allows us to avoid manually writing the constants inline
360 //since type IDs are generated from this class anyway, I'm OK with this hack
361 //nonetheless, we should try to get rid of it in a future major version (e.g by using string type IDs)
362 $reflect = new \ReflectionClass(ItemTypeIds::class);
363 $typeId = $reflect->getConstant(mb_strtoupper($name));
364 if(!is_int($typeId)){
365 //this allows registering new stuff without adding new type ID constants
366 //this reduces the number of mandatory steps to test new features in local development
367 \GlobalLogger::get()->error(self::class . ": No constant type ID found for $name, generating a new one");
368 $typeId = ItemTypeIds::newId();
369 }
370
371 $item = $createItem(new IID($typeId));
372
373 self::_registryRegister($name, $item);
374
375 return $item;
376 }
