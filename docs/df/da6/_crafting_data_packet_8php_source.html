68 protected function decodePayload(PacketSerializer $in) : void{
69 $recipeCount = $in->getUnsignedVarInt();
70 for($i = 0; $i < $recipeCount; ++$i){
71 $recipeType = $in->getVarInt();
72
73 switch($recipeType){
74 case self::ENTRY_SHAPELESS:
75 case self::ENTRY_SHULKER_BOX:
76 case self::ENTRY_SHAPELESS_CHEMISTRY:
77 $this->recipesWithTypeIds[] = ShapelessRecipe::decode($recipeType, $in);
78 break;
79 case self::ENTRY_SHAPED:
80 case self::ENTRY_SHAPED_CHEMISTRY:
81 $this->recipesWithTypeIds[] = ShapedRecipe::decode($recipeType, $in);
82 break;
83 case self::ENTRY_FURNACE:
84 case self::ENTRY_FURNACE_DATA:
85 $this->recipesWithTypeIds[] = FurnaceRecipe::decode($recipeType, $in);
86 break;
87 case self::ENTRY_MULTI:
88 $this->recipesWithTypeIds[] = MultiRecipe::decode($recipeType, $in);
89 break;
90 default:
91 throw new PacketDecodeException("Unhandled recipe type $recipeType!"); //do not continue attempting to decode
92 }
93 }
94 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
95 $inputId = $in->getVarInt();
96 $inputMeta = $in->getVarInt();
97 $ingredientId = $in->getVarInt();
98 $ingredientMeta = $in->getVarInt();
99 $outputId = $in->getVarInt();
100 $outputMeta = $in->getVarInt();
101 $this->potionTypeRecipes[] = new PotionTypeRecipe($inputId, $inputMeta, $ingredientId, $ingredientMeta, $outputId, $outputMeta);
102 }
103 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
104 $input = $in->getVarInt();
105 $ingredient = $in->getVarInt();
106 $output = $in->getVarInt();
107 $this->potionContainerRecipes[] = new PotionContainerChangeRecipe($input, $ingredient, $output);
108 }
109 for($i = 0, $count = $in->getUnsignedVarInt(); $i < $count; ++$i){
110 $inputIdAndData = $in->getVarInt();
111 [$inputId, $inputMeta] = [$inputIdAndData >> 16, $inputIdAndData & 0x7fff];
112 $outputs = [];
113 for($j = 0, $outputCount = $in->getUnsignedVarInt(); $j < $outputCount; ++$j){
114 $outputItemId = $in->getVarInt();
115 $outputItemCount = $in->getVarInt();
116 $outputs[] = new MaterialReducerRecipeOutput($outputItemId, $outputItemCount);
117 }
118 $this->materialReducerRecipes[] = new MaterialReducerRecipe($inputId, $inputMeta, $outputs);
119 }
120 $this->cleanRecipes = $in->getBool();
121 }

123 protected function encodePayload(PacketSerializer $out) : void{
124 $out->putUnsignedVarInt(count($this->recipesWithTypeIds));
125 foreach($this->recipesWithTypeIds as $d){
126 $out->putVarInt($d->getTypeId());
127 $d->encode($out);
128 }
129 $out->putUnsignedVarInt(count($this->potionTypeRecipes));
130 foreach($this->potionTypeRecipes as $recipe){
131 $out->putVarInt($recipe->getInputItemId());
132 $out->putVarInt($recipe->getInputItemMeta());
133 $out->putVarInt($recipe->getIngredientItemId());
134 $out->putVarInt($recipe->getIngredientItemMeta());
135 $out->putVarInt($recipe->getOutputItemId());
136 $out->putVarInt($recipe->getOutputItemMeta());
137 }
138 $out->putUnsignedVarInt(count($this->potionContainerRecipes));
139 foreach($this->potionContainerRecipes as $recipe){
140 $out->putVarInt($recipe->getInputItemId());
141 $out->putVarInt($recipe->getIngredientItemId());
142 $out->putVarInt($recipe->getOutputItemId());
143 }
144 $out->putUnsignedVarInt(count($this->materialReducerRecipes));
145 foreach($this->materialReducerRecipes as $recipe){
146 $out->putVarInt(($recipe->getInputItemId() << 16) | $recipe->getInputItemMeta());
147 $out->putUnsignedVarInt(count($recipe->getOutputs()));
148 foreach($recipe->getOutputs() as $output){
149 $out->putVarInt($output->getItemId());
150 $out->putVarInt($output->getCount());
151 }
152 }
153 $out->putBool($this->cleanRecipes);
154 }
