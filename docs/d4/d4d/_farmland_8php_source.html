73 public function onRandomTick() : void{
74 $world = $this->position->getWorld();
75 if(!$this->canHydrate()){
76 if($this->wetness > 0){
77 $event = new FarmlandHydrationChangeEvent($this, $this->wetness, $this->wetness - 1);
78 $event->call();
79 if(!$event->isCancelled()){
80 $this->wetness = $event->getNewHydration();
81 $world->setBlock($this->position, $this, false);
82 }
83 }else{
84 $world->setBlock($this->position, VanillaBlocks::DIRT());
85 }
86 }elseif($this->wetness < self::MAX_WETNESS){
87 $event = new FarmlandHydrationChangeEvent($this, $this->wetness, self::MAX_WETNESS);
88 $event->call();
89 if(!$event->isCancelled()){
90 $this->wetness = $event->getNewHydration();
91 $world->setBlock($this->position, $this, false);
92 }
93 }
94 }
95
96 public function onEntityLand(Entity $entity) : ?float{
97 if($entity instanceof Living && lcg_value() < $entity->getFallDistance() - 0.5){
98 $ev = new EntityTrampleFarmlandEvent($entity, $this);
99 $ev->call();
100 if(!$ev->isCancelled()){
101 $this->getPosition()->getWorld()->setBlock($this->getPosition(), VanillaBlocks::DIRT());
102 }
103 }
104 return null;
105 }

107 protected function canHydrate() : bool{
108 //TODO: check rain
109 $start = $this->position->add(-4, 0, -4);
110 $end = $this->position->add(4, 1, 4);
111 for($y = $start->y; $y <= $end->y; ++$y){
112 for($z = $start->z; $z <= $end->z; ++$z){
113 for($x = $start->x; $x <= $end->x; ++$x){
114 if($this->position->getWorld()->getBlockAt($x, $y, $z) instanceof Water){
115 return true;
116 }
117 }
118 }
119 }
120
121 return false;
122 }
