264 public function getItemStack(\Closure $readExtraCrapInTheMiddle) : ItemStack{
265 $id = $this->getVarInt();
266 if($id === 0){
267 return ItemStack::null();
268 }
269
270 $count = $this->getLShort();
271 $meta = $this->getUnsignedVarInt();
272
273 $readExtraCrapInTheMiddle($this);
274
275 $blockRuntimeId = $this->getVarInt();
276 $extraData = self::decoder($this->getString(), 0, $this->context);
277 return (static function() use ($extraData, $id, $meta, $count, $blockRuntimeId) : ItemStack{
278 $nbtLen = $extraData->getLShort();
279
281 $compound = null;
282 if($nbtLen === 0xffff){
283 $nbtDataVersion = $extraData->getByte();
284 if($nbtDataVersion !== 1){
285 throw new PacketDecodeException("Unexpected NBT data version $nbtDataVersion");
286 }
287 $offset = $extraData->getOffset();
288 try{
289 $compound = (new LittleEndianNbtSerializer())->read($extraData->getBuffer(), $offset, 512)->mustGetCompoundTag();
290 }catch(NbtDataException $e){
291 throw PacketDecodeException::wrap($e, "Failed decoding NBT root");
292 }finally{
293 $extraData->setOffset($offset);
294 }
295 }elseif($nbtLen !== 0){
296 throw new PacketDecodeException("Unexpected fake NBT length $nbtLen");
297 }
298
299 $canPlaceOn = [];
300 for($i = 0, $canPlaceOnCount = $extraData->getLInt(); $i < $canPlaceOnCount; ++$i){
301 $canPlaceOn[] = $extraData->get($extraData->getLShort());
302 }
303
304 $canDestroy = [];
305 for($i = 0, $canDestroyCount = $extraData->getLInt(); $i < $canDestroyCount; ++$i){
306 $canDestroy[] = $extraData->get($extraData->getLShort());
307 }
308
309 $shieldBlockingTick = null;
310 if($id === $extraData->shieldItemRuntimeId){
311 $shieldBlockingTick = $extraData->getLLong();
312 }
313
314 if(!$extraData->feof()){
315 throw new PacketDecodeException("Unexpected trailing extradata for network item $id");
316 }
317
318 return new ItemStack($id, $meta, $count, $blockRuntimeId, $compound, $canPlaceOn, $canDestroy, $shieldBlockingTick);
319 })();
320 }

325 public function putItemStack(ItemStack $item, \Closure $writeExtraCrapInTheMiddle) : void{
326 if($item->getId() === 0){
327 $this->putVarInt(0);
328
329 return;
330 }
331
332 $this->putVarInt($item->getId());
333 $this->putLShort($item->getCount());
334 $this->putUnsignedVarInt($item->getMeta());
335
336 $writeExtraCrapInTheMiddle($this);
337
338 $this->putVarInt($item->getBlockRuntimeId());
339 $context = $this->context;
340 $this->putString((static function() use ($item, $context) : string{
341 $extraData = PacketSerializer::encoder($context);
342
343 $nbt = $item->getNbt();
344 if($nbt !== null){
345 $extraData->putLShort(0xffff);
346 $extraData->putByte(1); //TODO: NBT data version (?)
347 $extraData->put((new LittleEndianNbtSerializer())->write(new TreeRoot($nbt)));
348 }else{
349 $extraData->putLShort(0);
350 }
351
352 $extraData->putLInt(count($item->getCanPlaceOn()));
353 foreach($item->getCanPlaceOn() as $entry){
354 $extraData->putLShort(strlen($entry));
355 $extraData->put($entry);
356 }
357 $extraData->putLInt(count($item->getCanDestroy()));
358 foreach($item->getCanDestroy() as $entry){
359 $extraData->putLShort(strlen($entry));
360 $extraData->put($entry);
361 }
362
363 $blockingTick = $item->getShieldBlockingTick();
364 if($item->getId() === $extraData->shieldItemRuntimeId){
365 $extraData->putLLong($blockingTick ?? 0);
366 }
367 return $extraData->getBuffer();
368 })());
369 }
