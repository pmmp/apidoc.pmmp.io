274 public function getItemStack(\Closure $readExtraCrapInTheMiddle) : ItemStack{
275 $id = $this->getVarInt();
276 if($id === 0){
277 return ItemStack::null();
278 }
279
280 $count = $this->getLShort();
281 $meta = $this->getUnsignedVarInt();
282
283 $readExtraCrapInTheMiddle($this);
284
285 $blockRuntimeId = $this->getVarInt();
286 $extraData = self::decoder($this->getString(), 0, $this->context);
287 return (static function() use ($extraData, $id, $meta, $count, $blockRuntimeId) : ItemStack{
288 $nbtLen = $extraData->getLShort();
289
291 $compound = null;
292 if($nbtLen === 0xffff){
293 $nbtDataVersion = $extraData->getByte();
294 if($nbtDataVersion !== 1){
295 throw new PacketDecodeException("Unexpected NBT data version $nbtDataVersion");
296 }
297 $offset = $extraData->getOffset();
298 try{
299 $compound = (new LittleEndianNbtSerializer())->read($extraData->getBuffer(), $offset, 512)->mustGetCompoundTag();
300 }catch(NbtDataException $e){
301 throw PacketDecodeException::wrap($e, "Failed decoding NBT root");
302 }finally{
303 $extraData->setOffset($offset);
304 }
305 }elseif($nbtLen !== 0){
306 throw new PacketDecodeException("Unexpected fake NBT length $nbtLen");
307 }
308
309 $canPlaceOn = [];
310 for($i = 0, $canPlaceOnCount = $extraData->getLInt(); $i < $canPlaceOnCount; ++$i){
311 $canPlaceOn[] = $extraData->get($extraData->getLShort());
312 }
313
314 $canDestroy = [];
315 for($i = 0, $canDestroyCount = $extraData->getLInt(); $i < $canDestroyCount; ++$i){
316 $canDestroy[] = $extraData->get($extraData->getLShort());
317 }
318
319 $shieldBlockingTick = null;
320 if($id === $extraData->shieldItemRuntimeId){
321 $shieldBlockingTick = $extraData->getLLong();
322 }
323
324 if(!$extraData->feof()){
325 throw new PacketDecodeException("Unexpected trailing extradata for network item $id");
326 }
327
328 return new ItemStack($id, $meta, $count, $blockRuntimeId, $compound, $canPlaceOn, $canDestroy, $shieldBlockingTick);
329 })();
330 }

335 public function putItemStack(ItemStack $item, \Closure $writeExtraCrapInTheMiddle) : void{
336 if($item->getId() === 0){
337 $this->putVarInt(0);
338
339 return;
340 }
341
342 $this->putVarInt($item->getId());
343 $this->putLShort($item->getCount());
344 $this->putUnsignedVarInt($item->getMeta());
345
346 $writeExtraCrapInTheMiddle($this);
347
348 $this->putVarInt($item->getBlockRuntimeId());
349 $context = $this->context;
350 $this->putString((static function() use ($item, $context) : string{
351 $extraData = PacketSerializer::encoder($context);
352
353 $nbt = $item->getNbt();
354 if($nbt !== null){
355 $extraData->putLShort(0xffff);
356 $extraData->putByte(1); //TODO: NBT data version (?)
357 $extraData->put((new LittleEndianNbtSerializer())->write(new TreeRoot($nbt)));
358 }else{
359 $extraData->putLShort(0);
360 }
361
362 $extraData->putLInt(count($item->getCanPlaceOn()));
363 foreach($item->getCanPlaceOn() as $entry){
364 $extraData->putLShort(strlen($entry));
365 $extraData->put($entry);
366 }
367 $extraData->putLInt(count($item->getCanDestroy()));
368 foreach($item->getCanDestroy() as $entry){
369 $extraData->putLShort(strlen($entry));
370 $extraData->put($entry);
371 }
372
373 $blockingTick = $item->getShieldBlockingTick();
374 if($item->getId() === $extraData->shieldItemRuntimeId){
375 $extraData->putLLong($blockingTick ?? 0);
376 }
377 return $extraData->getBuffer();
378 })());
379 }
