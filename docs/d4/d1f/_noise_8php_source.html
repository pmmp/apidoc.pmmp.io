151 public function noise2D($x, $z, $normalized = false){
152 $result = 0;
153 $amp = 1;
154 $freq = 1;
155 $max = 0;
156
157 $x *= $this->expansion;
158 $z *= $this->expansion;
159
160 for($i = 0; $i < $this->octaves; ++$i){
161 $result += $this->getNoise2D($x * $freq, $z * $freq) * $amp;
162 $max += $amp;
163 $freq *= 2;
164 $amp *= $this->persistence;
165 }
166
167 if($normalized === true){
168 $result /= $max;

277 public function getFastNoise3D(int $xSize, int $ySize, int $zSize, int $xSamplingRate, int $ySamplingRate, int $zSamplingRate, int $x, int $y, int $z) : array{
278
279 assert($xSamplingRate !== 0, new \InvalidArgumentException("xSamplingRate cannot be 0"));

287 $noiseArray = array_fill(0, $xSize + 1, array_fill(0, $zSize + 1, []));
288
289 for($xx = 0; $xx <= $xSize; $xx += $xSamplingRate){
290 for($zz = 0; $zz <= $zSize; $zz += $zSamplingRate){
291 for($yy = 0; $yy <= $ySize; $yy += $ySamplingRate){
292 $noiseArray[$xx][$zz][$yy] = $this->noise3D($x + $xx, $y + $yy, $z + $zz, true);

294 }
295 }

308 for($xx = 0; $xx < $xSize; ++$xx){
309 $nx = (int) ($xx / $xSamplingRate) * $xSamplingRate;
310 $nnx = $nx + $xSamplingRate;
311
312 $dx1 = (($nnx - $xx) / ($nnx - $nx));
313 $dx2 = (($xx - $nx) / ($nnx - $nx));
314
315 for($zz = 0; $zz < $zSize; ++$zz){
316 $nz = (int) ($zz / $zSamplingRate) * $zSamplingRate;
317 $nnz = $nz + $zSamplingRate;
318

320 $dz2 = ($zz - $nz) / ($nnz - $nz);
321
322 for($yy = 0; $yy < $ySize; ++$yy){
323 if($xx % $xSamplingRate !== 0 || $zz % $zSamplingRate !== 0 || $yy % $ySamplingRate !== 0){
324 $ny = (int) ($yy / $ySamplingRate) * $ySamplingRate;
325 $nny = $ny + $ySamplingRate;
326
327 $dy1 = (($nny - $yy) / ($nny - $ny));
328 $dy2 = (($yy - $ny) / ($nny - $ny));
329
330 $noiseArray[$xx][$zz][$yy] = $dz1 * (
331 $dy1 * (
332 $dx1 * $noiseArray[$nx][$nz][$ny] + $dx2 * $noiseArray[$nnx][$nz][$ny]
333 ) + $dy2 * (
334 $dx1 * $noiseArray[$nx][$nz][$nny] + $dx2 * $noiseArray[$nnx][$nz][$nny]
335 )
336 ) + $dz2 * (
337 $dy1 * (
338 $dx1 * $noiseArray[$nx][$nnz][$ny] + $dx2 * $noiseArray[$nnx][$nnz][$ny]
339 ) + $dy2 * (
340 $dx1 * $noiseArray[$nx][$nnz][$nny] + $dx2 * $noiseArray[$nnx][$nnz][$nny]

345 }
