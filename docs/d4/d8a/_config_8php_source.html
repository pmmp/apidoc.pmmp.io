151 private function load(string $file, int $type = Config::DETECT, array $default = []) : void{
152 $this->file = $file;
153
154 $this->type = $type;
155 if($this->type === Config::DETECT){
156 $extension = strtolower(Path::getExtension($this->file));
157 if(isset(Config::$formats[$extension])){
158 $this->type = Config::$formats[$extension];
159 }else{
160 throw new \InvalidArgumentException("Cannot detect config type of " . $this->file);
161 }
162 }
163
164 if(!file_exists($file)){
165 $this->config = $default;
166 $this->save();
167 }else{
168 $content = file_get_contents($this->file);
169 if($content === false){
170 throw new \RuntimeException("Unable to load config file");
171 }
172 switch($this->type){
173 case Config::PROPERTIES:
174 $config = self::parseProperties($content);
175 break;
176 case Config::JSON:
177 try{
178 $config = json_decode($content, true, flags: JSON_THROW_ON_ERROR);
179 }catch(\JsonException $e){
180 throw ConfigLoadException::wrap($this->file, $e);
181 }
182 break;
183 case Config::YAML:
184 $content = self::fixYAMLIndexes($content);
185 try{
186 $config = ErrorToExceptionHandler::trap(fn() => yaml_parse($content));
187 }catch(\ErrorException $e){
188 throw ConfigLoadException::wrap($this->file, $e);
189 }
190 break;
191 case Config::SERIALIZED:
192 try{
193 $config = ErrorToExceptionHandler::trap(fn() => unserialize($content));
194 }catch(\ErrorException $e){
195 throw ConfigLoadException::wrap($this->file, $e);
196 }
197 break;
198 case Config::ENUM:
199 $config = array_fill_keys(self::parseList($content), true);
200 break;
201 default:
202 throw new \InvalidArgumentException("Invalid config type specified");
203 }
204 if(!is_array($config)){
205 throw new ConfigLoadException("Failed to load config $this->file: Expected array for base type, but got " . get_debug_type($config));
206 }
207 $this->config = $config;
208 if($this->fillDefaults($default, $this->config) > 0){
209 $this->save();
210 }
211 }

224 public function save() : void{
225 $content = null;
226 switch($this->type){
227 case Config::PROPERTIES:
228 $content = self::writeProperties($this->config);
229 break;
230 case Config::JSON:
231 $content = json_encode($this->config, $this->jsonOptions | JSON_THROW_ON_ERROR);
232 break;
233 case Config::YAML:
234 $content = yaml_emit($this->config, YAML_UTF8_ENCODING);
235 break;
236 case Config::SERIALIZED:
237 $content = serialize($this->config);
238 break;
239 case Config::ENUM:
240 $content = self::writeList(array_keys($this->config));
241 break;
242 default:
243 throw new AssumptionFailedError("Config type is unknown, has not been set or not detected");
244 }
245
246 Filesystem::safeFilePutContents($this->file, $content);
247
248 $this->changed = false;
249 }

352 public function setNested($key, $value) : void{
353 $vars = explode(".", $key);
354 $base = array_shift($vars);
355
356 if(!isset($this->config[$base])){
357 $this->config[$base] = [];
358 }
359
360 $base = &$this->config[$base];
361
362 while(count($vars) > 0){
363 $baseKey = array_shift($vars);
364 if(!isset($base[$baseKey])){
365 $base[$baseKey] = [];
366 }
367 $base = &$base[$baseKey];
368 }
369
370 $base = $value;
371 $this->nestedCache = [];
372 $this->changed = true;
373 }

381 public function getNested($key, $default = null){
382 if(isset($this->nestedCache[$key])){
383 return $this->nestedCache[$key];
384 }
385
386 $vars = explode(".", $key);
387 $base = array_shift($vars);
388 if(isset($this->config[$base])){
389 $base = $this->config[$base];
390 }else{
391 return $default;
392 }
393
394 while(count($vars) > 0){
395 $baseKey = array_shift($vars);
396 if(is_array($base) && isset($base[$baseKey])){
397 $base = $base[$baseKey];
398 }else{
399 return $default;
400 }
401 }
402
403 return $this->nestedCache[$key] = $base;
404 }

406 public function removeNested(string $key) : void{
407 $this->nestedCache = [];
408 $this->changed = true;
409
410 $vars = explode(".", $key);
411
412 $currentNode = &$this->config;
413 while(count($vars) > 0){
414 $nodeName = array_shift($vars);
415 if(isset($currentNode[$nodeName])){
416 if(count($vars) === 0){ //final node
417 unset($currentNode[$nodeName]);
418 }elseif(is_array($currentNode[$nodeName])){
419 $currentNode = &$currentNode[$nodeName];
420 }
421 }else{
422 break;
423 }
424 }

504 private function fillDefaults(array $default, &$data) : int{
505 $changed = 0;
506 foreach(Utils::stringifyKeys($default) as $k => $v){
507 if(is_array($v)){
508 if(!isset($data[$k]) || !is_array($data[$k])){
509 $data[$k] = [];
510 }
511 $changed += $this->fillDefaults($v, $data[$k]);
512 }elseif(!isset($data[$k])){
513 $data[$k] = $v;
514 ++$changed;
515 }
516 }
517
518 if($changed > 0){
519 $this->changed = true;
520 }
521
522 return $changed;
523 }

554 $content = "#Properties Config file\r\n#" . date("D M j H:i:s T Y") . "\r\n";

569 public static function parseProperties(string $content) : array{
570 $result = [];
571 if(preg_match_all('/^\s*([a-zA-Z0-9\-_\.]+)[ \t]*=([^\r\n]*)/um', $content, $matches) > 0){ //false or 0 matches
572 foreach($matches[1] as $i => $k){
573 $v = trim($matches[2][$i]);
574 switch(strtolower($v)){
575 case "on":
576 case "true":
577 case "yes":
578 $v = true;
579 break;
580 case "off":
581 case "false":
582 case "no":
583 $v = false;
584 break;
585 default:
586 $v = match($v){
587 (string) ((int) $v) => (int) $v,
588 (string) ((float) $v) => (float) $v,
589 default => $v,
590 };
591 break;
592 }
593 $result[(string) $k] = $v;
594 }
595 }
596
597 return $result;
