827 protected static function register(string $name, \Closure $createBlock, ?string $tileClass = null) : Block{
828 //this sketchy hack allows us to avoid manually writing the constants inline
829 //since type IDs are generated from this class anyway, I'm OK with this hack
830 //nonetheless, we should try to get rid of it in a future major version (e.g by using string type IDs)
831 $reflect = new \ReflectionClass(BlockTypeIds::class);
832 $typeId = $reflect->getConstant(mb_strtoupper($name));
833 if(!is_int($typeId)){
834 //this allows registering new stuff without adding new type ID constants
835 //this reduces the number of mandatory steps to test new features in local development
836 \GlobalLogger::get()->error(self::class . ": No constant type ID found for $name, generating a new one");
837 $typeId = BlockTypeIds::newId();
838 }
839 $block = $createBlock(new BID($typeId, $tileClass));
840 self::_registryRegister($name, $block);
841
842 return $block;
843 }
