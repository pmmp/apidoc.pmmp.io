205 public static function decodeMushroomBlock(RedMushroomBlock $block, BlockStateReader $in) : Block{
206 switch($type = $in->readBoundedInt(BlockStateNames::HUGE_MUSHROOM_BITS, 0, 15)){
207 case BlockLegacyMetadata::MUSHROOM_BLOCK_ALL_STEM: return VanillaBlocks::ALL_SIDED_MUSHROOM_STEM();
208 case BlockLegacyMetadata::MUSHROOM_BLOCK_STEM: return VanillaBlocks::MUSHROOM_STEM();
209 default:
210 //invalid types get left as default
211 $type = MushroomBlockTypeIdMap::getInstance()->fromId($type);
212 return $type !== null ? $block->setMushroomBlockType($type) : $block;
213 }
214 }

253 public static function decodeWall(Wall $block, BlockStateReader $in) : Wall{
254 $block->setPost($in->readBool(BlockStateNames::WALL_POST_BIT));
255 $block->setConnection(Facing::NORTH, $in->readWallConnectionType(BlockStateNames::WALL_CONNECTION_TYPE_NORTH));
256 $block->setConnection(Facing::SOUTH, $in->readWallConnectionType(BlockStateNames::WALL_CONNECTION_TYPE_SOUTH));
257 $block->setConnection(Facing::WEST, $in->readWallConnectionType(BlockStateNames::WALL_CONNECTION_TYPE_WEST));
258 $block->setConnection(Facing::EAST, $in->readWallConnectionType(BlockStateNames::WALL_CONNECTION_TYPE_EAST));
259
260 return $block;
261 }
