74 protected function matchRecipeItems(array $txItems, array $recipeItems, bool $wildcards, int $iterations = 0) : int{
75 if(count($recipeItems) === 0){
76 throw new TransactionValidationException("No recipe items given");
77 }
78 if(count($txItems) === 0){
79 throw new TransactionValidationException("No transaction items given");
80 }
81
82 while(count($recipeItems) > 0){
84 $recipeItem = array_pop($recipeItems);
85 $needCount = $recipeItem->getCount();
86 foreach($recipeItems as $i => $otherRecipeItem){
87 if($otherRecipeItem->canStackWith($recipeItem)){ //make sure they have the same wildcards set
88 $needCount += $otherRecipeItem->getCount();
89 unset($recipeItems[$i]);
90 }
91 }
92
93 $haveCount = 0;
94 foreach($txItems as $j => $txItem){
95 if($txItem->equals($recipeItem, !$wildcards || !$recipeItem->hasAnyDamageValue(), !$wildcards || $recipeItem->hasNamedTag())){
96 $haveCount += $txItem->getCount();
97 unset($txItems[$j]);
98 }
99 }
100
101 if($haveCount % $needCount !== 0){
102 //wrong count for this output, should divide exactly
103 throw new TransactionValidationException("Expected an exact multiple of required $recipeItem (given: $haveCount, needed: $needCount)");
104 }
105
106 $multiplier = intdiv($haveCount, $needCount);
107 if($multiplier < 1){
108 throw new TransactionValidationException("Expected more than zero items matching $recipeItem (given: $haveCount, needed: $needCount)");
109 }
110 if($iterations === 0){
111 $iterations = $multiplier;
112 }elseif($multiplier !== $iterations){
113 //wrong count for this output, should match previous outputs
114 throw new TransactionValidationException("Expected $recipeItem x$iterations, but found x$multiplier");
115 }
116 }
117
118 if(count($txItems) > 0){
119 //all items should be destroyed in this process
120 throw new TransactionValidationException("Expected 0 ingredients left over, have " . count($txItems));
121 }
122
123 return $iterations;
124 }

126 public function validate() : void{
127 $this->squashDuplicateSlotChanges();
128 if(count($this->actions) < 1){
129 throw new TransactionValidationException("Transaction must have at least one action to be executable");
130 }
131
132 $this->matchItems($this->outputs, $this->inputs);
133
134 $failed = 0;
135 foreach($this->craftingManager->matchRecipeByOutputs($this->outputs) as $recipe){
136 try{
137 //compute number of times recipe was crafted
138 $this->repetitions = $this->matchRecipeItems($this->outputs, $recipe->getResultsFor($this->source->getCraftingGrid()), false);
139 //assert that $repetitions x recipe ingredients should be consumed
140 $this->matchRecipeItems($this->inputs, $recipe->getIngredientList(), true, $this->repetitions);
141
142 //Success!
143 $this->recipe = $recipe;
144 break;
145 }catch(TransactionValidationException $e){
146 //failed
147 ++$failed;
148 }
149 }
150
151 if($this->recipe === null){
152 throw new TransactionValidationException("Unable to match a recipe to transaction (tried to match against $failed recipes)");
153 }
154 }
