100 private function prepareNodes() : LightPropagationContext{
101 $context = new LightPropagationContext();
102 foreach($this->updateNodes as $blockHash => [$x, $y, $z, $newLevel]){
103 if($this->subChunkExplorer->moveTo($x, $y, $z) !== SubChunkExplorerStatus::INVALID){
104 $lightArray = $this->getCurrentLightArray();
105 $oldLevel = $lightArray->get($x & SubChunk::COORD_MASK, $y & SubChunk::COORD_MASK, $z & SubChunk::COORD_MASK);
106
107 if($oldLevel !== $newLevel){
108 $lightArray->set($x & SubChunk::COORD_MASK, $y & SubChunk::COORD_MASK, $z & SubChunk::COORD_MASK, $newLevel);
109 if($oldLevel < $newLevel){ //light increased
110 $context->spreadVisited[$blockHash] = true;
111 $context->spreadQueue->enqueue([$x, $y, $z]);
112 }else{ //light removed
113 $context->removalVisited[$blockHash] = true;
114 $context->removalQueue->enqueue([$x, $y, $z, $oldLevel]);
115 }
116 }
117 }
118 }
119 return $context;

122 public function execute() : int{
123 $context = $this->prepareNodes();
124
125 $touched = 0;
126 while(!$context->removalQueue->isEmpty()){
127 $touched++;
128 [$x, $y, $z, $oldAdjacentLight] = $context->removalQueue->dequeue();
129
130 foreach(self::ADJACENTS as [$ox, $oy, $oz]){
131 $cx = $x + $ox;
132 $cy = $y + $oy;
133 $cz = $z + $oz;
134
135 if($this->subChunkExplorer->moveTo($cx, $cy, $cz) !== SubChunkExplorerStatus::INVALID){
136 $this->computeRemoveLight($cx, $cy, $cz, $oldAdjacentLight, $context);
137 }elseif($this->getEffectiveLight($cx, $cy, $cz) > 0 && !isset($context->spreadVisited[$index = World::blockHash($cx, $cy, $cz)])){
138 $context->spreadVisited[$index] = true;
139 $context->spreadQueue->enqueue([$cx, $cy, $cz]);
140 }
141 }
142 }
143
144 while(!$context->spreadQueue->isEmpty()){
145 $touched++;
146 [$x, $y, $z] = $context->spreadQueue->dequeue();
147
148 unset($context->spreadVisited[World::blockHash($x, $y, $z)]);
149
150 $newAdjacentLight = $this->getEffectiveLight($x, $y, $z);
151 if($newAdjacentLight <= 0){
152 continue;
153 }
154
155 foreach(self::ADJACENTS as [$ox, $oy, $oz]){
156 $cx = $x + $ox;
157 $cy = $y + $oy;
158 $cz = $z + $oz;
159
160 if($this->subChunkExplorer->moveTo($cx, $cy, $cz) !== SubChunkExplorerStatus::INVALID){
161 $this->computeSpreadLight($cx, $cy, $cz, $newAdjacentLight, $context);
162 }
163 }
164 }
165
166 return $touched;
167 }

169 protected function computeRemoveLight(int $x, int $y, int $z, int $oldAdjacentLevel, LightPropagationContext $context) : void{
170 $lightArray = $this->getCurrentLightArray();
171 $lx = $x & SubChunk::COORD_MASK;
172 $ly = $y & SubChunk::COORD_MASK;
173 $lz = $z & SubChunk::COORD_MASK;
174 $current = $lightArray->get($lx, $ly, $lz);
175
176 if($current !== 0 && $current < $oldAdjacentLevel){
177 $lightArray->set($lx, $ly, $lz, 0);
178
179 if(!isset($context->removalVisited[$index = World::blockHash($x, $y, $z)])){
180 $context->removalVisited[$index] = true;
181 if($current > 1){
182 $context->removalQueue->enqueue([$x, $y, $z, $current]);
183 }
184 }
185 }elseif($current >= $oldAdjacentLevel){
186 if(!isset($context->spreadVisited[$index = World::blockHash($x, $y, $z)])){
187 $context->spreadVisited[$index] = true;
188 $context->spreadQueue->enqueue([$x, $y, $z]);
189 }
190 }
191 }

193 protected function computeSpreadLight(int $x, int $y, int $z, int $newAdjacentLevel, LightPropagationContext $context) : void{
194 $lightArray = $this->getCurrentLightArray();
195 $lx = $x & SubChunk::COORD_MASK;
196 $ly = $y & SubChunk::COORD_MASK;
197 $lz = $z & SubChunk::COORD_MASK;
198 $current = $lightArray->get($lx, $ly, $lz);
199 $potentialLight = $newAdjacentLevel - $this->lightFilters[$this->subChunkExplorer->currentSubChunk->getFullBlock($lx, $ly, $lz)];
200
201 if($current < $potentialLight){
202 $lightArray->set($lx, $ly, $lz, $potentialLight);
203
204 if(!isset($context->spreadVisited[$index = World::blockHash($x, $y, $z)]) && $potentialLight > 1){
205 $context->spreadVisited[$index] = true;
206 $context->spreadQueue->enqueue([$x, $y, $z]);
207 }
208 }
