107 public function __construct(Player $player, NetworkSession $session){
108 $this->player = $player;
109 $this->session = $session;

111 $this->containerOpenCallbacks = new ObjectSet();
112 $this->containerOpenCallbacks->add(\Closure::fromCallable([self::class, 'createContainerOpen']));

117 $this->add(ContainerIds::UI, $this->player->getCursorInventory());

120 $this->syncSelectedHotbarSlot();

144 public function onTransactionStart(InventoryTransaction $tx) : void{
145 foreach($tx->getActions() as $action){
146 if($action instanceof SlotChangeAction && ($windowId = $this->getWindowId($action->getInventory())) !== null){
147 //in some cases the inventory might not have a window ID, but still be referenced by a transaction (e.g. crafting grid changes), so we can't unconditionally record the change here or we might leak things
148 $this->initiatedSlotChanges[$windowId][$action->getSlot()] = $action->getTargetItem();
149 }
150 }

157 public function addPredictedSlotChanges(array $networkInventoryActions) : void{
158 foreach($networkInventoryActions as $action){
159 if($action->sourceType === NetworkInventoryAction::SOURCE_CONTAINER && isset($this->windowMap[$action->windowId])){
160 //this won't cover stuff like crafting grid due to too much magic
161 try{
162 $item = TypeConverter::getInstance()->netItemStackToCore($action->newItem->getItemStack());
163 }catch(TypeConversionException $e){
164 throw new PacketHandlingException($e->getMessage(), 0, $e);
165 }
166 $this->initiatedSlotChanges[$action->windowId][$action->inventorySlot] = $item;
167 }
168 }

171 public function onCurrentWindowChange(Inventory $inventory) : void{
172 $this->onCurrentWindowRemove();
173 $this->add($this->lastInventoryNetworkId = max(ContainerIds::FIRST, ($this->lastInventoryNetworkId + 1) % self::RESERVED_WINDOW_ID_RANGE_START), $inventory);
174
175 foreach($this->containerOpenCallbacks as $callback){
176 $pks = $callback($this->lastInventoryNetworkId, $inventory);
177 if($pks !== null){
178 foreach($pks as $pk){
179 $this->session->sendDataPacket($pk);
180 }
181 $this->syncContents($inventory);
182 return;
183 }
184 }
185 throw new \LogicException("Unsupported inventory type");
186 }

195 protected static function createContainerOpen(int $id, Inventory $inv) : ?array{
196 //TODO: we should be using some kind of tagging system to identify the types. Instanceof is flaky especially
197 //if the class isn't final, not to mention being inflexible.
198 if($inv instanceof BlockInventory){
199 $blockPosition = BlockPosition::fromVector3($inv->getHolder());
200 $windowType = match(true){
201 $inv instanceof LoomInventory => WindowTypes::LOOM,
202 $inv instanceof FurnaceInventory => match($inv->getFurnaceType()->id()){
203 FurnaceType::FURNACE()->id() => WindowTypes::FURNACE,
204 FurnaceType::BLAST_FURNACE()->id() => WindowTypes::BLAST_FURNACE,
205 FurnaceType::SMOKER()->id() => WindowTypes::SMOKER,
206 default => throw new AssumptionFailedError("Unreachable")
207 },
208 $inv instanceof EnchantInventory => WindowTypes::ENCHANTMENT,
209 $inv instanceof BrewingStandInventory => WindowTypes::BREWING_STAND,
210 $inv instanceof AnvilInventory => WindowTypes::ANVIL,
211 $inv instanceof HopperInventory => WindowTypes::HOPPER,
212 $inv instanceof CraftingTableInventory => WindowTypes::WORKBENCH,
213 default => WindowTypes::CONTAINER
214 };
215 return [ContainerOpenPacket::blockInv($id, $windowType, $blockPosition)];
216 }
217 return null;
218 }

220 public function onClientOpenMainInventory() : void{
221 $id = self::HARDCODED_INVENTORY_WINDOW_ID;
222 if(!isset($this->openHardcodedWindows[$id])){
223 //TODO: HACK! this restores 1.14ish behaviour, but this should be able to be listened to and
224 //controlled by plugins. However, the player is always a subscriber to their own inventory so it
225 //doesn't integrate well with the regular container system right now.
226 $this->openHardcodedWindows[$id] = true;
227 $this->session->sendDataPacket(ContainerOpenPacket::entityInv(
228 InventoryManager::HARDCODED_INVENTORY_WINDOW_ID,
229 WindowTypes::INVENTORY,
230 $this->player->getId()
231 ));
232 }

242 public function onClientRemoveWindow(int $id) : void{
243 if(isset($this->openHardcodedWindows[$id])){
244 unset($this->openHardcodedWindows[$id]);
245 }elseif($id === $this->lastInventoryNetworkId){
246 $this->remove($id);
247 $this->player->removeCurrentWindow();
248 }else{
249 $this->session->getLogger()->debug("Attempted to close inventory with network ID $id, but current is $this->lastInventoryNetworkId");
250 }
251
252 //Always send this, even if no window matches. If we told the client to close a window, it will behave as if it
253 //initiated the close and expect an ack.
254 $this->session->sendDataPacket(ContainerClosePacket::create($id, false));
255 }

257 public function syncSlot(Inventory $inventory, int $slot) : void{
258 $windowId = $this->getWindowId($inventory);
259 if($windowId !== null){
260 $currentItem = $inventory->getItem($slot);
261 $clientSideItem = $this->initiatedSlotChanges[$windowId][$slot] ?? null;
262 if($clientSideItem === null || !$clientSideItem->equalsExact($currentItem)){
263 $itemStackWrapper = ItemStackWrapper::legacy(TypeConverter::getInstance()->coreItemStackToNet($currentItem));
264 if($windowId === ContainerIds::OFFHAND){
265 //TODO: HACK!
266 //The client may sometimes ignore the InventorySlotPacket for the offhand slot.
267 //This can cause a lot of problems (totems, arrows, and more...).
268 //The workaround is to send an InventoryContentPacket instead
269 //BDS (Bedrock Dedicated Server) also seems to work this way.
270 $this->session->sendDataPacket(InventoryContentPacket::create($windowId, [$itemStackWrapper]));
271 }else{
272 $this->session->sendDataPacket(InventorySlotPacket::create($windowId, $slot, $itemStackWrapper));
273 }
274 }
275 unset($this->initiatedSlotChanges[$windowId][$slot]);
276 }
277 }

279 public function syncContents(Inventory $inventory) : void{
280 $windowId = $this->getWindowId($inventory);
281 if($windowId !== null){
282 unset($this->initiatedSlotChanges[$windowId]);
283 $typeConverter = TypeConverter::getInstance();
284 if($windowId === ContainerIds::UI){
285 //TODO: HACK!
286 //Since 1.13, cursor is now part of a larger "UI inventory", and sending contents for this larger inventory does
287 //not work the way it's intended to. Even if it did, it would be necessary to send all 51 slots just to update
288 //this one, which is just not worth it.
289 //This workaround isn't great, but it's at least simple.
290 $this->session->sendDataPacket(InventorySlotPacket::create(
291 $windowId,
292 0,
293 ItemStackWrapper::legacy($typeConverter->coreItemStackToNet($inventory->getItem(0)))
294 ));
295 }else{
296 $this->session->sendDataPacket(InventoryContentPacket::create($windowId, array_map(function(Item $itemStack) use ($typeConverter) : ItemStackWrapper{
297 return ItemStackWrapper::legacy($typeConverter->coreItemStackToNet($itemStack));
298 }, $inventory->getContents(true))));
299 }
300 }
301 }

309 public function syncMismatchedPredictedSlotChanges() : void{
310 foreach($this->initiatedSlotChanges as $windowId => $slots){
311 if(!isset($this->windowMap[$windowId])){
312 continue;
313 }
314 $inventory = $this->windowMap[$windowId];
315
316 foreach($slots as $slot => $expectedItem){
317 if(!$inventory->slotExists($slot)){
318 continue; //TODO: size desync ???
319 }
320 $actualItem = $inventory->getItem($slot);
321 if(!$actualItem->equalsExact($expectedItem)){
322 $this->session->getLogger()->debug("Detected prediction mismatch in inventory " . get_class($inventory) . "#" . spl_object_id($inventory) . " slot $slot");
323 $this->syncSlot($inventory, $slot);
324 }
325 }
326 }
327
328 $this->initiatedSlotChanges = [];
329 }

342 public function syncSelectedHotbarSlot() : void{
343 $selected = $this->player->getInventory()->getHeldItemIndex();
344 if($selected !== $this->clientSelectedHotbarSlot){
345 $this->session->sendDataPacket(MobEquipmentPacket::create(
346 $this->player->getId(),
347 ItemStackWrapper::legacy(TypeConverter::getInstance()->coreItemStackToNet($this->player->getInventory()->getItemInHand())),
348 $selected,
349 $selected,
350 ContainerIds::INVENTORY
351 ));
352 $this->clientSelectedHotbarSlot = $selected;
353 }
