129 public function onNearbyBlockChange() : void{
130 $changed = false;
131
132 $up = $this->getSide(Facing::UP);
133 //check which faces have corresponding vines in the block above
134 $supportedFaces = $up instanceof Vine ? array_intersect_key($this->faces, $up->faces) : [];
135
136 foreach($this->faces as $face){
137 if(!isset($supportedFaces[$face]) && !$this->getSide($face)->isSolid()){
138 unset($this->faces[$face]);
139 $changed = true;
140 }
141 }
142
143 if($changed){
144 if(count($this->faces) === 0){
145 $this->position->getWorld()->useBreakOn($this->position);
146 }else{
147 $this->position->getWorld()->setBlock($this->position, $this);
148 }
149 }
