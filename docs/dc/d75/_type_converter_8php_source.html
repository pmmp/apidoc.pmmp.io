113 public function coreItemStackToRecipeIngredient(Item $itemStack) : RecipeIngredient{
114 if($itemStack->isNull()){
115 return new RecipeIngredient(null, 0);
116 }
117 if($itemStack->hasAnyDamageValue()){
118 [$id, ] = ItemTranslator::getInstance()->toNetworkId($itemStack->getId(), 0);
119 $meta = 0x7fff;
120 }else{
121 [$id, $meta] = ItemTranslator::getInstance()->toNetworkId($itemStack->getId(), $itemStack->getMeta());
122 }
123 return new RecipeIngredient(new IntIdMetaItemDescriptor($id, $meta), $itemStack->getCount());
124 }

126 public function recipeIngredientToCoreItemStack(RecipeIngredient $ingredient) : Item{
127 $descriptor = $ingredient->getDescriptor();
128 if($descriptor === null){
129 return VanillaItems::AIR();
130 }
131 if($descriptor instanceof IntIdMetaItemDescriptor){
132 [$id, $meta] = ItemTranslator::getInstance()->fromNetworkIdWithWildcardHandling($descriptor->getId(), $descriptor->getMeta());
133 return ItemFactory::getInstance()->get($id, $meta, $ingredient->getCount());
134 }
135 if($descriptor instanceof StringIdMetaItemDescriptor){
136 $intId = GlobalItemTypeDictionary::getInstance()->getDictionary()->fromStringId($descriptor->getId());
137 [$id, $meta] = ItemTranslator::getInstance()->fromNetworkIdWithWildcardHandling($intId, $descriptor->getMeta());
138 return ItemFactory::getInstance()->get($id, $meta, $ingredient->getCount());
139 }
140
141 throw new \LogicException("Unsupported conversion of recipe ingredient to core item stack");
142 }

144 public function coreItemStackToNet(Item $itemStack) : ItemStack{
145 if($itemStack->isNull()){
146 return ItemStack::null();
147 }
148 $nbt = null;
149 if($itemStack->hasNamedTag()){
150 $nbt = clone $itemStack->getNamedTag();
151 }
152
153 $isBlockItem = $itemStack->getId() < 256;
154
155 $idMeta = ItemTranslator::getInstance()->toNetworkIdQuiet($itemStack->getId(), $itemStack->getMeta());
156 if($idMeta === null){
157 //Display unmapped items as INFO_UPDATE, but stick something in their NBT to make sure they don't stack with
158 //other unmapped items.
159 [$id, $meta] = ItemTranslator::getInstance()->toNetworkId(ItemIds::INFO_UPDATE, 0);
160 if($nbt === null){
161 $nbt = new CompoundTag();
162 }
163 $nbt->setInt(self::PM_ID_TAG, $itemStack->getId());
164 $nbt->setInt(self::PM_META_TAG, $itemStack->getMeta());
165 }else{
166 [$id, $meta] = $idMeta;
167
168 if($itemStack instanceof Durable && $itemStack->getDamage() > 0){
169 if($nbt !== null){
170 if(($existing = $nbt->getTag(self::DAMAGE_TAG)) !== null){
171 $nbt->removeTag(self::DAMAGE_TAG);
172 $nbt->setTag(self::DAMAGE_TAG_CONFLICT_RESOLUTION, $existing);
173 }
174 }else{
175 $nbt = new CompoundTag();
176 }
177 $nbt->setInt(self::DAMAGE_TAG, $itemStack->getDamage());
178 $meta = 0;
179 }elseif($isBlockItem && $itemStack->getMeta() !== 0){
180 //TODO HACK: This foul-smelling code ensures that we can correctly deserialize an item when the
181 //client sends it back to us, because as of 1.16.220, blockitems quietly discard their metadata
182 //client-side. Aside from being very annoying, this also breaks various server-side behaviours.
183 if($nbt === null){
184 $nbt = new CompoundTag();
185 }
186 $nbt->setInt(self::PM_META_TAG, $itemStack->getMeta());
187 $meta = 0;
188 }
189 }
190
191 $blockRuntimeId = 0;
192 if($isBlockItem){
193 $block = $itemStack->getBlock();
194 if($block->getId() !== BlockLegacyIds::AIR){
195 $blockRuntimeId = RuntimeBlockMapping::getInstance()->toRuntimeId($block->getFullId());
196 }
197 }
198
199 return new ItemStack(
200 $id,
201 $meta,
202 $itemStack->getCount(),
203 $blockRuntimeId,
204 $nbt,
205 [],
206 [],
207 $id === $this->shieldRuntimeId ? 0 : null
208 );
209 }

214 public function netItemStackToCore(ItemStack $itemStack) : Item{
215 if($itemStack->getId() === 0){
216 return VanillaItems::AIR();
217 }
218 $compound = $itemStack->getNbt();
219
220 [$id, $meta] = ItemTranslator::getInstance()->fromNetworkId($itemStack->getId(), $itemStack->getMeta());
221
222 if($compound !== null){
223 $compound = clone $compound;
224 if(($idTag = $compound->getTag(self::PM_ID_TAG)) instanceof IntTag){
225 $id = $idTag->getValue();
226 $compound->removeTag(self::PM_ID_TAG);
227 }
228 if(($damageTag = $compound->getTag(self::DAMAGE_TAG)) instanceof IntTag){
229 $meta = $damageTag->getValue();
230 $compound->removeTag(self::DAMAGE_TAG);
231 if(($conflicted = $compound->getTag(self::DAMAGE_TAG_CONFLICT_RESOLUTION)) !== null){
232 $compound->removeTag(self::DAMAGE_TAG_CONFLICT_RESOLUTION);
233 $compound->setTag(self::DAMAGE_TAG, $conflicted);
234 }
235 }elseif(($metaTag = $compound->getTag(self::PM_META_TAG)) instanceof IntTag){
236 //TODO HACK: This foul-smelling code ensures that we can correctly deserialize an item when the
237 //client sends it back to us, because as of 1.16.220, blockitems quietly discard their metadata
238 //client-side. Aside from being very annoying, this also breaks various server-side behaviours.
239 $meta = $metaTag->getValue();
240 $compound->removeTag(self::PM_META_TAG);
241 }
242 if($compound->count() === 0){
243 $compound = null;
244 }
245 }
246 if($id < -0x8000 || $id >= 0x7fff){
247 throw new TypeConversionException("Item ID must be in range " . -0x8000 . " ... " . 0x7fff . " (received $id)");
248 }
249 if($meta < 0 || $meta >= 0x7fff){ //this meta value may have been restored from the NBT
250 throw new TypeConversionException("Item meta must be in range 0 ... " . 0x7fff . " (received $meta)");
251 }
252
253 try{
254 return ItemFactory::getInstance()->get(
255 $id,
256 $meta,
257 $itemStack->getCount(),
258 $compound
259 );
260 }catch(NbtException $e){
261 throw TypeConversionException::wrap($e, "Bad itemstack NBT data");
262 }
263 }

268 public function createInventoryAction(NetworkInventoryAction $action, Player $player, InventoryManager $inventoryManager) : ?InventoryAction{
269 if($action->oldItem->getItemStack()->equals($action->newItem->getItemStack())){
270 //filter out useless noise in 1.13
271 return null;
272 }
273 try{
274 $old = $this->netItemStackToCore($action->oldItem->getItemStack());
275 }catch(TypeConversionException $e){
276 throw TypeConversionException::wrap($e, "Inventory action: oldItem");
277 }
278 try{
279 $new = $this->netItemStackToCore($action->newItem->getItemStack());
280 }catch(TypeConversionException $e){
281 throw TypeConversionException::wrap($e, "Inventory action: newItem");
282 }
283 switch($action->sourceType){
284 case NetworkInventoryAction::SOURCE_CONTAINER:
285 if($action->windowId === ContainerIds::UI && $action->inventorySlot === UIInventorySlotOffset::CREATED_ITEM_OUTPUT){
286 return null; //useless noise
287 }
288 $located = $inventoryManager->locateWindowAndSlot($action->windowId, $action->inventorySlot);
289 if($located !== null){
290 [$window, $slot] = $located;
291 return new SlotChangeAction($window, $slot, $old, $new);
292 }
293
294 throw new TypeConversionException("No open container with window ID $action->windowId");
295 case NetworkInventoryAction::SOURCE_WORLD:
296 if($action->inventorySlot !== NetworkInventoryAction::ACTION_MAGIC_SLOT_DROP_ITEM){
297 throw new TypeConversionException("Only expecting drop-item world actions from the client!");
298 }
299
300 return new DropItemAction($new);
301 case NetworkInventoryAction::SOURCE_CREATIVE:
302 switch($action->inventorySlot){
303 case NetworkInventoryAction::ACTION_MAGIC_SLOT_CREATIVE_DELETE_ITEM:
304 return new DestroyItemAction($new);
305 case NetworkInventoryAction::ACTION_MAGIC_SLOT_CREATIVE_CREATE_ITEM:
306 return new CreateItemAction($old);
307 default:
308 throw new TypeConversionException("Unexpected creative action type $action->inventorySlot");
309
310 }
311 case NetworkInventoryAction::SOURCE_TODO:
312 //These are used to balance a transaction that involves special actions, like crafting, enchanting, etc.
313 //The vanilla server just accepted these without verifying them. We don't need to care about them since
314 //we verify crafting by checking for imbalances anyway.
315 return null;
316 default:
317 throw new TypeConversionException("Unknown inventory source type $action->sourceType");
318 }
319 }
