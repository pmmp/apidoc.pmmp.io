89 protected function findLog(Vector3 $pos, array &$visited = [], int $distance = 0) : bool{
90 $index = World::blockHash($pos->x, $pos->y, $pos->z);
91 if(isset($visited[$index])){
92 return false;
93 }
94 $visited[$index] = true;
95
96 $block = $this->position->getWorld()->getBlock($pos);
97 if($block instanceof Wood){ //type doesn't matter
98 return true;
99 }
100
101 if($block->getId() === $this->getId() && $distance <= 4){
102 foreach(Facing::ALL as $side){
103 if($this->findLog($pos->getSide($side), $visited, $distance + 1)){
104 return true;
105 }
106 }
107 }
108
109 return false;
110 }

123 public function onRandomTick() : void{

125 $ev = new LeavesDecayEvent($this);
126 $ev->call();
127 $world = $this->position->getWorld();
128 if($ev->isCancelled() || $this->findLog($this->position)){
129 $this->checkDecay = false;
130 $world->setBlock($this->position, $this, false);
