100 private static function pack(array $items) : array{
102 $result = [];
103
104 foreach($items as $i => $item){
105 foreach($result as $otherItem){
106 if($item->canStackWith($otherItem)){
107 $otherItem->setCount($otherItem->getCount() + $item->getCount());
108 continue 2;
109 }
110 }
111
112 //No matching item found
113 $result[] = clone $item;
114 }
115
116 return $result;
117 }

122 private static function hashOutputs(array $outputs) : string{
123 $outputs = self::pack($outputs);
124 usort($outputs, [self::class, "sort"]);
125 $result = new BinaryStream();
126 foreach($outputs as $o){
127 //count is not written because the outputs might be from multiple repetitions of a single recipe
128 //this reduces the accuracy of the hash, but it won't matter in most cases.
129 $result->putVarInt($o->getId());
130 $result->putVarInt($o->getMeta());
131 $result->put((new LittleEndianNbtSerializer())->write(new TreeRoot($o->getNamedTag())));
132 }
133
134 return $result->getBuffer();
135 }
