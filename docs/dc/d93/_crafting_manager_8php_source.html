109 private static function pack(array $items) : array{
111 $result = [];
112
113 foreach($items as $i => $item){
114 foreach($result as $otherItem){
115 if($item->canStackWith($otherItem)){
116 $otherItem->setCount($otherItem->getCount() + $item->getCount());
117 continue 2;
118 }
119 }
120
121 //No matching item found
122 $result[] = clone $item;
123 }
124
125 return $result;
126 }

131 private static function hashOutputs(array $outputs) : string{
132 $outputs = self::pack($outputs);
133 usort($outputs, [self::class, "sort"]);
134 $result = new BinaryStream();
135 foreach($outputs as $o){
136 //count is not written because the outputs might be from multiple repetitions of a single recipe
137 //this reduces the accuracy of the hash, but it won't matter in most cases.
138 $result->putVarInt($o->getId());
139 $result->putVarInt($o->getMeta());
140 $result->put((new LittleEndianNbtSerializer())->write(new TreeRoot($o->getNamedTag())));
141 }
142
143 return $result->getBuffer();
144 }
