58 public function __construct(string $canonicalBlockStatesFile, string $r12ToCurrentBlockMapFile){
59 $stream = PacketSerializer::decoder(
60 Utils::assumeNotFalse(file_get_contents($canonicalBlockStatesFile), "Missing required resource file"),
61 0,
62 new PacketSerializerContext(GlobalItemTypeDictionary::getInstance()->getDictionary())
63 );
64 $list = [];

73 private function setupLegacyMappings(string $r12ToCurrentBlockMapFile) : void{
74 $legacyIdMap = LegacyBlockIdToStringIdMap::getInstance();
76 $legacyStateMap = [];
77 $legacyStateMapReader = PacketSerializer::decoder(
78 Utils::assumeNotFalse(file_get_contents($r12ToCurrentBlockMapFile), "Missing required resource file"),
79 0,
80 new PacketSerializerContext(GlobalItemTypeDictionary::getInstance()->getDictionary())
81 );
82 $nbtReader = new NetworkNbtSerializer();
83 while(!$legacyStateMapReader->feof()){
84 $id = $legacyStateMapReader->getString();
85 $meta = $legacyStateMapReader->getLShort();
86
87 $offset = $legacyStateMapReader->getOffset();
88 $state = $nbtReader->read($legacyStateMapReader->getBuffer(), $offset)->mustGetCompoundTag();
89 $legacyStateMapReader->setOffset($offset);
90 $legacyStateMap[] = new R12ToCurrentBlockMapEntry($id, $meta, $state);
91 }
92
96 $idToStatesMap = [];
97 foreach($this->bedrockKnownStates as $k => $state){
98 $idToStatesMap[$state->getString("name")][] = $k;
99 }
100 foreach($legacyStateMap as $pair){
101 $id = $legacyIdMap->stringToLegacy($pair->getId());
102 if($id === null){
103 throw new \RuntimeException("No legacy ID matches " . $pair->getId());
104 }
105 $data = $pair->getMeta();
106 if($data > 15){
107 //we can't handle metadata with more than 4 bits
108 continue;
109 }
110 $mappedState = $pair->getBlockState();
111 $mappedName = $mappedState->getString("name");
112 if(!isset($idToStatesMap[$mappedName])){
113 throw new \RuntimeException("Mapped new state does not appear in network table");
114 }
115 foreach($idToStatesMap[$mappedName] as $k){
116 $networkState = $this->bedrockKnownStates[$k];
117 if($mappedState->equals($networkState)){
118 $this->registerMapping($k, $id, $data);
119 continue 2;
120 }
121 }
122 throw new \RuntimeException("Mapped new state does not appear in network table");
123 }
124 }

134 private function registerMapping(int $staticRuntimeId, int $legacyId, int $legacyMeta) : void{
135 $this->legacyToRuntimeMap[($legacyId << Block::INTERNAL_METADATA_BITS) | $legacyMeta] = $staticRuntimeId;
136 $this->runtimeToLegacyMap[$staticRuntimeId] = ($legacyId << Block::INTERNAL_METADATA_BITS) | $legacyMeta;
137 }
