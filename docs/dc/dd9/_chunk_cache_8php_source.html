49 public static function getInstance(World $world, Compressor $compressor) : self{
50 $worldId = spl_object_id($world);
51 $compressorId = spl_object_id($compressor);
52 if(!isset(self::$instances[$worldId])){
53 self::$instances[$worldId] = [];
54 $world->addOnUnloadCallback(static function() use ($worldId) : void{
55 foreach(self::$instances[$worldId] as $cache){
56 $cache->caches = [];
57 }
58 unset(self::$instances[$worldId]);
59 \GlobalLogger::get()->debug("Destroyed chunk packet caches for world#$worldId");
60 });
61 }
62 if(!isset(self::$instances[$worldId][$compressorId])){
63 \GlobalLogger::get()->debug("Created new chunk packet cache (world#$worldId, compressor#$compressorId)");
64 self::$instances[$worldId][$compressorId] = new self($world, $compressor);
65 }
66 return self::$instances[$worldId][$compressorId];
67 }

69 public static function pruneCaches() : void{
70 foreach(self::$instances as $compressorMap){
71 foreach($compressorMap as $chunkCache){
72 foreach($chunkCache->caches as $chunkHash => $promise){
73 if($promise->hasResult()){
74 //Do not clear promises that are not yet fulfilled; they will have requesters waiting on them
75 unset($chunkCache->caches[$chunkHash]);
76 }
77 }
78 }
79 }

98 public function request(int $chunkX, int $chunkZ) : CompressBatchPromise{
99 $this->world->registerChunkListener($this, $chunkX, $chunkZ);
100 $chunk = $this->world->getChunk($chunkX, $chunkZ);
101 if($chunk === null){
102 throw new \InvalidArgumentException("Cannot request an unloaded chunk");
103 }
104 $chunkHash = World::chunkHash($chunkX, $chunkZ);
105
106 if(isset($this->caches[$chunkHash])){
107 ++$this->hits;
108 return $this->caches[$chunkHash];
109 }
110
111 ++$this->misses;
112
113 $this->world->timings->syncChunkSendPrepare->startTiming();
114 try{
115 $this->caches[$chunkHash] = new CompressBatchPromise();
116
117 $this->world->getServer()->getAsyncPool()->submitTask(
118 new ChunkRequestTask(
119 $chunkX,
120 $chunkZ,
121 $chunk,
122 $this->caches[$chunkHash],
123 $this->compressor,
124 function() use ($chunkHash, $chunkX, $chunkZ) : void{
125 $this->world->getLogger()->error("Failed preparing chunk $chunkX $chunkZ, retrying");
126
127 if(isset($this->caches[$chunkHash])){
128 $this->restartPendingRequest($chunkX, $chunkZ);
129 }
130 }
131 )
132 );
133
134 return $this->caches[$chunkHash];
135 }finally{
136 $this->world->timings->syncChunkSendPrepare->stopTiming();
137 }
138 }

140 private function destroy(int $chunkX, int $chunkZ) : bool{
141 $chunkHash = World::chunkHash($chunkX, $chunkZ);
142 $existing = $this->caches[$chunkHash] ?? null;
143 unset($this->caches[$chunkHash]);
144
145 return $existing !== null;
146 }
147
153 private function restartPendingRequest(int $chunkX, int $chunkZ) : void{
154 $chunkHash = World::chunkHash($chunkX, $chunkZ);
155 $existing = $this->caches[$chunkHash] ?? null;
156 if($existing === null || $existing->hasResult()){
157 throw new \InvalidArgumentException("Restart can only be applied to unresolved promises");
158 }
159 $existing->cancel();
160 unset($this->caches[$chunkHash]);
161
162 $this->request($chunkX, $chunkZ)->onResolve(...$existing->getResolveCallbacks());
163 }
164
168 private function destroyOrRestart(int $chunkX, int $chunkZ) : void{
169 $cache = $this->caches[World::chunkHash($chunkX, $chunkZ)] ?? null;
170 if($cache !== null){
171 if(!$cache->hasResult()){
172 //some requesters are waiting for this chunk, so their request needs to be fulfilled
173 $this->restartPendingRequest($chunkX, $chunkZ);
174 }else{
175 //dump the cache, it'll be regenerated the next time it's requested
176 $this->destroy($chunkX, $chunkZ);
177 }
178 }
179 }

216 public function calculateCacheSize() : int{
217 $result = 0;
