184 public static function getScheduledTaskTimings(TaskHandler $task, int $period) : TimingsHandler{
185 $name = "Task: " . $task->getOwnerName() . " Runnable: " . $task->getTaskName();
186
187 if($period > 0){
188 $name .= "(interval:" . $period . ")";
189 }else{
190 $name .= "(Single)";
191 }
192
193 if(!isset(self::$pluginTaskTimingMap[$name])){
194 self::$pluginTaskTimingMap[$name] = new TimingsHandler($name, self::$schedulerSync);
195 }
196
197 return self::$pluginTaskTimingMap[$name];
198 }

200 public static function getEntityTimings(Entity $entity) : TimingsHandler{
