43 public function __construct(string $name){
44 parent::__construct(
45 $name,
46 KnownTranslationFactory::pocketmine_command_clear_description(),
47 KnownTranslationFactory::pocketmine_command_clear_usage()
48 );

52 public function execute(CommandSender $sender, string $commandLabel, array $args){
53 if(!$this->testPermission($sender)){
54 return true;
55 }
56
57 if(count($args) > 3){
58 throw new InvalidCommandSyntaxException();
59 }
60
61 $target = $this->fetchPermittedPlayerTarget($sender, $args[0] ?? null, DefaultPermissionNames::COMMAND_CLEAR_SELF, DefaultPermissionNames::COMMAND_CLEAR_OTHER);
62 if($target === null){
63 return true;
64 }
65
66 $targetItem = null;
67 $maxCount = -1;
68 if(isset($args[1])){
69 try{
70 $targetItem = StringToItemParser::getInstance()->parse($args[1]) ?? LegacyStringToItemParser::getInstance()->parse($args[1]);
71
72 if(isset($args[2])){
73 $targetItem->setCount($maxCount = $this->getInteger($sender, $args[2], -1));
74 }
75 }catch(LegacyStringToItemParserException $e){
76 //vanilla checks this at argument parsing layer, can't come up with a better alternative
77 $sender->sendMessage(KnownTranslationFactory::commands_give_item_notFound($args[1])->prefix(TextFormat::RED));
78 return true;
79 }
80 }
81
85 $inventories = [
86 $target->getInventory(),
87 $target->getCursorInventory(),
88 $target->getArmorInventory(),
89 $target->getOffHandInventory()
90 ];
91
92 // Checking player's inventory for all the items matching the criteria
93 if($targetItem !== null && $maxCount === 0){
94 $count = $this->countItems($inventories, $targetItem);
95 if($count > 0){
96 $sender->sendMessage(KnownTranslationFactory::commands_clear_testing($target->getName(), (string) $count));
97 }else{
98 $sender->sendMessage(KnownTranslationFactory::commands_clear_failure_no_items($target->getName())->prefix(TextFormat::RED));
99 }
100
101 return true;
102 }
103
104 $clearedCount = 0;
105 if($targetItem === null){
106 // Clear all items from the inventories
107 $clearedCount += $this->countItems($inventories, null);
108 foreach($inventories as $inventory){
109 $inventory->clearAll();
110 }
111 }else{
112 // Clear the item from target's inventory irrelevant of the count
113 if($maxCount === -1){
114 $clearedCount += $this->countItems($inventories, $targetItem);
115 foreach($inventories as $inventory){
116 $inventory->remove($targetItem);
117 }
118 }else{
119 // Clear the item from target's inventory up to maxCount
120 foreach($inventories as $inventory){
121 foreach($inventory->all($targetItem) as $index => $item){
122 // The count to reduce from the item and max count
123 $reductionCount = min($item->getCount(), $maxCount);
124 $item->pop($reductionCount);
125 $clearedCount += $reductionCount;
126 $inventory->setItem($index, $item);
127
128 $maxCount -= $reductionCount;
129 if($maxCount <= 0){
130 break 2;
131 }
132 }
133 }
134 }
135 }
136
137 if($clearedCount > 0){
138 Command::broadcastCommandMessage($sender, KnownTranslationFactory::commands_clear_success($target->getName(), (string) $clearedCount));
139 }else{
140 $sender->sendMessage(KnownTranslationFactory::commands_clear_failure_no_items($target->getName())->prefix(TextFormat::RED));
141 }
142
143 return true;
144 }

149 protected function countItems(array $inventories, ?Item $target) : int{
150 $count = 0;
151 foreach($inventories as $inventory){
152 $contents = $target !== null ? $inventory->all($target) : $inventory->getContents();
153 foreach($contents as $item){
154 $count += $item->getCount();
155 }
156 }
157 return $count;
158 }
