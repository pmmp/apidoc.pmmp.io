61 public static function defaultFromString($value) : string{
62 if(is_bool($value)){
63 if($value){
64 return "true";
65 }else{
66 return "false";
67 }
68 }
69 $lower = strtolower($value);
70 if(isset(self::DEFAULT_STRING_MAP[$lower])){
71 return self::DEFAULT_STRING_MAP[$lower];
72 }
73
74 throw new PermissionParserException("Unknown permission default name \"$value\"");
75 }

85 public static function loadPermissions(array $data, string $default = self::DEFAULT_FALSE) : array{
86 $result = [];
87 foreach(Utils::stringifyKeys($data) as $name => $entry){
88 $desc = null;
89 if(isset($entry["default"])){
90 $default = PermissionParser::defaultFromString($entry["default"]);
91 }
92
93 if(isset($entry["children"])){
94 throw new PermissionParserException("Nested permission declarations are no longer supported. Declare each permission separately.");
95 }
96
97 if(isset($entry["description"])){
98 $desc = $entry["description"];
99 }
100
101 $result[$default][] = new Permission($name, $desc);
102 }
103 return $result;
