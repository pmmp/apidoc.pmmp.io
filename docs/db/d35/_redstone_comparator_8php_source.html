40class RedstoneComparator extends Flowable{
41 use HorizontalFacingTrait;
42 use AnalogRedstoneSignalEmitterTrait;
43 use PoweredByRedstoneTrait;
44
45 protected BlockIdentifierFlattened $idInfoFlattened;
46
47 protected bool $isSubtractMode = false;
48
49 public function __construct(BlockIdentifierFlattened $idInfo, string $name, BlockBreakInfo $breakInfo){
50 $this->idInfoFlattened = $idInfo;
51 parent::__construct($idInfo, $name, $breakInfo);
52 }
53
54 public function getId() : int{
55 return $this->powered ? $this->idInfoFlattened->getSecondId() : parent::getId();
56 }
57
58 public function readStateFromData(int $id, int $stateMeta) : void{
59 $this->facing = BlockDataSerializer::readLegacyHorizontalFacing($stateMeta & 0x03);
60 $this->isSubtractMode = ($stateMeta & BlockLegacyMetadata::REDSTONE_COMPARATOR_FLAG_SUBTRACT) !== 0;
61 $this->powered = ($id === $this->idInfoFlattened->getSecondId() || ($stateMeta & BlockLegacyMetadata::REDSTONE_COMPARATOR_FLAG_POWERED) !== 0);
62 }
63
64 public function writeStateToMeta() : int{
65 return BlockDataSerializer::writeLegacyHorizontalFacing($this->facing) |
66 ($this->isSubtractMode ? BlockLegacyMetadata::REDSTONE_COMPARATOR_FLAG_SUBTRACT : 0) |
67 ($this->powered ? BlockLegacyMetadata::REDSTONE_COMPARATOR_FLAG_POWERED : 0);
68 }
69
70 public function getStateBitmask() : int{
71 return 0b1111;
72 }
73
74 public function readStateFromWorld() : void{
75 parent::readStateFromWorld();
76 $tile = $this->position->getWorld()->getTile($this->position);
77 if($tile instanceof Comparator){
78 $this->signalStrength = $tile->getSignalStrength();
79 }
80 }
81
82 public function writeStateToWorld() : void{
83 parent::writeStateToWorld();
84 $tile = $this->position->getWorld()->getTile($this->position);
85 assert($tile instanceof Comparator);
86 $tile->setSignalStrength($this->signalStrength);
87 }
88
89 public function isSubtractMode() : bool{
90 return $this->isSubtractMode;
91 }
92
94 public function setSubtractMode(bool $isSubtractMode) : self{
95 $this->isSubtractMode = $isSubtractMode;
96 return $this;
97 }
98
102 protected function recalculateCollisionBoxes() : array{
103 return [AxisAlignedBB::one()->trim(Facing::UP, 7 / 8)];
104 }
105
106 public function place(BlockTransaction $tx, Item $item, Block $blockReplace, Block $blockClicked, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
107 if($this->canBeSupportedBy($blockReplace->getSide(Facing::DOWN))){
108 if($player !== null){
109 $this->facing = Facing::opposite($player->getHorizontalFacing());
110 }
111 return parent::place($tx, $item, $blockReplace, $blockClicked, $face, $clickVector, $player);
112 }
113
114 return false;
115 }
116
117 public function onInteract(Item $item, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
118 $this->isSubtractMode = !$this->isSubtractMode;
119 $this->position->getWorld()->setBlock($this->position, $this);
120 return true;
121 }
122
123 public function onNearbyBlockChange() : void{
124 if(!$this->canBeSupportedBy($this->getSide(Facing::DOWN))){
125 $this->position->getWorld()->useBreakOn($this->position);
126 }
127 }
128
129 private function canBeSupportedBy(Block $block) : bool{
130 return !$block->getSupportType(Facing::UP)->equals(SupportType::NONE());
131 }
