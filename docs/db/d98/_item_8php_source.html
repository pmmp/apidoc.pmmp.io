271 protected function deserializeCompoundTag(CompoundTag $tag) : void{
272 $this->customName = "";
273 $this->lore = [];
274
275 $display = $tag->getCompoundTag(self::TAG_DISPLAY);
276 if($display !== null){
277 $this->customName = $display->getString(self::TAG_DISPLAY_NAME, $this->customName);
278 $lore = $display->getListTag(self::TAG_DISPLAY_LORE);
279 if($lore !== null && $lore->getTagType() === NBT::TAG_String){
281 foreach($lore as $t){
282 $this->lore[] = $t->getValue();
283 }
284 }
285 }
286
287 $this->removeEnchantments();
288 $enchantments = $tag->getListTag(self::TAG_ENCH);
289 if($enchantments !== null && $enchantments->getTagType() === NBT::TAG_Compound){
291 foreach($enchantments as $enchantment){
292 $magicNumber = $enchantment->getShort("id", -1);
293 $level = $enchantment->getShort("lvl", 0);
294 if($level <= 0){
295 continue;
296 }
297 $type = EnchantmentIdMap::getInstance()->fromId($magicNumber);
298 if($type !== null){
299 $this->addEnchantment(new EnchantmentInstance($type, $level));
300 }
301 }
302 }
303
304 $this->blockEntityTag = $tag->getCompoundTag(self::TAG_BLOCK_ENTITY_TAG);
305
306 $this->canPlaceOn = [];
307 $canPlaceOn = $tag->getListTag("CanPlaceOn");
308 if($canPlaceOn !== null && $canPlaceOn->getTagType() === NBT::TAG_String){
310 foreach($canPlaceOn as $entry){
311 $this->canPlaceOn[$entry->getValue()] = $entry->getValue();
312 }
313 }
314 $this->canDestroy = [];
315 $canDestroy = $tag->getListTag("CanDestroy");
316 if($canDestroy !== null && $canDestroy->getTagType() === NBT::TAG_String){
318 foreach($canDestroy as $entry){
319 $this->canDestroy[$entry->getValue()] = $entry->getValue();
320 }
321 }
322 }

324 protected function serializeCompoundTag(CompoundTag $tag) : void{
325 $display = $tag->getCompoundTag(self::TAG_DISPLAY) ?? new CompoundTag();
326
327 $this->hasCustomName() ?
328 $display->setString(self::TAG_DISPLAY_NAME, $this->getCustomName()) :
329 $display->removeTag(self::TAG_DISPLAY_NAME);
330
331 if(count($this->lore) > 0){
332 $loreTag = new ListTag();
333 foreach($this->lore as $line){
334 $loreTag->push(new StringTag($line));
335 }
336 $display->setTag(self::TAG_DISPLAY_LORE, $loreTag);
337 }else{
338 $display->removeTag(self::TAG_DISPLAY_LORE);
339 }
340 $display->count() > 0 ?
341 $tag->setTag(self::TAG_DISPLAY, $display) :
342 $tag->removeTag(self::TAG_DISPLAY);
343
344 if($this->hasEnchantments()){
345 $ench = new ListTag();
346 foreach($this->getEnchantments() as $enchantmentInstance){
347 $ench->push(CompoundTag::create()
348 ->setShort("id", EnchantmentIdMap::getInstance()->toId($enchantmentInstance->getType()))
349 ->setShort("lvl", $enchantmentInstance->getLevel())
350 );
351 }
352 $tag->setTag(self::TAG_ENCH, $ench);
353 }else{
354 $tag->removeTag(self::TAG_ENCH);
355 }
356
357 ($blockData = $this->getCustomBlockData()) !== null ?
358 $tag->setTag(self::TAG_BLOCK_ENTITY_TAG, clone $blockData) :
359 $tag->removeTag(self::TAG_BLOCK_ENTITY_TAG);
360
361 if(count($this->canPlaceOn) > 0){
362 $canPlaceOn = new ListTag();
363 foreach($this->canPlaceOn as $item){
364 $canPlaceOn->push(new StringTag($item));
365 }
366 $tag->setTag("CanPlaceOn", $canPlaceOn);
367 }else{
368 $tag->removeTag("CanPlaceOn");
369 }
370 if(count($this->canDestroy) > 0){
371 $canDestroy = new ListTag();
372 foreach($this->canDestroy as $item){
373 $canDestroy->push(new StringTag($item));
374 }
375 $tag->setTag("CanDestroy", $canDestroy);
376 }else{
377 $tag->removeTag("CanDestroy");
378 }

635 $nbt = "";

678 public static function nbtDeserialize(CompoundTag $tag) : Item{
679 if($tag->getTag("id") === null || $tag->getTag("Count") === null){
680 return VanillaItems::AIR();
681 }
682
683 $count = Binary::unsignByte($tag->getByte("Count"));
684 $meta = $tag->getShort("Damage", 0);
685
686 $idTag = $tag->getTag("id");
687 if($idTag instanceof ShortTag){
688 $item = ItemFactory::getInstance()->get($idTag->getValue(), $meta, $count);
689 }elseif($idTag instanceof StringTag){ //PC item save format
690 try{
691 $item = LegacyStringToItemParser::getInstance()->parse($idTag->getValue() . ":$meta");
692 }catch(LegacyStringToItemParserException $e){
693 //TODO: improve error handling
694 return VanillaItems::AIR();
695 }
696 $item->setCount($count);
697 }else{
698 throw new SavedDataLoadingException("Item CompoundTag ID must be an instance of StringTag or ShortTag, " . get_class($idTag) . " given");
699 }
700
701 $itemNBT = $tag->getCompoundTag("tag");
702 if($itemNBT !== null){
703 $item->setNamedTag(clone $itemNBT);
704 }
705
706 return $item;
707 }
