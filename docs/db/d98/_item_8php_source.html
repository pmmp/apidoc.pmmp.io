300 protected function deserializeCompoundTag(CompoundTag $tag) : void{
301 $this->customName = "";
302 $this->lore = [];
303
304 $display = $tag->getCompoundTag(self::TAG_DISPLAY);
305 if($display !== null){
306 $this->customName = $display->getString(self::TAG_DISPLAY_NAME, $this->customName);
307 $lore = $display->getListTag(self::TAG_DISPLAY_LORE);
308 if($lore !== null && $lore->getTagType() === NBT::TAG_String){
310 foreach($lore as $t){
311 $this->lore[] = $t->getValue();
312 }
313 }
314 }
315
316 $this->removeEnchantments();
317 $enchantments = $tag->getListTag(self::TAG_ENCH);
318 if($enchantments !== null && $enchantments->getTagType() === NBT::TAG_Compound){
320 foreach($enchantments as $enchantment){
321 $magicNumber = $enchantment->getShort(self::TAG_ENCH_ID, -1);
322 $level = $enchantment->getShort(self::TAG_ENCH_LVL, 0);
323 if($level <= 0){
324 continue;
325 }
326 $type = EnchantmentIdMap::getInstance()->fromId($magicNumber);
327 if($type !== null){
328 $this->addEnchantment(new EnchantmentInstance($type, $level));
329 }
330 }
331 }
332
333 $this->blockEntityTag = $tag->getCompoundTag(self::TAG_BLOCK_ENTITY_TAG);
334
335 $this->canPlaceOn = [];
336 $canPlaceOn = $tag->getListTag(self::TAG_CAN_PLACE_ON);
337 if($canPlaceOn !== null && $canPlaceOn->getTagType() === NBT::TAG_String){
339 foreach($canPlaceOn as $entry){
340 $this->canPlaceOn[$entry->getValue()] = $entry->getValue();
341 }
342 }
343 $this->canDestroy = [];
344 $canDestroy = $tag->getListTag(self::TAG_CAN_DESTROY);
345 if($canDestroy !== null && $canDestroy->getTagType() === NBT::TAG_String){
347 foreach($canDestroy as $entry){
348 $this->canDestroy[$entry->getValue()] = $entry->getValue();
349 }
350 }
351
352 $this->keepOnDeath = $tag->getByte(self::TAG_KEEP_ON_DEATH, 0) !== 0;
353 }

355 protected function serializeCompoundTag(CompoundTag $tag) : void{
356 $display = $tag->getCompoundTag(self::TAG_DISPLAY);
357
358 if($this->customName !== ""){
359 $display ??= new CompoundTag();
360 $display->setString(self::TAG_DISPLAY_NAME, $this->customName);
361 }else{
362 $display?->removeTag(self::TAG_DISPLAY_NAME);
363 }
364
365 if(count($this->lore) > 0){
366 $loreTag = new ListTag();
367 foreach($this->lore as $line){
368 $loreTag->push(new StringTag($line));
369 }
370 $display ??= new CompoundTag();
371 $display->setTag(self::TAG_DISPLAY_LORE, $loreTag);
372 }else{
373 $display?->removeTag(self::TAG_DISPLAY_LORE);
374 }
375 $display !== null && $display->count() > 0 ?
376 $tag->setTag(self::TAG_DISPLAY, $display) :
377 $tag->removeTag(self::TAG_DISPLAY);
378
379 if(count($this->enchantments) > 0){
380 $ench = new ListTag();
381 $enchantmentIdMap = EnchantmentIdMap::getInstance();
382 foreach($this->enchantments as $enchantmentInstance){
383 $ench->push(CompoundTag::create()
384 ->setShort(self::TAG_ENCH_ID, $enchantmentIdMap->toId($enchantmentInstance->getType()))
385 ->setShort(self::TAG_ENCH_LVL, $enchantmentInstance->getLevel())
386 );
387 }
388 $tag->setTag(self::TAG_ENCH, $ench);
389 }else{
390 $tag->removeTag(self::TAG_ENCH);
391 }
392
393 $this->blockEntityTag !== null ?
394 $tag->setTag(self::TAG_BLOCK_ENTITY_TAG, clone $this->blockEntityTag) :
395 $tag->removeTag(self::TAG_BLOCK_ENTITY_TAG);
396
397 if(count($this->canPlaceOn) > 0){
398 $canPlaceOn = new ListTag();
399 foreach($this->canPlaceOn as $item){
400 $canPlaceOn->push(new StringTag($item));
401 }
402 $tag->setTag(self::TAG_CAN_PLACE_ON, $canPlaceOn);
403 }else{
404 $tag->removeTag(self::TAG_CAN_PLACE_ON);
405 }
406 if(count($this->canDestroy) > 0){
407 $canDestroy = new ListTag();
408 foreach($this->canDestroy as $item){
409 $canDestroy->push(new StringTag($item));
410 }
411 $tag->setTag(self::TAG_CAN_DESTROY, $canDestroy);
412 }else{
413 $tag->removeTag(self::TAG_CAN_DESTROY);
414 }
415
416 if($this->keepOnDeath){
417 $tag->setByte(self::TAG_KEEP_ON_DEATH, 1);
418 }else{
419 $tag->removeTag(self::TAG_KEEP_ON_DEATH);
420 }

730 public static function nbtDeserialize(CompoundTag $tag) : Item{
731 if($tag->getTag(self::TAG_ID) === null || $tag->getTag(self::TAG_COUNT) === null){
732 return VanillaItems::AIR();
733 }
734
735 $count = Binary::unsignByte($tag->getByte(self::TAG_COUNT));
736 $meta = $tag->getShort(self::TAG_DAMAGE, 0);
737
738 $idTag = $tag->getTag(self::TAG_ID);
739 if($idTag instanceof ShortTag){
740 $item = ItemFactory::getInstance()->get($idTag->getValue(), $meta, $count);
741 }elseif($idTag instanceof StringTag){ //PC item save format
742 try{
743 $item = LegacyStringToItemParser::getInstance()->parse($idTag->getValue() . ":$meta");
744 }catch(LegacyStringToItemParserException $e){
745 //TODO: improve error handling
746 return VanillaItems::AIR();
747 }
748 $item->setCount($count);
749 }else{
750 throw new SavedDataLoadingException("Item CompoundTag ID must be an instance of StringTag or ShortTag, " . get_class($idTag) . " given");
751 }
752
753 $itemNBT = $tag->getCompoundTag(self::TAG_TAG);
754 if($itemNBT !== null){
755 $item->setNamedTag(clone $itemNBT);
756 }
757
758 return $item;
759 }
