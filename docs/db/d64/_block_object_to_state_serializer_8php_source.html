233 public function map(Block $block, \Closure $serializer) : void{
234 if(isset($this->serializers[$block->getTypeId()])){
235 throw new \InvalidArgumentException("Block type ID " . $block->getTypeId() . " already has a serializer registered");
236 }
237 $this->serializers[$block->getTypeId()] = $serializer;
238 }

262 public function serializeBlock(Block $blockState) : BlockStateData{
263 $typeId = $blockState->getTypeId();
264
265 $locatedSerializer = $this->serializers[$typeId] ?? null;
266 if($locatedSerializer === null){
267 throw new BlockStateSerializeException("No serializer registered for " . get_class($blockState) . " with type ID $typeId");
268 }
269
278 $serializer = $locatedSerializer;
279
281 $writer = $serializer($blockState);
282 return $writer->getBlockStateData();
283 }
