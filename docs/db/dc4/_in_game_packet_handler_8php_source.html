174 private function resolveOnOffInputFlags(int $inputFlags, int $startFlag, int $stopFlag) : ?bool{
175 $enabled = ($inputFlags & (1 << $startFlag)) !== 0;
176 $disabled = ($inputFlags & (1 << $stopFlag)) !== 0;
177 if($enabled !== $disabled){
178 return $enabled;
179 }
180 //neither flag was set, or both were set
181 return null;
182 }

184 public function handlePlayerAuthInput(PlayerAuthInputPacket $packet) : bool{
185 $rawPos = $packet->getPosition();
186 $rawYaw = $packet->getYaw();
187 $rawPitch = $packet->getPitch();
188 foreach([$rawPos->x, $rawPos->y, $rawPos->z, $rawYaw, $packet->getHeadYaw(), $rawPitch] as $float){
189 if(is_infinite($float) || is_nan($float)){
190 $this->session->getLogger()->debug("Invalid movement received, contains NAN/INF components");
191 return false;
192 }
193 }
194
195 if($rawYaw !== $this->lastPlayerAuthInputYaw || $rawPitch !== $this->lastPlayerAuthInputPitch){
196 $this->lastPlayerAuthInputYaw = $rawYaw;
197 $this->lastPlayerAuthInputPitch = $rawPitch;
198
199 $yaw = fmod($rawYaw, 360);
200 $pitch = fmod($rawPitch, 360);
201 if($yaw < 0){
202 $yaw += 360;
203 }
204
205 $this->player->setRotation($yaw, $pitch);
206 }
207
208 $hasMoved = $this->lastPlayerAuthInputPosition === null || !$this->lastPlayerAuthInputPosition->equals($rawPos);
209 $newPos = $rawPos->round(4)->subtract(0, 1.62, 0);
210
211 if($this->forceMoveSync && $hasMoved){
212 $curPos = $this->player->getLocation();
213
214 if($newPos->distanceSquared($curPos) > 1){ //Tolerate up to 1 block to avoid problems with client-sided physics when spawning in blocks
215 $this->session->getLogger()->debug("Got outdated pre-teleport movement, received " . $newPos . ", expected " . $curPos);
216 //Still getting movements from before teleport, ignore them
217 return false;
218 }
219
220 // Once we get a movement within a reasonable distance, treat it as a teleport ACK and remove position lock
221 $this->forceMoveSync = false;
222 }
223
224 $inputFlags = $packet->getInputFlags();
225 if($inputFlags !== $this->lastPlayerAuthInputFlags){
226 $this->lastPlayerAuthInputFlags = $inputFlags;
227
228 $sneaking = $this->resolveOnOffInputFlags($inputFlags, PlayerAuthInputFlags::START_SNEAKING, PlayerAuthInputFlags::STOP_SNEAKING);
229 $sprinting = $this->resolveOnOffInputFlags($inputFlags, PlayerAuthInputFlags::START_SPRINTING, PlayerAuthInputFlags::STOP_SPRINTING);
230 $swimming = $this->resolveOnOffInputFlags($inputFlags, PlayerAuthInputFlags::START_SWIMMING, PlayerAuthInputFlags::STOP_SWIMMING);
231 $gliding = $this->resolveOnOffInputFlags($inputFlags, PlayerAuthInputFlags::START_GLIDING, PlayerAuthInputFlags::STOP_GLIDING);
232 $mismatch =
233 ($sneaking !== null && !$this->player->toggleSneak($sneaking)) |
234 ($sprinting !== null && !$this->player->toggleSprint($sprinting)) |
235 ($swimming !== null && !$this->player->toggleSwim($swimming)) |
236 ($gliding !== null && !$this->player->toggleGlide($gliding));
237 if((bool) $mismatch){
238 $this->player->sendData([$this->player]);
239 }
240
241 if($packet->hasFlag(PlayerAuthInputFlags::START_JUMPING)){
242 $this->player->jump();
243 }
244 }
245
246 if(!$this->forceMoveSync && $hasMoved){
247 $this->lastPlayerAuthInputPosition = $rawPos;
248 //TODO: this packet has WAYYYYY more useful information that we're not using
249 $this->player->handleMovement($newPos);
250 }
251
252 $packetHandled = true;
253
254 $blockActions = $packet->getBlockActions();
255 if($blockActions !== null){
256 if(count($blockActions) > 100){
257 throw new PacketHandlingException("Too many block actions in PlayerAuthInputPacket");
258 }
259 foreach($blockActions as $k => $blockAction){
260 $actionHandled = false;
261 if($blockAction instanceof PlayerBlockActionStopBreak){
262 $actionHandled = $this->handlePlayerActionFromData($blockAction->getActionType(), new BlockPosition(0, 0, 0), Facing::DOWN);
263 }elseif($blockAction instanceof PlayerBlockActionWithBlockInfo){
264 $actionHandled = $this->handlePlayerActionFromData($blockAction->getActionType(), $blockAction->getBlockPosition(), $blockAction->getFace());
265 }
266
267 if(!$actionHandled){
268 $packetHandled = false;
269 $this->session->getLogger()->debug("Unhandled player block action at offset $k in PlayerAuthInputPacket");
270 }
271 }
272 }
273
274 $useItemTransaction = $packet->getItemInteractionData();
275 if($useItemTransaction !== null){
276 if(count($useItemTransaction->getTransactionData()->getActions()) > 100){
277 throw new PacketHandlingException("Too many actions in item use transaction");
278 }
279
280 $this->inventoryManager->setCurrentItemStackRequestId($useItemTransaction->getRequestId());
281 $this->inventoryManager->addRawPredictedSlotChanges($useItemTransaction->getTransactionData()->getActions());
282 if(!$this->handleUseItemTransaction($useItemTransaction->getTransactionData())){
283 $packetHandled = false;
284 $this->session->getLogger()->debug("Unhandled transaction in PlayerAuthInputPacket (type " . $useItemTransaction->getTransactionData()->getActionType() . ")");
285 }else{
286 $this->inventoryManager->syncMismatchedPredictedSlotChanges();
287 }
288 $this->inventoryManager->setCurrentItemStackRequestId(null);
289 }
290
291 $itemStackRequest = $packet->getItemStackRequest();
292 if($itemStackRequest !== null){
293 $result = $this->handleSingleItemStackRequest($itemStackRequest);
294 $this->session->sendDataPacket(ItemStackResponsePacket::create([$result]));
295 }
296
297 return $packetHandled;
298 }

325 public function handleInventoryTransaction(InventoryTransactionPacket $packet) : bool{
326 $result = true;
327
328 if(count($packet->trData->getActions()) > 50){
329 throw new PacketHandlingException("Too many actions in inventory transaction");
330 }
331
332 $this->inventoryManager->setCurrentItemStackRequestId($packet->requestId);
333 $this->inventoryManager->addRawPredictedSlotChanges($packet->trData->getActions());
334
335 if($packet->trData instanceof NormalTransactionData){
336 $result = $this->handleNormalTransaction($packet->trData, $packet->requestId);
337 }elseif($packet->trData instanceof MismatchTransactionData){
338 $this->session->getLogger()->debug("Mismatch transaction received");
339 $this->inventoryManager->requestSyncAll();
340 $result = true;
341 }elseif($packet->trData instanceof UseItemTransactionData){
342 $result = $this->handleUseItemTransaction($packet->trData);
343 }elseif($packet->trData instanceof UseItemOnEntityTransactionData){
344 $result = $this->handleUseItemOnEntityTransaction($packet->trData);
345 }elseif($packet->trData instanceof ReleaseItemTransactionData){
346 $result = $this->handleReleaseItemTransaction($packet->trData);
347 }
348
349 $this->inventoryManager->syncMismatchedPredictedSlotChanges();
350 $this->inventoryManager->setCurrentItemStackRequestId(null);
351 return $result;
352 }

354 private function executeInventoryTransaction(InventoryTransaction $transaction, int $requestId) : bool{
355 $this->player->setUsingItem(false);
356
357 $this->inventoryManager->setCurrentItemStackRequestId($requestId);
358 $this->inventoryManager->addTransactionPredictedSlotChanges($transaction);
359 try{
360 $transaction->execute();
361 }catch(TransactionValidationException $e){
362 $this->inventoryManager->requestSyncAll();
363 $logger = $this->session->getLogger();
364 $logger->debug("Invalid inventory transaction $requestId: " . $e->getMessage());
365
366 return false;
367 }catch(TransactionCancelledException){
368 $this->session->getLogger()->debug("Inventory transaction $requestId cancelled by a plugin");
369
370 return false;
371 }finally{
372 $this->inventoryManager->syncMismatchedPredictedSlotChanges();
373 $this->inventoryManager->setCurrentItemStackRequestId(null);
374 }
375
376 return true;
377 }

379 private function handleNormalTransaction(NormalTransactionData $data, int $itemStackRequestId) : bool{
380 //When the ItemStackRequest system is used, this transaction type is only used for dropping items by pressing Q.
381 //I don't know why they don't just use ItemStackRequest for that too, which already supports dropping items by
382 //clicking them outside an open inventory menu, but for now it is what it is.
383 //Fortunately, this means we can be extremely strict about the validation criteria.
384
385 if(count($data->getActions()) > 2){
386 throw new PacketHandlingException("Expected exactly 2 actions for dropping an item");
387 }
388
389 $sourceSlot = null;
390 $clientItemStack = null;
391 $droppedCount = null;
392
393 foreach($data->getActions() as $networkInventoryAction){
394 if($networkInventoryAction->sourceType === NetworkInventoryAction::SOURCE_WORLD && $networkInventoryAction->inventorySlot == NetworkInventoryAction::ACTION_MAGIC_SLOT_DROP_ITEM){
395 $droppedCount = $networkInventoryAction->newItem->getItemStack()->getCount();
396 if($droppedCount <= 0){
397 throw new PacketHandlingException("Expected positive count for dropped item");
398 }
399 }elseif($networkInventoryAction->sourceType === NetworkInventoryAction::SOURCE_CONTAINER && $networkInventoryAction->windowId === ContainerIds::INVENTORY){
400 //mobile players can drop an item from a non-selected hotbar slot
401 $sourceSlot = $networkInventoryAction->inventorySlot;
402 $clientItemStack = $networkInventoryAction->oldItem->getItemStack();
403 }else{
404 throw new PacketHandlingException("Unexpected action type in drop item transaction");
405 }
406 }
407 if($sourceSlot === null || $clientItemStack === null || $droppedCount === null){
408 throw new PacketHandlingException("Missing information in drop item transaction, need source slot, client item stack and dropped count");
409 }
410
411 $inventory = $this->player->getInventory();
412
413 if(!$inventory->slotExists($sourceSlot)){
414 return false; //TODO: size desync??
415 }
416
417 $sourceSlotItem = $inventory->getItem($sourceSlot);
418 $serverItemStack = TypeConverter::getInstance()->coreItemStackToNet($sourceSlotItem);
419 //because the client doesn't tell us the expected itemstack ID, we have to deep-compare our known
420 //itemstack info with the one the client sent. This is costly, but we don't have any other option :(
421 if(!$serverItemStack->equals($clientItemStack)){
422 return false;
423 }
424
425 //this modifies $sourceSlotItem
426 $droppedItem = $sourceSlotItem->pop($droppedCount);
427
428 $builder = new TransactionBuilder();
429 $builder->getInventory($inventory)->setItem($sourceSlot, $sourceSlotItem);
430 $builder->addAction(new DropItemAction($droppedItem));
431
432 $transaction = new InventoryTransaction($this->player, $builder->generateActions());
433 return $this->executeInventoryTransaction($transaction, $itemStackRequestId);
434 }

436 private function handleUseItemTransaction(UseItemTransactionData $data) : bool{
437 $this->player->selectHotbarSlot($data->getHotbarSlot());
438
439 switch($data->getActionType()){
440 case UseItemTransactionData::ACTION_CLICK_BLOCK:
441 //TODO: start hack for client spam bug
442 $clickPos = $data->getClickPosition();
443 $spamBug = ($this->lastRightClickData !== null &&
444 microtime(true) - $this->lastRightClickTime < 0.1 && //100ms
445 $this->lastRightClickData->getPlayerPosition()->distanceSquared($data->getPlayerPosition()) < 0.00001 &&
446 $this->lastRightClickData->getBlockPosition()->equals($data->getBlockPosition()) &&
447 $this->lastRightClickData->getClickPosition()->distanceSquared($clickPos) < 0.00001 //signature spam bug has 0 distance, but allow some error
448 );
449 //get rid of continued spam if the player clicks and holds right-click
450 $this->lastRightClickData = $data;
451 $this->lastRightClickTime = microtime(true);
452 if($spamBug){
453 return true;
454 }
455 //TODO: end hack for client spam bug
456
457 self::validateFacing($data->getFace());
458
459 $blockPos = $data->getBlockPosition();
460 $vBlockPos = new Vector3($blockPos->getX(), $blockPos->getY(), $blockPos->getZ());
461 if(!$this->player->interactBlock($vBlockPos, $data->getFace(), $clickPos)){
462 $this->onFailedBlockAction($vBlockPos, $data->getFace());
463 }
464 return true;
465 case UseItemTransactionData::ACTION_BREAK_BLOCK:
466 $blockPos = $data->getBlockPosition();
467 $vBlockPos = new Vector3($blockPos->getX(), $blockPos->getY(), $blockPos->getZ());
468 if(!$this->player->breakBlock($vBlockPos)){
469 $this->onFailedBlockAction($vBlockPos, null);
470 }
471 return true;
472 case UseItemTransactionData::ACTION_CLICK_AIR:
473 if($this->player->isUsingItem()){
474 if(!$this->player->consumeHeldItem()){
475 $hungerAttr = $this->player->getAttributeMap()->get(Attribute::HUNGER) ?? throw new AssumptionFailedError();
476 $hungerAttr->markSynchronized(false);
477 }
478 return true;
479 }
480 $this->player->useHeldItem();
481 return true;
482 }
483
484 return false;
485 }

547 private function handleSingleItemStackRequest(ItemStackRequest $request) : ItemStackResponse{
548 if(count($request->getActions()) > 20){
549 //TODO: we can probably lower this limit, but this will do for now
550 throw new PacketHandlingException("Too many actions in ItemStackRequest");
551 }
552 $executor = new ItemStackRequestExecutor($this->player, $this->inventoryManager, $request);
553 try{
554 $transaction = $executor->generateInventoryTransaction();
555 $result = $this->executeInventoryTransaction($transaction, $request->getRequestId());
556 }catch(ItemStackRequestProcessException $e){
557 $result = false;
558 $this->session->getLogger()->debug("ItemStackRequest #" . $request->getRequestId() . " failed: " . $e->getMessage());
559 $this->session->getLogger()->debug(implode("\n", Utils::printableExceptionInfo($e)));
560 $this->inventoryManager->requestSyncAll();
561 }
562
563 if(!$result){
564 return new ItemStackResponse(ItemStackResponse::RESULT_ERROR, $request->getRequestId());
565 }
566 return $executor->buildItemStackResponse();
567 }

602 public function handleInteract(InteractPacket $packet) : bool{
603 if($packet->action === InteractPacket::ACTION_MOUSEOVER){
604 //TODO HACK: silence useless spam (MCPE 1.8)
605 //due to some messy Mojang hacks, it sends this when changing the held item now, which causes us to think
606 //the inventory was closed when it wasn't.
607 //this is also sent whenever entity metadata updates, which can get really spammy.
608 //TODO: implement handling for this where it matters
609 return true;
610 }

818 private function checkBookText(string $string, string $fieldName, int $softLimit, int $hardLimit, bool &$cancel) : string{
819 if(strlen($string) > $hardLimit){
820 throw new PacketHandlingException(sprintf("Book %s must be at most %d bytes, but have %d bytes", $fieldName, $hardLimit, strlen($string)));
821 }
822
823 $result = TextFormat::clean($string, false);

826 $cancel = true;

828 }

833 public function handleBookEdit(BookEditPacket $packet) : bool{
834 //TODO: break this up into book API things
835 $oldBook = $this->player->getInventory()->getItem($packet->inventorySlot);
836 if(!($oldBook instanceof WritableBook)){
837 return false;
838 }
839
840 $newBook = clone $oldBook;
841 $modifiedPages = [];
842 $cancel = false;
843 switch($packet->type){
844 case BookEditPacket::TYPE_REPLACE_PAGE:
845 $text = self::checkBookText($packet->text, "page text", 256, WritableBookPage::PAGE_LENGTH_HARD_LIMIT_BYTES, $cancel);
846 $newBook->setPageText($packet->pageNumber, $text);
847 $modifiedPages[] = $packet->pageNumber;
848 break;
849 case BookEditPacket::TYPE_ADD_PAGE:
850 if(!$newBook->pageExists($packet->pageNumber)){
851 //this may only come before a page which already exists
852 //TODO: the client can send insert-before actions on trailing client-side pages which cause odd behaviour on the server
853 return false;
854 }
855 $text = self::checkBookText($packet->text, "page text", 256, WritableBookPage::PAGE_LENGTH_HARD_LIMIT_BYTES, $cancel);
856 $newBook->insertPage($packet->pageNumber, $text);
857 $modifiedPages[] = $packet->pageNumber;
858 break;
859 case BookEditPacket::TYPE_DELETE_PAGE:
860 if(!$newBook->pageExists($packet->pageNumber)){
861 return false;
862 }
863 $newBook->deletePage($packet->pageNumber);
864 $modifiedPages[] = $packet->pageNumber;
865 break;
866 case BookEditPacket::TYPE_SWAP_PAGES:
867 if(!$newBook->pageExists($packet->pageNumber) || !$newBook->pageExists($packet->secondaryPageNumber)){
868 //the client will create pages on its own without telling us until it tries to switch them
869 $newBook->addPage(max($packet->pageNumber, $packet->secondaryPageNumber));
870 }
871 $newBook->swapPages($packet->pageNumber, $packet->secondaryPageNumber);
872 $modifiedPages = [$packet->pageNumber, $packet->secondaryPageNumber];
873 break;
874 case BookEditPacket::TYPE_SIGN_BOOK:
875 $title = self::checkBookText($packet->title, "title", 16, Limits::INT16_MAX, $cancel);
876 //this one doesn't have a limit in vanilla, so we have to improvise
877 $author = self::checkBookText($packet->author, "author", 256, Limits::INT16_MAX, $cancel);
878
879 $newBook = VanillaItems::WRITTEN_BOOK()
880 ->setPages($oldBook->getPages())
881 ->setAuthor($author)
882 ->setTitle($title)
883 ->setGeneration(WrittenBook::GENERATION_ORIGINAL);
884 break;
885 default:
886 return false;
887 }
888
889 //for redundancy, in case of protocol changes, we don't want to pass these directly
890 $action = match($packet->type){
891 BookEditPacket::TYPE_REPLACE_PAGE => PlayerEditBookEvent::ACTION_REPLACE_PAGE,
892 BookEditPacket::TYPE_ADD_PAGE => PlayerEditBookEvent::ACTION_ADD_PAGE,
893 BookEditPacket::TYPE_DELETE_PAGE => PlayerEditBookEvent::ACTION_DELETE_PAGE,
894 BookEditPacket::TYPE_SWAP_PAGES => PlayerEditBookEvent::ACTION_SWAP_PAGES,
895 BookEditPacket::TYPE_SIGN_BOOK => PlayerEditBookEvent::ACTION_SIGN_BOOK,
896 default => throw new AssumptionFailedError("We already filtered unknown types in the switch above")
897 };
898
899 /*
900 * Plugins may have created books with more than 50 pages; we allow plugins to do this, but not players.
901 * Don't allow the page count to grow past 50, but allow deleting, swapping or altering text of existing pages.
902 */
903 $oldPageCount = count($oldBook->getPages());
904 $newPageCount = count($newBook->getPages());
905 if(($newPageCount > $oldPageCount && $newPageCount > 50)){
906 $this->session->getLogger()->debug("Cancelled book edit due to adding too many pages (new page count would be $newPageCount)");
907 $cancel = true;
908 }
909
910 $event = new PlayerEditBookEvent($this->player, $oldBook, $newBook, $action, $modifiedPages);
911 if($cancel){
912 $event->cancel();
913 }
914
915 $event->call();
916 if($event->isCancelled()){
917 return true;
918 }
919
920 $this->player->getInventory()->setItem($packet->inventorySlot, $event->getNewBook());
921
922 return true;
923 }
