176 private function resolveOnOffInputFlags(int $inputFlags, int $startFlag, int $stopFlag) : ?bool{
177 $enabled = ($inputFlags & (1 << $startFlag)) !== 0;
178 $disabled = ($inputFlags & (1 << $stopFlag)) !== 0;
179 if($enabled !== $disabled){
180 return $enabled;
181 }
182 //neither flag was set, or both were set
183 return null;

186 public function handlePlayerAuthInput(PlayerAuthInputPacket $packet) : bool{
187 $rawPos = $packet->getPosition();
188 $rawYaw = $packet->getYaw();
189 $rawPitch = $packet->getPitch();
190 foreach([$rawPos->x, $rawPos->y, $rawPos->z, $rawYaw, $packet->getHeadYaw(), $rawPitch] as $float){
191 if(is_infinite($float) || is_nan($float)){
192 $this->session->getLogger()->debug("Invalid movement received, contains NAN/INF components");
193 return false;
194 }
195 }
196
197 if($rawYaw !== $this->lastPlayerAuthInputYaw || $rawPitch !== $this->lastPlayerAuthInputPitch){
198 $this->lastPlayerAuthInputYaw = $rawYaw;
199 $this->lastPlayerAuthInputPitch = $rawPitch;
200
201 $yaw = fmod($rawYaw, 360);
202 $pitch = fmod($rawPitch, 360);
203 if($yaw < 0){
204 $yaw += 360;
205 }
206
207 $this->player->setRotation($yaw, $pitch);
208 }
209
210 $hasMoved = $this->lastPlayerAuthInputPosition === null || !$this->lastPlayerAuthInputPosition->equals($rawPos);
211 $newPos = $rawPos->round(4)->subtract(0, 1.62, 0);
212
213 if($this->forceMoveSync && $hasMoved){
214 $curPos = $this->player->getLocation();
215
216 if($newPos->distanceSquared($curPos) > 1){ //Tolerate up to 1 block to avoid problems with client-sided physics when spawning in blocks
217 $this->session->getLogger()->debug("Got outdated pre-teleport movement, received " . $newPos . ", expected " . $curPos);
218 //Still getting movements from before teleport, ignore them
219 return false;
220 }
221
222 // Once we get a movement within a reasonable distance, treat it as a teleport ACK and remove position lock
223 $this->forceMoveSync = false;
224 }
225
226 $inputFlags = $packet->getInputFlags();
227 if($inputFlags !== $this->lastPlayerAuthInputFlags){
228 $this->lastPlayerAuthInputFlags = $inputFlags;
229
230 $sneaking = $this->resolveOnOffInputFlags($inputFlags, PlayerAuthInputFlags::START_SNEAKING, PlayerAuthInputFlags::STOP_SNEAKING);
231 $sprinting = $this->resolveOnOffInputFlags($inputFlags, PlayerAuthInputFlags::START_SPRINTING, PlayerAuthInputFlags::STOP_SPRINTING);
232 $swimming = $this->resolveOnOffInputFlags($inputFlags, PlayerAuthInputFlags::START_SWIMMING, PlayerAuthInputFlags::STOP_SWIMMING);
233 $gliding = $this->resolveOnOffInputFlags($inputFlags, PlayerAuthInputFlags::START_GLIDING, PlayerAuthInputFlags::STOP_GLIDING);
234 $mismatch =
235 ($sneaking !== null && !$this->player->toggleSneak($sneaking)) |
236 ($sprinting !== null && !$this->player->toggleSprint($sprinting)) |
237 ($swimming !== null && !$this->player->toggleSwim($swimming)) |
238 ($gliding !== null && !$this->player->toggleGlide($gliding));
239 if((bool) $mismatch){
240 $this->player->sendData([$this->player]);
241 }
242
243 if($packet->hasFlag(PlayerAuthInputFlags::START_JUMPING)){
244 $this->player->jump();
245 }
246 }
247
248 if(!$this->forceMoveSync && $hasMoved){
249 $this->lastPlayerAuthInputPosition = $rawPos;
250 //TODO: this packet has WAYYYYY more useful information that we're not using
251 $this->player->handleMovement($newPos);
252 }
253
254 $packetHandled = true;
255
256 $useItemTransaction = $packet->getItemInteractionData();
257 if($useItemTransaction !== null){
258 if(!$this->handleUseItemTransaction($useItemTransaction->getTransactionData())){
259 $packetHandled = false;
260 $this->session->getLogger()->debug("Unhandled transaction in PlayerAuthInputPacket (type " . $useItemTransaction->getTransactionData()->getActionType() . ")");
261 }else{
262 $this->inventoryManager->syncMismatchedPredictedSlotChanges();
263 }
264 }
265
266 $blockActions = $packet->getBlockActions();
267 if($blockActions !== null){
268 foreach($blockActions as $k => $blockAction){
269 $actionHandled = false;
270 if($blockAction instanceof PlayerBlockActionStopBreak){
271 $actionHandled = $this->handlePlayerActionFromData($blockAction->getActionType(), new BlockPosition(0, 0, 0), Facing::DOWN);
272 }elseif($blockAction instanceof PlayerBlockActionWithBlockInfo){
273 $actionHandled = $this->handlePlayerActionFromData($blockAction->getActionType(), $blockAction->getBlockPosition(), $blockAction->getFace());
274 }
275
276 if(!$actionHandled){
277 $packetHandled = false;
278 $this->session->getLogger()->debug("Unhandled player block action at offset $k in PlayerAuthInputPacket");
279 }
280 }
281 }
282
283 return $packetHandled;
284 }

311 public function handleInventoryTransaction(InventoryTransactionPacket $packet) : bool{
312 $result = true;
313
314 if($packet->trData instanceof NormalTransactionData){
315 $result = $this->handleNormalTransaction($packet->trData);
316 }elseif($packet->trData instanceof MismatchTransactionData){
317 $this->session->getLogger()->debug("Mismatch transaction received");
318 $this->inventoryManager->syncAll();
319 $result = true;
320 }elseif($packet->trData instanceof UseItemTransactionData){
321 $result = $this->handleUseItemTransaction($packet->trData);
322 }elseif($packet->trData instanceof UseItemOnEntityTransactionData){
323 $result = $this->handleUseItemOnEntityTransaction($packet->trData);
324 }elseif($packet->trData instanceof ReleaseItemTransactionData){
325 $result = $this->handleReleaseItemTransaction($packet->trData);
326 }
327
328 if(!$result){
329 $this->inventoryManager->syncAll();
330 }else{
331 $this->inventoryManager->syncMismatchedPredictedSlotChanges();
332 }
333 return $result;
334 }

336 private function handleNormalTransaction(NormalTransactionData $data) : bool{
338 $actions = [];
339
340 $isCraftingPart = false;
341 $converter = TypeConverter::getInstance();
342 foreach($data->getActions() as $networkInventoryAction){
343 if(
344 (
345 $networkInventoryAction->sourceType === NetworkInventoryAction::SOURCE_TODO && (
346 $networkInventoryAction->windowId === NetworkInventoryAction::SOURCE_TYPE_CRAFTING_RESULT ||
347 $networkInventoryAction->windowId === NetworkInventoryAction::SOURCE_TYPE_CRAFTING_USE_INGREDIENT
348 )
349 ) || (
350 $this->craftingTransaction !== null &&
351 !$networkInventoryAction->oldItem->getItemStack()->equals($networkInventoryAction->newItem->getItemStack()) &&
352 $networkInventoryAction->sourceType === NetworkInventoryAction::SOURCE_CONTAINER &&
353 $networkInventoryAction->windowId === ContainerIds::UI &&
354 $networkInventoryAction->inventorySlot === UIInventorySlotOffset::CREATED_ITEM_OUTPUT
355 )
356 ){
357 $isCraftingPart = true;
358 }
359
360 try{
361 $action = $converter->createInventoryAction($networkInventoryAction, $this->player, $this->inventoryManager);
362 if($action !== null){
363 $actions[] = $action;
364 }
365 }catch(TypeConversionException $e){
366 $this->session->getLogger()->debug("Error unpacking inventory action: " . $e->getMessage());
367 return false;
368 }
369 }
370
371 if($isCraftingPart){
372 if($this->craftingTransaction === null){
373 //TODO: this might not be crafting if there is a special inventory open (anvil, enchanting, loom etc)
374 $this->craftingTransaction = new CraftingTransaction($this->player, $this->player->getServer()->getCraftingManager(), $actions);
375 }else{
376 foreach($actions as $action){
377 $this->craftingTransaction->addAction($action);
378 }
379 }
380
381 try{
382 $this->craftingTransaction->validate();
383 }catch(TransactionValidationException $e){
384 //transaction is incomplete - crafting transaction comes in lots of little bits, so we have to collect
385 //all of the parts before we can execute it
386 return true;
387 }
388 $this->player->setUsingItem(false);
389 try{
390 $this->inventoryManager->onTransactionStart($this->craftingTransaction);
391 $this->craftingTransaction->execute();
392 }catch(TransactionException $e){
393 $this->session->getLogger()->debug("Failed to execute crafting transaction: " . $e->getMessage());
394
395 //TODO: only sync slots that the client tried to change
396 foreach($this->craftingTransaction->getInventories() as $inventory){
397 $this->inventoryManager->syncContents($inventory);
398 }
399 return false;
400 }finally{
401 $this->craftingTransaction = null;
402 }
403 }else{
404 //normal transaction fallthru
405 if($this->craftingTransaction !== null){
406 $this->session->getLogger()->debug("Got unexpected normal inventory action with incomplete crafting transaction, refusing to execute crafting");
407 $this->craftingTransaction = null;
408 return false;
409 }
410
411 if(count($actions) === 0){
412 //TODO: 1.13+ often sends transactions with nothing but useless crap in them, no need for the debug noise
413 return true;
414 }
415
416 $this->player->setUsingItem(false);
417 $transaction = new InventoryTransaction($this->player, $actions);
418 $this->inventoryManager->onTransactionStart($transaction);
419 try{
420 $transaction->execute();
421 }catch(TransactionException $e){
422 $logger = $this->session->getLogger();
423 $logger->debug("Failed to execute inventory transaction: " . $e->getMessage());
424 $logger->debug("Actions: " . json_encode($data->getActions()));
425
426 foreach($transaction->getInventories() as $inventory){
427 $this->inventoryManager->syncContents($inventory);
428 }
429
430 return false;
431 }
432 }
433
434 return true;
435 }

437 private function handleUseItemTransaction(UseItemTransactionData $data) : bool{
438 $this->player->selectHotbarSlot($data->getHotbarSlot());
439 $this->inventoryManager->addPredictedSlotChanges($data->getActions());
440
441 switch($data->getActionType()){
442 case UseItemTransactionData::ACTION_CLICK_BLOCK:
443 //TODO: start hack for client spam bug
444 $clickPos = $data->getClickPosition();
445 $spamBug = ($this->lastRightClickData !== null &&
446 microtime(true) - $this->lastRightClickTime < 0.1 && //100ms
447 $this->lastRightClickData->getPlayerPosition()->distanceSquared($data->getPlayerPosition()) < 0.00001 &&
448 $this->lastRightClickData->getBlockPosition()->equals($data->getBlockPosition()) &&
449 $this->lastRightClickData->getClickPosition()->distanceSquared($clickPos) < 0.00001 //signature spam bug has 0 distance, but allow some error
450 );
451 //get rid of continued spam if the player clicks and holds right-click
452 $this->lastRightClickData = $data;
453 $this->lastRightClickTime = microtime(true);
454 if($spamBug){
455 return true;
456 }
457 //TODO: end hack for client spam bug
458
459 self::validateFacing($data->getFace());
460
461 $blockPos = $data->getBlockPosition();
462 $vBlockPos = new Vector3($blockPos->getX(), $blockPos->getY(), $blockPos->getZ());
463 if(!$this->player->interactBlock($vBlockPos, $data->getFace(), $clickPos)){
464 $this->onFailedBlockAction($vBlockPos, $data->getFace());
465 }
466 return true;
467 case UseItemTransactionData::ACTION_BREAK_BLOCK:
468 $blockPos = $data->getBlockPosition();
469 $vBlockPos = new Vector3($blockPos->getX(), $blockPos->getY(), $blockPos->getZ());
470 if(!$this->player->breakBlock($vBlockPos)){
471 $this->onFailedBlockAction($vBlockPos, null);
472 }
473 return true;
474 case UseItemTransactionData::ACTION_CLICK_AIR:
475 if($this->player->isUsingItem()){
476 if(!$this->player->consumeHeldItem()){
477 $hungerAttr = $this->player->getAttributeMap()->get(Attribute::HUNGER) ?? throw new AssumptionFailedError();
478 $hungerAttr->markSynchronized(false);
479 }
480 return true;
481 }
482 $this->player->useHeldItem();
483 return true;
484 }
485
486 return false;
487 }

606 private function handlePlayerActionFromData(int $action, BlockPosition $blockPosition, int $face) : bool{
607 $pos = new Vector3($blockPosition->getX(), $blockPosition->getY(), $blockPosition->getZ());
608
609 switch($action){
610 case PlayerAction::START_BREAK:
611 self::validateFacing($face);
612 if(!$this->player->attackBlock($pos, $face)){
613 $this->onFailedBlockAction($pos, $face);
614 }
615
616 break;
617
618 case PlayerAction::ABORT_BREAK:
619 case PlayerAction::STOP_BREAK:
620 $this->player->stopBreakBlock($pos);
621 break;
622 case PlayerAction::START_SLEEPING:
623 //unused
624 break;
625 case PlayerAction::STOP_SLEEPING:
626 $this->player->stopSleep();
627 break;
628 case PlayerAction::CRACK_BREAK:
629 self::validateFacing($face);
630 $this->player->continueBreakBlock($pos, $face);
631 break;
632 case PlayerAction::INTERACT_BLOCK: //TODO: ignored (for now)
633 break;
634 case PlayerAction::CREATIVE_PLAYER_DESTROY_BLOCK:
635 //TODO: do we need to handle this?
636 break;

785 private function checkBookText(string $string, string $fieldName, int $softLimit, int $hardLimit, bool &$cancel) : string{
786 if(strlen($string) > $hardLimit){
787 throw new PacketHandlingException(sprintf("Book %s must be at most %d bytes, but have %d bytes", $fieldName, $hardLimit, strlen($string)));
788 }
789
790 $result = TextFormat::clean($string, false);

792 if(strlen($result) > $softLimit * 4 || mb_strlen($result, 'UTF-8') > $softLimit){
793 $cancel = true;

795 }
796
797 return $result;

800 public function handleBookEdit(BookEditPacket $packet) : bool{
801 //TODO: break this up into book API things
802 $oldBook = $this->player->getInventory()->getItem($packet->inventorySlot);
803 if(!($oldBook instanceof WritableBook)){
804 return false;
805 }
806
807 $newBook = clone $oldBook;
808 $modifiedPages = [];
809 $cancel = false;
810 switch($packet->type){
811 case BookEditPacket::TYPE_REPLACE_PAGE:
812 $text = self::checkBookText($packet->text, "page text", 256, WritableBookPage::PAGE_LENGTH_HARD_LIMIT_BYTES, $cancel);
813 $newBook->setPageText($packet->pageNumber, $text);
814 $modifiedPages[] = $packet->pageNumber;
815 break;
816 case BookEditPacket::TYPE_ADD_PAGE:
817 if(!$newBook->pageExists($packet->pageNumber)){
818 //this may only come before a page which already exists
819 //TODO: the client can send insert-before actions on trailing client-side pages which cause odd behaviour on the server
820 return false;
821 }
822 $text = self::checkBookText($packet->text, "page text", 256, WritableBookPage::PAGE_LENGTH_HARD_LIMIT_BYTES, $cancel);
823 $newBook->insertPage($packet->pageNumber, $text);
824 $modifiedPages[] = $packet->pageNumber;
825 break;
826 case BookEditPacket::TYPE_DELETE_PAGE:
827 if(!$newBook->pageExists($packet->pageNumber)){
828 return false;
829 }
830 $newBook->deletePage($packet->pageNumber);
831 $modifiedPages[] = $packet->pageNumber;
832 break;
833 case BookEditPacket::TYPE_SWAP_PAGES:
834 if(!$newBook->pageExists($packet->pageNumber) || !$newBook->pageExists($packet->secondaryPageNumber)){
835 //the client will create pages on its own without telling us until it tries to switch them
836 $newBook->addPage(max($packet->pageNumber, $packet->secondaryPageNumber));
837 }
838 $newBook->swapPages($packet->pageNumber, $packet->secondaryPageNumber);
839 $modifiedPages = [$packet->pageNumber, $packet->secondaryPageNumber];
840 break;
841 case BookEditPacket::TYPE_SIGN_BOOK:
842 $title = self::checkBookText($packet->title, "title", 16, Limits::INT16_MAX, $cancel);
843 //this one doesn't have a limit in vanilla, so we have to improvise
844 $author = self::checkBookText($packet->author, "author", 256, Limits::INT16_MAX, $cancel);
845
846 $newBook = VanillaItems::WRITTEN_BOOK()
847 ->setPages($oldBook->getPages())
848 ->setAuthor($author)
849 ->setTitle($title)
850 ->setGeneration(WrittenBook::GENERATION_ORIGINAL);
851 break;
852 default:
853 return false;
854 }
855
856 //for redundancy, in case of protocol changes, we don't want to pass these directly
857 $action = match($packet->type){
858 BookEditPacket::TYPE_REPLACE_PAGE => PlayerEditBookEvent::ACTION_REPLACE_PAGE,
859 BookEditPacket::TYPE_ADD_PAGE => PlayerEditBookEvent::ACTION_ADD_PAGE,
860 BookEditPacket::TYPE_DELETE_PAGE => PlayerEditBookEvent::ACTION_DELETE_PAGE,
861 BookEditPacket::TYPE_SWAP_PAGES => PlayerEditBookEvent::ACTION_SWAP_PAGES,
862 BookEditPacket::TYPE_SIGN_BOOK => PlayerEditBookEvent::ACTION_SIGN_BOOK,
863 default => throw new AssumptionFailedError("We already filtered unknown types in the switch above")
864 };
865
866 /*
867 * Plugins may have created books with more than 50 pages; we allow plugins to do this, but not players.
868 * Don't allow the page count to grow past 50, but allow deleting, swapping or altering text of existing pages.
869 */
870 $oldPageCount = count($oldBook->getPages());
871 $newPageCount = count($newBook->getPages());
872 if(($newPageCount > $oldPageCount && $newPageCount > 50)){
873 $this->session->getLogger()->debug("Cancelled book edit due to adding too many pages (new page count would be $newPageCount)");
874 $cancel = true;
875 }
876
877 $event = new PlayerEditBookEvent($this->player, $oldBook, $newBook, $action, $modifiedPages);
878 if($cancel){
879 $event->cancel();
880 }
881
882 $event->call();
883 if($event->isCancelled()){
884 return true;
885 }
886
887 $this->player->getInventory()->setItem($packet->inventorySlot, $event->getNewBook());
888
889 return true;
890 }

902 private static function stupid_json_decode(string $json, bool $assoc = false){
903 if(preg_match('/^\[(.+)\]$/s', $json, $matches) > 0){
904 $raw = $matches[1];
905 $lastComma = -1;
906 $newParts = [];
907 $inQuotes = false;
908 for($i = 0, $len = strlen($raw); $i <= $len; ++$i){
909 if($i === $len || ($raw[$i] === "," && !$inQuotes)){
910 $part = substr($raw, $lastComma + 1, $i - ($lastComma + 1));
911 if(trim($part) === ""){ //regular parts will have quotes or something else that makes them non-empty
912 $part = '""';
913 }
914 $newParts[] = $part;
915 $lastComma = $i;
916 }elseif($raw[$i] === '"'){
917 if(!$inQuotes){
918 $inQuotes = true;
919 }else{
920 $backslashes = 0;
921 for(; $backslashes < $i && $raw[$i - $backslashes - 1] === "\\"; ++$backslashes){}
922 if(($backslashes % 2) === 0){ //unescaped quote
923 $inQuotes = false;
924 }
925 }
926 }
927 }
928
929 $fixed = "[" . implode(",", $newParts) . "]";
930 try{
931 return json_decode($fixed, $assoc, self::MAX_FORM_RESPONSE_DEPTH, JSON_THROW_ON_ERROR);
932 }catch(\JsonException $e){
933 throw PacketHandlingException::wrap($e, "Failed to fix JSON (original: $json, modified: $fixed)");
934 }
935 }
936
937 try{
938 return json_decode($json, $assoc, self::MAX_FORM_RESPONSE_DEPTH, JSON_THROW_ON_ERROR);
939 }catch(\JsonException $e){
940 throw PacketHandlingException::wrap($e);
941 }
942 }
