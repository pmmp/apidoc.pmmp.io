59 final protected function handleRakNetConnectionPacket(string $packet) : void{
60 $id = ord($packet[0]);
61 if($id === MessageIdentifiers::ID_CONNECTION_REQUEST){
62 $dataPacket = new ConnectionRequest();
63 $dataPacket->decode(new PacketSerializer($packet));
64 $this->queueConnectedPacket(ConnectionRequestAccepted::create(
65 $this->address,
66 [],
67 $dataPacket->sendPingTime,
68 $this->getRakNetTimeMS()
69 ), PacketReliability::UNRELIABLE, 0, true);
70 }elseif($id === MessageIdentifiers::ID_NEW_INCOMING_CONNECTION){
71 $dataPacket = new NewIncomingConnection();
72 $dataPacket->decode(new PacketSerializer($packet));
73
74 if($dataPacket->address->getPort() === $this->server->getPort() or !$this->server->portChecking){
75 $this->state = self::STATE_CONNECTED; //FINALLY!
76 $this->server->openSession($this);
77
78 //$this->handlePong($dataPacket->sendPingTime, $dataPacket->sendPongTime); //can't use this due to system-address count issues in MCPE >.<
79 $this->sendPing();
80 }
81 }
