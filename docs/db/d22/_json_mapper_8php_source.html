168 // Store the property inspection results so we don't have to do it
169 // again for subsequent objects of the same type

181 'JSON property "' . $key . '" does not exist'
182 . ' in object of type ' . $strClassName
183 );

185 $undefinedPropertyKey = call_user_func(
186 $this->undefinedPropertyHandler,

188 );

194 } else {

196 'info',
197 'Property {property} does not exist in {class}',
198 array('property' => $key, 'class' => $strClassName)
199 );
200 }
201

203 continue;
204 }
205 }
206
207 if ($accessor === null) {
208 if ($this->bExceptionOnUndefinedProperty) {
209 throw new JsonMapper_Exception(
210 'JSON property "' . $key . '" has no public setter method'
211 . ' in object of type ' . $strClassName
212 );
213 }

215 'info',
216 'Property {property} has no public setter method in {class}',

218 );

220 }
221

225 continue;

241 continue;

244 continue;
245 } else if ($this->isSimpleType($type)

265 );

285 }

287 $array = array();
288 $subtype = $type;
289 } else {
290 if (is_a($type, 'ArrayAccess', true)) {
291 $array = $this->createInstance($type, false, $jvalue);
292 }
293 }
294
295 if ($array !== null) {
296 if (!is_array($jvalue) && $this->isFlatType(gettype($jvalue))) {
297 throw new JsonMapper_Exception(
298 'JSON property "' . $key . '" must be an array, '
299 . gettype($jvalue) . ' given'
300 );
301 }
302

304 $subtype = $this->getFullNamespace($cleanSubtype, $strNs);
305 $child = $this->mapArray($jvalue, $array, $subtype, $key);
306 } else if ($this->isFlatType(gettype($jvalue))) {
307 //use constructor parameter if we have a class
308 // but only a flat type (i.e. string, int)

312 . gettype($jvalue) . ' given'
313 );

315 $child = $this->createInstance($type, true, $jvalue);
316 } else {
317 $child = $this->createInstance($type, false, $jvalue);
318 $this->map($jvalue, $child);
319 }
320 $this->setProperty($object, $accessor, $child);
321 }

431 {
432 $originalClass = $class;
433 foreach ($json as $key => $jvalue) {
434 $class = $this->getMappedType($originalClass, $jvalue);
435 if ($class === null) {
436 $array[$key] = $jvalue;
437 } else if ($this->isArrayOfType($class)) {
438 $array[$key] = $this->mapArray(
439 $jvalue,
440 array(),
441 substr($class, 0, -2)
442 );
443 } else if ($this->isFlatType(gettype($jvalue))) {
444 //use constructor parameter if we have a class
445 // but only a flat type (i.e. string, int)
446 if ($this->isSimpleType($class)) {
447 settype($jvalue, $class);
448 $array[$key] = $jvalue;
449 } else {
450 $array[$key] = $this->createInstance(
451 $class, true, $jvalue
452 );
453 }
454 } else if ($this->isFlatType($class)) {
455 throw new JsonMapper_Exception(

488 protected function inspectProperty(ReflectionClass $rc, $name)
489 {
490 //try setter method first
491 $setter = 'set' . $this->getCamelCaseName($name);
492
493 if ($rc->hasMethod($setter)) {
494 $rmeth = $rc->getMethod($setter);
495 if ($rmeth->isPublic() || $this->bIgnoreVisibility) {
496 $isNullable = false;
497 $rparams = $rmeth->getParameters();
498 if (count($rparams) > 0) {
499 $isNullable = $rparams[0]->allowsNull();
500 $ptype = $rparams[0]->getType();
501 if ($ptype !== null) {
502 $typeName = $this->stringifyReflectionType($ptype);
503 //allow overriding an "array" type hint
504 // with a more specific class in the docblock
505 if ($typeName !== 'array') {
506 return array(
507 true, $rmeth,
508 $typeName,
509 $isNullable,
510 );
511 }
512 }
513 }
514
515 $docblock = $rmeth->getDocComment();
516 $annotations = static::parseAnnotations($docblock);
517
518 if (!isset($annotations['param'][0])) {
519 return array(true, $rmeth, null, $isNullable);
520 }
521 list($type) = explode(' ', trim($annotations['param'][0]));
522 return array(true, $rmeth, $type, $this->isNullable($type));
523 }
524 }
525
526 //now try to set the property directly
527 //we have to look it up in the class hierarchy
528 $class = $rc;
529 $rprop = null;
530 do {
531 if ($class->hasProperty($name)) {
532 $rprop = $class->getProperty($name);
533 }
534 } while ($rprop === null && $class = $class->getParentClass());
535
536 if ($rprop === null) {
537 //case-insensitive property matching
538 foreach ($rc->getProperties() as $p) {
539 if ((strcasecmp($p->name, $name) === 0)) {
540 $rprop = $p;
541 break;
542 }
543 }
544 }
545 if ($rprop !== null) {
546 if ($rprop->isPublic() || $this->bIgnoreVisibility) {
547 $docblock = $rprop->getDocComment();
548 $annotations = static::parseAnnotations($docblock);
549
550 if (!isset($annotations['var'][0])) {
551 // If there is no annotations (higher priority) inspect
552 // if there's a scalar type being defined
553 if (PHP_VERSION_ID >= 70400 && $rprop->hasType()) {
554 $rPropType = $rprop->getType();
555 $propTypeName = $this->stringifyReflectionType($rPropType);
556 if ($this->isSimpleType($propTypeName)) {
557 return array(
558 true,
559 $rprop,
560 $propTypeName,
561 $rPropType->allowsNull()
562 );
563 }
564
565 return array(
566 true,
567 $rprop,
568 '\\' . ltrim($propTypeName, '\\'),
569 $rPropType->allowsNull()
570 );
571 }
572
573 return array(true, $rprop, null, false);
574 }
575
576 //support "@var type description"
577 list($type) = explode(' ', $annotations['var'][0]);
578
579 return array(true, $rprop, $type, $this->isNullable($type));
580 } else {
581 //no setter, private property
582 return array(true, null, null, false);
583 }
584 }
585
586 //no setter, no property
587 return array(false, null, null, false);
588 }

874 protected static function parseAnnotations($docblock)
875 {
876 $annotations = array();
877 // Strip away the docblock header and footer
878 // to ease parsing of one line annotations
879 $docblock = substr($docblock, 3, -2);
880
881 $re = '/@(?P<name>[A-Za-z_-]+)(?:[ \t]+(?P<value>.*?))?[ \t]*\r?$/m';
882 if (preg_match_all($re, $docblock, $matches)) {
883 $numMatches = count($matches[0]);

886 $annotations[$matches['name'][$i]][] = $matches['value'][$i];
