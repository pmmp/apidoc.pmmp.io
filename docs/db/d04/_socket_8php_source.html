51 public function __construct(InternetAddress $bindAddress){
52 $this->bindAddress = $bindAddress;
53 $socket = @socket_create($bindAddress->getVersion() === 4 ? AF_INET : AF_INET6, SOCK_DGRAM, SOL_UDP);
54 if($socket === false){
55 throw new \RuntimeException("Failed to create socket: " . trim(socket_strerror(socket_last_error())));
56 }
57 $this->socket = $socket;
58
59 if($bindAddress->getVersion() === 6){
60 socket_set_option($this->socket, IPPROTO_IPV6, IPV6_V6ONLY, 1); //Don't map IPv4 to IPv6, the implementation can create another RakLib instance to handle IPv4
61 }
62
63 if(@socket_bind($this->socket, $bindAddress->getIp(), $bindAddress->getPort()) === true){
64 $this->setSendBuffer(1024 * 1024 * 8)->setRecvBuffer(1024 * 1024 * 8);
65 }else{
66 $error = socket_last_error($this->socket);
67 if($error === SOCKET_EADDRINUSE){ //platform error messages aren't consistent
68 throw new SocketException("Failed to bind socket: Something else is already running on $bindAddress", $error);
69 }
70 throw new SocketException("Failed to bind to " . $bindAddress . ": " . trim(socket_strerror($error)), $error);
71 }
72 socket_set_nonblock($this->socket);
73 }

100 public function readPacket(?string &$source, ?int &$port) : ?string{
101 $buffer = "";
102 if(@socket_recvfrom($this->socket, $buffer, 65535, 0, $source, $port) === false){
103 $errno = socket_last_error($this->socket);
104 if($errno === SOCKET_EWOULDBLOCK){
105 return null;
106 }
107 throw new SocketException("Failed to recv (errno $errno): " . trim(socket_strerror($errno)), $errno);
108 }

115 public function writePacket(string $buffer, string $dest, int $port) : int{
116 $result = @socket_sendto($this->socket, $buffer, strlen($buffer), 0, $dest, $port);
117 if($result === false){

119 throw new SocketException("Failed to send to $dest $port (errno $errno): " . trim(socket_strerror($errno)), $errno);
120 }
121 return $result;
