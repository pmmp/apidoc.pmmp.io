108 private function init(ServerConfigGroup $config) : void{
109 $this->memoryLimit = $config->getPropertyInt("memory.main-limit", 0) * 1024 * 1024;
110
111 $defaultMemory = 1024;
112
113 if(preg_match("/([0-9]+)([KMGkmg])/", $config->getConfigString("memory-limit", ""), $matches) > 0){
114 $m = (int) $matches[1];
115 if($m <= 0){
116 $defaultMemory = 0;
117 }else{
118 switch(mb_strtoupper($matches[2])){
119 case "K":
120 $defaultMemory = intdiv($m, 1024);
121 break;
122 case "M":
123 $defaultMemory = $m;
124 break;
125 case "G":
126 $defaultMemory = $m * 1024;
127 break;
128 default:
129 $defaultMemory = $m;
130 break;
131 }
132 }
133 }
134
135 $hardLimit = $config->getPropertyInt("memory.main-hard-limit", $defaultMemory);
136
137 if($hardLimit <= 0){
138 ini_set("memory_limit", '-1');
139 }else{
140 ini_set("memory_limit", $hardLimit . "M");
141 }
142
143 $this->globalMemoryLimit = $config->getPropertyInt("memory.global-limit", 0) * 1024 * 1024;
144 $this->checkRate = $config->getPropertyInt("memory.check-rate", 20);
145 $this->continuousTrigger = $config->getPropertyBool("memory.continuous-trigger", true);
146 $this->continuousTriggerRate = $config->getPropertyInt("memory.continuous-trigger-rate", 30);
147
148 $this->garbageCollectionPeriod = $config->getPropertyInt("memory.garbage-collection.period", 36000);
149 $this->garbageCollectionTrigger = $config->getPropertyBool("memory.garbage-collection.low-memory-trigger", true);
150 $this->garbageCollectionAsync = $config->getPropertyBool("memory.garbage-collection.collect-async-worker", true);

152 $this->lowMemChunkRadiusOverride = $config->getPropertyInt("memory.max-chunks.chunk-radius", 4);

157

184 public function trigger(int $memory, int $limit, bool $global = false, int $triggerCount = 0) : void{
185 $this->logger->debug(sprintf("%sLow memory triggered, limit %gMB, using %gMB",
186 $global ? "Global " : "", round(($limit / 1024) / 1024, 2), round(($memory / 1024) / 1024, 2)));
187 if($this->lowMemClearWorldCache){
188 foreach($this->server->getWorldManager()->getWorlds() as $world){
189 $world->clearCache(true);
190 }
191 ChunkCache::pruneCaches();
192 }
193
194 if($this->lowMemChunkGC){
195 foreach($this->server->getWorldManager()->getWorlds() as $world){
196 $world->doChunkGarbageCollection();
197 }
198 }
199
200 $ev = new LowMemoryEvent($memory, $limit, $global, $triggerCount);
201 $ev->call();
202
203 $cycles = 0;
204 if($this->garbageCollectionTrigger){
205 $cycles = $this->triggerGarbageCollector();
206 }
207
208 $this->logger->debug(sprintf("Freed %gMB, $cycles cycles", round(($ev->getMemoryFreed() / 1024) / 1024, 2)));
209 }

214 public function check() : void{
215 Timings::$memoryManager->startTiming();
216
217 if(($this->memoryLimit > 0 || $this->globalMemoryLimit > 0) && ++$this->checkTicker >= $this->checkRate){
218 $this->checkTicker = 0;
219 $memory = Process::getAdvancedMemoryUsage();
220 $trigger = false;
221 if($this->memoryLimit > 0 && $memory[0] > $this->memoryLimit){
222 $trigger = 0;
223 }elseif($this->globalMemoryLimit > 0 && $memory[1] > $this->globalMemoryLimit){
224 $trigger = 1;
225 }
226
227 if($trigger !== false){
228 if($this->lowMemory && $this->continuousTrigger){
229 if(++$this->continuousTriggerTicker >= $this->continuousTriggerRate){
230 $this->continuousTriggerTicker = 0;
231 $this->trigger($memory[$trigger], $this->memoryLimit, $trigger > 0, ++$this->continuousTriggerCount);
232 }
233 }else{
234 $this->lowMemory = true;
235 $this->continuousTriggerCount = 0;
236 $this->trigger($memory[$trigger], $this->memoryLimit, $trigger > 0);
237 }
238 }else{
239 $this->lowMemory = false;
240 }
241 }
242
243 if($this->garbageCollectionPeriod > 0 && ++$this->garbageCollectionTicker >= $this->garbageCollectionPeriod){
244 $this->garbageCollectionTicker = 0;
245 $this->triggerGarbageCollector();
246 }
247
248 Timings::$memoryManager->stopTiming();
249 }

251 public function triggerGarbageCollector() : int{
252 Timings::$garbageCollector->startTiming();
253
254 if($this->garbageCollectionAsync){
255 $pool = $this->server->getAsyncPool();
256 if(($w = $pool->shutdownUnusedWorkers()) > 0){
257 $this->logger->debug("Shut down $w idle async pool workers");
258 }
259 foreach($pool->getRunningWorkers() as $i){
260 $pool->submitTaskToWorker(new GarbageCollectionTask(), $i);
261 }
262 }
263
264 $cycles = gc_collect_cycles();
265 gc_mem_caches();
266
267 Timings::$garbageCollector->stopTiming();
268
269 return $cycles;
270 }

293 public static function dumpMemory($startingObject, string $outputFolder, int $maxNesting, int $maxStringSize, \Logger $logger) : void{
294 $hardLimit = Utils::assumeNotFalse(ini_get('memory_limit'), "memory_limit INI directive should always exist");
295 ini_set('memory_limit', '-1');
296 gc_disable();
297
298 if(!file_exists($outputFolder)){
299 mkdir($outputFolder, 0777, true);
300 }
301
302 $obData = Utils::assumeNotFalse(fopen(Path::join($outputFolder, "objects.js"), "wb+"));
303
304 $objects = [];
305
306 $refCounts = [];
307
308 $instanceCounts = [];
309
310 $staticProperties = [];
311 $staticCount = 0;
312
313 $functionStaticVars = [];
314 $functionStaticVarsCount = 0;
315
316 foreach(get_declared_classes() as $className){
317 $reflection = new \ReflectionClass($className);
318 $staticProperties[$className] = [];
319 foreach($reflection->getProperties() as $property){
320 if(!$property->isStatic() || $property->getDeclaringClass()->getName() !== $className){
321 continue;
322 }
323
324 if(!$property->isPublic()){
325 $property->setAccessible(true);
326 }
327 if(!$property->isInitialized()){
328 continue;
329 }
330
331 $staticCount++;
332 $staticProperties[$className][$property->getName()] = self::continueDump($property->getValue(), $objects, $refCounts, 0, $maxNesting, $maxStringSize);
333 }
334
335 if(count($staticProperties[$className]) === 0){
336 unset($staticProperties[$className]);
337 }
338
339 foreach($reflection->getMethods() as $method){
340 if($method->getDeclaringClass()->getName() !== $reflection->getName()){
341 continue;
342 }
343 $methodStatics = [];
344 foreach($method->getStaticVariables() as $name => $variable){
345 $methodStatics[$name] = self::continueDump($variable, $objects, $refCounts, 0, $maxNesting, $maxStringSize);
346 }
347 if(count($methodStatics) > 0){
348 $functionStaticVars[$className . "::" . $method->getName()] = $methodStatics;
349 $functionStaticVarsCount += count($functionStaticVars);
350 }
351 }
352 }
353
354 file_put_contents(Path::join($outputFolder, "staticProperties.js"), json_encode($staticProperties, JSON_UNESCAPED_SLASHES | JSON_PRETTY_PRINT | JSON_THROW_ON_ERROR));
355 $logger->info("Wrote $staticCount static properties");
356
357 $globalVariables = [];
358 $globalCount = 0;
359
360 $ignoredGlobals = [
361 'GLOBALS' => true,
362 '_SERVER' => true,
363 '_REQUEST' => true,
364 '_POST' => true,
365 '_GET' => true,
366 '_FILES' => true,
367 '_ENV' => true,
368 '_COOKIE' => true,
369 '_SESSION' => true
370 ];
371
372 foreach(Utils::stringifyKeys($GLOBALS) as $varName => $value){
373 if(isset($ignoredGlobals[$varName])){
374 continue;
375 }
376
377 $globalCount++;
378 $globalVariables[$varName] = self::continueDump($value, $objects, $refCounts, 0, $maxNesting, $maxStringSize);
379 }
380
381 file_put_contents(Path::join($outputFolder, "globalVariables.js"), json_encode($globalVariables, JSON_UNESCAPED_SLASHES | JSON_PRETTY_PRINT | JSON_THROW_ON_ERROR));
382 $logger->info("Wrote $globalCount global variables");
383
384 foreach(get_defined_functions()["user"] as $function){
385 $reflect = new \ReflectionFunction($function);
386
387 $vars = [];
388 foreach($reflect->getStaticVariables() as $varName => $variable){
389 $vars[$varName] = self::continueDump($variable, $objects, $refCounts, 0, $maxNesting, $maxStringSize);
390 }
391 if(count($vars) > 0){
392 $functionStaticVars[$function] = $vars;
393 $functionStaticVarsCount += count($vars);
394 }
395 }
396 file_put_contents(Path::join($outputFolder, 'functionStaticVars.js'), json_encode($functionStaticVars, JSON_UNESCAPED_SLASHES | JSON_PRETTY_PRINT | JSON_THROW_ON_ERROR));
397 $logger->info("Wrote $functionStaticVarsCount function static variables");
398
399 $data = self::continueDump($startingObject, $objects, $refCounts, 0, $maxNesting, $maxStringSize);
400
401 do{
402 $continue = false;
403 foreach($objects as $hash => $object){
404 if(!is_object($object)){
405 continue;
406 }
407 $continue = true;
408
409 $className = get_class($object);
410 if(!isset($instanceCounts[$className])){
411 $instanceCounts[$className] = 1;
412 }else{
413 $instanceCounts[$className]++;
414 }
415
416 $objects[$hash] = true;
417 $info = [
418 "information" => "$hash@$className",
419 ];
420 if($object instanceof \Closure){
421 $info["definition"] = Utils::getNiceClosureName($object);
422 $info["referencedVars"] = [];
423 $reflect = new \ReflectionFunction($object);
424 if(($closureThis = $reflect->getClosureThis()) !== null){
425 $info["this"] = self::continueDump($closureThis, $objects, $refCounts, 0, $maxNesting, $maxStringSize);
426 }
427
428 foreach($reflect->getStaticVariables() as $name => $variable){
429 $info["referencedVars"][$name] = self::continueDump($variable, $objects, $refCounts, 0, $maxNesting, $maxStringSize);
430 }
431 }else{
432 $reflection = new \ReflectionObject($object);
433
434 $info["properties"] = [];
435
436 for($original = $reflection; $reflection !== false; $reflection = $reflection->getParentClass()){
437 foreach($reflection->getProperties() as $property){
438 if($property->isStatic()){
439 continue;
440 }
441
442 $name = $property->getName();
443 if($reflection !== $original){
444 if($property->isPrivate()){
445 $name = $reflection->getName() . ":" . $name;
446 }else{
447 continue;
448 }
449 }
450 if(!$property->isPublic()){
451 $property->setAccessible(true);
452 }
453 if(!$property->isInitialized($object)){
454 continue;
455 }
456
457 $info["properties"][$name] = self::continueDump($property->getValue($object), $objects, $refCounts, 0, $maxNesting, $maxStringSize);
458 }
459 }
460 }
461
462 fwrite($obData, json_encode($info, JSON_UNESCAPED_SLASHES | JSON_THROW_ON_ERROR) . "\n");
463 }
464
465 }while($continue);
466
467 $logger->info("Wrote " . count($objects) . " objects");
468
469 fclose($obData);
470
471 file_put_contents(Path::join($outputFolder, "serverEntry.js"), json_encode($data, JSON_UNESCAPED_SLASHES | JSON_PRETTY_PRINT | JSON_THROW_ON_ERROR));
472 file_put_contents(Path::join($outputFolder, "referenceCounts.js"), json_encode($refCounts, JSON_UNESCAPED_SLASHES | JSON_PRETTY_PRINT | JSON_THROW_ON_ERROR));
473
474 arsort($instanceCounts, SORT_NUMERIC);
475 file_put_contents(Path::join($outputFolder, "instanceCounts.js"), json_encode($instanceCounts, JSON_UNESCAPED_SLASHES | JSON_PRETTY_PRINT | JSON_THROW_ON_ERROR));
476
477 $logger->info("Finished!");
478
479 ini_set('memory_limit', $hardLimit);
480 gc_enable();
481 }

490 private static function continueDump($from, array &$objects, array &$refCounts, int $recursion, int $maxNesting, int $maxStringSize){
491 if($maxNesting <= 0){
492 return "(error) NESTING LIMIT REACHED";
493 }
494
495 --$maxNesting;
496
497 if(is_object($from)){
498 if(!isset($objects[$hash = spl_object_hash($from)])){
499 $objects[$hash] = $from;
500 $refCounts[$hash] = 0;
501 }
502
503 ++$refCounts[$hash];
504
505 $data = "(object) $hash";
506 }elseif(is_array($from)){
507 if($recursion >= 5){
508 return "(error) ARRAY RECURSION LIMIT REACHED";
509 }
510 $data = [];
511 $numeric = 0;
512 foreach($from as $key => $value){
513 $data[$numeric] = [
514 "k" => self::continueDump($key, $objects, $refCounts, $recursion + 1, $maxNesting, $maxStringSize),
515 "v" => self::continueDump($value, $objects, $refCounts, $recursion + 1, $maxNesting, $maxStringSize),
516 ];
517 $numeric++;
518 }
519 }elseif(is_string($from)){
520 $data = "(string) len(" . strlen($from) . ") " . substr(Utils::printable($from), 0, $maxStringSize);
521 }elseif(is_resource($from)){
522 $data = "(resource) " . print_r($from, true);
523 }else{
524 $data = $from;
525 }
526
527 return $data;
528 }
