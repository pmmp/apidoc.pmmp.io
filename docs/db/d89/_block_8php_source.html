311 public function generateStatePermutations() : \Generator{
312 //TODO: this bruteforce approach to discovering all valid states is very inefficient for larger state data sizes
313 //at some point we'll need to find a better way to do this
314 $bits = $this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits;
315 if($bits > Block::INTERNAL_STATE_DATA_BITS){
316 throw new \LogicException("Block state data cannot use more than " . Block::INTERNAL_STATE_DATA_BITS . " bits");
317 }
318 for($blockItemStateData = 0; $blockItemStateData < (1 << $this->requiredBlockItemStateDataBits); ++$blockItemStateData){
319 $withType = clone $this;
320 try{
321 $withType->decodeBlockItemState($blockItemStateData);
322 $encoded = $withType->encodeBlockItemState();
323 if($encoded !== $blockItemStateData){
324 throw new \LogicException(static::class . "::decodeBlockItemState() accepts invalid inputs (returned $encoded for input $blockItemStateData)");
325 }
326 }catch(InvalidSerializedRuntimeDataException){ //invalid property combination, leave it
327 continue;
328 }
329
330 for($blockOnlyStateData = 0; $blockOnlyStateData < (1 << $this->requiredBlockOnlyStateDataBits); ++$blockOnlyStateData){
331 $withState = clone $withType;
332 try{
333 $withState->decodeBlockOnlyState($blockOnlyStateData);
334 $encoded = $withState->encodeBlockOnlyState();
335 if($encoded !== $blockOnlyStateData){
336 throw new \LogicException(static::class . "::decodeBlockOnlyState() accepts invalid inputs (returned $encoded for input $blockOnlyStateData)");
337 }
338 }catch(InvalidSerializedRuntimeDataException){ //invalid property combination, leave it
339 continue;
340 }
341
342 yield $withState;
343 }
344 }

365 public function writeStateToWorld() : void{
366 $world = $this->position->getWorld();
367 $chunk = $world->getOrLoadChunkAtPosition($this->position);
368 if($chunk === null){
369 throw new AssumptionFailedError("World::setBlock() should have loaded the chunk before calling this method");
370 }
371 $chunk->setBlockStateId($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
372
373 $tileType = $this->idInfo->getTileClass();
374 $oldTile = $world->getTile($this->position);
375 if($oldTile !== null){
376 if($tileType === null || !($oldTile instanceof $tileType)){
377 $oldTile->close();
378 $oldTile = null;
379 }elseif($oldTile instanceof Spawnable){
380 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
381 }
382 }
383 if($oldTile === null && $tileType !== null){
388 $tile = new $tileType($world, $this->position->asVector3());
389 $world->addTile($tile);
390 }

936 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
937 $bbs = $this->getCollisionBoxes();
938 if(count($bbs) === 0){
939 return null;
940 }
941
943 $currentHit = null;
945 $currentDistance = PHP_INT_MAX;
946
947 foreach($bbs as $bb){
948 $nextHit = $bb->calculateIntercept($pos1, $pos2);
949 if($nextHit === null){
950 continue;
951 }
952
953 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
954 if($nextDistance < $currentDistance){
955 $currentHit = $nextHit;
956 $currentDistance = $nextDistance;
957 }
958 }
959
960 return $currentHit;
961 }
