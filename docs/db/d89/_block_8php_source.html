146 public function writeStateToWorld() : void{
147 $this->position->getWorld()->getOrLoadChunkAtPosition($this->position)->setFullBlock($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getFullId());
148
149 $tileType = $this->idInfo->getTileClass();
150 $oldTile = $this->position->getWorld()->getTile($this->position);
151 if($oldTile !== null){
152 if($tileType === null || !($oldTile instanceof $tileType)){
153 $oldTile->close();
154 $oldTile = null;
155 }elseif($oldTile instanceof Spawnable){
156 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
157 }
158 }
159 if($oldTile === null && $tileType !== null){
164 $tile = new $tileType($this->position->getWorld(), $this->position->asVector3());
165 $this->position->getWorld()->addTile($tile);
166 }

624 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
625 $bbs = $this->getCollisionBoxes();
626 if(count($bbs) === 0){
627 return null;
628 }
629
631 $currentHit = null;
633 $currentDistance = PHP_INT_MAX;
634
635 foreach($bbs as $bb){
636 $nextHit = $bb->calculateIntercept($pos1, $pos2);
637 if($nextHit === null){
638 continue;
639 }
640
641 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
642 if($nextDistance < $currentDistance){
643 $currentHit = $nextHit;
644 $currentDistance = $nextDistance;
645 }
646 }
647
648 return $currentHit;
649 }
