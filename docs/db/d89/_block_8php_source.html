151 public function getStateId() : int{
152 $typeId = $this->getTypeId();
153 //TODO: this XOR mask improves hashtable distribution, but it's only effective if the number of unique block
154 //type IDs is larger than the number of available state data bits. We should probably hash (e.g. using xxhash)
155 //the type ID to create a better mask.
156 //Alternatively, we could hash the whole state ID, but this is currently problematic, since we currently need
157 //to be able to recover the state data from the state ID because of UnknownBlock.
158 return ($typeId << self::INTERNAL_STATE_DATA_BITS) | ($this->encodeFullState() ^ ($typeId & self::INTERNAL_STATE_DATA_MASK));
159 }

209 $reader = new RuntimeDataReader($this->requiredBlockItemStateDataBits, $data);

211 $this->describeBlockItemState($reader);
212 $readBits = $reader->getOffset();
213 if($this->requiredBlockItemStateDataBits !== $readBits){
214 throw new \LogicException(get_class($this) . ": Exactly $this->requiredBlockItemStateDataBits bits of block-item state data were provided, but $readBits were read");

228 private function decodeFullState(int $data) : void{
229 $reader = new RuntimeDataReader($this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits, $data);
230 $this->decodeBlockItemState($reader->readInt($this->requiredBlockItemStateDataBits));
231 $this->decodeBlockOnlyState($reader->readInt($this->requiredBlockOnlyStateDataBits));
232 }

234 private function encodeBlockItemState() : int{
235 $writer = new RuntimeDataWriter($this->requiredBlockItemStateDataBits);
236
237 $this->describeBlockItemState($writer);
238 $writtenBits = $writer->getOffset();

240 throw new \LogicException(get_class($this) . ": Exactly $this->requiredBlockItemStateDataBits bits of block-item state data were expected, but $writtenBits were written");

258 private function encodeFullState() : int{
259 $writer = new RuntimeDataWriter($this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits);
260 $writer->writeInt($this->requiredBlockItemStateDataBits, $this->encodeBlockItemState());
261 $writer->writeInt($this->requiredBlockOnlyStateDataBits, $this->encodeBlockOnlyState());
262
263 return $writer->getValue();

297 public function generateStatePermutations() : \Generator{
298 //TODO: this bruteforce approach to discovering all valid states is very inefficient for larger state data sizes
299 //at some point we'll need to find a better way to do this
300 $bits = $this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits;
301 if($bits > Block::INTERNAL_STATE_DATA_BITS){
302 throw new \LogicException("Block state data cannot use more than " . Block::INTERNAL_STATE_DATA_BITS . " bits");
303 }
304 for($stateData = 0; $stateData < (1 << $bits); ++$stateData){
305 $v = clone $this;
306 try{
307 $v->decodeFullState($stateData);
308 if($v->encodeFullState() !== $stateData){
309 throw new \LogicException(static::class . "::decodeStateData() accepts invalid state data (returned " . $v->encodeFullState() . " for input $stateData)");
310 }
311 }catch(InvalidSerializedRuntimeDataException){ //invalid property combination, leave it
312 continue;
313 }
314
315 yield $v;
316 }

341 public function writeStateToWorld() : void{
342 $world = $this->position->getWorld();
343 $chunk = $world->getOrLoadChunkAtPosition($this->position);
344 if($chunk === null){
345 throw new AssumptionFailedError("World::setBlock() should have loaded the chunk before calling this method");
346 }
347 $chunk->setBlockStateId($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
348
349 $tileType = $this->idInfo->getTileClass();
350 $oldTile = $world->getTile($this->position);
351 if($oldTile !== null){
352 if($tileType === null || !($oldTile instanceof $tileType)){
353 $oldTile->close();
354 $oldTile = null;
355 }elseif($oldTile instanceof Spawnable){
356 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
357 }
358 }
359 if($oldTile === null && $tileType !== null){
364 $tile = new $tileType($world, $this->position->asVector3());
365 $world->addTile($tile);
366 }

881 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
882 $bbs = $this->getCollisionBoxes();
883 if(count($bbs) === 0){
884 return null;
885 }
886
888 $currentHit = null;
890 $currentDistance = PHP_INT_MAX;
891
892 foreach($bbs as $bb){
893 $nextHit = $bb->calculateIntercept($pos1, $pos2);
894 if($nextHit === null){
895 continue;
896 }
897
898 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
899 if($nextDistance < $currentDistance){
900 $currentHit = $nextHit;
901 $currentDistance = $nextDistance;
902 }
903 }
904
905 return $currentHit;
906 }
