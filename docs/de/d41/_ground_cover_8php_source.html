38 public function populate(ChunkManager $world, int $chunkX, int $chunkZ, Random $random) : void{
39 $chunk = $world->getChunk($chunkX, $chunkZ);
40 $factory = BlockFactory::getInstance();
41 $biomeRegistry = BiomeRegistry::getInstance();
42 for($x = 0; $x < Chunk::EDGE_LENGTH; ++$x){
43 for($z = 0; $z < Chunk::EDGE_LENGTH; ++$z){
44 $biome = $biomeRegistry->getBiome($chunk->getBiomeId($x, $z));
45 $cover = $biome->getGroundCover();
46 if(count($cover) > 0){
47 $diffY = 0;
48 if(!$cover[0]->isSolid()){
49 $diffY = 1;
50 }
51
52 $startY = 127;
53 for(; $startY > 0; --$startY){
54 if(!$factory->fromFullBlock($chunk->getFullBlock($x, $startY, $z))->isTransparent()){
55 break;
56 }
57 }
58 $startY = min(127, $startY + $diffY);
59 $endY = $startY - count($cover);
60 for($y = $startY; $y > $endY && $y >= 0; --$y){
61 $b = $cover[$startY - $y];
62 $id = $factory->fromFullBlock($chunk->getFullBlock($x, $y, $z));
63 if($id->getId() === BlockLegacyIds::AIR && $b->isSolid()){
64 break;
65 }
66 if($b->canBeFlowedInto() && $id instanceof Liquid){
67 continue;
68 }
69
70 $chunk->setFullBlock($x, $y, $z, $b->getFullId());
71 }
72 }
73 }
74 }
