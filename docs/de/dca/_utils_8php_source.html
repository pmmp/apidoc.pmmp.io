121 public static function getNiceClosureName(\Closure $closure) : string{
122 $func = new \ReflectionFunction($closure);
123 if(!str_ends_with($func->getName(), '{closure}')){
124 //closure wraps a named function, can be done with reflection or fromCallable()
125 //isClosure() is useless here because it just tells us if $func is reflecting a Closure object
126
127 $scope = $func->getClosureScopeClass();
128 if($scope !== null){ //class method
129 return
130 $scope->getName() .
131 ($func->getClosureThis() !== null ? "->" : "::") .
132 $func->getName(); //name doesn't include class in this case
133 }
134
135 //non-class function
136 return $func->getName();
137 }
138 $filename = $func->getFileName();
139
140 return "closure@" . ($filename !== false ?
141 Filesystem::cleanPath($filename) . "#L" . $func->getStartLine() :
142 "internal"
143 );
144 }

197 public static function getMachineUniqueId(string $extra = "") : UuidInterface{
198 if(self::$serverUniqueId !== null && $extra === ""){
199 return self::$serverUniqueId;
200 }
201
202 $machine = php_uname("a");
203 $cpuinfo = @file("/proc/cpuinfo");
204 if($cpuinfo !== false){
205 $cpuinfoLines = preg_grep("/(model name|Processor|Serial)/", $cpuinfo);
206 if($cpuinfoLines === false){
207 throw new AssumptionFailedError("Pattern is valid, so this shouldn't fail ...");
208 }
209 $machine .= implode("", $cpuinfoLines);
210 }
211 $machine .= sys_get_temp_dir();
212 $machine .= $extra;
213 $os = Utils::getOS();
214 if($os === Utils::OS_WINDOWS){
215 @exec("ipconfig /ALL", $mac);
216 $mac = implode("\n", $mac);
217 if(preg_match_all("#Physical Address[. ]{1,}: ([0-9A-F\\-]{17})#", $mac, $matches) > 0){
218 foreach($matches[1] as $i => $v){
219 if($v == "00-00-00-00-00-00"){
220 unset($matches[1][$i]);
221 }
222 }
223 $machine .= implode(" ", $matches[1]); //Mac Addresses
224 }
225 }elseif($os === Utils::OS_LINUX){
226 if(file_exists("/etc/machine-id")){
227 $machine .= file_get_contents("/etc/machine-id");
228 }else{
229 @exec("ifconfig 2>/dev/null", $mac);
230 $mac = implode("\n", $mac);
231 if(preg_match_all("#HWaddr[ \t]{1,}([0-9a-f:]{17})#", $mac, $matches) > 0){
232 foreach($matches[1] as $i => $v){
233 if($v == "00:00:00:00:00:00"){
234 unset($matches[1][$i]);
235 }
236 }
237 $machine .= implode(" ", $matches[1]); //Mac Addresses
238 }
239 }
240 }elseif($os === Utils::OS_ANDROID){
241 $machine .= @file_get_contents("/system/build.prop");
242 }elseif($os === Utils::OS_MACOS){
243 $machine .= shell_exec("system_profiler SPHardwareDataType | grep UUID");
244 }
245 $data = $machine . PHP_MAXPATHLEN;
246 $data .= PHP_INT_MAX;
247 $data .= PHP_INT_SIZE;
248 $data .= get_current_user();
249 foreach(get_loaded_extensions() as $ext){
250 $data .= $ext . ":" . phpversion($ext);
251 }
252
253 //TODO: use of NIL as namespace is a hack; it works for now, but we should have a proper namespace UUID
254 $uuid = Uuid::uuid3(Uuid::NIL, $data);
255
256 if($extra === ""){
257 self::$serverUniqueId = $uuid;
258 }
259
260 return $uuid;
261 }

273 public static function getOS(bool $recalculate = false) : string{
274 if(self::$os === null || $recalculate){
275 $uname = php_uname("s");
276 if(stripos($uname, "Darwin") !== false){
277 if(str_starts_with(php_uname("m"), "iP")){
278 self::$os = self::OS_IOS;
279 }else{
280 self::$os = self::OS_MACOS;
281 }
282 }elseif(stripos($uname, "Win") !== false || $uname === "Msys"){
283 self::$os = self::OS_WINDOWS;
284 }elseif(stripos($uname, "Linux") !== false){
285 if(@file_exists("/system/build.prop")){
286 self::$os = self::OS_ANDROID;
287 }else{
288 self::$os = self::OS_LINUX;
289 }
290 }elseif(stripos($uname, "BSD") !== false || $uname === "DragonFly"){
291 self::$os = self::OS_BSD;
292 }else{
293 self::$os = self::OS_UNKNOWN;
294 }
295 }
296
297 return self::$os;
298 }

300 public static function getCoreCount(bool $recalculate = false) : int{
301 if(self::$cpuCores !== null && !$recalculate){
302 return self::$cpuCores;
303 }
304
305 $processors = 0;
306 switch(Utils::getOS()){
307 case Utils::OS_LINUX:
308 case Utils::OS_ANDROID:
309 if(($cpuinfo = @file('/proc/cpuinfo')) !== false){
310 foreach($cpuinfo as $l){
311 if(preg_match('/^processor[ \t]*:[ \t]*[0-9]+$/m', $l) > 0){
312 ++$processors;
313 }
314 }
315 }elseif(($cpuPresent = @file_get_contents("/sys/devices/system/cpu/present")) !== false){
316 if(preg_match("/^([0-9]+)\\-([0-9]+)$/", trim($cpuPresent), $matches) > 0){
317 $processors = ((int) $matches[2]) - ((int) $matches[1]);
318 }
319 }
320 break;
321 case Utils::OS_BSD:
322 case Utils::OS_MACOS:
323 $processors = (int) shell_exec("sysctl -n hw.ncpu");
324 break;
325 case Utils::OS_WINDOWS:
326 $processors = (int) getenv("NUMBER_OF_PROCESSORS");
327 break;
328 }
329 return self::$cpuCores = $processors;
330 }

335 public static function hexdump(string $bin) : string{
336 $output = "";
337 $bin = str_split($bin, 16);
338 foreach($bin as $counter => $line){
339 $hex = chunk_split(chunk_split(str_pad(bin2hex($line), 32, " ", STR_PAD_RIGHT), 2, " "), 24, " ");
340 $ascii = preg_replace('#([^\x20-\x7E])#', ".", $line);
341 $output .= str_pad(dechex($counter << 4), 4, "0", STR_PAD_LEFT) . " " . $hex . " " . $ascii . PHP_EOL;
342 }
343
344 return $output;
345 }

360 public static function javaStringHash(string $string) : int{
361 $hash = 0;
362 for($i = 0, $len = strlen($string); $i < $len; $i++){
363 $ord = ord($string[$i]);
364 if(($ord & 0x80) !== 0){
365 $ord -= 0x100;
366 }
367 $hash = 31 * $hash + $ord;
368 $hash &= 0xFFFFFFFF;
369 }
370 return $hash;
371 }

377 ob_start();
378 debug_zval_dump($value);

380 if($contents === false) throw new AssumptionFailedError("ob_get_contents() should never return false here");

383

413 if($trace === null){
414 $trace = $e->getTrace();

418 $lines[] = "--- Stack trace ---";
419 foreach(Utils::printableTrace($trace) as $line){
420 $lines[] = " " . $line;
421 }
422 for($prev = $e->getPrevious(); $prev !== null; $prev = $prev->getPrevious()){
423 $lines[] = "--- Previous ---";
424 $lines[] = self::printableExceptionMessage($prev);
425 foreach(Utils::printableTrace($prev->getTrace()) as $line){
426 $lines[] = " " . $line;
427 }
428 }
429 $lines[] = "--- End of exception information ---";
430 return $lines;

433 private static function stringifyValueForTrace(mixed $value, int $maxStringLength) : string{
434 return match(true){
435 is_object($value) => "object " . self::getNiceClassName($value) . "#" . spl_object_id($value),
436 is_array($value) => "array[" . count($value) . "]",
437 is_string($value) => "string[" . strlen($value) . "] " . substr(Utils::printable($value), 0, $maxStringLength),
438 is_bool($value) => $value ? "true" : "false",
439 is_int($value) => "int " . $value,
440 is_float($value) => "float " . $value,
441 $value === null => "null",
442 default => gettype($value) . " " . Utils::printable((string) $value)
443 };

452 public static function printableTrace(array $trace, int $maxStringLength = 80) : array{
453 $messages = [];
454 for($i = 0; isset($trace[$i]); ++$i){
455 $params = "";
456 if(isset($trace[$i]["args"]) || isset($trace[$i]["params"])){
457 if(isset($trace[$i]["args"])){
458 $args = $trace[$i]["args"];
459 }else{
460 $args = $trace[$i]["params"];
461 }
464 $paramsList = [];
465 $offset = 0;
466 foreach($args as $argId => $value){
467 $paramsList[] = ($argId === $offset ? "" : "$argId: ") . self::stringifyValueForTrace($value, $maxStringLength);
468 $offset++;
469 }
470 $params = implode(", ", $paramsList);
471 }
472 $messages[] = "#$i " . (isset($trace[$i]["file"]) ? Filesystem::cleanPath($trace[$i]["file"]) : "") . "(" . (isset($trace[$i]["line"]) ? $trace[$i]["line"] : "") . "): " . (isset($trace[$i]["class"]) ? $trace[$i]["class"] . (($trace[$i]["type"] === "dynamic" || $trace[$i]["type"] === "->") ? "->" : "::") : "") . $trace[$i]["function"] . "(" . Utils::printable($params) . ")";
473 }
474 return $messages;
475 }
