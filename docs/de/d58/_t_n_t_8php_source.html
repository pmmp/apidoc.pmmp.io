112 public function ignite(int $fuse = 80) : void{

115 $mot = (new Random())->nextSignedFloat() * M_PI * 2;
116
117 $tnt = new PrimedTNT(Location::fromObject($this->position->add(0.5, 0, 0.5), $this->position->getWorld()));
118 $tnt->setFuse($fuse);
