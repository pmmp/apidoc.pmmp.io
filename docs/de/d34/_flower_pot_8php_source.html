94 public function onInteract(Item $item, int $face, Vector3 $clickVector, ?Player $player = null, array &$returnedItems = []) : bool{
95 $world = $this->position->getWorld();
96 $plant = $item->getBlock();
97 if($this->plant !== null){
98 if($this->isValidPlant($plant)){
99 //for some reason, vanilla doesn't remove the contents of the pot if the held item is plantable
100 //and will also cause a new plant to be placed if clicking on the side
101 return false;
102 }
103
104 $removedItems = [$this->plant->asItem()];
105 if($player !== null){
106 //this one just has to be a weirdo :(
107 //this is the only block that directly adds items to the player inventory instead of just dropping items
108 $removedItems = $player->getInventory()->addItem(...$removedItems);
109 }
110 foreach($removedItems as $drops){
111 $world->dropItem($this->position->add(0.5, 0.5, 0.5), $drops);
112 }
113
114 $this->setPlant(null);
115 $world->setBlock($this->position, $this);
116 return true;
117 }elseif($this->isValidPlant($plant)){
118 $this->setPlant($plant);
119 $item->pop();
120 $world->setBlock($this->position, $this);
121
122 return true;
123 }
124
125 return false;
126 }
