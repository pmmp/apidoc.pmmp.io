124 public function onInteract(Item $item, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
125 $world = $this->position->getWorld();
126 $plant = $item->getBlock();
127 if($this->plant !== null){
128 if($this->isValidPlant($plant)){
129 //for some reason, vanilla doesn't remove the contents of the pot if the held item is plantable
130 //and will also cause a new plant to be placed if clicking on the side
131 return false;
132 }
133
134 $removedItems = [$this->plant->asItem()];
135 if($player !== null){
136 //this one just has to be a weirdo :(
137 //this is the only block that directly adds items to the player inventory instead of just dropping items
138 $removedItems = $player->getInventory()->addItem(...$removedItems);
139 }
140 foreach($removedItems as $drops){
141 $world->dropItem($this->position->add(0.5, 0.5, 0.5), $drops);
142 }
143
144 $this->setPlant(null);
145 $world->setBlock($this->position, $this);
146 return true;
147 }elseif($this->isValidPlant($plant)){
148 $this->setPlant($plant);
149 $item->pop();
150 $world->setBlock($this->position, $this);
151
152 return true;
153 }
154
155 return false;
156 }
