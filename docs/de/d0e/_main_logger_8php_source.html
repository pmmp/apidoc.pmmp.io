182 protected function send($message, $level, $prefix, $color) : void{
183 $time = new \DateTime('now', new \DateTimeZone($this->timezone));
184
185 $thread = \Thread::getCurrentThread();
186 if($thread === null){
187 $threadName = $this->mainThreadName . " thread";
188 }elseif($thread instanceof Thread || $thread instanceof Worker){
189 $threadName = $thread->getThreadName() . " thread";
190 }else{
191 $threadName = (new \ReflectionClass($thread))->getShortName() . " thread";
192 }
193
194 $message = sprintf($this->format, $time->format("H:i:s.v"), $color, $threadName, $prefix, TextFormat::clean($message, false));
195
196 if(!Terminal::isInit()){
197 Terminal::init($this->useFormattingCodes); //lazy-init colour codes because we don't know if they've been registered on this thread
198 }
199
200 $this->synchronized(function() use ($message, $level, $time) : void{
201 Terminal::writeLine($message);
202 $this->logWriterThread->write($time->format("Y-m-d") . " " . TextFormat::clean($message) . PHP_EOL);
203
204 foreach($this->attachments as $attachment){
205 $attachment->log($level, $message);
206 }
207 });
208 }
