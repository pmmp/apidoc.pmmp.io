57 protected function checkForHarden() : bool{
58 if($this->falling){
59 return false;
60 }
61 $colliding = null;
62 foreach(Facing::ALL as $side){
63 if($side === Facing::DOWN){
64 continue;
65 }
66 $blockSide = $this->getSide($side);
67 if($blockSide instanceof Water){
68 $colliding = $blockSide;
69 break;
70 }
71 }
72
73 if($colliding !== null){
74 if($this->decay === 0){
75 $this->liquidCollide($colliding, VanillaBlocks::OBSIDIAN());
76 return true;
77 }elseif($this->decay <= 4){
78 $this->liquidCollide($colliding, VanillaBlocks::COBBLESTONE());
79 return true;
80 }
81 }
82 return false;
83 }
