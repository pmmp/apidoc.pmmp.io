141 public function readChunk(int $x, int $z) : ?string{
142 $index = self::getChunkOffset($x, $z);
143
144 $this->lastUsed = time();
145
146 if($this->locationTable[$index] === null){
147 return null;
148 }
149
150 fseek($this->filePointer, $this->locationTable[$index]->getFirstSector() << 12);
151
152 /*
153 * this might cause us to read some junk, but under normal circumstances it won't be any more than 4096 bytes wasted.
154 * doing this in a single call is faster than making two seeks and reads to fetch the chunk.
155 * this relies on the assumption that the end of the file is always padded to a multiple of 4096 bytes.
156 */
157 $bytesToRead = $this->locationTable[$index]->getSectorCount() << 12;
158 $payload = fread($this->filePointer, $bytesToRead);
159
160 if($payload === false || strlen($payload) !== $bytesToRead){
161 throw new CorruptedChunkException("Corrupted chunk detected (unexpected EOF, truncated or non-padded chunk found)");
162 }
163 $stream = new BinaryStream($payload);
164
165 try{
166 $length = $stream->getInt();
167 if($length <= 0){ //TODO: if we reached here, the locationTable probably needs updating
168 return null;
169 }
170
171 $compression = $stream->getByte();
172 if($compression !== self::COMPRESSION_ZLIB && $compression !== self::COMPRESSION_GZIP){
173 throw new CorruptedChunkException("Invalid compression type (got $compression, expected " . self::COMPRESSION_ZLIB . " or " . self::COMPRESSION_GZIP . ")");
174 }
175
176 return $stream->get($length - 1); //length prefix includes the compression byte
177 }catch(BinaryDataException $e){
178 throw new CorruptedChunkException("Corrupted chunk detected: " . $e->getMessage(), 0, $e);
179 }
180 }

189 private function disposeGarbageArea(RegionLocationTableEntry $oldLocation) : void{
190 /* release the area containing the old copy to the garbage pool */
191 $this->garbageTable->add($oldLocation);
192
193 $endGarbage = $this->garbageTable->end();
194 $nextSector = $this->nextSector;
195 for(; $endGarbage !== null && $endGarbage->getLastSector() + 1 === $nextSector; $endGarbage = $this->garbageTable->end()){
196 $nextSector = $endGarbage->getFirstSector();
197 $this->garbageTable->remove($endGarbage);
198 }
199
200 if($nextSector !== $this->nextSector){
201 $this->nextSector = $nextSector;
202 ftruncate($this->filePointer, $this->nextSector << 12);
203 }
204 }

210 public function writeChunk(int $x, int $z, string $chunkData) : void{
211 $this->lastUsed = time();
212
213 $length = strlen($chunkData) + 1;
214 if($length + 4 > self::MAX_SECTOR_LENGTH){
215 throw new ChunkException("Chunk is too big! " . ($length + 4) . " > " . self::MAX_SECTOR_LENGTH);
216 }
217
218 $newSize = (int) ceil(($length + 4) / 4096);
219 $index = self::getChunkOffset($x, $z);
220
221 /*
222 * look for an unused area big enough to hold this data
223 * this is corruption-resistant (it leaves the old data intact if a failure occurs when writing new data), and
224 * also allows the file to become more compact across consecutive writes without introducing a dedicated garbage
225 * collection mechanism.
226 */
227 $newLocation = $this->garbageTable->allocate($newSize);
228
229 /* if no gaps big enough were found, append to the end of the file instead */
230 if($newLocation === null){
231 $newLocation = new RegionLocationTableEntry($this->nextSector, $newSize, time());
232 $this->bumpNextFreeSector($newLocation);
233 }
234
235 /* write the chunk data into the chosen location */
236 fseek($this->filePointer, $newLocation->getFirstSector() << 12);
237 fwrite($this->filePointer, str_pad(Binary::writeInt($length) . chr(self::COMPRESSION_ZLIB) . $chunkData, $newSize << 12, "\x00", STR_PAD_RIGHT));
238
239 /*
240 * update the file header - we do this after writing the main data, so that if a failure occurs while writing,
241 * the header will still point to the old (intact) copy of the chunk, instead of a potentially broken new
242 * version of the file (e.g. partially written).
243 */
244 $oldLocation = $this->locationTable[$index];
245 $this->locationTable[$index] = $newLocation;
246 $this->writeLocationIndex($index);
247
248 if($oldLocation !== null){
249 $this->disposeGarbageArea($oldLocation);
250 }
251 }

258 $oldLocation = $this->locationTable[$index];

269 protected static function getChunkOffset(int $x, int $z) : int{
270 if($x < 0 || $x > 31 || $z < 0 || $z > 31){
271 throw new \InvalidArgumentException("Invalid chunk position in region, expected x/z in range 0-31, got x=$x, z=$z");

297 protected function loadLocationTable() : void{
298 fseek($this->filePointer, 0);
299
300 $headerRaw = fread($this->filePointer, self::REGION_HEADER_LENGTH);
301 if($headerRaw === false || strlen($headerRaw) !== self::REGION_HEADER_LENGTH){
302 throw new CorruptedRegionException("Corrupted region header (unexpected end of file)");
303 }
304
306 $data = unpack("N*", $headerRaw);
307
308 for($i = 0; $i < 1024; ++$i){
309 $index = $data[$i + 1];
310 $offset = $index >> 8;
311 $sectorCount = $index & 0xff;
312 $timestamp = $data[$i + 1025];
313
314 if($offset === 0 || $sectorCount === 0){
315 $this->locationTable[$i] = null;
316 }elseif($offset >= self::FIRST_SECTOR){
317 $this->bumpNextFreeSector($this->locationTable[$i] = new RegionLocationTableEntry($offset, $sectorCount, $timestamp));
318 }else{
319 self::getChunkCoords($i, $chunkXX, $chunkZZ);
320 throw new CorruptedRegionException("Invalid region header entry for x=$chunkXX z=$chunkZZ, offset overlaps with header");
321 }
322 }
323
324 $this->checkLocationTableValidity();
325
326 $this->garbageTable = RegionGarbageMap::buildFromLocationTable($this->locationTable);
327
328 fseek($this->filePointer, 0);
329 }

334 private function checkLocationTableValidity() : void{
336 $usedOffsets = [];
337
338 $fileSize = filesize($this->filePath);
339 if($fileSize === false) throw new AssumptionFailedError("filesize() should not return false here");
340 for($i = 0; $i < 1024; ++$i){
341 $entry = $this->locationTable[$i];
342 if($entry === null){
343 continue;
344 }
345
346 self::getChunkCoords($i, $x, $z);
347 $offset = $entry->getFirstSector();
348 $fileOffset = $offset << 12;
349
350 //TODO: more validity checks
351
352 if($fileOffset >= $fileSize){
353 throw new CorruptedRegionException("Region file location offset x=$x,z=$z points to invalid file location $fileOffset");
354 }
355 if(isset($usedOffsets[$offset])){
356 self::getChunkCoords($usedOffsets[$offset], $existingX, $existingZ);
357 throw new CorruptedRegionException("Found two chunk offsets (chunk1: x=$existingX,z=$existingZ, chunk2: x=$x,z=$z) pointing to the file location $fileOffset");
358 }
359 $usedOffsets[$offset] = $i;
360 }
361 ksort($usedOffsets, SORT_NUMERIC);
362 $prevLocationIndex = null;
363 foreach($usedOffsets as $startOffset => $locationTableIndex){
364 if($this->locationTable[$locationTableIndex] === null){
365 continue;
366 }
367 if($prevLocationIndex !== null){
368 assert($this->locationTable[$prevLocationIndex] !== null);
369 if($this->locationTable[$locationTableIndex]->overlaps($this->locationTable[$prevLocationIndex])){
370 self::getChunkCoords($locationTableIndex, $chunkXX, $chunkZZ);
371 self::getChunkCoords($prevLocationIndex, $prevChunkXX, $prevChunkZZ);
372 throw new CorruptedRegionException("Overlapping chunks detected in region header (chunk1: x=$chunkXX,z=$chunkZZ, chunk2: x=$prevChunkXX,z=$prevChunkZZ)");
373 }
374 }
375 $prevLocationIndex = $locationTableIndex;
376 }

380 $entry = $this->locationTable[$index];
381 fseek($this->filePointer, $index << 2);

400 public function generateSectorMap(string $usedChar, string $freeChar) : string{
401 $result = str_repeat($freeChar, $this->nextSector);
402 for($i = 0; $i < self::FIRST_SECTOR; ++$i){
403 $result[$i] = $usedChar;
404 }
405 foreach($this->locationTable as $locationTableEntry){
406 if($locationTableEntry === null){
407 continue;
408 }
409 foreach($locationTableEntry->getUsedSectors() as $sectorIndex){
410 if($sectorIndex >= strlen($result)){
411 throw new AssumptionFailedError("This should never happen...");
412 }
413 if($result[$sectorIndex] === $usedChar){
414 throw new AssumptionFailedError("Overlap detected");
415 }
416 $result[$sectorIndex] = $usedChar;
417 }
418 }
419 return $result;
420 }

425 public function getProportionUnusedSpace() : float{
