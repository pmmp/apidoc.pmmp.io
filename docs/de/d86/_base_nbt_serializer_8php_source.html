46 private function readRoot(int $maxDepth) : TreeRoot{
47 $type = $this->readByte();
48 if($type === NBT::TAG_End){
49 throw new NbtDataException("Found TAG_End at the start of buffer");
50 }
51
52 $rootName = $this->readString();

63 public function read(string $buffer, int &$offset = 0, int $maxDepth = 0) : TreeRoot{
