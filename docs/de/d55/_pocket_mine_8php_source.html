68 function check_platform_dependencies(){
69 if(version_compare(MIN_PHP_VERSION, PHP_VERSION) > 0){
70 //If PHP version isn't high enough, anything below might break, so don't bother checking it.
71 return [
72 "PHP >= " . MIN_PHP_VERSION . " is required, but you have PHP " . PHP_VERSION . "."
73 ];
74 }
75
76 $messages = [];
77
78 if(PHP_INT_SIZE < 8){
79 $messages[] = "32-bit systems/PHP are no longer supported. Please upgrade to a 64-bit system, or use a 64-bit PHP binary if this is a 64-bit system.";
80 }
81
82 if(php_sapi_name() !== "cli"){
83 $messages[] = "Only PHP CLI is supported.";
84 }
85
86 $extensions = [
87 "chunkutils2" => "PocketMine ChunkUtils v2",
88 "curl" => "cURL",
89 "crypto" => "php-crypto",
90 "ctype" => "ctype",
91 "date" => "Date",
92 "gmp" => "GMP",
93 "hash" => "Hash",
94 "igbinary" => "igbinary",
95 "json" => "JSON",
96 "leveldb" => "LevelDB",
97 "mbstring" => "Multibyte String",
98 "morton" => "morton",
99 "openssl" => "OpenSSL",
100 "pcre" => "PCRE",
101 "phar" => "Phar",
102 "pthreads" => "pthreads",
103 "reflection" => "Reflection",
104 "sockets" => "Sockets",
105 "spl" => "SPL",
106 "yaml" => "YAML",
107 "zip" => "Zip",
108 "zlib" => "Zlib"
109 ];
110
111 foreach($extensions as $ext => $name){
112 if(!extension_loaded($ext)){
113 $messages[] = "Unable to find the $name ($ext) extension.";
114 }
115 }
116
117 if(($pthreads_version = phpversion("pthreads")) !== false){
118 if(substr_count($pthreads_version, ".") < 2){
119 $pthreads_version = "0.$pthreads_version";
120 }
121 if(version_compare($pthreads_version, "4.0.0") < 0 || version_compare($pthreads_version, "5.0.0") > 0){
122 $messages[] = "pthreads ^4.0.0 is required, while you have $pthreads_version.";
123 }
124 }
125
126 if(($leveldb_version = phpversion("leveldb")) !== false){
127 if(version_compare($leveldb_version, "0.2.1") < 0){
128 $messages[] = "php-leveldb >= 0.2.1 is required, while you have $leveldb_version.";
129 }
130 if(!defined('LEVELDB_ZLIB_RAW_COMPRESSION')){
131 $messages[] = "Given version of php-leveldb doesn't support ZLIB_RAW compression (use https://github.com/pmmp/php-leveldb)";
132 }
133 }
134
135 $chunkutils2_version = phpversion("chunkutils2");
136 $wantedVersionLock = "0.3";
137 $wantedVersionMin = "$wantedVersionLock.0";
138 if($chunkutils2_version !== false && (
139 version_compare($chunkutils2_version, $wantedVersionMin) < 0 ||
140 preg_match("/^" . preg_quote($wantedVersionLock, "/") . "\.\d+(?:-dev)?$/", $chunkutils2_version) === 0 //lock in at ^0.2, optionally at a patch release
141 )){
142 $messages[] = "chunkutils2 ^$wantedVersionMin is required, while you have $chunkutils2_version.";
143 }
144
145 if(extension_loaded("pocketmine")){
146 $messages[] = "The native PocketMine extension is no longer supported.";
147 }
148
149 if(!defined('AF_INET6')){
150 $messages[] = "IPv6 support is required, but your PHP binary was built without IPv6 support.";
151 }
152
153 return $messages;
154 }

159 function emit_performance_warnings(\Logger $logger){
160 if(PHP_DEBUG !== 0){
161 $logger->warning("This PHP binary was compiled in debug mode. This has a major impact on performance.");
162 }
163 if(extension_loaded("xdebug")){
164 $logger->warning("Xdebug extension is enabled. This has a major impact on performance.");
165 }
166 if(((int) ini_get('zend.assertions')) !== -1){
167 $logger->warning("Debugging assertions are enabled. This may degrade performance. To disable them, set `zend.assertions = -1` in php.ini.");
168 }
169 if(\Phar::running(true) === ""){
170 $logger->warning("Non-packaged installation detected. This will degrade autoloading speed and make startup times longer.");
171 }
172 if(function_exists('opcache_get_status') && ($opcacheStatus = opcache_get_status(false)) !== false){
173 $jitEnabled = $opcacheStatus["jit"]["on"] ?? false;
174 if($jitEnabled !== false){
175 $logger->warning(<<<'JIT_WARNING'
176
177
178 --------------------------------------- ! WARNING ! ---------------------------------------
179 You're using PHP 8.0 with JIT enabled. This provides significant performance improvements.
180 HOWEVER, it is EXPERIMENTAL, and has already been seen to cause weird and unexpected bugs.
181 Proceed with caution.
182 If you want to report any bugs, make sure to mention that you are using PHP 8.0 with JIT.
183 To turn off JIT, change `opcache.jit` to `0` in your php.ini file.
184 -------------------------------------------------------------------------------------------
185
186JIT_WARNING
187);
188 }
189 }

206 function server(){
207 if(count($messages = check_platform_dependencies()) > 0){
208 echo PHP_EOL;
209 $binary = version_compare(PHP_VERSION, "5.4") >= 0 ? PHP_BINARY : "unknown";
210 critical_error("Selected PHP binary does not satisfy some requirements.");
211 foreach($messages as $m){
212 echo " - $m" . PHP_EOL;
213 }
214 critical_error("PHP binary used: " . $binary);
215 critical_error("Loaded php.ini: " . (($file = php_ini_loaded_file()) !== false ? $file : "none"));
216 $phprc = getenv("PHPRC");
217 critical_error("Value of PHPRC environment variable: " . ($phprc === false ? "" : $phprc));
218 critical_error("Please recompile PHP with the needed configuration, or refer to the installation instructions at http://pmmp.rtfd.io/en/rtfd/installation.html.");
219 echo PHP_EOL;
220 exit(1);
221 }
222 unset($messages);
223
224 error_reporting(-1);
225 set_ini_entries();
226
227 $bootstrap = dirname(__FILE__, 2) . '/vendor/autoload.php';
228 if(!is_file($bootstrap)){
229 critical_error("Composer autoloader not found at " . $bootstrap);
230 critical_error("Please install/update Composer dependencies or use provided builds.");
231 exit(1);
232 }
233 require_once($bootstrap);
234
235 $composerGitHash = InstalledVersions::getReference('pocketmine/pocketmine-mp');
236 if($composerGitHash !== null){
237 //we can't verify dependency versions if we were installed without using git
238 $currentGitHash = explode("-", VersionInfo::GIT_HASH())[0];
239 if($currentGitHash !== $composerGitHash){
240 critical_error("Composer dependencies and/or autoloader are out of sync.");
241 critical_error("- Current revision is $currentGitHash");
242 critical_error("- Composer dependencies were last synchronized for revision $composerGitHash");
243 critical_error("Out-of-sync Composer dependencies may result in crashes and classes not being found.");
244 critical_error("Please synchronize Composer dependencies before running the server.");
245 exit(1);
246 }
247 }
248 if(extension_loaded('parallel')){
249 \parallel\bootstrap(\pocketmine\COMPOSER_AUTOLOADER_PATH);
250 }
251
252 ErrorToExceptionHandler::set();
253
254 $opts = getopt("", ["data:", "plugins:", "no-wizard", "enable-ansi", "disable-ansi"]);
255
256 $cwd = Utils::assumeNotFalse(realpath(Utils::assumeNotFalse(getcwd())));
257 $dataPath = isset($opts["data"]) ? $opts["data"] . DIRECTORY_SEPARATOR : $cwd . DIRECTORY_SEPARATOR;
258 $pluginPath = isset($opts["plugins"]) ? $opts["plugins"] . DIRECTORY_SEPARATOR : $cwd . DIRECTORY_SEPARATOR . "plugins" . DIRECTORY_SEPARATOR;
259 Filesystem::addCleanedPath($pluginPath, Filesystem::CLEAN_PATH_PLUGINS_PREFIX);
260
261 if(!file_exists($dataPath)){
262 mkdir($dataPath, 0777, true);
263 }
264
265 $lockFilePath = Path::join($dataPath, 'server.lock');
266 if(($pid = Filesystem::createLockFile($lockFilePath)) !== null){
267 critical_error("Another " . VersionInfo::NAME . " instance (PID $pid) is already using this folder (" . realpath($dataPath) . ").");
268 critical_error("Please stop the other server first before running a new one.");
269 exit(1);
270 }
271
272 //Logger has a dependency on timezone
273 Timezone::init();
274
275 if(isset($opts["enable-ansi"])){
276 Terminal::init(true);
277 }elseif(isset($opts["disable-ansi"])){
278 Terminal::init(false);
279 }else{
280 Terminal::init();
281 }
282
283 $logger = new MainLogger(Path::join($dataPath, "server.log"), Terminal::hasFormattingCodes(), "Server", new \DateTimeZone(Timezone::get()));
284 \GlobalLogger::set($logger);
285
286 emit_performance_warnings($logger);
287
288 $exitCode = 0;
289 do{
290 if(!file_exists(Path::join($dataPath, "server.properties")) && !isset($opts["no-wizard"])){
291 $installer = new SetupWizard($dataPath);
292 if(!$installer->run()){
293 $exitCode = -1;
294 break;
295 }
296 }
297
298 /*
299 * We now use the Composer autoloader, but this autoloader is still for loading plugins.
300 */
301 $autoloader = new \BaseClassLoader();
302 $autoloader->register(false);
303
304 new Server($autoloader, $logger, $dataPath, $pluginPath);
305
306 $logger->info("Stopping other threads");
307
308 $killer = new ServerKiller(8);
309 $killer->start(PTHREADS_INHERIT_NONE);
310 usleep(10000); //Fixes ServerKiller not being able to start on single-core machines
311
312 if(ThreadManager::getInstance()->stopAll() > 0){
313 $logger->debug("Some threads could not be stopped, performing a force-kill");
314 Process::kill(Process::pid(), true);
315 }
316 }while(false);
317
318 $logger->shutdownLogWriterThread();
319
320 echo Terminal::$FORMAT_RESET . PHP_EOL;
321
322 Filesystem::releaseLockFile($lockFilePath);
323
324 exit($exitCode);
325 }
