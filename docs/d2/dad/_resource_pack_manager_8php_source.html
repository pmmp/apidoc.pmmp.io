63 public function __construct(string $path, \Logger $logger){
64 $this->path = $path;
65
66 if(!file_exists($this->path)){
67 $logger->debug("Resource packs path $path does not exist, creating directory");
68 mkdir($this->path);
69 }elseif(!is_dir($this->path)){
70 throw new \InvalidArgumentException("Resource packs path $path exists and is not a directory");
71 }
72
73 $resourcePacksYml = Path::join($this->path, "resource_packs.yml");
74 if(!file_exists($resourcePacksYml)){
75 copy(Path::join(\pocketmine\RESOURCE_PATH, "resource_packs.yml"), $resourcePacksYml);
76 }
77
78 $resourcePacksConfig = new Config($resourcePacksYml, Config::YAML, []);
79
80 $this->serverForceResources = (bool) $resourcePacksConfig->get("force_resources", false);
81
82 $logger->info("Loading resource packs...");
83
84 $resourceStack = $resourcePacksConfig->get("resource_stack", []);
85 if(!is_array($resourceStack)){
86 throw new \InvalidArgumentException("\"resource_stack\" key should contain a list of pack names");
87 }
88
89 foreach($resourceStack as $pos => $pack){
90 if(!is_string($pack) && !is_int($pack) && !is_float($pack)){
91 $logger->critical("Found invalid entry in resource pack list at offset $pos of type " . gettype($pack));
92 continue;
93 }
94 $pack = (string) $pack;
95 try{
96 $packPath = Path::join($this->path, $pack);
97 if(!file_exists($packPath)){
98 throw new ResourcePackException("File or directory not found");
99 }
100 if(is_dir($packPath)){
101 throw new ResourcePackException("Directory resource packs are unsupported");
102 }
103
104 $newPack = null;
105 //Detect the type of resource pack.
106 $info = new \SplFileInfo($packPath);
107 switch($info->getExtension()){
108 case "zip":
109 case "mcpack":
110 $newPack = new ZippedResourcePack($packPath);
111 break;
112 }
113
114 if($newPack instanceof ResourcePack){
115 $this->resourcePacks[] = $newPack;
116 $index = strtolower($newPack->getPackId());
117 $this->uuidList[$index] = $newPack;
118
119 $keyPath = Path::join($this->path, $pack . ".key");
120 if(file_exists($keyPath)){
121 try{
122 $this->encryptionKeys[$index] = ErrorToExceptionHandler::trapAndRemoveFalse(
123 fn() => file_get_contents($keyPath)
124 );
125 }catch(\ErrorException $e){
126 throw new ResourcePackException("Could not read encryption key file: " . $e->getMessage(), 0, $e);
127 }
128 }
129 }else{
130 throw new ResourcePackException("Format not recognized");
131 }
132 }catch(ResourcePackException $e){
133 $logger->critical("Could not load resource pack \"$pack\": " . $e->getMessage());
134 }
135 }
136
137 $logger->debug("Successfully loaded " . count($this->resourcePacks) . " resource packs");
138 }
