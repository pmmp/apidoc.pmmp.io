55 public function __construct(string $path, \Logger $logger){
56 $this->path = $path;
57
58 if(!file_exists($this->path)){
59 $logger->debug("Resource packs path $path does not exist, creating directory");
60 mkdir($this->path);
61 }elseif(!is_dir($this->path)){
62 throw new \InvalidArgumentException("Resource packs path $path exists and is not a directory");
63 }
64
65 $resourcePacksYml = Path::join($this->path, "resource_packs.yml");
66 if(!file_exists($resourcePacksYml)){
67 copy(Path::join(\pocketmine\RESOURCE_PATH, "resource_packs.yml"), $resourcePacksYml);
68 }
69
70 $resourcePacksConfig = new Config($resourcePacksYml, Config::YAML, []);
71
72 $this->serverForceResources = (bool) $resourcePacksConfig->get("force_resources", false);
73
74 $logger->info("Loading resource packs...");
75
76 $resourceStack = $resourcePacksConfig->get("resource_stack", []);
77 if(!is_array($resourceStack)){
78 throw new \InvalidArgumentException("\"resource_stack\" key should contain a list of pack names");
79 }
80
81 foreach($resourceStack as $pos => $pack){
82 if(!is_string($pack) && !is_int($pack) && !is_float($pack)){
83 $logger->critical("Found invalid entry in resource pack list at offset $pos of type " . gettype($pack));
84 continue;
85 }
86 $pack = (string) $pack;
87 try{
88 $packPath = Path::join($this->path, $pack);
89 if(!file_exists($packPath)){
90 throw new ResourcePackException("File or directory not found");
91 }
92 if(is_dir($packPath)){
93 throw new ResourcePackException("Directory resource packs are unsupported");
94 }
95
96 $newPack = null;
97 //Detect the type of resource pack.
98 $info = new \SplFileInfo($packPath);
99 switch($info->getExtension()){
100 case "zip":
101 case "mcpack":
102 $newPack = new ZippedResourcePack($packPath);
103 break;
104 }
105
106 if($newPack instanceof ResourcePack){
107 $this->resourcePacks[] = $newPack;
108 $this->uuidList[strtolower($newPack->getPackId())] = $newPack;
109 }else{
110 throw new ResourcePackException("Format not recognized");
111 }
112 }catch(ResourcePackException $e){
113 $logger->critical("Could not load resource pack \"$pack\": " . $e->getMessage());
114 }
115 }
116
117 $logger->debug("Successfully loaded " . count($this->resourcePacks) . " resource packs");
118 }
