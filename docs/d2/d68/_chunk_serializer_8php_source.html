54 public static function getSubChunkCount(Chunk $chunk) : int{
55 for($y = Chunk::MAX_SUBCHUNK_INDEX, $count = count($chunk->getSubChunks()); $y >= Chunk::MIN_SUBCHUNK_INDEX; --$y, --$count){
56 if($chunk->getSubChunk($y)->isEmptyFast()){
57 continue;
58 }
59 return $count;
60 }
61
62 return 0;
63 }

65 public static function serializeFullChunk(Chunk $chunk, RuntimeBlockMapping $blockMapper, PacketSerializerContext $encoderContext, ?string $tiles = null) : string{
66 $stream = PacketSerializer::encoder($encoderContext);
67
68 //TODO: HACK! fill in fake subchunks to make up for the new negative space client-side
69 for($y = 0; $y < self::LOWER_PADDING_SIZE; $y++){
70 $stream->putByte(8); //subchunk version 8
71 $stream->putByte(0); //0 layers - client will treat this as all-air
72 }
73
74 $subChunkCount = self::getSubChunkCount($chunk);
75 for($y = Chunk::MIN_SUBCHUNK_INDEX, $writtenCount = 0; $writtenCount < $subChunkCount; ++$y, ++$writtenCount){
76 self::serializeSubChunk($chunk->getSubChunk($y), $blockMapper, $stream, false);
77 }
78
79 //TODO: right now we don't support 3D natively, so we just 3Dify our 2D biomes so they fill the column
80 $encodedBiomePalette = self::serializeBiomesAsPalette($chunk);
81 $stream->put(str_repeat($encodedBiomePalette, 25));
82
83 $stream->putByte(0); //border block array count
84 //Border block entry format: 1 byte (4 bits X, 4 bits Z). These are however useless since they crash the regular client.
85
86 if($tiles !== null){
87 $stream->put($tiles);
88 }else{
89 $stream->put(self::serializeTiles($chunk));
90 }
91 return $stream->getBuffer();
92 }

94 public static function serializeSubChunk(SubChunk $subChunk, RuntimeBlockMapping $blockMapper, PacketSerializer $stream, bool $persistentBlockStates) : void{
95 $layers = $subChunk->getBlockLayers();
96 $stream->putByte(8); //version
97
98 $stream->putByte(count($layers));
99
100 foreach($layers as $blocks){
101 $bitsPerBlock = $blocks->getBitsPerBlock();
102 $words = $blocks->getWordArray();
103 $stream->putByte(($bitsPerBlock << 1) | ($persistentBlockStates ? 0 : 1));
104 $stream->put($words);
105 $palette = $blocks->getPalette();
106
107 if($bitsPerBlock !== 0){
108 //these LSHIFT by 1 uvarints are optimizations: the client expects zigzag varints here
109 //but since we know they are always unsigned, we can avoid the extra fcall overhead of
110 //zigzag and just shift directly.
111 $stream->putUnsignedVarInt(count($palette) << 1); //yes, this is intentionally zigzag
112 }
113 if($persistentBlockStates){
114 $nbtSerializer = new NetworkNbtSerializer();
115 foreach($palette as $p){
116 $stream->put($nbtSerializer->write(new TreeRoot($blockMapper->getBedrockKnownStates()[$blockMapper->toRuntimeId($p)])));
117 }
118 }else{
119 foreach($palette as $p){
120 $stream->put(Binary::writeUnsignedVarInt($blockMapper->toRuntimeId($p) << 1));
121 }
122 }
123 }
124 }

128 foreach($chunk->getTiles() as $tile){
129 if($tile instanceof Spawnable){

131 }
132 }

137 private static function serializeBiomesAsPalette(Chunk $chunk) : string{
138 $biomeIdMap = LegacyBiomeIdToStringIdMap::getInstance();
139 $biomePalette = new PalettedBlockArray($chunk->getBiomeId(0, 0));
140 for($x = 0; $x < 16; ++$x){
141 for($z = 0; $z < 16; ++$z){
142 $biomeId = $chunk->getBiomeId($x, $z);
143 if($biomeIdMap->legacyToString($biomeId) === null){
144 //make sure we aren't sending bogus biomes - the 1.18.0 client crashes if we do this
145 $biomeId = BiomeIds::OCEAN;
146 }
147 for($y = 0; $y < 16; ++$y){
148 $biomePalette->set($x, $y, $z, $biomeId);
149 }
150 }
151 }
152
153 $biomePaletteBitsPerBlock = $biomePalette->getBitsPerBlock();
154 $encodedBiomePalette =
155 chr(($biomePaletteBitsPerBlock << 1) | 1) . //the last bit is non-persistence (like for blocks), though it has no effect on biomes since they always use integer IDs
156 $biomePalette->getWordArray();
157
158 //these LSHIFT by 1 uvarints are optimizations: the client expects zigzag varints here
159 //but since we know they are always unsigned, we can avoid the extra fcall overhead of
160 //zigzag and just shift directly.
161 $biomePaletteArray = $biomePalette->getPalette();
162 if($biomePaletteBitsPerBlock !== 0){
163 $encodedBiomePalette .= Binary::writeUnsignedVarInt(count($biomePaletteArray) << 1);
164 }
165 foreach($biomePaletteArray as $p){
166 $encodedBiomePalette .= Binary::writeUnsignedVarInt($p << 1);
167 }
168
169 return $encodedBiomePalette;
170 }
