101 public static function getServiceUrl() : string{
102 $socket = Utils::assumeNotFalse(@socket_create(AF_INET, SOCK_DGRAM, SOL_UDP), fn() => "Socket error: " . trim(socket_strerror(socket_last_error())));
103 Utils::assumeNotFalse(@socket_set_option($socket, SOL_SOCKET, SO_RCVTIMEO, ["sec" => 3, "usec" => 0]), "Socket error: " . trim(socket_strerror(socket_last_error($socket))));
104 $contents =
105 "M-SEARCH * HTTP/1.1\r\n" .
106 "MX: 2\r\n" .
107 "HOST: 239.255.255.250:1900\r\n" .
108 "MAN: \"ssdp:discover\"\r\n" .
109 "ST: upnp:rootdevice\r\n\r\n";
110 $location = null;
111 for($i = 0; $i < self::MAX_DISCOVERY_ATTEMPTS; ++$i){
112 $sendbyte = @socket_sendto($socket, $contents, strlen($contents), 0, "239.255.255.250", 1900);
113 if($sendbyte === false){
114 throw new UPnPException("Socket error: " . trim(socket_strerror(socket_last_error($socket))));
115 }
116 if($sendbyte !== strlen($contents)){
117 throw new UPnPException("Socket error: Unable to send the entire contents.");
118 }
119 while(true){
120 if(@socket_recvfrom($socket, $buffer, 1024, 0, $responseHost, $responsePort) === false){
121 if(socket_last_error($socket) === SOCKET_ETIMEDOUT){
122 continue 2;
123 }
124 throw new UPnPException("Socket error: " . trim(socket_strerror(socket_last_error($socket))));
125 }
126 $pregResult = preg_match('/location\s*:\s*(.+)\n/i', $buffer, $matches);
127 if($pregResult === false){
128 //TODO: replace with preg_last_error_msg() in PHP 8.
129 throw self::makePcreError();
130 }
131 if($pregResult !== 0){ //this might be garbage from somewhere other than the router
132 $location = trim($matches[1]);
133 break 2;
134 }
135 }
136 }
137 socket_close($socket);
138 if($location === null){
139 throw new UPnPException("Unable to find the router. Ensure that network discovery is enabled in Control Panel.");
140 }
141 $url = parse_url($location);
142 if($url === false){
143 throw new UPnPException("Failed to parse the router's url: {$location}");
144 }
145 if(!isset($url['host'])){
146 throw new UPnPException("Failed to recognize the host name from the router's url: {$location}");
147 }
148 $urlHost = $url['host'];
149 if(!isset($url['port'])){
150 throw new UPnPException("Failed to recognize the port number from the router's url: {$location}");
151 }
152 $urlPort = $url['port'];
153 $response = Internet::getURL($location, 3, [], $err);
154 if($response === null){
155 throw new UPnPException("Unable to access XML: {$err}");
156 }
157 if($response->getCode() !== 200){
158 throw new UPnPException("Unable to access XML: {$response->getBody()}");
159 }
160
161 $defaultInternalError = libxml_use_internal_errors(true);
162 try{
163 $root = new \SimpleXMLElement($response->getBody());
164 }catch(\Exception $e){
165 throw new UPnPException("Broken XML.");
166 }
167 libxml_use_internal_errors($defaultInternalError);
168 $root->registerXPathNamespace("upnp", "urn:schemas-upnp-org:device-1-0");
169 $xpathResult = Utils::assumeNotFalse($root->xpath(
170 '//upnp:device[upnp:deviceType="urn:schemas-upnp-org:device:InternetGatewayDevice:1"]' .
171 '/upnp:deviceList/upnp:device[upnp:deviceType="urn:schemas-upnp-org:device:WANDevice:1"]' .
172 '/upnp:deviceList/upnp:device[upnp:deviceType="urn:schemas-upnp-org:device:WANConnectionDevice:1"]' .
173 '/upnp:serviceList/upnp:service[upnp:serviceType="urn:schemas-upnp-org:service:WANIPConnection:1"]' .
174 '/upnp:controlURL'
175 ), "xpath query is borked");
176
177 if($xpathResult === null || count($xpathResult) === 0){
178 throw new UPnPException("Your router does not support portforwarding");
179 }
180 $controlURL = (string) $xpathResult[0];
181 $serviceURL = sprintf("%s:%d/%s", $urlHost, $urlPort, $controlURL);
182 return $serviceURL;
183 }

188 public static function portForward(string $serviceURL, string $internalIP, int $internalPort, int $externalPort) : void{
189 $body =
190 '<u:AddPortMapping xmlns:u="urn:schemas-upnp-org:service:WANIPConnection:1">' .
191 '<NewRemoteHost></NewRemoteHost>' .
192 '<NewExternalPort>' . $externalPort . '</NewExternalPort>' .
193 '<NewProtocol>UDP</NewProtocol>' .
194 '<NewInternalPort>' . $internalPort . '</NewInternalPort>' .
195 '<NewInternalClient>' . $internalIP . '</NewInternalClient>' .
196 '<NewEnabled>1</NewEnabled>' .
197 '<NewPortMappingDescription>PocketMine-MP</NewPortMappingDescription>' .
198 '<NewLeaseDuration>0</NewLeaseDuration>' .
199 '</u:AddPortMapping>';
200
201 $contents =
202 '<?xml version="1.0"?>' .
203 '<s:Envelope xmlns:s="http://schemas.xmlsoap.org/soap/envelope/" s:encodingStyle="http://schemas.xmlsoap.org/soap/encoding/">' .
204 '<s:Body>' . $body . '</s:Body></s:Envelope>';
205
206 $headers = [
207 'Content-Type: text/xml',
208 'SOAPAction: "urn:schemas-upnp-org:service:WANIPConnection:1#AddPortMapping"'
209 ];
210
211 if(Internet::postURL($serviceURL, $contents, 3, $headers, $err) === null){
212 throw new UPnPException("Failed to portforward using UPnP: " . $err);
213 }
214 }
