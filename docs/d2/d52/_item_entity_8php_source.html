80 protected function initEntity(CompoundTag $nbt) : void{
81 parent::initEntity($nbt);
82
83 $this->setMaxHealth(5);
84 $this->setHealth($nbt->getShort("Health", (int) $this->getHealth()));
85
86 $age = $nbt->getShort("Age", 0);
87 if($age === -32768){
88 $this->despawnDelay = self::NEVER_DESPAWN;
89 }else{
90 $this->despawnDelay = max(0, self::DEFAULT_DESPAWN_DELAY - $age);
91 }
92 $this->pickupDelay = $nbt->getShort("PickupDelay", $this->pickupDelay);
93 $this->owner = $nbt->getString("Owner", $this->owner);

99 protected function entityBaseTick(int $tickDiff = 1) : bool{
100 if($this->closed){
101 return false;
102 }
103
104 $hasUpdate = parent::entityBaseTick($tickDiff);
105
106 if(!$this->isFlaggedForDespawn() && $this->pickupDelay !== self::NEVER_DESPAWN){ //Infinite delay
107 $this->pickupDelay -= $tickDiff;
108 if($this->pickupDelay < 0){
109 $this->pickupDelay = 0;
110 }
111 if($this->hasMovementUpdate() && $this->despawnDelay % self::MERGE_CHECK_PERIOD === 0){
112 $mergeable = [$this]; //in case the merge target ends up not being this
113 $mergeTarget = $this;
114 foreach($this->getWorld()->getNearbyEntities($this->boundingBox->expandedCopy(0.5, 0.5, 0.5), $this) as $entity){
115 if(!$entity instanceof ItemEntity || $entity->isFlaggedForDespawn()){
116 continue;
117 }
118
119 if($entity->isMergeable($this)){
120 $mergeable[] = $entity;
121 if($entity->item->getCount() > $mergeTarget->item->getCount()){
122 $mergeTarget = $entity;
123 }
124 }
125 }
126 foreach($mergeable as $itemEntity){
127 if($itemEntity !== $mergeTarget){
128 $itemEntity->tryMergeInto($mergeTarget);
129 }
130 }
131 }
132
133 $this->despawnDelay -= $tickDiff;
134 if($this->despawnDelay <= 0){
135 $ev = new ItemDespawnEvent($this);
136 $ev->call();
137 if($ev->isCancelled()){
138 $this->despawnDelay = self::DEFAULT_DESPAWN_DELAY;
139 }else{
140 $this->flagForDespawn();
141 $hasUpdate = true;
142 }
143 }
144 }
145
146 return $hasUpdate;
147 }

152 public function isMergeable(ItemEntity $entity) : bool{
153 $item = $entity->item;
154 return $entity !== $this && $entity->pickupDelay !== self::NEVER_DESPAWN && $item->canStackWith($this->item) && $item->getCount() + $this->item->getCount() <= $item->getMaxStackSize();
155 }

160 public function tryMergeInto(ItemEntity $consumer) : bool{
161 if(!$this->isMergeable($consumer)){
162 return false;
163 }
164
165 $ev = new ItemMergeEvent($this, $consumer);
166 $ev->call();
167
168 if($ev->isCancelled()){
169 return false;
170 }
171
172 $consumer->setStackSize($consumer->item->getCount() + $this->item->getCount());
173 $this->flagForDespawn();
174 $consumer->pickupDelay = max($consumer->pickupDelay, $this->pickupDelay);
175 $consumer->despawnDelay = max($consumer->despawnDelay, $this->despawnDelay);
176
177 return true;
178 }

287 public function onCollideWithPlayer(Player $player) : void{
288 if($this->getPickupDelay() !== 0){
289 return;
290 }
291
292 $item = $this->getItem();
293 $playerInventory = match(true){
294 $player->getOffHandInventory()->getItem(0)->canStackWith($item) && $player->getOffHandInventory()->getAddableItemQuantity($item) > 0 => $player->getOffHandInventory(),
295 $player->getInventory()->getAddableItemQuantity($item) > 0 => $player->getInventory(),
296 default => null
297 };
298
299 $ev = new EntityItemPickupEvent($player, $this, $item, $playerInventory);
300 if($player->hasFiniteResources() && $playerInventory === null){
301 $ev->cancel();
302 }
303
304 $ev->call();
305 if($ev->isCancelled()){
306 return;
307 }
308
309 foreach($this->getViewers() as $viewer){
310 $viewer->getNetworkSession()->onPlayerPickUpItem($player, $this);
311 }
312
313 $inventory = $ev->getInventory();
314 if($inventory !== null){
315 foreach($inventory->addItem($ev->getItem()) as $remains){
316 $this->getWorld()->dropItem($this->location, $remains, new Vector3(0, 0, 0));
317 }
318 }
319 $this->flagForDespawn();
320 }
