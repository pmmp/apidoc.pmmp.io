109 public function onRandomTick() : void{
110 if(!$this->getSide(Facing::DOWN)->isSameType($this)){
111 $world = $this->position->getWorld();
112 if($this->age === self::MAX_AGE){
113 for($y = 1; $y < 3; ++$y){
114 if(!$world->isInWorld($this->position->x, $this->position->y + $y, $this->position->z)){
115 break;
116 }
117 $b = $world->getBlockAt($this->position->x, $this->position->y + $y, $this->position->z);
118 if($b->getId() === BlockLegacyIds::AIR){
119 $ev = new BlockGrowEvent($b, VanillaBlocks::CACTUS());
120 $ev->call();
121 if($ev->isCancelled()){
122 break;
123 }
124 $world->setBlock($b->position, $ev->getNewState());
125 }else{
126 break;
127 }
128 }
129 $this->age = 0;
130 $world->setBlock($this->position, $this);
131 }else{
132 ++$this->age;
133 $world->setBlock($this->position, $this);
134 }
135 }
136 }
