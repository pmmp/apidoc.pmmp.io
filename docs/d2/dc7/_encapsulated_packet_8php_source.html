52 public static function fromBinary(BinaryStream $stream) : EncapsulatedPacket{
53 $packet = new EncapsulatedPacket();
54
55 $flags = $stream->getByte();
56 $packet->reliability = $reliability = ($flags & self::RELIABILITY_FLAGS) >> self::RELIABILITY_SHIFT;
57 $hasSplit = ($flags & self::SPLIT_FLAG) > 0;
58
59 $length = (int) ceil($stream->getShort() / 8);
60 if($length === 0){
61 throw new BinaryDataException("Encapsulated payload length cannot be zero");
62 }
63
64 if(PacketReliability::isReliable($reliability)){
65 $packet->messageIndex = $stream->getLTriad();
66 }
67
68 if(PacketReliability::isSequenced($reliability)){
69 $packet->sequenceIndex = $stream->getLTriad();
70 }
71
72 if(PacketReliability::isSequencedOrOrdered($reliability)){
73 $packet->orderIndex = $stream->getLTriad();
74 $packet->orderChannel = $stream->getByte();
75 }
76
77 if($hasSplit){
78 $splitCount = $stream->getInt();
79 $splitID = $stream->getShort();
80 $splitIndex = $stream->getInt();
81 $packet->splitInfo = new SplitPacketInfo($splitID, $splitIndex, $splitCount);
82 }
83
84 $packet->buffer = $stream->get($length);
85 return $packet;
86 }
87
88 public function toBinary() : string{
89 return
90 chr(($this->reliability << self::RELIABILITY_SHIFT) | ($this->splitInfo !== null ? self::SPLIT_FLAG : 0)) .
91 Binary::writeShort(strlen($this->buffer) << 3) .
92 (PacketReliability::isReliable($this->reliability) ? Binary::writeLTriad($this->messageIndex) : "") .
93 (PacketReliability::isSequenced($this->reliability) ? Binary::writeLTriad($this->sequenceIndex) : "") .
94 (PacketReliability::isSequencedOrOrdered($this->reliability) ? Binary::writeLTriad($this->orderIndex) . chr($this->orderChannel) : "") .
95 ($this->splitInfo !== null ? Binary::writeInt($this->splitInfo->getTotalPartCount()) . Binary::writeShort($this->splitInfo->getId()) . Binary::writeInt($this->splitInfo->getPartIndex()) : "")
96 . $this->buffer;
97 }
98
99 public function getTotalLength() : int{
100 return
101 1 + //reliability
102 2 + //length
103 (PacketReliability::isReliable($this->reliability) ? 3 : 0) + //message index
104 (PacketReliability::isSequenced($this->reliability) ? 3 : 0) + //sequence index
105 (PacketReliability::isSequencedOrOrdered($this->reliability) ? 3 + 1 : 0) + //order index (3) + order channel (1)
106 ($this->splitInfo !== null ? 4 + 2 + 4 : 0) + //split count (4) + split ID (2) + split index (4)
107 strlen($this->buffer);
108 }
109
110 public function __toString() : string{
111 return $this->toBinary();
112 }
