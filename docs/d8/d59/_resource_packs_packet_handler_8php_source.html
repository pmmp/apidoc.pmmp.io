133 public function handleResourcePackClientResponse(ResourcePackClientResponsePacket $packet) : bool{
134 switch($packet->status){
135 case ResourcePackClientResponsePacket::STATUS_REFUSED:
136 //TODO: add lang strings for this
137 $this->session->disconnect("Refused resource packs", "You must accept resource packs to join this server.", true);
138 break;
139 case ResourcePackClientResponsePacket::STATUS_SEND_PACKS:
140 foreach($packet->packIds as $uuid){
141 //dirty hack for mojang's dirty hack for versions
142 $splitPos = strpos($uuid, "_");
143 if($splitPos !== false){
144 $uuid = substr($uuid, 0, $splitPos);
145 }
146 $pack = $this->getPackById($uuid);
147
148 if(!($pack instanceof ResourcePack)){
149 //Client requested a resource pack but we don't have it available on the server
150 $this->disconnectWithError("Unknown pack $uuid requested, available packs: " . implode(", ", array_keys($this->resourcePacksById)));
151 return false;
152 }
153
154 $this->session->sendDataPacket(ResourcePackDataInfoPacket::create(
155 $pack->getPackId(),
156 self::PACK_CHUNK_SIZE,
157 (int) ceil($pack->getPackSize() / self::PACK_CHUNK_SIZE),
158 $pack->getPackSize(),
159 $pack->getSha256(),
160 false,
161 ResourcePackType::RESOURCES //TODO: this might be an addon (not behaviour pack), needed to properly support client-side custom items
162 ));
163 }
164 $this->session->getLogger()->debug("Player requested download of " . count($packet->packIds) . " resource packs");
165
166 break;
167 case ResourcePackClientResponsePacket::STATUS_HAVE_ALL_PACKS:
168 $stack = array_map(static function(ResourcePack $pack) : ResourcePackStackEntry{
169 return new ResourcePackStackEntry($pack->getPackId(), $pack->getPackVersion(), ""); //TODO: subpacks
170 }, $this->resourcePackStack);
171
172 //we support chemistry blocks by default, the client should already have this installed
173 $stack[] = new ResourcePackStackEntry("0fba4063-dba1-4281-9b89-ff9390653530", "1.0.0", "");
174
175 //we don't force here, because it doesn't have user-facing effects
176 //but it does have an annoying side-effect when true: it makes
177 //the client remove its own non-server-supplied resource packs.
178 $this->session->sendDataPacket(ResourcePackStackPacket::create($stack, [], false, ProtocolInfo::MINECRAFT_VERSION_NETWORK, new Experiments([], false), false));
179 $this->session->getLogger()->debug("Applying resource pack stack");
180 break;
181 case ResourcePackClientResponsePacket::STATUS_COMPLETED:
182 $this->session->getLogger()->debug("Resource packs sequence completed");
183 ($this->completionCallback)();
184 break;
185 default:
186 return false;
187 }
188
189 return true;
190 }

192 public function handleResourcePackChunkRequest(ResourcePackChunkRequestPacket $packet) : bool{
193 $pack = $this->getPackById($packet->packId);
194 if(!($pack instanceof ResourcePack)){
195 $this->disconnectWithError("Invalid request for chunk $packet->chunkIndex of unknown pack $packet->packId, available packs: " . implode(", ", array_keys($this->resourcePacksById)));
196 return false;
197 }
198
199 $packId = $pack->getPackId(); //use this because case may be different
200
201 if(isset($this->downloadedChunks[$packId][$packet->chunkIndex])){
202 $this->disconnectWithError("Duplicate request for chunk $packet->chunkIndex of pack $packet->packId");
203 return false;
204 }
205
206 $offset = $packet->chunkIndex * self::PACK_CHUNK_SIZE;
207 if($offset < 0 || $offset >= $pack->getPackSize()){
208 $this->disconnectWithError("Invalid out-of-bounds request for chunk $packet->chunkIndex of $packet->packId: offset $offset, file size " . $pack->getPackSize());
209 return false;
210 }
211
212 if(!isset($this->downloadedChunks[$packId])){
213 $this->downloadedChunks[$packId] = [$packet->chunkIndex => true];
214 }else{
215 $this->downloadedChunks[$packId][$packet->chunkIndex] = true;
216 }
217
218 $this->requestQueue->enqueue([$pack, $packet->chunkIndex]);
219 $this->processChunkRequestQueue();
220
221 return true;
222 }
223
224 private function processChunkRequestQueue() : void{
225 if($this->activeRequests >= self::MAX_CONCURRENT_CHUNK_REQUESTS || $this->requestQueue->isEmpty()){
226 return;
227 }
232 [$pack, $chunkIndex] = $this->requestQueue->dequeue();
233
234 $packId = $pack->getPackId();
235 $offset = $chunkIndex * self::PACK_CHUNK_SIZE;
236 $chunkData = $pack->getPackChunk($offset, self::PACK_CHUNK_SIZE);
237 $this->activeRequests++;
238 $this->session
239 ->sendDataPacketWithReceipt(ResourcePackChunkDataPacket::create($packId, $chunkIndex, $offset, $chunkData))
240 ->onCompletion(
241 function() : void{
242 $this->activeRequests--;
243 $this->processChunkRequestQueue();
244 },
245 function() : void{
246 //this may have been rejected because of a disconnection - this will do nothing in that case
247 $this->disconnectWithError("Plugin interrupted sending of resource packs");
248 }
249 );
250 }
