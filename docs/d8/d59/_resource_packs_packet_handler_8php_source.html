82 public function handleResourcePackClientResponse(ResourcePackClientResponsePacket $packet) : bool{
83 switch($packet->status){
84 case ResourcePackClientResponsePacket::STATUS_REFUSED:
85 //TODO: add lang strings for this
86 $this->session->disconnect("You must accept resource packs to join this server.", true);
87 break;
88 case ResourcePackClientResponsePacket::STATUS_SEND_PACKS:
89 foreach($packet->packIds as $uuid){
90 //dirty hack for mojang's dirty hack for versions
91 $splitPos = strpos($uuid, "_");
92 if($splitPos !== false){
93 $uuid = substr($uuid, 0, $splitPos);
94 }
95 $pack = $this->resourcePackManager->getPackById($uuid);
96
97 if(!($pack instanceof ResourcePack)){
98 //Client requested a resource pack but we don't have it available on the server
99 $this->disconnectWithError("Unknown pack $uuid requested, available packs: " . implode(", ", $this->resourcePackManager->getPackIdList()));
100 return false;
101 }
102
103 $this->session->sendDataPacket(ResourcePackDataInfoPacket::create(
104 $pack->getPackId(),
105 self::PACK_CHUNK_SIZE,
106 (int) ceil($pack->getPackSize() / self::PACK_CHUNK_SIZE),
107 $pack->getPackSize(),
108 $pack->getSha256(),
109 false,
110 ResourcePackType::RESOURCES //TODO: this might be an addon (not behaviour pack), needed to properly support client-side custom items
111 ));
112 }
113 $this->session->getLogger()->debug("Player requested download of " . count($packet->packIds) . " resource packs");
114
115 break;
116 case ResourcePackClientResponsePacket::STATUS_HAVE_ALL_PACKS:
117 $stack = array_map(static function(ResourcePack $pack) : ResourcePackStackEntry{
118 return new ResourcePackStackEntry($pack->getPackId(), $pack->getPackVersion(), ""); //TODO: subpacks
119 }, $this->resourcePackManager->getResourceStack());
120
121 //we support chemistry blocks by default, the client should already have this installed
122 $stack[] = new ResourcePackStackEntry("0fba4063-dba1-4281-9b89-ff9390653530", "1.0.0", "");
123
124 //we don't force here, because it doesn't have user-facing effects
125 //but it does have an annoying side-effect when true: it makes
126 //the client remove its own non-server-supplied resource packs.
127 $this->session->sendDataPacket(ResourcePackStackPacket::create($stack, [], false, ProtocolInfo::MINECRAFT_VERSION_NETWORK, new Experiments([], false)));
128 $this->session->getLogger()->debug("Applying resource pack stack");
129 break;
130 case ResourcePackClientResponsePacket::STATUS_COMPLETED:
131 $this->session->getLogger()->debug("Resource packs sequence completed");
132 ($this->completionCallback)();
133 break;
134 default:
135 return false;
136 }
137
138 return true;
139 }

141 public function handleResourcePackChunkRequest(ResourcePackChunkRequestPacket $packet) : bool{
142 $pack = $this->resourcePackManager->getPackById($packet->packId);
143 if(!($pack instanceof ResourcePack)){
144 $this->disconnectWithError("Invalid request for chunk $packet->chunkIndex of unknown pack $packet->packId, available packs: " . implode(", ", $this->resourcePackManager->getPackIdList()));
145 return false;
146 }
147
148 $packId = $pack->getPackId(); //use this because case may be different
149
150 if(isset($this->downloadedChunks[$packId][$packet->chunkIndex])){
151 $this->disconnectWithError("Duplicate request for chunk $packet->chunkIndex of pack $packet->packId");
152 return false;
153 }
154
155 $offset = $packet->chunkIndex * self::PACK_CHUNK_SIZE;
156 if($offset < 0 || $offset >= $pack->getPackSize()){
157 $this->disconnectWithError("Invalid out-of-bounds request for chunk $packet->chunkIndex of $packet->packId: offset $offset, file size " . $pack->getPackSize());
158 return false;
159 }
160
161 if(!isset($this->downloadedChunks[$packId])){
162 $this->downloadedChunks[$packId] = [$packet->chunkIndex => true];
163 }else{
164 $this->downloadedChunks[$packId][$packet->chunkIndex] = true;
165 }
166
167 $this->session->sendDataPacket(ResourcePackChunkDataPacket::create($packId, $packet->chunkIndex, $offset, $pack->getPackChunk($offset, self::PACK_CHUNK_SIZE)));
168
169 return true;
170 }
