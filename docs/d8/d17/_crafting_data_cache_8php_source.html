58 public function getCache(CraftingManager $manager) : CraftingDataPacket{
59 $id = spl_object_id($manager);
60 if(!isset($this->caches[$id])){
61 $manager->getDestructorCallbacks()->add(function() use ($id) : void{
62 unset($this->caches[$id]);
63 });
64 $manager->getRecipeRegisteredCallbacks()->add(function() use ($id) : void{
65 unset($this->caches[$id]);
66 });
67 $this->caches[$id] = $this->buildCraftingDataCache($manager);
68 }
69 return $this->caches[$id];
70 }

75 private function buildCraftingDataCache(CraftingManager $manager) : CraftingDataPacket{
76 Timings::$craftingDataCacheRebuild->startTiming();
77
78 $counter = 0;
79 $nullUUID = Uuid::fromString(Uuid::NIL);
80 $converter = TypeConverter::getInstance();
81 $recipesWithTypeIds = [];
82 foreach($manager->getShapelessRecipes() as $list){
83 foreach($list as $recipe){
84 $recipesWithTypeIds[] = new ProtocolShapelessRecipe(
85 CraftingDataPacket::ENTRY_SHAPELESS,
86 Binary::writeInt(++$counter),
87 array_map(function(Item $item) use ($converter) : RecipeIngredient{
88 return $converter->coreItemStackToRecipeIngredient($item);
89 }, $recipe->getIngredientList()),
90 array_map(function(Item $item) use ($converter) : ItemStack{
91 return $converter->coreItemStackToNet($item);
92 }, $recipe->getResults()),
93 $nullUUID,
94 CraftingRecipeBlockName::CRAFTING_TABLE,
95 50,
96 $counter
97 );
98 }
99 }
100 foreach($manager->getShapedRecipes() as $list){
101 foreach($list as $recipe){
102 $inputs = [];
103
104 for($row = 0, $height = $recipe->getHeight(); $row < $height; ++$row){
105 for($column = 0, $width = $recipe->getWidth(); $column < $width; ++$column){
106 $inputs[$row][$column] = $converter->coreItemStackToRecipeIngredient($recipe->getIngredient($column, $row));
107 }
108 }
109 $recipesWithTypeIds[] = $r = new ProtocolShapedRecipe(
110 CraftingDataPacket::ENTRY_SHAPED,
111 Binary::writeInt(++$counter),
112 $inputs,
113 array_map(function(Item $item) use ($converter) : ItemStack{
114 return $converter->coreItemStackToNet($item);
115 }, $recipe->getResults()),
116 $nullUUID,
117 CraftingRecipeBlockName::CRAFTING_TABLE,
118 50,
119 $counter
120 );
121 }
122 }
123
124 foreach(FurnaceType::getAll() as $furnaceType){
125 $typeTag = match($furnaceType->id()){
126 FurnaceType::FURNACE()->id() => FurnaceRecipeBlockName::FURNACE,
127 FurnaceType::BLAST_FURNACE()->id() => FurnaceRecipeBlockName::BLAST_FURNACE,
128 FurnaceType::SMOKER()->id() => FurnaceRecipeBlockName::SMOKER,
129 default => throw new AssumptionFailedError("Unreachable"),
130 };
131 foreach($manager->getFurnaceRecipeManager($furnaceType)->getAll() as $recipe){
132 $input = $converter->coreItemStackToNet($recipe->getInput());
133 $recipesWithTypeIds[] = new ProtocolFurnaceRecipe(
134 CraftingDataPacket::ENTRY_FURNACE_DATA,
135 $input->getId(),
136 $input->getMeta(),
137 $converter->coreItemStackToNet($recipe->getResult()),
138 $typeTag
139 );
140 }
141 }
142
143 $potionTypeRecipes = [];
144 foreach($manager->getPotionTypeRecipes() as $recipes){
145 foreach($recipes as $recipe){
146 $input = $converter->coreItemStackToNet($recipe->getInput());
147 $ingredient = $converter->coreItemStackToNet($recipe->getIngredient());
148 $output = $converter->coreItemStackToNet($recipe->getOutput());
149 $potionTypeRecipes[] = new ProtocolPotionTypeRecipe(
150 $input->getId(),
151 $input->getMeta(),
152 $ingredient->getId(),
153 $ingredient->getMeta(),
154 $output->getId(),
155 $output->getMeta()
156 );
157 }
158 }
159
160 $potionContainerChangeRecipes = [];
161 $itemTranslator = ItemTranslator::getInstance();
162 foreach($manager->getPotionContainerChangeRecipes() as $recipes){
163 foreach($recipes as $recipe){
164 $input = $itemTranslator->toNetworkId($recipe->getInputItemId(), 0);
165 $ingredient = $itemTranslator->toNetworkId($recipe->getIngredient()->getId(), 0);
166 $output = $itemTranslator->toNetworkId($recipe->getOutputItemId(), 0);
167 $potionContainerChangeRecipes[] = new ProtocolPotionContainerChangeRecipe(
168 $input[0],
169 $ingredient[0],
170 $output[0]
171 );
172 }
173 }
174
175 Timings::$craftingDataCacheRebuild->stopTiming();
176 return CraftingDataPacket::create($recipesWithTypeIds, $potionTypeRecipes, $potionContainerChangeRecipes, [], true);
177 }
