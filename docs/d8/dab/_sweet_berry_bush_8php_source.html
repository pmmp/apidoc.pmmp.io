40class SweetBerryBush extends Flowable{
41 public const STAGE_SAPLING = 0;
42 public const STAGE_BUSH_NO_BERRIES = 1;
43 public const STAGE_BUSH_SOME_BERRIES = 2;
44 public const STAGE_MATURE = 3;
45
46 protected int $age = self::STAGE_SAPLING;
47
48 protected function writeStateToMeta() : int{
49 return $this->age;
50 }
51
52 public function readStateFromData(int $id, int $stateMeta) : void{
53 $this->age = BlockDataSerializer::readBoundedInt("stage", $stateMeta, self::STAGE_SAPLING, self::STAGE_MATURE);
54 }
55
56 public function getStateBitmask() : int{
57 return 0b111;
58 }
59
60 public function getAge() : int{ return $this->age; }
61
63 public function setAge(int $age) : self{
64 if($age < self::STAGE_SAPLING || $age > self::STAGE_MATURE){
65 throw new \InvalidArgumentException("Age must be in range 0-3");
66 }
67 $this->age = $age;
68 return $this;
69 }
70
71 public function getBerryDropAmount() : int{
72 if($this->age === self::STAGE_MATURE){
73 return mt_rand(2, 3);
74 }elseif($this->age >= self::STAGE_BUSH_SOME_BERRIES){
75 return mt_rand(1, 2);
76 }
77 return 0;
78 }
79
80 protected function canBeSupportedBy(Block $block) : bool{
81 $id = $block->getId();
82 return $id === BlockLegacyIds::GRASS || $id === BlockLegacyIds::DIRT || $id === BlockLegacyIds::PODZOL;
83 }
84
85 public function place(BlockTransaction $tx, Item $item, Block $blockReplace, Block $blockClicked, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
86 if(!$this->canBeSupportedBy($blockReplace->getSide(Facing::DOWN))){
87 return false;
88 }
89 return parent::place($tx, $item, $blockReplace, $blockClicked, $face, $clickVector, $player);
90 }
91
92 public function onInteract(Item $item, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
93 if($this->age < self::STAGE_MATURE && $item instanceof Fertilizer){
94 $block = clone $this;
95 $block->age++;
96
97 $ev = new BlockGrowEvent($this, $block);
98 $ev->call();
99
100 if(!$ev->isCancelled()){
101 $this->position->getWorld()->setBlock($this->position, $ev->getNewState());
102 $item->pop();
103 }
104
105 }elseif(($dropAmount = $this->getBerryDropAmount()) > 0){
106 $this->position->getWorld()->setBlock($this->position, $this->setAge(self::STAGE_BUSH_NO_BERRIES));
107 $this->position->getWorld()->dropItem($this->position, $this->asItem()->setCount($dropAmount));
108 }
109
110 return true;
111 }
112
113 public function asItem() : Item{
114 return VanillaItems::SWEET_BERRIES();
115 }
116
117 public function getDropsForCompatibleTool(Item $item) : array{
118 if(($dropAmount = $this->getBerryDropAmount()) > 0){
119 return [
120 $this->asItem()->setCount($dropAmount)
121 ];
122 }
123
124 return [];
125 }

138 if($this->age < self::STAGE_MATURE && mt_rand(0, 2) === 1){
139 $block = clone $this;
140 ++$block->age;
141 $ev = new BlockGrowEvent($this, $block);
142 $ev->call();
143 if(!$ev->isCancelled()){
144 $this->position->getWorld()->setBlock($this->position, $ev->getNewState());
145 }
146 }
147 }

149 public function hasEntityCollision() : bool{
150 return true;
151 }
152
153 public function onEntityInside(Entity $entity) : bool{
154 //TODO: in MCPE, this only triggers if moving while inside the bush block - we don't have the system to deal
155 //with that reliably right now
156 if($this->age >= self::STAGE_BUSH_NO_BERRIES && $entity instanceof Living){

158 }
159 return true;
160 }
