163 private function registerYamlCommands() : void{
164 $pluginCmds = [];
165
166 foreach(Utils::stringifyKeys($this->getDescription()->getCommands()) as $key => $data){
167 if(strpos($key, ":") !== false){
168 $this->logger->error($this->server->getLanguage()->translate(KnownTranslationFactory::pocketmine_plugin_commandError($key, $this->getDescription()->getFullName(), ":")));
169 continue;
170 }
171
172 $newCmd = new PluginCommand($key, $this, $this);
173 if(($description = $data->getDescription()) !== null){
174 $newCmd->setDescription($description);
175 }
176
177 if(($usageMessage = $data->getUsageMessage()) !== null){
178 $newCmd->setUsage($usageMessage);
179 }
180
181 $aliasList = [];
182 foreach($data->getAliases() as $alias){
183 if(strpos($alias, ":") !== false){
184 $this->logger->error($this->server->getLanguage()->translate(KnownTranslationFactory::pocketmine_plugin_aliasError($alias, $this->getDescription()->getFullName(), ":")));
185 continue;
186 }
187 $aliasList[] = $alias;
188 }
189
190 $newCmd->setAliases($aliasList);
191
192 $newCmd->setPermission($data->getPermission());
193
194 if(($permissionDeniedMessage = $data->getPermissionDeniedMessage()) !== null){
195 $newCmd->setPermissionMessage($permissionDeniedMessage);
196 }
197
198 $pluginCmds[] = $newCmd;
199 }
200
201 if(count($pluginCmds) > 0){
202 $this->server->getCommandMap()->registerAll($this->getDescription()->getName(), $pluginCmds);
203 }
204 }

243 public function saveResource(string $filename, bool $replace = false) : bool{
244 if(trim($filename) === ""){
245 return false;
246 }
247
248 if(($resource = $this->getResource($filename)) === null){
249 return false;
250 }
251
252 $out = Path::join($this->dataFolder, $filename);
253 if(!file_exists(dirname($out))){
254 mkdir(dirname($out), 0755, true);
255 }
256
257 if(file_exists($out) && !$replace){
258 return false;
259 }
260
261 $fp = fopen($out, "wb");
262 if($fp === false) throw new AssumptionFailedError("fopen() should not fail with wb flags");
263
264 $ret = stream_copy_to_stream($resource, $fp) > 0;
265 fclose($fp);
266 fclose($resource);
267 return $ret;
268 }
