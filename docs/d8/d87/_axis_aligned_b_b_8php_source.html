420 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
421 $v1 = $pos1->getIntermediateWithXValue($pos2, $this->minX);
422 $v2 = $pos1->getIntermediateWithXValue($pos2, $this->maxX);
423 $v3 = $pos1->getIntermediateWithYValue($pos2, $this->minY);
424 $v4 = $pos1->getIntermediateWithYValue($pos2, $this->maxY);
425 $v5 = $pos1->getIntermediateWithZValue($pos2, $this->minZ);
426 $v6 = $pos1->getIntermediateWithZValue($pos2, $this->maxZ);
427
428 if($v1 !== null and !$this->isVectorInYZ($v1)){
429 $v1 = null;
430 }
431
432 if($v2 !== null and !$this->isVectorInYZ($v2)){
433 $v2 = null;
434 }
435
436 if($v3 !== null and !$this->isVectorInXZ($v3)){
437 $v3 = null;
438 }
439
440 if($v4 !== null and !$this->isVectorInXZ($v4)){
441 $v4 = null;
442 }
443
444 if($v5 !== null and !$this->isVectorInXY($v5)){
445 $v5 = null;
446 }
447
448 if($v6 !== null and !$this->isVectorInXY($v6)){
449 $v6 = null;
450 }
451
452 $vector = null;
453 $distance = PHP_INT_MAX;
454
455 foreach([$v1, $v2, $v3, $v4, $v5, $v6] as $v){
456 if($v !== null and ($d = $pos1->distanceSquared($v)) < $distance){
457 $vector = $v;
458 $distance = $d;
459 }
460 }
461
462 if($vector === null){
463 return null;
464 }
465
466 $f = -1;
467
468 if($vector === $v1){
469 $f = Facing::WEST;
470 }elseif($vector === $v2){
471 $f = Facing::EAST;
472 }elseif($vector === $v3){
473 $f = Facing::DOWN;
474 }elseif($vector === $v4){
475 $f = Facing::UP;
476 }elseif($vector === $v5){
477 $f = Facing::NORTH;
478 }elseif($vector === $v6){
479 $f = Facing::SOUTH;
480 }
481
482 return new RayTraceResult($this, $f, $vector);
483 }
