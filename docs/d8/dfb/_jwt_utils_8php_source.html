70 public static function split(string $jwt) : array{
71 $v = explode(".", $jwt);
72 if(count($v) !== 3){
73 throw new JwtException("Expected exactly 3 JWT parts, got " . count($v));
74 }
75 return [$v[0], $v[1], $v[2]]; //workaround phpstan bug
76 }

86 public static function parse(string $token) : array{
87 $v = self::split($token);
88 $header = json_decode(self::b64UrlDecode($v[0]), true);
89 if(!is_array($header)){
90 throw new JwtException("Failed to decode JWT header JSON: " . json_last_error_msg());
91 }
92 $body = json_decode(self::b64UrlDecode($v[1]), true);
93 if(!is_array($body)){
94 throw new JwtException("Failed to decode JWT payload JSON: " . json_last_error_msg());
95 }
96 $signature = self::b64UrlDecode($v[2]);
97 return [$header, $body, $signature];
98 }

103 public static function verify(string $jwt, \OpenSSLAsymmetricKey $signingKey) : bool{
104 [$header, $body, $signature] = self::split($jwt);
105
106 $plainSignature = self::b64UrlDecode($signature);
107 if(strlen($plainSignature) !== 96){
108 throw new JwtException("JWT signature has unexpected length, expected 96, got " . strlen($plainSignature));
109 }
110
111 [$rString, $sString] = str_split($plainSignature, 48);
112 $convert = fn(string $str) => gmp_strval(gmp_import($str, 1, GMP_BIG_ENDIAN | GMP_MSW_FIRST), 10);
113
114 $sequence = new Sequence(
115 new Integer($convert($rString)),
116 new Integer($convert($sString))
117 );
118
119 $v = openssl_verify(
120 $header . '.' . $body,
121 $sequence->getBinary(),
122 $signingKey,
123 OPENSSL_ALGO_SHA384
124 );
125 switch($v){
126 case 0: return false;
127 case 1: return true;
128 case -1: throw new JwtException("Error verifying JWT signature: " . openssl_error_string());
129 default: throw new AssumptionFailedError("openssl_verify() should only return -1, 0 or 1");
130 }
131 }

137 public static function create(array $header, array $claims, \OpenSSLAsymmetricKey $signingKey) : string{
138 $jwtBody = JwtUtils::b64UrlEncode(json_encode($header, JSON_THROW_ON_ERROR)) . "." . JwtUtils::b64UrlEncode(json_encode($claims, JSON_THROW_ON_ERROR));
139
140 openssl_sign(
141 $jwtBody,
142 $rawDerSig,
143 $signingKey,
144 OPENSSL_ALGO_SHA384
145 );
146
147 try{
148 $asnObject = Sequence::fromBinary($rawDerSig);
149 }catch(ParserException $e){
150 throw new AssumptionFailedError("Failed to parse OpenSSL signature: " . $e->getMessage(), 0, $e);
151 }
152 if(count($asnObject) !== 2){
153 throw new AssumptionFailedError("OpenSSL produced invalid signature, expected exactly 2 parts");
154 }
155 [$r, $s] = [$asnObject[0], $asnObject[1]];
156 if(!($r instanceof Integer) || !($s instanceof Integer)){
157 throw new AssumptionFailedError("OpenSSL produced invalid signature, expected 2 INTEGER parts");
158 }
159 $rString = $r->getContent();
160 $sString = $s->getContent();
161
162 $toBinary = fn($str) => str_pad(
163 gmp_export(gmp_init($str, 10), 1, GMP_BIG_ENDIAN | GMP_MSW_FIRST),
164 48,
165 "\x00",
166 STR_PAD_LEFT
167 );
168 $jwtSig = JwtUtils::b64UrlEncode($toBinary($rString) . $toBinary($sString));
169
170 return "$jwtBody.$jwtSig";
171 }

173 public static function b64UrlEncode(string $str) : string{
174 return rtrim(strtr(base64_encode($str), '+/', '-_'), '=');
175 }

177 public static function b64UrlDecode(string $str) : string{
178 if(($len = strlen($str) % 4) !== 0){
179 $str .= str_repeat('=', 4 - $len);
180 }
181 $decoded = base64_decode(strtr($str, '-_', '+/'), true);
182 if($decoded === false){
183 throw new JwtException("Malformed base64url encoded payload could not be decoded");
184 }
185 return $decoded;
186 }

188 public static function emitDerPublicKey(\OpenSSLAsymmetricKey $opensslKey) : string{
189 $details = Utils::assumeNotFalse(openssl_pkey_get_details($opensslKey), "Failed to get details from OpenSSL key resource");
191 $pemKey = $details['key'];
192 if(preg_match("@^-----BEGIN[A-Z\d ]+PUBLIC KEY-----\n([A-Za-z\d+/\n]+)\n-----END[A-Z\d ]+PUBLIC KEY-----\n$@", $pemKey, $matches) === 1){
193 $derKey = base64_decode(str_replace("\n", "", $matches[1]), true);
194 if($derKey !== false){
195 return $derKey;
196 }
197 }
198 throw new AssumptionFailedError("OpenSSL resource contains invalid public key");
199 }

201 public static function parseDerPublicKey(string $derKey) : \OpenSSLAsymmetricKey{
202 $signingKeyOpenSSL = openssl_pkey_get_public(sprintf("-----BEGIN PUBLIC KEY-----\n%s\n-----END PUBLIC KEY-----\n", base64_encode($derKey)));
203 if($signingKeyOpenSSL === false){
204 throw new JwtException("OpenSSL failed to parse key: " . openssl_error_string());
205 }
206 return $signingKeyOpenSSL;
