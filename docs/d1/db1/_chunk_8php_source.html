71 public function __construct(array $subChunks, BiomeArray $biomeIds, bool $terrainPopulated){
72 $this->subChunks = new \SplFixedArray(Chunk::MAX_SUBCHUNKS);
73
74 foreach($this->subChunks as $y => $null){
75 $this->subChunks[$y] = $subChunks[$y + self::MIN_SUBCHUNK_INDEX] ?? new SubChunk(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS, []);
76 }
77
78 $val = (self::MAX_SUBCHUNK_INDEX + 1) * SubChunk::EDGE_LENGTH;
79 $this->heightMap = HeightArray::fill($val); //TODO: what about lazily initializing this?
80 $this->biomeIds = $biomeIds;

121 public function getHighestBlockAt(int $x, int $z) : ?int{
122 for($y = self::MAX_SUBCHUNK_INDEX; $y >= self::MIN_SUBCHUNK_INDEX; --$y){
123 $height = $this->getSubChunk($y)->getHighestBlockAt($x, $z);

193 public function addTile(Tile $tile) : void{
194 if($tile->isClosed()){
195 throw new \InvalidArgumentException("Attempted to add a garbage closed Tile to a chunk");
196 }
197
198 $pos = $tile->getPosition();
199 if(isset($this->tiles[$index = Chunk::blockHash($pos->x, $pos->y, $pos->z)]) && $this->tiles[$index] !== $tile){
200 throw new \InvalidArgumentException("Another tile is already at this location");
201 }
202 $this->tiles[$index] = $tile;
203 }

339 public static function blockHash(int $x, int $y, int $z) : int{
340 return ($y << (2 * SubChunk::COORD_BIT_SIZE)) |
341 (($z & SubChunk::COORD_MASK) << SubChunk::COORD_BIT_SIZE) |
342 ($x & SubChunk::COORD_MASK);
