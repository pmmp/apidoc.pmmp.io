47 public function setAxis(int $axis) : self{
48 if($axis !== Axis::X && $axis !== Axis::Z){
49 throw new \InvalidArgumentException("Axis must be X or Z only");

55 public function place(BlockTransaction $tx, Item $item, Block $blockReplace, Block $blockClicked, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
56 if(!$this->canBeSupportedBy($tx->fetchBlock($blockReplace->getPosition()->down()))){
57 return false;
58 }
59 if($player !== null){
60 $playerBlockPos = $player->getPosition()->floor();
61 $directionVector = $blockReplace->getPosition()->subtractVector($playerBlockPos)->normalize();
62 $angle = rad2deg(atan2($directionVector->getZ(), $directionVector->getX()));
63
64 if($angle <= 45 || 315 <= $angle || (135 <= $angle && $angle <= 225)){
65 //TODO: This produces Z axis 75% of the time, because any negative angle will produce Z axis.
66 //This is a bug in vanilla. https://bugs.mojang.com/browse/MCPE-125311
67 $this->axis = Axis::Z;
68 }
69 }
70
71 $this->dead = !$this->isCoveredWithWater();
72
73 return parent::place($tx, $item, $blockReplace, $blockClicked, $face, $clickVector, $player);
74 }
75
76 public function onNearbyBlockChange() : void{
77 $world = $this->position->getWorld();
78 if(!$this->canBeSupportedBy($world->getBlock($this->position->down()))){
79 $world->useBreakOn($this->position);
80 }else{
81 parent::onNearbyBlockChange();
82 }
83 }

87 }
