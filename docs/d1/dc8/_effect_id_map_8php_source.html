82 public function register(int $mcpeId, Effect $effect) : void{
83 $this->idToEffect[$mcpeId] = $effect;
84 $this->effectToId[spl_object_id($effect)] = $mcpeId;
85 }
86
87 public function fromId(int $id) : ?Effect{
88 //we might not have all the effect IDs registered
89 return $this->idToEffect[$id] ?? null;
90 }
91
92 public function toId(Effect $effect) : int{
93 if(!array_key_exists(spl_object_id($effect), $this->effectToId)){
94 //this should never happen, so we treat it as an exceptional condition
95 throw new \InvalidArgumentException("Effect does not have a mapped ID");
96 }
97 return $this->effectToId[spl_object_id($effect)];
98 }
