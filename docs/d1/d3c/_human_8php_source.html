122 ($skinDataTag = $skinTag->getTag("Data")) instanceof StringTag ? $skinDataTag->getValue() : $skinTag->getByteArray("Data"), //old data (this used to be saved as a StringTag in older versions of PM)

244 protected function initEntity(CompoundTag $nbt) : void{
245 parent::initEntity($nbt);
246
247 $this->hungerManager = new HungerManager($this);
248 $this->xpManager = new ExperienceManager($this);
249
250 $this->inventory = new PlayerInventory($this);
251 $syncHeldItem = function() : void{
252 foreach($this->getViewers() as $viewer){
253 $viewer->getNetworkSession()->onMobMainHandItemChange($this);
254 }
255 };
256 $this->inventory->getListeners()->add(new CallbackInventoryListener(
257 function(Inventory $unused, int $slot, Item $unused2) use ($syncHeldItem) : void{
258 if($slot === $this->inventory->getHeldItemIndex()){
259 $syncHeldItem();
260 }
261 },
262 function(Inventory $unused, array $oldItems) use ($syncHeldItem) : void{
263 if(array_key_exists($this->inventory->getHeldItemIndex(), $oldItems)){
264 $syncHeldItem();
265 }
266 }
267 ));
268 $this->offHandInventory = new PlayerOffHandInventory($this);
269 $this->enderInventory = new PlayerEnderInventory($this);
270 $this->initHumanData($nbt);
271
272 $inventoryTag = $nbt->getListTag("Inventory");
273 if($inventoryTag !== null){
274 $inventoryItems = [];
275 $armorInventoryItems = [];
276
278 foreach($inventoryTag as $i => $item){
279 $slot = $item->getByte("Slot");
280 if($slot >= 0 && $slot < 9){ //Hotbar
281 //Old hotbar saving stuff, ignore it
282 }elseif($slot >= 100 && $slot < 104){ //Armor
283 $armorInventoryItems[$slot - 100] = Item::nbtDeserialize($item);
284 }elseif($slot >= 9 && $slot < $this->inventory->getSize() + 9){
285 $inventoryItems[$slot - 9] = Item::nbtDeserialize($item);
286 }
287 }
288
289 self::populateInventoryFromListTag($this->inventory, $inventoryItems);
290 self::populateInventoryFromListTag($this->armorInventory, $armorInventoryItems);
291 }
292 $offHand = $nbt->getCompoundTag("OffHandItem");
293 if($offHand !== null){
294 $this->offHandInventory->setItem(0, Item::nbtDeserialize($offHand));
295 }
296 $this->offHandInventory->getListeners()->add(CallbackInventoryListener::onAnyChange(function() : void{
297 foreach($this->getViewers() as $viewer){
298 $viewer->getNetworkSession()->onMobOffHandItemChange($this);
299 }
300 }));
301
302 $enderChestInventoryTag = $nbt->getListTag("EnderChestInventory");
303 if($enderChestInventoryTag !== null){
304 $enderChestInventoryItems = [];
305
307 foreach($enderChestInventoryTag as $i => $item){
308 $enderChestInventoryItems[$item->getByte("Slot")] = Item::nbtDeserialize($item);
309 }
310 self::populateInventoryFromListTag($this->enderInventory, $enderChestInventoryItems);
311 }
312
313 $this->inventory->setHeldItemIndex($nbt->getInt("SelectedInventorySlot", 0));
314 $this->inventory->getHeldItemIndexChangeListeners()->add(function(int $oldIndex) : void{
315 foreach($this->getViewers() as $viewer){
316 $viewer->getNetworkSession()->onMobMainHandItemChange($this);
317 }
318 });
319
320 $this->hungerManager->setFood((float) $nbt->getInt("foodLevel", (int) $this->hungerManager->getFood()));
321 $this->hungerManager->setExhaustion($nbt->getFloat("foodExhaustionLevel", $this->hungerManager->getExhaustion()));
322 $this->hungerManager->setSaturation($nbt->getFloat("foodSaturationLevel", $this->hungerManager->getSaturation()));
323 $this->hungerManager->setFoodTickTimer($nbt->getInt("foodTickTimer", $this->hungerManager->getFoodTickTimer()));
324
325 $this->xpManager->setXpAndProgressNoEvent(
326 $nbt->getInt("XpLevel", 0),
327 $nbt->getFloat("XpP", 0.0));
328 $this->xpManager->setLifetimeTotalXp($nbt->getInt("XpTotal", 0));
329
330 if(($xpSeedTag = $nbt->getTag("XpSeed")) instanceof IntTag){
331 $this->xpSeed = $xpSeedTag->getValue();
332 }else{
333 $this->xpSeed = random_int(Limits::INT32_MIN, Limits::INT32_MAX);
334 }
335 }

350 public function applyDamageModifiers(EntityDamageEvent $source) : void{
351 parent::applyDamageModifiers($source);
352
353 $type = $source->getCause();
354 if($type !== EntityDamageEvent::CAUSE_SUICIDE && $type !== EntityDamageEvent::CAUSE_VOID
355 && ($this->inventory->getItemInHand() instanceof Totem || $this->offHandInventory->getItem(0) instanceof Totem)){
356
357 $compensation = $this->getHealth() - $source->getFinalDamage() - 1;
358 if($compensation < 0){
359 $source->setModifier($compensation, EntityDamageEvent::MODIFIER_TOTEM);
360 }
361 }

364 protected function applyPostDamageEffects(EntityDamageEvent $source) : void{
365 parent::applyPostDamageEffects($source);
366 $totemModifier = $source->getModifier(EntityDamageEvent::MODIFIER_TOTEM);
367 if($totemModifier < 0){ //Totem prevented death
368 $this->effectManager->clear();
369
370 $this->effectManager->add(new EffectInstance(VanillaEffects::REGENERATION(), 40 * 20, 1));
371 $this->effectManager->add(new EffectInstance(VanillaEffects::FIRE_RESISTANCE(), 40 * 20, 1));
372 $this->effectManager->add(new EffectInstance(VanillaEffects::ABSORPTION(), 5 * 20, 1));
373
374 $this->broadcastAnimation(new TotemUseAnimation($this));
375 $this->broadcastSound(new TotemUseSound());
376
377 $hand = $this->inventory->getItemInHand();
378 if($hand instanceof Totem){
379 $hand->pop(); //Plugins could alter max stack size
380 $this->inventory->setItemInHand($hand);
381 }elseif(($offHand = $this->offHandInventory->getItem(0)) instanceof Totem){
382 $offHand->pop();
383 $this->offHandInventory->setItem(0, $offHand);
384 }
385 }

396 public function saveNBT() : CompoundTag{
397 $nbt = parent::saveNBT();
398
399 $nbt->setInt("foodLevel", (int) $this->hungerManager->getFood());
400 $nbt->setFloat("foodExhaustionLevel", $this->hungerManager->getExhaustion());
401 $nbt->setFloat("foodSaturationLevel", $this->hungerManager->getSaturation());
402 $nbt->setInt("foodTickTimer", $this->hungerManager->getFoodTickTimer());
403
404 $nbt->setInt("XpLevel", $this->xpManager->getXpLevel());
405 $nbt->setFloat("XpP", $this->xpManager->getXpProgress());
406 $nbt->setInt("XpTotal", $this->xpManager->getLifetimeTotalXp());
407 $nbt->setInt("XpSeed", $this->xpSeed);
408
409 $inventoryTag = new ListTag([], NBT::TAG_Compound);
410 $nbt->setTag("Inventory", $inventoryTag);
411 if($this->inventory !== null){
412 //Normal inventory
413 $slotCount = $this->inventory->getSize() + $this->inventory->getHotbarSize();
414 for($slot = $this->inventory->getHotbarSize(); $slot < $slotCount; ++$slot){
415 $item = $this->inventory->getItem($slot - 9);
416 if(!$item->isNull()){
417 $inventoryTag->push($item->nbtSerialize($slot));
418 }
419 }
420
421 //Armor
422 for($slot = 100; $slot < 104; ++$slot){
423 $item = $this->armorInventory->getItem($slot - 100);
424 if(!$item->isNull()){
425 $inventoryTag->push($item->nbtSerialize($slot));
426 }
427 }
428
429 $nbt->setInt("SelectedInventorySlot", $this->inventory->getHeldItemIndex());
430 }
431 $offHandItem = $this->offHandInventory->getItem(0);
432 if(!$offHandItem->isNull()){
433 $nbt->setTag("OffHandItem", $offHandItem->nbtSerialize());
434 }
435
436 if($this->enderInventory !== null){
438 $items = [];
439
440 $slotCount = $this->enderInventory->getSize();
441 for($slot = 0; $slot < $slotCount; ++$slot){
442 $item = $this->enderInventory->getItem($slot);
443 if(!$item->isNull()){
444 $items[] = $item->nbtSerialize($slot);
445 }
446 }
447
448 $nbt->setTag("EnderChestInventory", new ListTag($items, NBT::TAG_Compound));
449 }
450
451 if($this->skin !== null){
452 $nbt->setTag("Skin", CompoundTag::create()
453 ->setString("Name", $this->skin->getSkinId())
454 ->setByteArray("Data", $this->skin->getSkinData())
455 ->setByteArray("CapeData", $this->skin->getCapeData())
456 ->setString("GeometryName", $this->skin->getGeometryName())
457 ->setByteArray("GeometryData", $this->skin->getGeometryData())
458 );
459 }
460
461 return $nbt;
462 }

491 array_fill(0, UpdateAbilitiesPacketLayer::NUMBER_OF_ABILITIES, false),
