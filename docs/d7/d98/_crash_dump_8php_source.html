178 private function baseCrash() : void{
179 global $lastExceptionError, $lastError;
180
181 if(isset($lastExceptionError)){
182 $error = $lastExceptionError;
183 }else{
184 $error = error_get_last();
185 if($error === null){
186 throw new \RuntimeException("Crash error information missing - did something use exit()?");
187 }
188 $error["trace"] = Utils::currentTrace(3); //Skipping CrashDump->baseCrash, CrashDump->construct, Server->crashDump
189 $error["fullFile"] = $error["file"];
190 $error["file"] = Filesystem::cleanPath($error["file"]);
191 try{
192 $error["type"] = ErrorTypeToStringMap::get($error["type"]);
193 }catch(\InvalidArgumentException $e){
194 //pass
195 }
196 if(($pos = strpos($error["message"], "\n")) !== false){
197 $error["message"] = substr($error["message"], 0, $pos);
198 }
199 }
200 $error["message"] = mb_scrub($error["message"], 'UTF-8');
201
202 if(isset($lastError)){
203 if(isset($lastError["trace"])){
204 $lastError["trace"] = Utils::printableTrace($lastError["trace"]);
205 }
206 $this->data->lastError = $lastError;
207 $this->data->lastError["message"] = mb_scrub($this->data->lastError["message"], 'UTF-8');
208 }
209
210 $this->data->error = $error;
211 unset($this->data->error["fullFile"]);
212 unset($this->data->error["trace"]);
213
214 $this->data->plugin_involvement = self::PLUGIN_INVOLVEMENT_NONE;
215 if(!$this->determinePluginFromFile($error["fullFile"], true)){ //fatal errors won't leave any stack trace
216 foreach($error["trace"] as $frame){
217 if(!isset($frame["file"])){
218 continue; //PHP core
219 }
220 if($this->determinePluginFromFile($frame["file"], false)){
221 break;
222 }
223 }
224 }
225
226 if($this->server->getConfigGroup()->getPropertyBool("auto-report.send-code", true) && file_exists($error["fullFile"])){
227 $file = @file($error["fullFile"], FILE_IGNORE_NEW_LINES);
228 if($file !== false){
229 for($l = max(0, $error["line"] - 10); $l < $error["line"] + 10 && isset($file[$l]); ++$l){
230 $this->data->code[$l + 1] = $file[$l];
231 }
232 }
233 }
234
235 $this->data->trace = Utils::printableTrace($error["trace"]);
236 }

238 private function determinePluginFromFile(string $filePath, bool $crashFrame) : bool{
239 $frameCleanPath = Filesystem::cleanPath($filePath);
240 if(strpos($frameCleanPath, Filesystem::CLEAN_PATH_SRC_PREFIX) !== 0){
241 if($crashFrame){
242 $this->data->plugin_involvement = self::PLUGIN_INVOLVEMENT_DIRECT;
243 }else{
244 $this->data->plugin_involvement = self::PLUGIN_INVOLVEMENT_INDIRECT;
245 }
246
247 if(file_exists($filePath)){
248 $reflection = new \ReflectionClass(PluginBase::class);
249 $file = $reflection->getProperty("file");
250 $file->setAccessible(true);
251 foreach($this->server->getPluginManager()->getPlugins() as $plugin){
252 $filePath = Filesystem::cleanPath($file->getValue($plugin));
253 if(strpos($frameCleanPath, $filePath) === 0){
254 $this->data->plugin = $plugin->getName();
255 break;
256 }
257 }
258 }
259 return true;
260 }
261 return false;
262 }
