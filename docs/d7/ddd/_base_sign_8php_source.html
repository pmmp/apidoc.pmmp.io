119 public function updateText(Player $author, SignText $text) : bool{
120 $size = 0;
121 foreach($text->getLines() as $line){
122 $size += strlen($line);
123 }
124 if($size > 1000){
125 throw new \UnexpectedValueException($author->getName() . " tried to write $size bytes of text onto a sign (bigger than max 1000)");
126 }
127 $ev = new SignChangeEvent($this, $author, new SignText(array_map(function(string $line) : string{
128 return TextFormat::clean($line, false);
129 }, $text->getLines())));
130 if($this->editorEntityRuntimeId === null || $this->editorEntityRuntimeId !== $author->getId()){
131 $ev->cancel();
132 }
133 $ev->call();
134 if(!$ev->isCancelled()){
135 $this->setText($ev->getNewText());
136 $this->position->getWorld()->setBlock($this->position, $this);
137 return true;
138 }
139
140 return false;
