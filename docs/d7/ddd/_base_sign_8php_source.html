150 public function onInteract(Item $item, int $face, Vector3 $clickVector, ?Player $player = null, array &$returnedItems = []) : bool{
151 if($player === null){
152 return false;
153 }
154 $dyeColor = $item instanceof Dye ? $item->getColor() : match($item->getTypeId()){
155 ItemTypeIds::BONE_MEAL => DyeColor::WHITE(),
156 ItemTypeIds::LAPIS_LAZULI => DyeColor::BLUE(),
157 ItemTypeIds::COCOA_BEANS => DyeColor::BROWN(),
158 default => null
159 };
160 if($dyeColor !== null){
161 $color = $dyeColor->equals(DyeColor::BLACK()) ? new Color(0, 0, 0) : $dyeColor->getRgbValue();
162 if($color->toARGB() === $this->text->getBaseColor()->toARGB()){
163 return false;
164 }
165
166 if($this->doSignChange(new SignText($this->text->getLines(), $color, $this->text->isGlowing()), $player, $item)){
167 $this->position->getWorld()->addSound($this->position, new DyeUseSound());
168 return true;
169 }
170 }elseif($item->getTypeId() === ItemTypeIds::INK_SAC){
171 return $this->changeSignGlowingState(false, $player, $item);
172 }elseif($item->getTypeId() === ItemTypeIds::GLOW_INK_SAC){
173 return $this->changeSignGlowingState(true, $player, $item);
174 }
175 return false;
176 }

210 public function updateText(Player $author, SignText $text) : bool{
211 $size = 0;
212 foreach($text->getLines() as $line){
213 $size += strlen($line);
214 }
215 if($size > 1000){
216 throw new \UnexpectedValueException($author->getName() . " tried to write $size bytes of text onto a sign (bigger than max 1000)");
217 }
218 $ev = new SignChangeEvent($this, $author, new SignText(array_map(function(string $line) : string{
219 return TextFormat::clean($line, false);
220 }, $text->getLines())));
221 if($this->editorEntityRuntimeId === null || $this->editorEntityRuntimeId !== $author->getId()){
222 $ev->cancel();
223 }
224 $ev->call();
225 if(!$ev->isCancelled()){
226 $this->setText($ev->getNewText());
227 $this->setEditorEntityRuntimeId(null);
228 $this->position->getWorld()->setBlock($this->position, $this);
229 return true;
230 }
231
232 return false;
