77 private function runSubprocess(\Threaded $buffer, ?SleeperNotifier $notifier) : void{
78 $server = stream_socket_server("tcp://127.0.0.1:0");
79 if($server === false){
80 throw new \RuntimeException("Failed to open console reader socket server");
81 }
82 $address = Utils::assumeNotFalse(stream_socket_get_name($server, false), "stream_socket_get_name() shouldn't return false here");
83
84 //Windows sucks, and likes to corrupt UTF-8 file paths when they travel to the subprocess, so we base64 encode
85 //the path to avoid the problem. This is an abysmally shitty hack, but here we are :(
86 $sub = Utils::assumeNotFalse(proc_open(
87 [PHP_BINARY, '-dopcache.enable_cli=0', '-r', sprintf('require base64_decode("%s", true);', base64_encode(Path::join(__DIR__, 'ConsoleReaderChildProcess.php'))), $address],
88 [
89 2 => fopen("php://stderr", "w"),
90 ],
91 $pipes
92 ), "Something has gone horribly wrong");
93
94 $client = stream_socket_accept($server, 15);
95 if($client === false){
96 throw new AssumptionFailedError("stream_socket_accept() returned false");
97 }
98 stream_socket_shutdown($server, STREAM_SHUT_RDWR);
99 while(!$this->isKilled){
100 $r = [$client];
101 $w = null;
102 $e = null;
103 if(stream_select($r, $w, $e, 0, 200000) === 1){
104 $command = fgets($client);
105 if($command === false){
106 //subprocess died for some reason; this could be someone killed it manually from outside (e.g.
107 //mistyped PID) or it might be a ctrl+c signal to this process that the child is handling
108 //differently (different signal handlers).
109 //since we have no way to know the difference, we just kill the sub and start a new one.
110 break;
111 }
112
113 $command = preg_replace("#\\x1b\\x5b([^\\x1b]*\\x7e|[\\x40-\\x50])#", "", trim($command)) ?? throw new AssumptionFailedError("This regex is assumed to be valid");
114 $command = preg_replace('/[[:cntrl:]]/', '', $command) ?? throw new AssumptionFailedError("This regex is assumed to be valid");
115 if($command === ""){
116 continue;
117 }
118 $buffer[] = $command;
119 if($notifier !== null){
120 $notifier->wakeupSleeper();
121 }
122 }
123 }
124
125 //we have no way to signal to the subprocess to shut down gracefully; besides, Windows sucks, and the subprocess
126 //gets stuck in a blocking fgets() read because stream_select() is a hunk of junk (hence the separate process in
127 //the first place).
128 proc_terminate($sub);
129 proc_close($sub);
130 stream_socket_shutdown($client, STREAM_SHUT_RDWR);
131 }
