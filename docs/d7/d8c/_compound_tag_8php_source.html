149 private function getTagValue(string $name, string $expectedClass, $default = null){
150 $tag = $this->getTag($name);
151 if($tag instanceof $expectedClass){
152 return $tag->getValue();
153 }
154 if($tag !== null){
155 throw new UnexpectedTagTypeException("Expected a tag of type $expectedClass, got " . get_class($tag));
156 }
157
158 if($default === null){
159 throw new NoSuchTagException("Tag \"$name\" does not exist");
160 }
161
162 return $default;

287 public static function read(NbtStreamReader $reader, ReaderTracker $tracker) : self{
288 $result = new self;
289 $tracker->protectDepth(static function() use($reader, $tracker, $result) : void{
290 for($type = $reader->readByte(); $type !== NBT::TAG_End; $type = $reader->readByte()){
291 $name = $reader->readString();
292 $tag = NBT::createTag($type, $reader, $tracker);
293 if($result->getTag($name) !== null){
294 //this is technically a corruption case, but it's very common on older PM worlds (pretty much every
295 //furnace in PM worlds prior to 2017 is affected), and since we can't extricate this borked data
296 //from the rest in Anvil/McRegion worlds, we can't barf on this - it would result in complete loss
297 //of the chunk.
298 //TODO: add a flag to enable throwing on this (strict mode)
299 continue;
300 }
301 $result->setTag($name, $tag);
302 }
303 });
304 return $result;
305 }

307 public function write(NbtStreamWriter $writer) : void{
308 foreach($this->value as $name => $tag){
309 if(is_int($name)){
310 //PHP sucks
311 //we only cast on seeing an int, because forcibly casting other types might conceal bugs.
312 $name = (string) $name;
313 }
314 $writer->writeByte($tag->getType());
315 $writer->writeString($name);
316 $tag->write($writer);
317 }
318 $writer->writeByte(NBT::TAG_End);
319 }
