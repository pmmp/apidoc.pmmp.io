99 public function register(string $className, array $saveNames = []) : void{
100 Utils::testValidInstance($className, Tile::class);
101
102 $shortName = (new \ReflectionClass($className))->getShortName();
103 if(!in_array($shortName, $saveNames, true)){
104 $saveNames[] = $shortName;
105 }
106
107 foreach($saveNames as $name){
108 $this->knownTiles[$name] = $className;
109 }
110
111 $this->saveNames[$className] = reset($saveNames);
112 }
113
118 public function createFromData(World $world, CompoundTag $nbt) : ?Tile{
119 try{
120 $type = $nbt->getString(Tile::TAG_ID, "");
121 if(!isset($this->knownTiles[$type])){
122 return null;
123 }
124 $class = $this->knownTiles[$type];
125 assert(is_a($class, Tile::class, true));
130 $tile = new $class($world, new Vector3($nbt->getInt(Tile::TAG_X), $nbt->getInt(Tile::TAG_Y), $nbt->getInt(Tile::TAG_Z)));
131 $tile->readSaveData($nbt);
132 }catch(NbtException $e){
133 throw new SavedDataLoadingException($e->getMessage(), 0, $e);
134 }
135
136 return $tile;
137 }

142 public function getSaveId(string $class) : string{
143 if(isset($this->saveNames[$class])){
144 return $this->saveNames[$class];
145 }
146 throw new \InvalidArgumentException("Tile $class is not registered");
147 }
