161 public function getFlowVector() : Vector3{
162 if($this->flowVector !== null){
163 return $this->flowVector;
164 }
165
166 $vX = $vY = $vZ = 0;
167
168 $decay = $this->getEffectiveFlowDecay($this);
169
170 $world = $this->position->getWorld();
171
172 foreach(Facing::HORIZONTAL as $j){
173 $x = $this->position->x;
174 $y = $this->position->y;
175 $z = $this->position->z;
176
177 match($j){
178 Facing::WEST => --$x,
179 Facing::EAST => ++$x,
180 Facing::NORTH => --$z,
181 Facing::SOUTH => ++$z
182 };
183
184 $sideBlock = $world->getBlockAt($x, $y, $z);
185 $blockDecay = $this->getEffectiveFlowDecay($sideBlock);
186
187 if($blockDecay < 0){
188 if(!$sideBlock->canBeFlowedInto()){
189 continue;
190 }
191
192 $blockDecay = $this->getEffectiveFlowDecay($world->getBlockAt($x, $y - 1, $z));
193
194 if($blockDecay >= 0){
195 $realDecay = $blockDecay - ($decay - 8);
196 $vX += ($x - $this->position->x) * $realDecay;
197 $vY += ($y - $this->position->y) * $realDecay;
198 $vZ += ($z - $this->position->z) * $realDecay;
199 }
200
201 continue;
202 }else{
203 $realDecay = $blockDecay - $decay;
204 $vX += ($x - $this->position->x) * $realDecay;
205 $vY += ($y - $this->position->y) * $realDecay;
206 $vZ += ($z - $this->position->z) * $realDecay;
207 }
208 }
209
210 $vector = new Vector3($vX, $vY, $vZ);
211
212 if($this->falling){
213 foreach(Facing::HORIZONTAL as $facing){
214 $pos = $this->position->getSide($facing);
215 if(
216 !$this->canFlowInto($world->getBlockAt($pos->x, $pos->y, $pos->z)) ||
217 !$this->canFlowInto($world->getBlockAt($pos->x, $pos->y + 1, $pos->z))
218 ){
219 $vector = $vector->normalize()->add(0, -6, 0);
220 break;
221 }
222 }
223 }
224
225 return $this->flowVector = $vector->normalize();
226 }

258 public function onScheduledUpdate() : void{
259 $multiplier = $this->getFlowDecayPerBlock();
260
261 $world = $this->position->getWorld();
262
263 if(!$this->isSource()){
264 $smallestFlowDecay = -100;
265 $this->adjacentSources = 0;
266 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x, $this->position->y, $this->position->z - 1), $smallestFlowDecay);
267 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x, $this->position->y, $this->position->z + 1), $smallestFlowDecay);
268 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x - 1, $this->position->y, $this->position->z), $smallestFlowDecay);
269 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x + 1, $this->position->y, $this->position->z), $smallestFlowDecay);
270
271 $newDecay = $smallestFlowDecay + $multiplier;
272 $falling = false;
273
274 if($newDecay > self::MAX_DECAY || $smallestFlowDecay < 0){
275 $newDecay = -1;
276 }
277
278 if($this->getEffectiveFlowDecay($world->getBlockAt($this->position->x, $this->position->y + 1, $this->position->z)) >= 0){
279 $falling = true;
280 }
281
282 $minAdjacentSources = $this->getMinAdjacentSourcesToFormSource();
283 if($minAdjacentSources !== null && $this->adjacentSources >= $minAdjacentSources){
284 $bottomBlock = $world->getBlockAt($this->position->x, $this->position->y - 1, $this->position->z);
285 if($bottomBlock->isSolid() || ($bottomBlock instanceof Liquid && $bottomBlock->hasSameTypeId($this) && $bottomBlock->isSource())){
286 $newDecay = 0;
287 $falling = false;
288 }
289 }
290
291 if($falling !== $this->falling || (!$falling && $newDecay !== $this->decay)){
292 if(!$falling && $newDecay < 0){
293 $world->setBlock($this->position, VanillaBlocks::AIR());
294 return;
295 }
296
297 $this->falling = $falling;
298 $this->decay = $falling ? 0 : $newDecay;
299 $world->setBlock($this->position, $this); //local block update will cause an update to be scheduled
300 }
301 }
302
303 $bottomBlock = $world->getBlockAt($this->position->x, $this->position->y - 1, $this->position->z);
304
305 $this->flowIntoBlock($bottomBlock, 0, true);
306
307 if($this->isSource() || !$bottomBlock->canBeFlowedInto()){
308 if($this->falling){
309 $adjacentDecay = 1; //falling liquid behaves like source block
310 }else{
311 $adjacentDecay = $this->decay + $multiplier;
312 }
313
314 if($adjacentDecay <= self::MAX_DECAY){
315 $calculator = new MinimumCostFlowCalculator($world, $this->getFlowDecayPerBlock(), $this->canFlowInto(...));
316 foreach($calculator->getOptimalFlowDirections($this->position->getFloorX(), $this->position->getFloorY(), $this->position->getFloorZ()) as $facing){
317 $this->flowIntoBlock($world->getBlock($this->position->getSide($facing)), $adjacentDecay, false);
318 }
319 }
320 }
321
322 $this->checkForHarden();
323 }

325 protected function flowIntoBlock(Block $block, int $newFlowDecay, bool $falling) : void{
326 if($this->canFlowInto($block) && !($block instanceof Liquid)){
327 $new = clone $this;
328 $new->falling = $falling;
329 $new->decay = $falling ? 0 : $newFlowDecay;
330
331 $ev = new BlockSpreadEvent($block, $this, $new);
332 $ev->call();
333 if(!$ev->isCancelled()){
334 $world = $this->position->getWorld();
335 if($block->getTypeId() !== BlockTypeIds::AIR){
336 $world->useBreakOn($block->position);
337 }
338
339 $world->setBlock($block->position, $ev->getNewState());
340 }
341 }

345 private function getSmallestFlowDecay(Block $block, int $decay) : int{
346 if(!($block instanceof Liquid) || !$block->hasSameTypeId($this)){
347 return $decay;
348 }
349
350 $blockDecay = $block->decay;
351
352 if($block->isSource()){
353 ++$this->adjacentSources;
354 }elseif($block->falling){
355 $blockDecay = 0;
356 }
357
358 return ($decay >= 0 && $blockDecay >= $decay) ? $decay : $blockDecay;
