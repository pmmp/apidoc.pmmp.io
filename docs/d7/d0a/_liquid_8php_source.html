172 public function getFlowVector() : Vector3{
173 if($this->flowVector !== null){
174 return $this->flowVector;
175 }
176
177 $vX = $vY = $vZ = 0;
178
179 $decay = $this->getEffectiveFlowDecay($this);
180
181 $world = $this->position->getWorld();
182
183 foreach(Facing::HORIZONTAL as $j){
184 $x = $this->position->x;
185 $y = $this->position->y;
186 $z = $this->position->z;
187
188 match($j){
189 Facing::WEST => --$x,
190 Facing::EAST => ++$x,
191 Facing::NORTH => --$z,
192 Facing::SOUTH => ++$z
193 };
194
195 $sideBlock = $world->getBlockAt($x, $y, $z);
196 $blockDecay = $this->getEffectiveFlowDecay($sideBlock);
197
198 if($blockDecay < 0){
199 if(!$sideBlock->canBeFlowedInto()){
200 continue;
201 }
202
203 $blockDecay = $this->getEffectiveFlowDecay($world->getBlockAt($x, $y - 1, $z));
204
205 if($blockDecay >= 0){
206 $realDecay = $blockDecay - ($decay - 8);
207 $vX += ($x - $this->position->x) * $realDecay;
208 $vY += ($y - $this->position->y) * $realDecay;
209 $vZ += ($z - $this->position->z) * $realDecay;
210 }
211
212 continue;
213 }else{
214 $realDecay = $blockDecay - $decay;
215 $vX += ($x - $this->position->x) * $realDecay;
216 $vY += ($y - $this->position->y) * $realDecay;
217 $vZ += ($z - $this->position->z) * $realDecay;
218 }
219 }
220
221 $vector = new Vector3($vX, $vY, $vZ);
222
223 if($this->falling){
224 foreach(Facing::HORIZONTAL as $facing){
225 $pos = $this->position->getSide($facing);
226 if(
227 !$this->canFlowInto($world->getBlockAt($pos->x, $pos->y, $pos->z)) ||
228 !$this->canFlowInto($world->getBlockAt($pos->x, $pos->y + 1, $pos->z))
229 ){
230 $vector = $vector->normalize()->add(0, -6, 0);
231 break;
232 }
233 }
234 }
235
236 return $this->flowVector = $vector->normalize();
237 }

269 public function onScheduledUpdate() : void{
270 $multiplier = $this->getFlowDecayPerBlock();
271
272 $world = $this->position->getWorld();
273
274 if(!$this->isSource()){
275 $smallestFlowDecay = -100;
276 $this->adjacentSources = 0;
277 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x, $this->position->y, $this->position->z - 1), $smallestFlowDecay);
278 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x, $this->position->y, $this->position->z + 1), $smallestFlowDecay);
279 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x - 1, $this->position->y, $this->position->z), $smallestFlowDecay);
280 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x + 1, $this->position->y, $this->position->z), $smallestFlowDecay);
281
282 $newDecay = $smallestFlowDecay + $multiplier;
283 $falling = false;
284
285 if($newDecay >= 8 || $smallestFlowDecay < 0){
286 $newDecay = -1;
287 }
288
289 if($this->getEffectiveFlowDecay($world->getBlockAt($this->position->x, $this->position->y + 1, $this->position->z)) >= 0){
290 $falling = true;
291 }
292
293 $minAdjacentSources = $this->getMinAdjacentSourcesToFormSource();
294 if($minAdjacentSources !== null && $this->adjacentSources >= $minAdjacentSources){
295 $bottomBlock = $world->getBlockAt($this->position->x, $this->position->y - 1, $this->position->z);
296 if($bottomBlock->isSolid() || ($bottomBlock instanceof Liquid && $bottomBlock->isSameType($this) && $bottomBlock->isSource())){
297 $newDecay = 0;
298 $falling = false;
299 }
300 }
301
302 if($falling !== $this->falling || (!$falling && $newDecay !== $this->decay)){
303 if(!$falling && $newDecay < 0){
304 $world->setBlock($this->position, VanillaBlocks::AIR());
305 return;
306 }
307
308 $this->falling = $falling;
309 $this->decay = $falling ? 0 : $newDecay;
310 $world->setBlock($this->position, $this); //local block update will cause an update to be scheduled
311 }
312 }
313
314 $bottomBlock = $world->getBlockAt($this->position->x, $this->position->y - 1, $this->position->z);
315
316 $this->flowIntoBlock($bottomBlock, 0, true);
317
318 if($this->isSource() || !$bottomBlock->canBeFlowedInto()){
319 if($this->falling){
320 $adjacentDecay = 1; //falling liquid behaves like source block
321 }else{
322 $adjacentDecay = $this->decay + $multiplier;
323 }
324
325 if($adjacentDecay < 8){
326 $calculator = new MinimumCostFlowCalculator($this->position->getWorld(), $this->getFlowDecayPerBlock(), \Closure::fromCallable([$this, 'canFlowInto']));
327 foreach($calculator->getOptimalFlowDirections($this->position->getFloorX(), $this->position->getFloorY(), $this->position->getFloorZ()) as $facing){
328 $this->flowIntoBlock($world->getBlock($this->position->getSide($facing)), $adjacentDecay, false);
329 }
330 }
331 }
332
333 $this->checkForHarden();
334 }

336 protected function flowIntoBlock(Block $block, int $newFlowDecay, bool $falling) : void{
337 if($this->canFlowInto($block) && !($block instanceof Liquid)){
338 $new = clone $this;
339 $new->falling = $falling;
340 $new->decay = $falling ? 0 : $newFlowDecay;
341
342 $ev = new BlockSpreadEvent($block, $this, $new);
343 $ev->call();
344 if(!$ev->isCancelled()){
345 if($block->getId() !== BlockLegacyIds::AIR){
346 $this->position->getWorld()->useBreakOn($block->position);
347 }
348
349 $this->position->getWorld()->setBlock($block->position, $ev->getNewState());
350 }
351 }

355 private function getSmallestFlowDecay(Block $block, int $decay) : int{
356 if(!($block instanceof Liquid) || !$block->isSameType($this)){
357 return $decay;
358 }
359
360 $blockDecay = $block->decay;
361
362 if($block->isSource()){
363 ++$this->adjacentSources;
364 }elseif($block->falling){
365 $blockDecay = 0;
366 }
367
368 return ($decay >= 0 && $blockDecay >= $decay) ? $decay : $blockDecay;
