69 protected function generateBaseChunk() : void{
70 $this->chunk = new Chunk([], BiomeArray::fill($this->options->getBiomeId()), false);
71
72 $structure = $this->options->getStructure();
73 $count = count($structure);
74 for($sy = 0; $sy < $count; $sy += SubChunk::EDGE_LENGTH){
75 $subchunk = $this->chunk->getSubChunk($sy >> SubChunk::COORD_BIT_SIZE);
76 for($y = 0; $y < SubChunk::EDGE_LENGTH && isset($structure[$y | $sy]); ++$y){
77 $id = $structure[$y | $sy];
78
79 for($Z = 0; $Z < SubChunk::EDGE_LENGTH; ++$Z){
80 for($X = 0; $X < SubChunk::EDGE_LENGTH; ++$X){
81 $subchunk->setFullBlock($X, $y, $Z, $id);
