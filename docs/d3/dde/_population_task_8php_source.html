71 public function onRun() : void{
72 $context = ThreadLocalGeneratorContext::fetch($this->worldId);
73 if($context === null){
74 throw new AssumptionFailedError("Generator context should have been initialized before any PopulationTask execution");
75 }
76 $generator = $context->getGenerator();

79 $chunk = $this->chunk !== null ? FastChunkSerializer::deserializeTerrain($this->chunk) : null;
80
82 $serialChunks = igbinary_unserialize($this->adjacentChunks);
83 $chunks = array_map(
84 fn(?string $serialized) => $serialized !== null ? FastChunkSerializer::deserializeTerrain($serialized) : null,
85 $serialChunks
86 );
87
88 self::setOrGenerateChunk($manager, $generator, $this->chunkX, $this->chunkZ, $chunk);
89
91 $resultChunks = []; //this is just to keep phpstan's type inference happy
92 foreach($chunks as $relativeChunkHash => $c){
93 World::getXZ($relativeChunkHash, $relativeX, $relativeZ);
94 $resultChunks[$relativeChunkHash] = self::setOrGenerateChunk($manager, $generator, $this->chunkX + $relativeX, $this->chunkZ + $relativeZ, $c);
95 }
96 $chunks = $resultChunks;
97
98 $generator->populateChunk($manager, $this->chunkX, $this->chunkZ);
99 $chunk = $manager->getChunk($this->chunkX, $this->chunkZ);
100 if($chunk === null){
101 throw new AssumptionFailedError("We just generated this chunk, so it must exist");
102 }
103 $chunk->setPopulated();
104
105 $this->chunk = FastChunkSerializer::serializeTerrain($chunk);
106
107 $serialChunks = [];
108 foreach($chunks as $relativeChunkHash => $c){
109 $serialChunks[$relativeChunkHash] = $c->isTerrainDirty() ? FastChunkSerializer::serializeTerrain($c) : null;
110 }
111 $this->adjacentChunks = igbinary_serialize($serialChunks) ?? throw new AssumptionFailedError("igbinary_serialize() returned null");
112 }

114 private static function setOrGenerateChunk(SimpleChunkManager $manager, Generator $generator, int $chunkX, int $chunkZ, ?Chunk $chunk) : Chunk{
115 $manager->setChunk($chunkX, $chunkZ, $chunk ?? new Chunk([], BiomeArray::fill(BiomeIds::OCEAN), false));
116 if($chunk === null){
117 $generator->generateChunk($manager, $chunkX, $chunkZ);
118 $chunk = $manager->getChunk($chunkX, $chunkZ);
119 if($chunk === null){
120 throw new AssumptionFailedError("We just set this chunk, so it must exist");
121 }
122 $chunk->setTerrainDirtyFlag(Chunk::DIRTY_FLAG_BLOCKS, true);
123 $chunk->setTerrainDirtyFlag(Chunk::DIRTY_FLAG_BIOMES, true);
124 }
125 return $chunk;
126 }

128 public function onCompletion() : void{

134
135 $chunk = $this->chunk !== null ?

143 $serialAdjacentChunks = igbinary_unserialize($this->adjacentChunks);

149 }
