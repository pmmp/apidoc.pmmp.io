78 public function writeFacingDirection(int $value) : self{
79 $this->writeInt(BlockStateNames::FACING_DIRECTION, match($value){
80 Facing::DOWN => 0,
81 Facing::UP => 1,
82 Facing::NORTH => 2,
83 Facing::SOUTH => 3,
84 Facing::WEST => 4,
85 Facing::EAST => 5,
86 default => throw new BlockStateSerializeException("Invalid Facing $value")
87 });
88 return $this;
89 }

92 public function writeEndRodFacingDirection(int $value) : self{
93 //end rods are stupid in bedrock and have everything except up/down the wrong way round
94 return $this->writeFacingDirection(Facing::axis($value) !== Axis::Y ? Facing::opposite($value) : $value);
95 }
