96 //TODO: this would be nicer if Vector3::getSide() accepted floats for distance

99 $randomDuration = (($this->flightTimeMultiplier + 1) * 10) + mt_rand(0, 12);

104 (Utils::getRandomFloat() - Utils::getRandomFloat()) * 0.0023,
105 0.05,
106 (Utils::getRandomFloat() - Utils::getRandomFloat()) * 0.0023

115 protected function deserializeCompoundTag(CompoundTag $tag) : void{
116 parent::deserializeCompoundTag($tag);
117
118 $fireworkData = $tag->getCompoundTag(self::TAG_FIREWORK_DATA);
119 if($fireworkData === null){
120 throw new SavedDataLoadingException("Missing firework data");
121 }
122
123 $this->setFlightTimeMultiplier($fireworkData->getByte(self::TAG_FLIGHT_TIME_MULTIPLIER, 1));
124
125 if(($explosions = $fireworkData->getListTag(self::TAG_EXPLOSIONS, CompoundTag::class)) !== null){
126 foreach($explosions as $explosion){
127 $this->explosions[] = FireworkRocketExplosion::fromCompoundTag($explosion);
128 }
129 }
130 }

132 protected function serializeCompoundTag(CompoundTag $tag) : void{

134

136 $fireworkData->setByte(self::TAG_FLIGHT_TIME_MULTIPLIER, $this->flightTimeMultiplier);
137 $fireworkData->setTag(self::TAG_EXPLOSIONS, new ListTag(array_map(fn(FireworkRocketExplosion $e) => $e->toCompoundTag(), $this->explosions)));
138
139 $tag->setTag(self::TAG_FIREWORK_DATA, $fireworkData);
