128 protected function decodePayload(ByteBufferReader $in) : void{
129 for($i = 0, $count = VarInt::readUnsignedInt($in); $i < $count; ++$i){

138 protected function encodePayload(ByteBufferWriter $out) : void{
139 VarInt::writeUnsignedInt($out, count($this->definitionData));

144 VarInt::writeUnsignedInt($out, count($this->strings));
145 foreach($this->strings as $string){
