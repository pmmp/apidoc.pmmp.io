435 public function register(Item $item, bool $override = false) : void{
436 $id = $item->getId();

468 public function get(int $id, int $meta = 0, int $count = 1, ?CompoundTag $tags = null) : Item{
470 $item = null;
471 if($meta !== -1){
472 if(isset($this->list[$offset = self::getListOffset($id, $meta)])){
473 $item = clone $this->list[$offset];
474 }elseif(isset($this->list[$zero = self::getListOffset($id, 0)]) && $this->list[$zero] instanceof Durable){
475 if($meta <= $this->list[$zero]->getMaxDurability()){
476 $item = clone $this->list[$zero];
477 $item->setDamage($meta);
478 }else{
479 $item = new Item(new IID($id, $meta));
480 }
481 }elseif($id < 256){ //intentionally includes negatives, for extended block IDs
482 //TODO: do not assume that item IDs and block IDs are the same or related
483 $item = new ItemBlock(new IID($id, $meta), BlockFactory::getInstance()->get(self::itemToBlockId($id), $meta & 0xf));
484 }
485 }
486
487 if($item === null){
488 //negative damage values will fallthru to here, to avoid crazy shit with crafting wildcard hacks
489 $item = new Item(new IID($id, $meta));
490 }
491
492 $item->setCount($count);
493 if($tags !== null){
494 $item->setNamedTag($tags);
495 }
496 return $item;
497 }

511 if($id < 256){

518 private static function getListOffset(int $id, int $variant) : int{
519 if($id < -0x8000 || $id > 0x7fff){
520 throw new \InvalidArgumentException("ID must be in range " . -0x8000 . " - " . 0x7fff);
521 }
522 return (($id & 0xffff) << 16) | ($variant & 0xffff);
